 18 // load the master sakai object to access all Sakai OAE API methods
 19 require(["jquery", "sakai/sakai.api.core"], function($, sakai){
 20 
 21     /**
 22      * @name sakai_global.featuredpeople
 23      *
 24      * @class featuredpeople
 25      *
 26      * @version 0.0.1
 27      * @param {String} tuid Unique id of the widget
 28      * @param {Boolean} showSettings Show the settings of the widget or not
 29      */
 30     sakai_global.featuredpeople = function(tuid, showSettings, pageData){
 31 
 32         var $rootel = $("#"+tuid);
 33 
 34         // Containers
 35         var $featuredpeopleContainer = $("#featuredpeople_container", $rootel);
 36 
 37         // Templates
 38         var featuredpeopleTemplate = "featuredpeople_template";
 39 
 40         var renderPeople = function(data){
 41             $featuredpeopleContainer.html(sakai.api.Util.TemplateRenderer(featuredpeopleTemplate, {
 42                 "data": data,
 43                 "category": pageData.category,
 44                 "title": pageData.title,
 45                 "sakai": sakai
 46             }));
 47             if (data.total > data.results.length){
 48                 $("#featuredpeople_showall", $rootel).show();
 49             }
 50         };
 51 
 52         var parsePeople = function(success, data){
 53             if (success) {
 54                 $.each(data.results, function(index, item){
 55                     if (item.picture) {
 56                         item.picture = "/~" + sakai.api.Util.safeURL(item.userid) + "/public/profile/" + $.parseJSON(item.picture).name;
 57                     } else {
 58                         item.picture = "/dev/images/default_User_icon_50x50.png";
 59                     }
 60                     item.baseHref = "/~" + sakai.api.Util.safeURL(item.userid);
 61                     if (item.userid === sakai.data.me.user.userid) {
 62                         item.baseHref = "/me";
 63                     }
 64                     item.name = sakai.api.User.getDisplayName(item);
 65                 });
 66                 renderPeople(data);
 67             }
 68         };
 69 
 70         var fetchPeople = function(){
 71             var q = "";
 72             if(pageData){
 73                 q = "directory/" + pageData.category.replace("-", "/");
 74             }
 75             sakai.api.Server.loadJSON("/var/search/bytag.json", parsePeople, {
 76                 page: 0,
 77                 items: 3,
 78                 tag: q,
 79                 type: "u"
 80             });
 81         };
 82 
 83         var doInit = function(){
 84             fetchPeople();
 85         };
 86 
 87         doInit();
 88 
 89     };
 90 
 91     sakai.api.Widgets.widgetLoader.informOnLoad("featuredpeople");
 92 });
