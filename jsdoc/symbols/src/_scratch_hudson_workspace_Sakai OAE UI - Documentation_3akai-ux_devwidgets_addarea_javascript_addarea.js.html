245         /*
246          * Create a new Sakai Document
247          * @param {String} title Title of the Sakai Document
248          * @param {String} permission ID of the permissions to be set
249          * @param {String} docStructure Default content to be set on the Sakai Doc
250          * @param {String|Boolean} preferredTitle Used to make a unique URL, can also be set to false 
251          * @param {Object} widgetContents Contains data about any widgets on the page
252          * @param {Boolean} nonEditable Defines if the Sakai doc will be editable or not
253          * @param {Function} callback Function to be executed after setting (or failing to set) the permissions
254          */
255         var createSakaiDoc = function(title, permission, docStructure, preferredTitle, nonEditable, callback) {
256             var realPermission = permission;
257             if (permission === 'advanced') {
258                 realPermission = 'private';
259             }
260             var parameters = {
261                 'sakai:pooled-content-file-name': title,
262                 'sakai:description': '',
263                 'sakai:permissions': permission,
264                 'sakai:copyright': 'creativecommons',
265                 'sakai:schemaversion': sakai.config.schemaVersion,
266                 'mimeType': 'x-sakai/document'
267             };
268 
269             // Prepare Sakai Doc
270             var structure0 = {},
271                 toCreate = {};
272             for (var i = 0; i < docStructure.length; i++) {
273                 var pageID = '',
274                     pageTitle = '';
275 
276                 // for multi-page creation, we don't care as much about the URL
277                 // since they're auto-generated pages
278                 if (docStructure.length > 1) {
279                     pageTitle = 'Page ' + (i+1);
280                     // if we wanted to make the URL names 'Page 1', etc, we'd make
281                     // that change here by..
282                     //   pageID = sakai.api.Util.makeSafeURL(pageId);
283                     // buf I figure since it's expected that the user will change these
284                     // pages names an id might be better in the URL at least until
285                     // we allow changing the lhnav title to change the url/data store of the page
286                     pageID = sakai.api.Util.generateWidgetId();
287                 // if we have a preferred title, let's use that
288                 } else if (preferredTitle) {
289                     pageTitle = title;
290                     pageID = sakai.api.Util.makeUniqueURL(preferredTitle, title, sakai_global.group.pubdata.structure0);
291                 // otherwise use the title of the page for the URL
292                 } else {
293                     pageTitle = title;
294                     pageID = sakai.api.Util.makeUniqueURL(title, null, sakai_global.group.pubdata.structure0);
295                 }
296                 var refID = sakai.api.Util.generateWidgetId();
297 
298                 structure0[pageID] = {
299                     '_title': pageTitle,
300                     '_order': i,
301                     '_ref': refID,
302                     '_nonEditable': nonEditable,
303                     'main': {
304                         '_title': pageTitle,
305                         '_order': 0,
306                         '_ref': refID,
307                         '_nonEditable': nonEditable
308                     }
309                 };
310                 toCreate[refID] = docStructure[i];
311             }
312 
313             parameters['structure0'] = JSON.stringify(structure0);
314             $.ajax({
315                 url: '/system/pool/createfile',
316                 type:'POST',
317                 data: parameters,
318                 dataType: 'json',
319                 success: function(data) {
320                     var poolId = data._contentItem.poolId;
321                     var struct = $.parseJSON(data._contentItem.item.structure0);
322                     var itemURLName = '';
323                     if (docStructure.length === 1) {
324                         for (itemURLName in struct) {
325                             if (struct.hasOwnProperty(itemURLName)) {
326                                 break;
327                             }
328                         }
329                     } else {
330                         itemURLName = sakai.api.Util.makeSafeURL(title);
331                     }
332                     var batchRequests = [];
333                     $.each(struct, function(i, obj) {
334                         batchRequests.push({
335                             url: '/p/' + poolId + '/' + obj._ref + '.save.json',
336                             method: 'POST',
337                             parameters: {
338                                 'sling:resourceType': 'sakai/pagecontent',
339                                 'sakai:pagecontent': JSON.stringify(toCreate[obj._ref]),
340                                 '_charset_': 'utf-8'
341                             }
342                         });
343                     });
344                     sakai.api.Server.saveJSON('/p/' + poolId, toCreate, function(success2, data2) {
345                         sakai.api.Server.batch(batchRequests, function(success3, data3) {
346                             if (success3) {
347                                 callback(poolId, itemURLName);
348                             }
349                         });
350                     });
351                 }
352             });
353         };

363         /*
364          * Set permissions on an existing Sakai Document
365          * @param {String} urlName Title of the Sakai Document
366          * @param {String} poolId ID of the document saved in the pool
367          * @param {String} docPermission ID of the permissions to be set
368          * @param {Boolean} existingNotMine Determines if the settings should be applied to an existing Sakai Doc that I own
369          * @param {Function} callback Function to be executed after setting (or failing to set) the permissions
370          */
371         var setSakaiDocPermissions = function(urlName, poolId, docPermission, existingNotMine, callback){
372             var filesArray = {};
373             var permission = "private";
374             if (docPermission === "public"){
375                 permission = "public";
376             } else if (docPermission === "everyone"){
377                 permission = "everyone";
378             }
379             filesArray[urlName] = {
380                 "hashpath": poolId,
381                 "permissions": permission
382             };
383             var viewRoles = [];
384             var editRoles = [];
385             var roles = fetchGroupRoles();
386             for (var i = 0; i < roles.length; i++){
387                 var role = roles[i];
388                 if (role.isManagerRole){
389                     editRoles.push(role.id);
390                 } else {
391                     viewRoles.push(role.id);
392                 }
393             }
394             if (existingNotMine) {
395                 var batchRequests = [];
396                 for (var j = 0; j < editRoles.length; j++) {
397                     batchRequests.push({
398                         "url": "/p/" + poolId + ".members.html",
399                         "method": "POST",
400                         "parameters": {
401                             ":viewer": sakai_global.group.groupId + "-" + editRoles[j]
402                         }
403                     });
404                 }
405                 if (docPermission !== "advanced") {
406                     for (var k = 0; k < viewRoles.length; k++) {
407                         batchRequests.push({
408                             "url": "/p/" + poolId + ".members.html",
409                             "method": "POST",
410                             "parameters": {
411                                 ":viewer": sakai_global.group.groupId + "-" + viewRoles[k]
412                             }
413                         });
414                     }
415                 }
416                 sakai.api.Server.batch(batchRequests, function(success, data){
417                     if (success) {
418                         callback(poolId);
419                     }
420                  });
421             } else {
422                 sakai.api.Content.setFilePermissions(filesArray, function(){
423                     var batchRequests = [];
424                     for (var l = 0; l < editRoles.length; l++) {
425                         batchRequests.push({
426                             "url": "/p/" + poolId + ".members.html",
427                             "method": "POST",
428                             "parameters": {
429                                 ":manager": sakai_global.group.groupId + "-" + editRoles[l]
430                             }
431                         });
432                     }
433                     if (docPermission !== "advanced") {
434                         for (var m = 0; m < viewRoles.length; m++) {
435                             batchRequests.push({
436                                 "url": "/p/" + poolId + ".members.html",
437                                 "method": "POST",
438                                 "parameters": {
439                                     ":viewer": sakai_global.group.groupId + "-" + viewRoles[m]
440                                 }
441                             });
442                         }
443                     }
444                     sakai.api.Server.batch(batchRequests, function(success, data){
445                         if (success) {
446                             callback(poolId);
447                         }
448                     });
449                 });
450             }
451         };

453         /*
454          * Add a Sakai Doc to a world
455          * @param {String} urlName Safe title of the Sakai Document as it's stored inside of the system
456          * @param {String} poolId ID of the document saved in the pool
457          * @param {String} docTitle Title of the Sakai Document
458          * @param {String} docPermission ID of the permissions to be set
459          * @param {Boolean} nonEditable Defines if the Sakai doc will be editable or not
460          * @param {Boolean} existingNotMine Determines if the settings should be applied to an existing Sakai Doc that I own
461          * @param {Function} callback Function to be executed after setting (or failing to set) the permissions
462          */
463         var addSakaiDocToWorld = function(urlName, poolId, docTitle, docPermission, nonEditable, existingNotMine, callback){
464             // Refetch docstructure information
465             $.ajax({
466                  url: "/~" + sakai_global.group.groupId + "/docstructure.infinity.json",
467                  cache: false,
468                  success: function(data){
469 
470                     var pubdata = sakai.api.Server.cleanUpSakaiDocObject(data);
471                     var newView = [];
472                     var newEdit = [];
473 
474                     if (docPermission === "public"){
475                         newView.push("everyone");
476                         newView.push("anonymous");
477                     } else if (docPermission === "everyone"){
478                         newView.push("everyone");
479                     }
480 
481                     var roles = fetchGroupRoles();
482                     for (var i = 0; i < roles.length; i++){
483                         var role = roles[i];
484                         if (role.isManagerRole){
485                             if (existingNotMine) {
486                                 newView.push("-" + role.id);
487                             } else {
488                                 newEdit.push("-" + role.id);
489                             }
490                         } else if (docPermission !== "advanced"){
491                             newView.push("-" + role.id);
492                         }
493                     }
494 
495                     pubdata.structure0[urlName] = {
496                         "_title": docTitle,
497                         "_order": _.size(pubdata.structure0),
498                         "_pid": poolId,
499                         "_view": JSON.stringify(newView),
500                         "_edit": JSON.stringify(newEdit),
501                         "_nonEditable": nonEditable
502                     };
503 
504                     // Store view and edit roles
505                     sakai_global.group.pubdata.structure0 = pubdata.structure0;
506                     sakai.api.Server.saveJSON("/~" + sakai_global.group.groupId + "/docstructure", {
507                         "structure0": JSON.stringify(pubdata.structure0)
508                     }, function(){
509                         callback(poolId, urlName);
510                     });
511                 }
512             });
513         };

741                 q: query,
742                 items: 50

754                 var sortOrder = $(".addarea_existing_container:visible").find(".addarea_existing_sort").val();
755                 data.results.sort(dateSort);
756                 if (sortOrder === "desc"){
757                     data.results.reverse();
758                 }
759                 // Check which items I can manage
760                 for (var i = 0; i < data.results.length; i++) {
761                     var manager = false;
762                     var managers = data.results[i]["sakai:pooled-content-manager"];
763                     for (var m = 0; m < managers.length; m++) {
764                         if (managers[m] === sakai.data.me.user.userid ||
765                         sakai.api.Groups.isCurrentUserAMember(managers[m], sakai.data.me)) {
766                             manager = true;
767                         }
768                     }
769                     data.results[i].canManage = manager;
770                 }
771                 var container = "#addarea_existing_everywhere_bottom";
772                 var context = "everywhere";
773                 if (library) {
774                     container = "#addarea_existing_mylibrary_bottom";
775                     context = "my_library";
776                 }
777                 // Render the results
778                 $(container).html(sakai.api.Util.TemplateRenderer("addarea_existing_bottom_template", {
779                         data: data.results,
780                         "context": context
781                     })
782                 );
