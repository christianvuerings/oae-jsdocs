 18 // load the master sakai object to access all Sakai OAE API methods
 19 require(["jquery", "sakai/sakai.api.core"], function($, sakai){
 20 
 21     /**
 22      * @name sakai_global.categories
 23      *
 24      * @class categories
 25      *
 26      * @version 0.0.1
 27      * @param {String} tuid Unique id of the widget
 28      * @param {Boolean} showSettings Show the settings of the widget or not
 29      */
 30     sakai_global.categories = function(tuid, showSettings){
 31 
 32         // Containers
 33         var $categoriesItemsContainer = $("#categories_items_container");
 34 
 35         // Templates
 36         var categoriesItemsTemplate = "categories_items_template";
 37 
 38         // Elements
 39         var $categoriesExpandContract = $("#categories_expand_contract");
 40 
 41         var directory = sakai.config.Directory;
 42         var categoriesToRender = [];
 43 
 44         /////////////
 45         // BINDING //
 46         /////////////
 47 
 48         /**
 49          * Expand or collapse the widget
 50          */
 51         var toggleWidgetvisibility = function(){
 52             $categoriesItemsContainer.toggle("display");
 53             $categoriesExpandContract.children("div").toggle();
 54         };
 55 
 56         var addBinding = function(){
 57             $categoriesExpandContract.bind("click", toggleWidgetvisibility);
 58         };
 59 
 60         /**
 61          * Add binding to the carousel action buttons after rendering and initializing the carousel
 62          * @param {Object} carousel reference to the carousel instance
 63          */
 64         var carouselBinding = function(carousel){
 65             $(".categories_items_scroll_scrollbutton.categories_items_scroll_deselected, #categories_view_next_raquo").live("click", function(){
 66                 $(".categories_items_scroll_scrollbutton.categories_items_scroll_deselected, #categories_view_next_raquo").die("click");
 67                 var clickedId = parseInt($(this)[0].id.split("scroll_")[1], 10);
 68                 if (clickedId) {
 69                     carousel.scroll(clickedId);
 70                 }
 71                 else {
 72                     carousel.next();
 73                 }
 74                 if ($(this)[0].id !== "categories_view_next_raquo") {
 75                     $(".categories_items_scroll_scrollbutton").addClass("categories_items_scroll_deselected");
 76                     $(".categories_items_scroll_selected").removeClass("categories_items_scroll_selected");
 77                     $(this).removeClass("categories_items_scroll_deselected");
 78                     $(this).addClass("categories_items_scroll_selected");
 79                 }
 80                 else {
 81                     var $this = $(".categories_items_scroll_selected");
 82                     if ($this.next().length) {
 83                         var $next = $this.next();
 84                         $next.addClass("categories_items_scroll_selected");
 85                         $next.removeClass("categories_items_scroll_deselected");
 86                     } else {
 87                         var $first = $(".categories_items_scroll_deselected:first");
 88                         $first.addClass("categories_items_scroll_selected");
 89                         $first.removeClass("categories_items_scroll_deselected");
 90                     }
 91                     $this.removeClass("categories_items_scroll_selected");
 92                     $this.addClass("categories_items_scroll_deselected");
 93                 }
 94                 return false;
 95             });
 96         };
 97 
 98         ////////////////////////////
 99         // CAROUSEL AND RENDERING //
100         ////////////////////////////
101 
102         /**
103          * Initialize the carousel after rendering the items
104          */
105         var addCarousel = function(){
106             $categoriesItemsContainer.jcarousel({
107                 animation: "slow",
108                 easing: "swing",
109                 scroll: 4,
110                 wrap: "circular",
111                 itemFirstInCallback: carouselBinding
112             });
113             $categoriesItemsContainer.css("display", "none");
114         };
115 
116         var renderCategories = function(){
117             sakai.api.Util.TemplateRenderer(categoriesItemsTemplate, {
118                 "directory": categoriesToRender,
119                 "sakai": sakai
120             }, $categoriesItemsContainer);
121             addCarousel();
122             $(".categories_widget").css("visibility", "visible");
123         };
124 
125         /**
126          * Parse the directory structure and extract some information from the featured content
127          * @param {Object} success true or false depending on the success of loading the featured content
128          * @param {Object} data contains featured content data
129          */   
130         var parseDirectory = function(success, data){
131             $.each(directory, function(i, toplevel){
132                 if (toplevel.divider){
133                     var toFillOut = 4 - (categoriesToRender.length % 4);
134                     if (toFillOut !== 4) {
135                         categoriesToRender.push({"spacer": true});
136                     }
137                 } else {
138                     toplevel.count = 0;
139                     if (data[i] && data[i].content && !$.isEmptyObject(data[i].content)){
140                         toplevel.content = data[i].content;
141                         toplevel.content.usedin = sakai.api.Content.getPlaceCount(toplevel.content);
142                         toplevel.content.commentcount = sakai.api.Content.getCommentCount(toplevel.content);
143                         toplevel.count = data[i]["sakai:tag-count"];
144                         var mimeType = sakai.api.Content.getMimeType(data[i].content);
145                         toplevel.mimeTypeDescription = sakai.api.i18n.getValueForKey(sakai.config.MimeTypes["other"].description);
146                         if (sakai.config.MimeTypes[mimeType]){
147                             toplevel.mimeTypeDescription = sakai.api.i18n.getValueForKey(sakai.config.MimeTypes[mimeType].description);
148                         }
149                     }
150                     toplevel.id = i;
151                     categoriesToRender.push(toplevel);
152                 }
153             });
154             var toFillOutExtra = 4 - (categoriesToRender.length % 4);
155             if (toFillOutExtra % 4) {
156                 for (var i = 0; i < toFillOutExtra; i++) {
157                     categoriesToRender.push({'spacer': true});
158                 }
159             }
160             renderCategories();
161         };
162 
163         /**
164          * Get a feed of content to display in the carousel
165          */
166         var getCategoryContent = function(){
167             sakai.api.Server.loadJSON("/tags/directory.tagged.json", parseDirectory, {});
168         };
169 
170         ////////////////
171         // INITIALIZE //
172         ////////////////
173 
174         var doInit = function(){
175             addBinding();
176             getCategoryContent();
177         };
178 
179         doInit();
180 
181     };
182 
183     sakai.api.Widgets.widgetLoader.informOnLoad("categories");
184 });
