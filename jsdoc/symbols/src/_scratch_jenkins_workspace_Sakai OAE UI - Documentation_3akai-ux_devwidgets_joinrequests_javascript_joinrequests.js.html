 23 /*global $ */
 24 
 25 require(["jquery", "sakai/sakai.api.core", "underscore"], function($, sakai, _) {
 26 
 27     /**
 28      * @name sakai_global.joinrequests
 29      *
 30      * @class joinrequests
 31      *
 32      * @description
 33      * The Join Requests widget manages join requests for a specific group. Its
 34      * primary placement is at the top of group_edit.html
 35      *
 36      * @version 0.0.1
 37      * @param {String} tuid Unique id of the widget
 38      * @param {Boolean} showSettings Show the settings of the widget or not
 39      */
 40     sakai_global.joinrequests = function (tuid, showSettings) {
 41 
 42         /////////////////////////////
 43         // Configuration variables //
 44         /////////////////////////////
 45 
 46         var numJoinrequests = 0,  // keeps track of the total number of requests
 47             groupid = "",
 48             joinGroupID = "",
 49             joinRole = "",
 50             groupData = {};
 51 
 52         // DOM elements
 53         var $rootel = $("#" + tuid);
 54         var $joinrequestsWidget = $(".joinrequests_widget", $rootel);
 55         var $joinrequests = $("#joinrequests_joinrequests", $rootel);
 56         var $joinrequestsTitle = $("#joinrequests_title", $rootel);
 57         var $joinrequestsError = $("#joinrequests_error", $rootel);
 58         var $joinrequestsSuccess = $("#joinrequests_success", $rootel);
 59         var $joinrequestsTemplate = $("#joinrequests_template", $rootel);
 60         var $addLink = $(".joinrequests_add_link");
 61         var $ignoreLink = $(".joinrequests_ignore_link");
 62         var $joinrequests_container = $("#joinrequests_container");
 63 
 64 
 65         /**
 66          * Renders the joinrequests widget with the given joinrequests array
 67          *
 68          * @param {Array} joinrequests Array of joinrequest Objects to display,
 69          * formatted for the UI.  Each joinrequest Object should contain:
 70          * {
 71          *    "userid": <userid>,
 72          *    "firstName": <user's first name>,
 73          *    "lastName": <user's last name>,
 74          *    "request_age": <request create-date, JS Date.toLocaleString() value>
 75          * }
 76          */
 77         var renderJoinRequests = function(joinrequests) {
 78             if (joinrequests) {
 79                 // populate template with data
 80                 var json = {
 81                     "joinrequests": joinrequests,
 82                     "joinrole": joinRole
 83                 };
 84                 $joinrequests.html(sakai.api.Util.TemplateRenderer($joinrequestsTemplate, json));
 85                 // show the widget
 86                 $joinrequestsWidget.show();
 87             }
 88         };
 89 
 90         /**
 91          * Fetches join requests from the server
 92          */
 93         var getJoinRequestsData = function(joinGroupID) {
 94             // get join requests from server
 95             sakai.api.Groups.getJoinRequests(joinGroupID, function (success, data) {
 96                 var joinrequests = [];
 97                 if (success) {
 98                     // process joinrequest data for UI
 99                     if (data && data.total && data.total > 0) {
100                         numJoinrequests = data.total;
101                         for (var i in data.results) {
102                             if (data.results.hasOwnProperty(i)) {
103                                 var jr = data.results[i];
104                                 var displayName = sakai.api.User.getDisplayName(jr);
105                                 if (automaticallyAcceptUser(jr.userid)) {
106                                     addUser(jr.userid, displayName);
107                                 } else {
108                                     joinrequests.push({
109                                         "userid": jr.userid,
110                                         "displayName": displayName,
111                                         "request_age": $.timeago(jr["_created"]),
112                                         "pic_src": sakai.api.User.getProfilePicture(jr)
113                                     });
114                                 }
115                             }
116                         }
117                     }
118                 }
119                 renderJoinRequests(joinrequests);
120             });
121         };
122 
123 
124         /**
125          * Returns whether or not the given user should automatically be accepted
126          * based on the 'accept' URL param.
127          *
128          * @param {String} userid The ID of the user that may need to be
129          * automatically accepted
130          * @return {Boolean} true if the given user's join request should be
131          * automatically accepted, false otherwise
132          */
133         var automaticallyAcceptUser = function(userid) {
134             if (_.isString(userid)) {
135                 return $.bbq.getState('accept') === userid;
136             } else {
137                 return false;
138             }
139         };
140 
141         // TODO when adding users we should refresh the entity widget so it stays
142         // in sync with the new participant count
143         var resetEntityCounts = function() {
144         };
145 
146         /**
147          * Adds a user to the current group
148          *
149          * @param {String} userid The ID of the user to add to the current group
150          */
151         var addUser = function(userid, displayName) {
152             // add user to group
153             var userToAdd = {
154                 "user": userid,
155                 "permission": groupData["sakai:joinRole"]
156             };
157             sakai.api.Groups.addUsersToGroup(groupid, [userToAdd], sakai.data.me, false, function(success) {
158                 if (success) {
159                     // show notification
160                     var name = displayName;
161                     if (!name) {
162                         name = $("#joinrequests_username_link_" + userid).html();
163                     }
164                     sakai.api.Util.notification.show($joinrequestsTitle.html(), name + " " + $joinrequestsSuccess.html());
165 
166                     // trigger the participants list to refresh
167                     $(window).trigger("usersselected.addpeople.sakai", [userToAdd]);
168 
169                     // remove join request from UI and server
170                     removeJoinRequest(userid);
171                 } else {
172                     sakai.api.Util.notification.show($joinrequestsTitle.html(), $joinrequestsError.html());
173                 }
174             });
175         };
176 
177 
178         /**
179          * Removes a join request from the UI and the server
180          *
181          * @param {String} userid The ID of the user whose join request to remove
182          */
183         var removeJoinRequest = function(userid) {
184             // remove join request from server
185             sakai.api.Groups.removeJoinRequest(userid, joinGroupID, function(success) {
186                 if (success) {
187                     $("#joinrequests_loading_" + userid).hide();
188                     // remove the UI joinrequest element
189                     $("#joinrequests_joinrequest_" + userid).fadeOut(function() {
190                         $(this).remove();
191                         numJoinrequests -= 1;
192                         if (numJoinrequests === 0) {
193                             sakai.api.Util.Modal.close($joinrequests_container);
194                         }
195                     });
196                 } else {
197                     sakai.api.Util.notification.show($joinrequestsTitle.html(), $joinrequestsError.html());
198                     hideSpinner(userid);
199                 }
200             });
201         };
202 
203 
204         /**
205          * Shows an AJAX loading spinner while the user's join request is processed
206          *
207          * @param {String} userid The ID of the user whose join request is being processed
208          */
209         var showSpinner = function(userid) {
210             $("#joinrequests_actions_" + userid).hide();
211             $("#joinrequests_loading_" + userid).show();
212         };
213 
214 
215         /**
216          * Hides the AJAX loading spinner once the user's join request is processed
217          *
218          * @param {String} userid The ID of the user whose join request has been processed
219          */
220         var hideSpinner = function(userid) {
221             $("#joinrequests_loading_" + userid).hide();
222             $("#joinrequests_actions_" + userid).show();
223         };
224 
225         /**
226          * Callback for onHide for the JQM
227          */
228         var handleJQMHide = function(h){
229             resetEntityCounts();
230             h.w.hide();
231             if (h.o) {
232                 h.o.remove();
233             }
234         };
235 
236         /////////////////////////////
237         // Event Bindings          //
238         /////////////////////////////
239 
240         // Add the specific user when the 'Add as a member' link is clicked
241         $addLink.live("click", function() {
242             var userid = this.id.split("_")[2];
243             showSpinner(userid);
244             addUser(userid);
245         });
246 
247         // Ignore the specific user when the 'Ignore' link is clicked
248         $ignoreLink.live("click", function() {
249             var userid = this.id.split("_")[2];
250             showSpinner(userid);
251             removeJoinRequest(userid);
252         });
253 
254 
255         /////////////////////////////
256         // Initialization          //
257         /////////////////////////////
258 
259         /**
260          * Initialize the modal dialog
261          */
262         var initializeJQM = function(){
263             sakai.api.Util.Modal.setup($joinrequests_container, {
264                 modal: true,
265                 overlay: 20,
266                 toTop: true,
267                 onHide: handleJQMHide
268             });
269         };
270 
271         var getJoinRoleTitle = function() {
272             var roles = $.parseJSON(groupData["sakai:roles"]),
273                 ret = "";
274             $.each(roles, function(i, role) {
275                 if (role.id === groupData["sakai:joinRole"]) {
276                     ret = role.titlePlural;
277                 }
278             });
279             return sakai.api.i18n.getValueForKey(ret);
280         };
281 
282         /**
283          * Initialization function run when the widget loads
284          */
285         var init = function() {
286             initializeJQM();
287             // _groupdata should be the group's authprofile
288             var _groupdata = sakai_global.group.groupData;
289             if (_groupdata && _groupdata['sakai:group-id']) {
290                 groupData = _groupdata;
291                 groupid = groupData['sakai:group-id'];
292                 if (groupData['sakai:joinRole']) {
293                     joinRole = getJoinRoleTitle();
294                     joinGroupID = groupid + '-' + groupData['sakai:joinRole'];
295                 }
296                 // get join request data
297                 getJoinRequestsData(joinGroupID);
298                 sakai.api.Util.Modal.open($joinrequests_container);
299             } else {
300                 debug.warn('The group\'s authprofile node wasn\'t passed in to init.joinrequests.sakai');
301             }
302         };
303 
304         $('.sakai_joinrequests_overlay').on('click', init);
305 
306         init();
307     };
308 
309     sakai.api.Widgets.widgetLoader.informOnLoad("joinrequests");
310 });
