 24 require(["jquery", "sakai/sakai.api.core"], function($, sakai) {
 25     /**
 26      * @name sakai_global.footer
 27      *
 28      * @class footer
 29      *
 30      * @description
 31      * Initialize the footer widget
 32      *
 33      * @version 0.0.1
 34      * @param {String} tuid Unique id of the widget
 35      * @param {Boolean} showSettings Show the settings of the widget or not
 36      */
 37     sakai_global.footer = function(tuid,showSettings){
 38 
 39 
 40         /////////////////////////////
 41         // Configuration variables //
 42         /////////////////////////////
 43 
 44         var doc_name;
 45         var $footer_debug_info = $("#footer_debug_info");
 46         var $footer_date_end = $("#footer_date_end");
 47         var $footer_root = $(".footer_main");
 48         var $footer_logo = $("#footer_logo_button");
 49         var $footer_www = $("#footer_www");
 50         var $footer_divider = $("#footer_divider");
 51         var $footer_phone = $("#footer_phone");
 52         var $footer_contactinfo = $("#footer_contactinfo");
 53         var $footer_contactinfo_template = $("#footer_contactinfo_template");
 54         var $footer_links_left = $("#footer_links");
 55         var $footer_links_left_template = $("#footer_links_template");
 56         var $footer_links_right = $("#footer_links_right");
 57         var $footer_links_right_template = $("#footer_links_right_template");
 58 
 59         //////////////////////
 60         // Helper functions //
 61         //////////////////////
 62 
 63         /**
 64          * This helper function will return the name of the current document (e.g. my_sakai.html)
 65          * @return {String} The name of the current document
 66          */
 67         var getDocName = function() {
 68             var url = document.URL;
 69             var slash = "/";
 70             if (url.match(/\\/)) {
 71                 slash = "\\";
 72             }
 73             return decodeURIComponent(url.substring(url.lastIndexOf(slash) + 1));
 74         };
 75 
 76         ////////////////////
 77         // Main functions //
 78         ////////////////////
 79 
 80         /**
 81          * Render the debug info
 82          * @param {Object} container jQuery selector where you want the debug info to appear in
 83          */
 84         var renderDebugInfo = function(container) {
 85 
 86             $.ajax({
 87                 url: "/var/scm-version.json",
 88                 type: "GET",
 89                 cache: false,
 90                 dataType: "json",
 91                 success: function(data){
 92                     // Construct debug info
 93                     var debug_text = "DEBUG:";
 94                     debug_text += " Nakamura Version: " + data["sakai:nakamura-version"];
 95                     getUxVersion(debug_text, container);
 96                 }
 97             });
 98         };
 99 
100         var getUxVersion = function(debug_text, container) {
101             $.ajax({
102                 url: "/var/ux-version/ux-version.json",
103                 type: "GET",
104                 cache: false,
105                 dataType: "json",
106                 success: function(data){
107                     debug_text += " | UX Version: " + data["sakai:ux-version"];
108                     debug_text += "<br/>DOC mod date: " + document.lastModified;
109                     debug_text += " | PLACE: " + (doc_name || "index.html");
110 
111                     // Put text into holding tag
112                     container.html(sakai.api.Security.saneHTML(debug_text));
113                 }
114             });
115         };
116 
117         var updateLocationLanguage = function(){
118             if (!sakai.data.me.user.anon) {
119                 $("#footer_location").text(sakai.data.me.user.locale.timezone.name);
120                 for (var i = 0, len = sakai.config.Languages.length; i < len; i++) {
121                     if (sakai.data.me.user.locale.country === sakai.config.Languages[i].country) {
122                         $("#footer_language").text(sakai.config.Languages[i].displayName);
123                         break;
124                     }
125                 }
126             }
127         };
128 
129 
130         /////////////////////////////
131         // Initialisation function //
132         /////////////////////////////
133 
134         /**
135          * Main initialization function for the footer widget
136          */
137         var doInit = function(){
138 
139             // Get the name of the current document
140             doc_name = getDocName();
141 
142             // Display debug info if set in config
143             if (sakai.config.displayDebugInfo === true) {
144 
145                 // Add binding to the image
146                 $footer_logo.toggle(function(){
147 
148                     // Render the debug info
149                     renderDebugInfo($footer_debug_info);
150 
151                     // Show the debug info
152                     $footer_debug_info.show();
153 
154                     // Update button title
155                     $footer_logo.attr('title', sakai.api.i18n.getValueForKey('HIDE_DEBUG_INFO', 'footer'));
156 
157                 },function(){
158 
159                     // Hide the debug info
160                     $footer_debug_info.hide();
161 
162                     // Update button title
163                     $footer_logo.attr('title', sakai.api.i18n.getValueForKey('SHOW_DEBUG_INFO', 'footer'));
164 
165                 }).addClass("footer_clickable");
166 
167             } else {
168                 // Disable and remove button title
169                 $footer_logo.removeAttr('title');
170                 $footer_logo.attr('disabled', 'disabled');
171             }
172 
173             if (!sakai.data.me.user.anon && (sakai.config.displayTimezone || sakai.config.displayLanguage)) {
174                 if (sakai.config.displayTimezone) {
175                     $("#footer_langdoc_loc").show();
176                 }
177                 if (sakai.config.displayLanguage) {
178                     $("#footer_langdoc_lang").show();
179                 }
180             }
181 
182             // Set the end year of the copyright notice
183             var d = new Date();
184             $footer_date_end.text(d.getFullYear());
185 
186             var leftLinksHTML = sakai.api.Util.TemplateRenderer($footer_links_left_template, {links:sakai.config.Footer.leftLinks});
187             leftLinksHTML = sakai.api.i18n.General.process(leftLinksHTML, "footer");
188             $footer_links_left.html(leftLinksHTML);
189 
190             var rightLinksHTML = sakai.api.Util.TemplateRenderer($footer_links_right_template, {links:sakai.config.Footer.rightLinks});
191             rightLinksHTML = sakai.api.i18n.General.process(rightLinksHTML, "footer");
192             $footer_links_right.html(rightLinksHTML);
193 
194             updateLocationLanguage();
195         };
196 
197         doInit();
198 
199     };
200 
201     sakai.api.Widgets.widgetLoader.informOnLoad("footer");
202 });
