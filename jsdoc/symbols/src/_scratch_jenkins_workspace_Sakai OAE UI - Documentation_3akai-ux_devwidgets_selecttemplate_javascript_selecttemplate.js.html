 25 require(["jquery", "sakai/sakai.api.core"], function($, sakai) {
 26 
 27     /**
 28      * @name sakai_global.newcreategroup
 29      *
 30      * @class newcreategroup
 31      *
 32      * @description
 33      * selecttemplate widget
 34      *
 35      * @version 0.0.1
 36      * @param {String} tuid Unique id of the widget
 37      * @param {Boolean} showSettings Show the settings of the widget or not
 38      */
 39     sakai_global.selecttemplate = function(tuid, showSettings){
 40 
 41         var $rootel = $("#" + tuid);
 42 
 43         var doInit = function(){
 44             sakai.api.Util.getTemplates(function(success, templates) {
 45                 if (success) {
 46                     var templatesToRender = false;
 47                     for (var i = 0; i < templates.length; i++) {
 48                         if (templates[i].id === tuid) {
 49                             templatesToRender = templates[i];
 50                             break;
 51                         }
 52                     }
 53                     if (templatesToRender) {
 54                         if (templatesToRender.templates.length === 1) {
 55                             renderCreateWorld(templatesToRender.id, templatesToRender.templates[0].id, true);
 56                         } else {
 57                             renderTemplateList(templatesToRender);
 58                         }
 59                     }
 60                 } else {
 61                     debug.error('Could not get the group templates');
 62                 }
 63             });
 64         };
 65 
 66         $(".selecttemplate_use_button", $rootel).live("click", function(){
 67             var clicked = $(this);
 68             if (clicked.data("templateid")){
 69                 renderCreateWorld(tuid, clicked.data("templateid"), false);
 70             }
 71         });
 72 
 73         var renderTemplateList = function(templates){
 74             $("#selecttemplate_container", $rootel).show();
 75             templates.sakai = sakai;
 76             templates.templates = templates.templates.sort(function(a,b) {
 77                 return a.order - b.order;
 78             });
 79             $("#selecttemplate_templatelist_container", $rootel).html(sakai.api.Util.TemplateRenderer("selecttemplate_templatelist_template", templates));
 80             $("#selecttemplate_type_name", $rootel).text(sakai.api.i18n.getValueForKey(templates.menuLabel || templates.title));
 81             $("#selecttemplate_createworld_container", $rootel).hide();
 82 
 83             $(".selecttemplate_preview_button", $rootel).live("click", function(){
 84                 var clicked = $(this);
 85                 if (clicked.data("templateid")){
 86                     renderPreview(tuid, clicked.data("templateid"), templates);
 87                 }
 88             });
 89         };
 90 
 91         var renderCreateWorld = function(category, id, singleTemplate){
 92             $("#selecttemplate_container", $rootel).hide();
 93             var tuid = sakai.api.Util.generateWidgetId();
 94             var toPassOn = {};
 95             toPassOn[tuid] = {
 96                 "category": category,
 97                 "id": id,
 98                 "singleTemplate": singleTemplate
 99             };
100             toPassOn[tuid + "addpeople"] = {
101                 "category": category,
102                 "id": id
103             };
104             $("#selecttemplate_createworld_container", $rootel).html(sakai.api.Util.TemplateRenderer("selecttemplate_createworld_template", {"tuid" : tuid}));
105             sakai.api.Widgets.widgetLoader.insertWidgets("selecttemplate_createworld_" + tuid, false, false, toPassOn);
106             $("#selecttemplate_createworld_container", $rootel).show();
107         };
108 
109         var renderPreview = function(category, id, templates){
110             var $selecttemplatePreviewDialog = $("#selecttemplate_preview_dialog", $rootel);
111             var $selecttemplatePreviewDialogContainer = $("#selecttemplate_preview_dialog_container", $rootel);
112             var selecttemplatePreviewDialogTemplate = "selecttemplate_preview_dialog_template";
113 
114             $selecttemplatePreviewDialog.jqm({
115                 modal: true,
116                 overlay: 20,
117                 toTop: true,
118                 top: "50px"
119             });
120 
121             var template;
122             for (var t in templates.templates) {
123                 if (templates.templates.hasOwnProperty(t)) {
124                     if (id === templates.templates[t].id) {
125                         template = templates.templates[t];
126                         break;
127                     }
128                 }
129             }
130             $selecttemplatePreviewDialog.jqmShow();
131             var json = {
132                 "template": template,
133                 "sakai": sakai
134             };
135             sakai.api.Util.TemplateRenderer(selecttemplatePreviewDialogTemplate, json, $selecttemplatePreviewDialogContainer);
136             $(".selecttemplate_use_button", "#selecttemplate_preview_dialog").live("click", function(){
137                 var clicked = $(this);
138                 if (clicked.data("templateid")){
139                     renderCreateWorld(tuid, clicked.data("templateid"), false);
140                     $selecttemplatePreviewDialog.jqmHide();
141                 }
142             });
143         };
144 
145         $(window).bind("hashchange", function(){
146             doInit();
147         });
148 
149         doInit();
150 
151     };
152 
153     sakai.api.Widgets.widgetLoader.informOnLoad("selecttemplate");
154 
155 });
