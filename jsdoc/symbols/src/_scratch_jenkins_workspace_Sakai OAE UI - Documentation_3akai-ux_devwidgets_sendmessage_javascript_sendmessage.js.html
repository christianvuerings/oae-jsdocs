 43             var toUser = false;  // configurable user to include as a message recipient
 44             var layover = true;        //    Will this widget be in a popup or inside another element.
 45             var callbackWhenDone = null;    //    Callback function for when the message gets sent
 46             var replyMessageID = null;

 50

 52             var messageForm = "#message_form";

 55             var messageFieldBody = "#comp-body";

 60             var errorClass = "sendmessage_error_message";
 61             var normalClass = "sendmessage_normal_message";

 65             var dialogClass = ".s3d-dialog";

 85             /**
 86              * This method will check if there are any required fields that are not filled in.
 87              * If a field is not filled in the invalidClass will be added to that field.
 88              * @return true = no errors, false = error
 89              */
 90             var checkFieldsForErrors = function(recipients) {
 91                 var subjectEl = $(messageFieldSubject);
 92                 var bodyEl = $(messageFieldBody);
 93                 var valid = true;
 94                 var subject = subjectEl.val();
 95                 var body = bodyEl.val();
 96 
 97                 subjectEl.removeClass(invalidClass);
 98                 bodyEl.removeClass(invalidClass);
 99 
100                 if (!subject) {
101                     valid = false;
102                     subjectEl.addClass(invalidClass);
103                 }
104                 if (!body) {
105                     valid = false;
106                     bodyEl.addClass(invalidClass);
107                 }
108                 // check if there are recipients
109                 if((recipients.length === 0 && !toUser.length) ||
110                     recipients.length === 1 && recipients[0] === "") {
111                     // no recipients are selected
112                     valid = false;
113                     // in the event allowOthers is false, the following will not be seen
114                     $(autoSuggestContainer).addClass(invalidClass);
115                     $(autoSuggestInput).addClass(invalidClass);
116                 }
117 
118                 return valid;
119             };

121             /**
122              * This will reset the whole widget to its default state.
123              * It will clear any values or texts that might have been entered.
124              */
125             var resetView = function() {
126                 $(dialogHeaderClass, $sendmessage_container).show();
127                 $sendmessage_container.addClass(dialogContainerClass);
128                 $(dialogBoxContainer).addClass(dialogBoxClass);
129                 $(messageDialogContainer).addClass(dialogClass.replace(/\./,''));
130                 $(messageDialogContainer).show();
131                 $(sendmessage_to).show();
132                 $(sendmessage_subject).show();
133                 $(sendmessage_body).find("label").show();
134                 // Clear the input fields
135                 $(messageFieldSubject + ", " + messageFieldBody).val('');
136 
137                 // remove autoSuggest if it exists
138                 sakai.api.Util.AutoSuggest.destroy($("#sendmessage_to_autoSuggest"));
139             };

141             /**
142              * Called when the request to the server has been answered
143              * @param {Boolean} succes    If the request failed or succeeded.
144              */
145             var showMessageSent = function(success) {
146                 // Depending on success we add the correct class and show the appropriate message.
147                 if (success) {
148                     var successMsg = $(notificationSuccess).text();
149                     sakai.api.Util.notification.show("", successMsg, sakai.api.Util.notification.type.INFORMATION);
150                 }
151                 else {
152                     var errorMsg = $(notificationError).text();
153                     sakai.api.Util.notification.show("", errorMsg, sakai.api.Util.notification.type.ERROR);
154                 }
155                 if ($(messageDialogContainer).hasClass('s3d-dialog')) {
156                     sakai.api.Util.Modal.close(messageDialogContainer);
157                 }
158 
159                 // If we have a valid callback function we call that
160                 // and dont show the message
161                 // If we dont have a callback we show a default message and fade out the layover.
162                 if (success && callbackWhenDone !== null) {
163                     callbackWhenDone(true);
164                 }
165 
166                 // Reset all the instance variables
167                 toUser = false;
168                 layover = true;
169                 callbackWhenDone = null;
170                 replyMessageID = null;
171             };

178             /**
179              * Initiates the 'To' field autoSuggest plugin with contact and group data
180              * @return None
181              */
182             var initAutoSuggest = function() {
183                 var preFill = [];
184                 if (toUser) {
185                     if ($.isPlainObject(toUser) && toUser.uuid) {
186                         preFill.push({
187                             "name": toUser.username,
188                             "value": toUser.uuid
189                         });
190                     } else if ($.isArray(toUser)) {
191                         $.each(toUser, function(i,usr) {
192                             preFill.push({
193                                 "name": usr.username,
194                                 "value": usr.uuid
195                             });
196                         });
197                     }
198                 }
199                 sakai.api.Util.AutoSuggest.setup($("#sendmessage_to_autoSuggest"), {
200                     "asHtmlID": "sendmessage_to_autoSuggest",
201                     startText: sakai.api.i18n.getValueForKey("ENTER_CONTACT_OR_GROUP_NAMES", "sendmessage"),
202                     keyDelay: "200",
203                     retrieveLimit: 10,
204                     preFill: preFill,
205                     searchObjProps: "name,value",
206                     formatList: function(data, elem) {
207                         // formats each line to be presented in autosuggest list
208                         // add the correct image, wrap name in a class
209                         var imgSrc = "/dev/images/user_avatar_icon_32x32.png";
210                         if(data.type === "group") {
211                             imgSrc = "/dev/images/group_avatar_icon_32x32.png";
212                         }
213                         var line_item = elem.html(
214                             '<img class="sm_suggestion_img" src="' + imgSrc + '" />' +
215                             '<span class="sm_suggestion_name">' + data.name + '</span>');
216                         return line_item;
217                     }
218                 });
219             };

225             var addSendMessageValidation = function() {
226                 var validateOpts = {
227                     submitHandler: sendMessage,
228                     'methods': {
229                         'requiredsuggest': {
230                             'method': function(value, element) {
231                                 return value.indexOf(
232                                     sakai.api.i18n.getValueForKey('ENTER_CONTACT_OR_GROUP_NAMES', 'sendmessage')) === -1 &&
233                                         $.trim($(element).next('input.as-values').val()).replace(/,/g, '') !== '';
234                             },
235                             'text': sakai.api.i18n.getValueForKey('AUTOSUGGEST_REQUIRED_ERROR')
236                         }
237                     }
238                 };
239                 sakai.api.Util.Forms.validate($sendmessage_form, validateOpts, true);
240             };

242             /**
243              * Initializes the sendmessage widget, optionally preloading the message
244              * with a recipient, subject and body. By default, the widget appears as
245              * a modal dialog. This function can be called from other widgets or pages.
246              * @param {Object|Array} userObj The user object containing the nescecary information {uuid:  "user1", username: "John Doe", type: "user"}, or a user profile
247              * @param {jQuery} $insertInId Insert the HTML into another element instead of showing it as a popup (String ID or jQuery)
248              * @param {Object} callback When the message is sent this function will be called. If no callback is provided a standard message will be shown that fades out.
249              * @param {String} subject The subject
250              * @param {String} body The body
251              * @param {Boolean} replyOnly hide the to: and subject: fields
252              * @param {String} replyID The ID of the message you're replying to
253              */
254             var initialize = function(userObj, $insertInId, callback, subject, body, replyOnly, replyID, buttonText) {
255                 layover = true;
256                 // Make sure that everything is standard.
257                 resetView();
258                 // The user we are sending a message to.
259                 if (userObj && (($.isPlainObject(userObj) && userObj.username) || $.isArray(userObj))) {
260                     toUser = userObj;
261                 } else {
262                     toUser = false;
263                 }
264 
265                 // Putting the subject and body which have been send in the textboxes
266                 if(body) {
267                     $(messageFieldBody).val(body);
268                 }
269                 if(subject) {
270                     $(messageFieldSubject).val(subject);
271                 }
272 
273                 if (replyOnly) {
274                     $(sendmessage_to).find("label").hide();
275                     $(sendmessage_subject).hide();
276                     $(sendmessage_body).find("label").hide();
277                 }
278                 if (replyID) {
279                     replyMessageID = replyID;
280                 } else {
281                     replyMessageID = null;
282                 }
283 
284                 if (buttonText) {
285                     $("#send_message span").text(buttonText);
286                 } else {
287                     $("#send_message span").text($("#sendmessage_default_button_text").text());
288                 }
289 
290                 // Maybe we dont want to display a popup but instead want to add it in another div.
291                 if ($insertInId) {
292                     if (!($insertInId instanceof jQuery)) {
293                         $insertInId = $(insertInId);
294                     }
295                     // Make sure this id exists.
296                     if ($insertInId.length > 0) {
297                         // The id exists!
298                         layover = false;
299 
300                         // Remove the dialog stuff.
301                         $(dialogHeaderClass, $sendmessage_container).hide();
302                         $sendmessage_container.removeClass(dialogContainerClass);
303                         $(messageDialogContainer).removeClass(dialogClass.replace(/\./,''));
304                         $(dialogBoxContainer).removeClass(dialogBoxClass);
305                         // Altough this isnt strictly nescecary it is cleaner.
306                         $rootel = $insertInId;
307                         $rootel.append($(messageDialogContainer));
308                         $sendmessage_form = $("#sendmessage_form", $rootel);
309                         bindEvents();
310                     }
311                 } else {
312                     $rootel = $("#"+tuid);
313                     $sendmessage_form = $("#sendmessage_form", $rootel);
314                     bindEvents();
315                 }
316 
317                 initAutoSuggest();
318                 addSendMessageValidation();
319                 // Store the callback
320                 if (callback) {
321                     callbackWhenDone = callback;
322                 }
323 
324                 // show popup
325                 if (layover) {
326                     var dialogOptions = {
327                         modal: true,
328                         overlay: 20,
329                         toTop: true
330                     };
331                     var openOptions = {
332                         bindKeyboardFocusIgnoreElements: 'a.as-close'
333                     };
334                     sakai.api.Util.Modal.setup(messageDialogContainer, dialogOptions);
335                     sakai.api.Util.Modal.open(messageDialogContainer, openOptions);
336                 }
337                 sakai.api.Util.Forms.clearValidation($sendmessage_form);
338             };

346             /**
347              * Callback function called after a call to send a message has completed
348              * @param {Boolean} success Status of the 'sendMessage' AJAX call
349              * @param {Object} data Data returned from the 'sendMessage' AJAX call
350              * @return None
351              */
352             var handleSentMessage = function(success, data) {
353                 if(success) {
354                     showMessageSent(success);
355                 } else {
356                     sakai.api.Util.notification.show(
357                         sakai.api.i18n.getValueForKey('SEND_MESSAGE', 'sendmessage'),
358                         sakai.api.i18n.getValueForKey('YOUR_MESSAGE_FAILED_DELIVERED', 'sendmessage'),
359                         sakai.api.Util.notification.type.ERROR);
360                 }
361                 $(buttonSendMessage).removeAttr("disabled");
362             };
363 
364             var sendMessage = function() {
365                 var recipients = [];
366                 // fetch list of selected recipients
367                 var recipientsString = $(autoSuggestValues).val();
368                 // autoSuggest adds unnecessary commas to the beginning and end
369                 // of the values string; remove them
370                 if(recipientsString[0] === ",") {
371                     recipientsString = recipientsString.slice(1);
372                 }
373                 if(recipientsString[recipientsString.length - 1] === ",") {
374                     recipientsString = recipientsString.slice(0, -1);
375                 }
376                 recipients = recipientsString.split(",");
377                 sakai.api.Communication.sendMessage(recipients, sakai.data.me, $(messageFieldSubject).val(), $(messageFieldBody).val(), "message", replyMessageID, handleSentMessage, true, "new_message");
378             };
379 
380             var bindEvents = function() {
381 
382                 ////////////////////////
383                 // jqModal functions  //
384                 ////////////////////////
385 
386                 $(send_message_cancel).die("click");
387                 $(send_message_cancel).live("click", function() {
388                     if ($(messageDialogContainer).hasClass('s3d-dialog')) {
389                         sakai.api.Util.Modal.close(messageDialogContainer);
390                     }
391                     if ($.isFunction(callbackWhenDone)) {
392                         callbackWhenDone(false);
393                     }
394                 });
395             };

401             $(document).on('initialize.sendmessage.sakai', function(e, userObj, insertInId, callback, subject, body, replyOnly, replyID, buttonText) {
402                 initialize(userObj, insertInId, callback, subject, body, replyOnly, replyID, buttonText);
403             });
404             $(document).on('click', '.sakai_sendmessage_overlay', function(ev) {
405                 var el = $(this);
406                 var person = false;
407                 var people = [];
408                 if (el.attr('sakai-entityid') && el.attr('sakai-entityname')) {
409                     var userIDArr = el.attr('sakai-entityid').split(',');
410                     var userNameArr = sakai.api.Security.safeOutput(el.attr('sakai-entityname')).split(',');
411                     for (var i = 0; i < userNameArr.length; i++) {
412                         people.push({
413                             'uuid': userIDArr[i],
414                             'username': userNameArr[i],
415                             'type': el.attr('sakai-entitytype') || 'user'
416                         });
417                     }
418                 }
419                 initialize(people);
420             });
421 
422             bindEvents();
423 
424         };
