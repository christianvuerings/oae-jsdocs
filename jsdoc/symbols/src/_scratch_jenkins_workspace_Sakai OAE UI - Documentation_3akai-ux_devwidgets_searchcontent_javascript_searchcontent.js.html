 18 // load the master sakai object to access all Sakai OAE API methods
 19 require(["jquery", "sakai/sakai.api.core", "/dev/javascript/search_util.js"], function($, sakai){
 20 
 21     /**
 22      * @name sakai.WIDGET_ID
 23      *
 24      * @class WIDGET_ID
 25      *
 26      * @description
 27      * WIDGET DESCRIPTION
 28      *
 29      * @version 0.0.1
 30      * @param {String} tuid Unique id of the widget
 31      * @param {Boolean} showSettings Show the settings of the widget or not
 32      */
 33     sakai_global.searchcontent = function(tuid, showSettings){
 34 
 35         //////////////////////
 36         // Config variables //
 37         //////////////////////
 38 
 39         // Search URL mapping
 40         var searchURLmap = {
 41             allfiles: sakai.config.URL.SEARCH_ALL_FILES,
 42             allfilesall: sakai.config.URL.SEARCH_ALL_FILES_ALL,
 43             pooledcontentmanager: sakai.config.URL.POOLED_CONTENT_MANAGER,
 44             pooledcontentmanagerall: sakai.config.URL.POOLED_CONTENT_MANAGER_ALL,
 45             pooledcontentviewer: sakai.config.URL.POOLED_CONTENT_VIEWER,
 46             pooledcontentviewerall: sakai.config.URL.POOLED_CONTENT_VIEWER_ALL
 47         };
 48 
 49         var infinityScroll = false;
 50 
 51         // CSS IDs
 52         var search = "#searchcontent";
 53         var $rootel = $("#" + tuid);
 54 
 55         var searchConfig = {
 56             search: "#searchcontent",
 57             global: {
 58                 resultTemp: search + "_result_temp",
 59                 numberFound: search + '_numberFound',
 60                 text: "#form .s3d-search-inputfield",
 61                 searchButton: "#form .s3d-search-button"
 62             },
 63             results: {
 64                 container: search + '_results_container',
 65                 resultsContainer: search + '_results',
 66                 resultsContainerAnonClass: 's3d-search-results-anon',
 67                 template: 'search_general_results_template',
 68                 noResultsTemplate: 'searchcontent_noresults_template'
 69             },
 70             facetedConfig: {
 71                 title: $("#searchcontent_result_title").html(),
 72                 value: "Content",
 73                 facets: {
 74                     "all": {
 75                         "category": $("#searchcontent_result_all_content").html(),
 76                         "searchurl": searchURLmap.allfiles,
 77                         "searchurlall": searchURLmap.allfilesall
 78                     }
 79                 }
 80             }
 81         };
 82 
 83         if (!sakai.data.me.user.anon) {
 84             searchConfig.facetedConfig.facets.manage = {
 85                 "category": $("#searchcontent_result_content_I_manage").html(),
 86                 "searchurl": searchURLmap.pooledcontentmanager,
 87                 "searchurlall": searchURLmap.pooledcontentmanagerall
 88             };
 89             searchConfig.facetedConfig.facets.member = {
 90                 "category": $("#searchcontent_result_content_I_m_a_viewer_of").html(),
 91                 "searchurl": searchURLmap.pooledcontentviewer,
 92                 "searchurlall": searchURLmap.pooledcontentviewerall
 93             };
 94         }
 95 
 96         ///////////////
 97         // Functions //
 98         ///////////////
 99 
100         /**
101          * Take a list of search results retrieved by the server and process them so they are
102          * ready to be run through the template
103          * @param {Object} results     List of results coming back from the infinite scroll plugin
104          * @param {Object} callback    Callback function from the infinite scroll plugin to call
105          */
106         var renderResults = function(results, callback){
107             // If we have results we add them to the object.
108             if (results && results.length) {
109                 sakai_global.data.search.prepareCMforRender(results, function(_results){
110                     callback(_results);
111                 });
112             } else {
113                 callback(results);
114             }
115         };
116 
117         /**
118          * This method will show all the appropriate elements for when a search is executed.
119          */
120         var showSearchContent = function(params){
121             // Set search box values
122             if (!params.q || (params.q === "*" || params.q === "**")) {
123                 $(searchConfig.global.text).val("");
124             } else {
125                 $(searchConfig.global.text).val(params.q);
126             }
127             $(searchConfig.results.container).html($(searchConfig.global.resultTemp).html());
128         };
129 
130         /**
131          * Render the default template when no results are found. This function will
132          * be called by the infinite scroll plugin
133          */
134         var handleEmptyResultList = function(){
135             $(searchConfig.global.numberFound).text("0");
136             $(searchConfig.results.container).html(sakai.api.Util.TemplateRenderer(searchConfig.results.noResultsTemplate, {sakai: sakai}));
137         };
138 
139         /**
140          * Kick off a search with a specific query and sort option. This function will
141          * initiate an infinite scroll for each search
142          */
143         var doSearch = function(){
144             var params = sakai_global.data.search.getQueryParams($rootel);
145             var urlsearchterm = sakai_global.data.search.processSearchString(params);
146             var tags = sakai_global.data.search.processRefineString(params);
147 
148             var facetedurl = "";
149             var facetedurlall = "";
150             if (params["facet"] && searchConfig.facetedConfig.facets[params["facet"]]) {
151                 facetedurl = searchConfig.facetedConfig.facets[params["facet"]].searchurl;
152                 facetedurlall = searchConfig.facetedConfig.facets[params["facet"]].searchurlall;
153             } else {
154                 for (var f in searchConfig.facetedConfig.facets) {
155                     facetedurl = searchConfig.facetedConfig.facets[f].searchurl;
156                     facetedurlall = searchConfig.facetedConfig.facets[f].searchurlall;
157                     break;
158                 }
159             }
160 
161             // Set all the input fields and paging correct.
162             showSearchContent(params);
163 
164             var url = '';
165             if ((urlsearchterm === '**' || urlsearchterm === '*') && params.refine === '') {
166                 url = facetedurlall;
167                 $(window).trigger('lhnav.addHashParam', [{'q': '', 'refine': ''}]);
168             } else {
169                 url = facetedurl.replace('.json', '.infinity.json');
170                 $(window).trigger('lhnav.addHashParam', [{'q': params.q, 'refine': params.refine}]);
171             }
172 
173             // Disable the previous infinite scroll
174             if (infinityScroll){
175                 infinityScroll.kill();
176             }
177             // Set up the infinite scroll for the list of search results
178             infinityScroll = $(searchConfig.results.container).infinitescroll(url, {
179                 'q': urlsearchterm,
180                 'tags': tags,
181                 'sortOn': params['sorton'],
182                 'sortOrder': params['sortby']
183             }, function(items, total){
184                 // Adjust display global total
185                 $(searchConfig.global.numberFound, $rootel).text("" + total);
186                 if (total === 1) {
187                     $(searchConfig.global.numberFound, $rootel).next('span.s3d-aural-text').text(
188                         '' + sakai.api.i18n.getValueForKey('CONTENT_ITEM_FOUND', 'searchcontent')
189                     );
190                 } else {
191                     $(searchConfig.global.numberFound, $rootel).next('span.s3d-aural-text').text(
192                         '' + sakai.api.i18n.getValueForKey('CONTENT_ITEMS_FOUND', 'searchcontent')
193                     );
194                 }
195                 return sakai.api.Util.TemplateRenderer(searchConfig.results.template, {
196                     "items": items,
197                     "sakai": sakai
198                 });
199             }, handleEmptyResultList, sakai.config.URL.INFINITE_LOADING_ICON, renderResults, function(){
200                 // Initialize content draggable
201                 sakai.api.Util.Draggable.setupDraggable({}, $(searchConfig.results.container));
202             }, false, function(data){
203                 // Generate refine by tags
204                 sakai_global.data.search.generateTagsRefineBy(data, params);
205             });
206         };
207 
208         /////////////////////////
209         // Initialise Function //
210         /////////////////////////
211 
212         if (sakai.data.me.user.anon){
213             $(searchConfig.results.resultsContainer).addClass(searchConfig.results.resultsContainerAnonClass);
214         }
215 
216         $(window).bind("hashchange", function(ev){
217             if ($.bbq.getState("l") === "content") {
218                 doSearch();
219             }
220         });
221 
222         $(window).bind("sakai.search.util.finish", function(ev, data){
223             if (data && data.tuid === tuid) {
224                 sakai.api.Widgets.widgetLoader.insertWidgets('searchcontent_widget', false, false, {
225                     "98384013291": {
226                         "facetedConfig": searchConfig.facetedConfig
227                     }
228                 });
229                 doSearch();
230             }
231         });
232 
233         $(window).trigger("sakai.search.util.init", [{"tuid": tuid}]);
234 
235     };
236 
237     // inform Sakai OAE that this widget has loaded and is ready to run
238     sakai.api.Widgets.widgetLoader.informOnLoad("searchcontent");
239 
240 });
