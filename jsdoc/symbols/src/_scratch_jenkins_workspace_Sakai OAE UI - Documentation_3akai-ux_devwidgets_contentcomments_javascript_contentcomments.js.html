139         /**
140          * Converts all HTML to flat text and converts \n to <br />
141          * @param {String} str
142          */
143         var tidyInput = function(str){
144             str = str.toString(); // in the event its not already a string, make it one
145             str = str.replace(/&/g, "&").replace(/</g, "<").replace(/>/g, ">");
146             str = str.replace(/\n/g, '<br />');
147             return str;
148         };

161         /**
162          * Show the contentcomments in a paged state or not
163          */
164         var displayCommentsPagedOrNot = function(){
165             jsonDisplay = {
166                 "comments": [],
167                 "settings": widgetSettings
168             };
169 
170             // sort contentcomments on create date
171             json.comments.sort(sortComments);
172 
173             // Loops through all the contentcomments and does the necessary changes to render the JSON-object
174             for (var i = 0; i < json.comments.length; i++) {
175                 jsonDisplay.comments[i] = {};
176                 var comment = json.comments[i];
177 
178                 comment.timeAgo = $.timeago(new Date(comment._created));
179                 comment.messageTxt = comment.comment;
180                 comment.message = tidyInput(comment.comment);
181                 comment.canEdit = false;
182                 comment["sakai:id"] = comment.commentId.substring((comment.commentId.lastIndexOf("/") + 1),comment.commentId.length);
183 
184                 var user = {};
185                 user.pictureUrl = sakai.config.URL.USER_DEFAULT_ICON_URL;
186                 // User
187                 // Puts the userinformation in a better structure for trimpath
188                 if (comment.userid) {
189                     if (contentData.isManager){
190                         comment.canDelete = true;
191                     }
192                     user.fullName = sakai.api.User.getDisplayName(comment);
193                     user.uid = comment.userid;
194                     // Check if the user has a picture
195                     var pictureUrl = sakai.api.Util.constructProfilePicture(comment);
196                     if (pictureUrl){
197                         user.pictureUrl = pictureUrl;
198                     }
199                     user.profile = "/~" + sakai.api.Util.safeURL(user.uid);
200                 }
201 
202                 comment.user = user;
203 
204                 jsonDisplay.comments[i] = comment;
205             }
206             jsonDisplay.sakai = sakai;
207             $(contentcommentsShowComments, rootel).html(sakai.api.Util.TemplateRenderer(contentcommentsShowCommentsTemplate, jsonDisplay));
208             // Render Math formulas in the text
209             sakai.api.Util.renderMath(tuid);
210         };

218             var showingComments = json.total;
219             if (widgetSettings.perPage < json.total) {
220                 showingComments = widgetSettings.perPage;

223             // Puts the number of contentcomments on the page
224             $(contentcommentsNumComments, rootel).html(json.total);
225             // Change to "comment" or "contentcomments"
226             if (json.total === 1) {
227                 $(contentcommentsCommentComments, rootel).text(sakai.api.i18n.getValueForKey("COMMENT_LC"));
228             }
229 
230             if (json.total > widgetSettings.perPage) {
231                 $(contentcommentsPager, rootel).show();
232             }
233             // Checks if the contentcomments undefined or if it's length is 0
234             displayCommentsPagedOrNot();

240         var getComments = function(){

242             var sortOrder = "desc";

277         var postComment = function(){
278             var comment = {
279                 // Replaces the \n (enters) with <br />
280                 "message": $(contentcommentsMessageTxt, rootel).val()
281             };
282             comment["sakai:type"] = "comment";
283 
284             var isLoggedIn = (me.user.anon && me.user.anon === true) ? false : true;
285             var allowPost = true;
286             // If the user is not loggedin but we allow anon contentcomments, we check some extra fields.
287             if (!isLoggedIn && widgetSettings['sakai:allowanonymous'] === true) {
288                 if (!isLoggedIn && widgetSettings['sakai:forcename']) {
289                     comment["sakai:name"] = $(contentcommentsNamePosterTxt, rootel).val();
290                     if (comment["sakai:name"].replace(/\s/g, "") === "") {
291                         allowPost = false;
292                     }
293                 }
294                 if (!isLoggedIn && widgetSettings['sakai:forcemail']) {
295                     comment["sakai:email"] = $(contentcommentsMailPosterTxt, rootel).val();
296                     if (comment["sakai:email"].replace(/\s/g, "") === "") {
297                         allowPost = false;
298                     }
299                 }
300             }
301             if (!isLoggedIn && widgetSettings['sakai:allowanonymous'] === false) {
302                 // This should not even happen.. Somebody is tinkering with the HTML.
303                 allowPost = false;
304                 sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("ANON_NOT_ALLOWED"),"",sakai.api.Util.notification.type.ERROR);
305             }
306 
307             var subject = "Comment";
308             //var to = "internal:w-" + widgeturl + "/message";
309 
310             var body = $(contentcommentsMessageTxt, rootel).val();
311             if (allowPost) {
312                 var message = {
313                     "_charset_":"utf-8",
314                     "comment": body
315                 };
316 
317                 var url = "/p/" + contentData.data["_path"] + ".comments";
318                 $.ajax({
319                     url: url,
320                     type: "POST",
321                     cache: false,
322                     success: function(data){
323                         // Clear the textboxes.
324                         $(contentcommentsMessageTxt, rootel).val("");
325                         $(contentcommentsNamePosterTxt, rootel).val("");
326                         $(contentcommentsMailPosterTxt, rootel).val("");
327                         // Add an acitivty
328                         sakai.api.Activity.createActivity("/p/" + contentData.data["_path"], "content", "default", {"sakai:activityMessage": "CONTENT_ADDED_COMMENT"}, function(responseData, success){
329                             if (success) {
330                                 // update the entity widget with the new activity
331                                 $window.trigger('updateContentActivity.entity.sakai', 'CONTENT_ADDED_COMMENT');
332                                 if (!rootel.parents(".collectionviewer_collection_item_comments").length){
333                                     $window.trigger('sakai.entity.updatecountcache', {increment: true});
334                                 }
335                             }
336                         });
337                         // Get the contentcomments.
338                         getComments();
339                     },
340                     error: function(xhr, textStatus, thrownError){
341                         if (xhr.status === 401) {
342                             sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("YOU_NOT_ALLOWED"),"",sakai.api.Util.notification.type.ERROR);
343                         }
344                         else {
345                             sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("FAILED_TO_SAVE"),"",sakai.api.Util.notification.type.ERROR);
346                         }
347                     },
348                     data: message
349                 });
350             }
351         };

395         /**
396          * fills up the settings JSON-object
397          * @return {Object} the settings JSON-object, returns {Boolean} false if input is invalid
398          */
399         var getCommentsSettings = function(){
400             var contentcomments = {};
401             contentcomments.contentcomments = [];
402 
403             // Checks if there's already some contentcomments placed on the widget
404             contentcomments.contentcomments = json.contentcomments || [];
405 
406             contentcomments.perPage = parseInt($(contentcommentsPageTxt, rootel).val(), 10);
407             if (isNaN(contentcomments.perPage)) {
408                 contentcomments.perPage = defaultPostsPerPage;
409             }
410 
411             if (contentcomments.perPage < 1) {
412                 sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("PLEASE_FILL_POSITIVE_NUM"),"",sakai.api.Util.notification.type.ERROR);
413                 return false;
414             }
415             // Check if a valid number is inserted
416             else
417                 if ($(contentcommentsPageTxt, rootel).val().search(/^\d*$/)) {
418                     sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("PLEASE_FILL_VALID_NUM"),"",sakai.api.Util.notification.type.ERROR);
419                     return false;
420                 }
421 
422 
423             contentcomments.direction = $("input[name=" + contentcommentsDirectionRbt + " ]:checked", rootel).val();
424 
425             // These properties are noy yet used in the contentcomments-widget, but are saved in JCR
426             contentcomments['sakai:allowanonymous'] = true;
427             if ($("#contentcomments_RequireLogInID", rootel).is(":checked")) {
428                 contentcomments['sakai:allowanonymous'] = false;
429             }
430             contentcomments['sakai:forcename'] = $(contentcommentsNameReqChk, rootel).attr("checked");
431             contentcomments['sakai:forcemail'] = $(contentcommentsEmailReqChk, rootel).attr("checked");
432             contentcomments['sakai:notification'] = $(contentcommentsSendMailChk, rootel).attr("checked");
433             contentcomments['sakai:notificationaddress'] = me.user.userid;
434             contentcomments['sling:resourceType'] = 'sakai/settings';
435             contentcomments['sakai:marker'] = tuid;
436             contentcomments['sakai:type'] = "comment";
437 
438             return contentcomments;
439         };

441         /**
442          * Makes sure that values that are supposed to be booleans, really are booleans.
443          * @param {String[]} arr Array of strings which holds keys for the widgetSettings variable that needs to be checked.
444          */
445         var cleanBooleanSettings = function(arr){
446             for (var i = 0; i < arr.length; i++) {
447                 var name = arr[i];
448                 widgetSettings[name] = (widgetSettings[name] && (widgetSettings[name] === true || widgetSettings[name] === "true" || widgetSettings[name] === 1)) ? true : false;
449             }

452         /**
453          * Gets the widget settings and shows the appropriate view.
454          */
455         var getWidgetSettings = function(){
456 
457             sakai.api.Widgets.loadWidgetData(tuid, function(success, data){
458                 if (success) {
459                     if (!data.message) {
460                         sakai.api.Widgets.saveWidgetData(tuid, {"message":{"sling:resourceType":"sakai/messagestore"}}, null);

463                     // Clean up some values so that true is really true and not "true" or 1 ...
464                     var keysToClean = ['sakai:forcename', 'sakai:forcemail', 'notification', 'sakai:allowanonymous'];
465                     cleanBooleanSettings(keysToClean);
466

468                     if (widgetSettings["sakai:allowanonymous"] === false && !isLoggedIn) {
469                         $(contentcommentsCommentBtn, rootel).parent().hide();
470                     }
471 
472                     if (showSettings) {
473                         showSettingScreen(true, data);
474                     } else {
475                         pagerClickHandler(1);
476                     }
477                 }
478                 else {
479                     if (showSettings) {
480                         showSettingScreen(false, data);
481                     } else {
482                         pagerClickHandler(1);
483                     }
484                 }
485             });
486 
487         };

573         /**
574          * Switch between main and settings page
575          * @param {Boolean} showSettings Show the settings of the widget or not

577         var doInit = function(){
578             // Temporarily set these here, always allowing comments
579             //contentData.data = contentData.data || {};
580             contentData.data["sakai:showcontentcomments"] = true;
581             contentData.data["sakai:allowcontentcomments"] = true;
582             $(contentcommentsEditorOptions, rootel).hide();
583             if (sakai_global.content_profile && contentData){
584                 contentPath = "/p/" + contentData.data["_path"];
585 
586                 // check if contentcomments are allowed or shown and display the checkbox options for the manager
587                 if (contentData.isManager){
588                     if (contentData.data["sakai:allowcontentcomments"] === false){
589                         $(contentcommentsAllowCheckbox, rootel).removeAttr("checked");
590                     } else {
591                         contentData.data["sakai:allowcontentcomments"] = true;
592                         $(contentcommentsAllowCheckbox, rootel).attr("checked", "checked");
593                     }
594                     if (contentData.data["sakai:showcontentcomments"] === false){
595                         $(contentcommentsShowCheckbox, rootel).removeAttr("checked");
596                         $(contentcommentsAllowCheckbox, rootel).removeAttr("checked");
597                         $(contentcommentsAllowCheckbox, rootel).attr("disabled", "disabled");
598                         showCommentsChecked = false;
599                     } else {
600                         contentData.data["sakai:showcontentcomments"] = true;
601                         $(contentcommentsShowCheckbox, rootel).attr("checked", "checked");
602                         $(contentcommentsAllowCheckbox, rootel).removeAttr("disabled");
603                     }
604                     $(contentcommentsEditorOptions, rootel).show();
605                 }
606             }
607             if (!showSettings) {
608                 // Show the main view.
609                 var picture = sakai.api.Util.constructProfilePicture(me.profile);
610                 sakai.api.Util.TemplateRenderer("#contentcomments_userCommentContainer_template", {picture: picture}, $(contentcommentsUserCommentContainer, rootel));
611                 $(contentcommentsSettingsContainer, rootel).hide();
612                 $(contentcommentsOutputContainer, rootel).show();
613                 var isLoggedIn = (me.user.anon && me.user.anon === true) ? false : true;
614                 if (!isLoggedIn) {
615                     $(contentcommentsUserCommentContainer, rootel).hide();
616                 }
617             }
618             addBinding();
619             checkCommentsPermissions(true);
620         };
