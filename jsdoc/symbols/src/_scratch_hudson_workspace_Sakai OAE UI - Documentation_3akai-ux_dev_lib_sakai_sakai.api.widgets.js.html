279                 /**
280                  * Locate a tag and remove it from the content
281                  * @param {String} content The complete content of a file (e.g. <div>...)
282                  * @param {String} tagName The name of the tag you want to remove (link/script)
283                  * @param {String} URLIdentifier The part that identifies the URL (href/src)
284                  */
285                 var locateTagAndRemove = function(content, tagName, URLIdentifier){
286                     var returnObject = {
287                         URL : [],
288                         content : content
289                     };
290                     var regexp = new RegExp('<'+tagName+'.*?'+URLIdentifier+'\\s?=\\s?["|'+'\''+']([^"]*)["|'+'\''+'].*/.*?>', "gi");
291                     var regexp_match_result = regexp.exec(content);
292                     while (regexp_match_result !== null) {
293                         returnObject.URL[returnObject.URL.length] = regexp_match_result[1]; // value of URLIdentifier attrib
294                         returnObject.content = returnObject.content.replace(regexp_match_result[0],""); // whole tag
295                         regexp_match_result = regexp.exec(content);
296                     }
297                     return returnObject;
298                 };

362                         var bundles = [];
363                         $.each(urls, function(index, url){ 
364                             var widgetname = batchWidgets[url];
365                             if ($.isPlainObject(sakai.widgets[widgetname].i18n)) {
366                                 // Add the default language bundle for the widget

512                  * Insert the widgets into the page
513                  * @param {String} containerId The id of the container element
514                  * @param {Boolean} showSettings Show the settings for the widget
515                  * @param {Object} widgetData Widget data associated to the loaded widgets

518                 var locateWidgets = function(containerId, showSettings, widgetData, context, callback) {
519 
520                     // Use document.getElementById() to avoid jQuery selector escaping issues with '/'
521                     var el = containerId ? document.getElementById(containerId) : $(document.body);
522 
523                     // Array of jQuery objects that contains all the elements in the with the widget selector class.
524                     var divarray = $(widgetSelector, el);
525 
526                     // Check if the showSettings variable is set, if not set the settings variable to false
527                     settings = showSettings || false;
528 
529                     // Object that will contain all the URLs + names of the widgets that need to be fetched with batch get
530                     var batchWidgets = {};
531 
532                     // Run over all the elements and load them
533                     for (var i = 0, j = divarray.length; i < j; i++){
534                         var id = divarray[i].id;
535                         var split = id.split("_");
536                         var widgetname = split[1];
537 
538                         // Set the id for the container of the widget
539                         var widgetid;
540                         if (split[2]){
541                             widgetid = split[2];
542                         } else if(widgetname) {
543                             widgetid = widgetname + "container" + Math.round(Math.random() * 10000000);
544                         }
545 
546                         // Check if the widget is an iframe widget
547                         if (sakai.widgets[widgetname]) {
548 
549                             // Set the placement for the widget
550                             var placement = "";
551                             if (split[3] !== undefined){
552                                 var length = split[0].length + 1 + widgetname.length + 1 + widgetid.length + 1;
553                                 placement = id.substring(length);
554                             } else if (context){
555                                 placement = context;
556                             }
557 
558                             // Check if the widget exists
559                             if (!widgetsInternal[widgetname]){
560                                 widgetsInternal[widgetname] = [];
561                             }
562 
563                             // Set the initial properties for the widget
564                             var index = widgetsInternal[widgetname].length;
565                             widgetsInternal[widgetname][index] = {
566                                 uid : widgetid,
567                                 placement : placement,
568                                 id : id,
569                                 widgetData: widgetData[widgetid] || false,
570                                 callback: callback
571                             };
572 
573                             var floating = "inline_class_widget_nofloat";
574                             if ($(divarray[i]).hasClass("block_image_left")){
575                                 floating = "inline_class_widget_leftfloat";
576                             } else if ($(divarray[i]).hasClass("block_image_right")){
577                                 floating = "inline_class_widget_rightfloat";
578                             }
579 
580                             widgetsInternal[widgetname][index].floating = floating;
581                             
582                         }
583                     }
584 
585                     for (i in widgetsInternal){
586                         if (widgetsInternal.hasOwnProperty(i)) {
587                             for (var ii = 0, jj = widgetsInternal[i].length; ii<jj; ii++) {
588 
589                                 // Replace all the widgets with id "widget_" to widgets with new id's
590                                 // and add set the appropriate float class
591                                 $(document.getElementById(widgetsInternal[i][ii].id)).replaceWith($('<div id="'+widgetsInternal[i][ii].uid+'" class="' + widgetsInternal[i][ii].floating + '"></div>'));
592                             }
593 
594                             var url = sakai.widgets[i].url;
595                             batchWidgets[url] = i; //i is the widgetname
596                         }
597                     }
598 
599                     // Load the HTML files for the widgets
600                     loadWidgetFiles(widgetsInternal, batchWidgets);
601 
602                 };

723         /**
724          * This binds to any links with a hash URL and handles the
725          * pushState for them
726          */
727         bindToHash : function() {
728             $("a[href^='#']").live("click", function(e) {
729                 var $target = $(e.currentTarget),
730                     args = $target.attr("href"),
731                     replace = $target.data("reset-hash"),
732                     remove = $target.data("remove-params"),
733                     stateToAdd = {}, currentState = {}, newState = {};
734                 // new state to push
735                 stateToAdd = $.deparam.fragment(args, true);
736                 // current window.location.hash
737                 currentState = $.deparam.fragment();
738                 // the link wants to remove params from the url
739                 if (remove) {
740                     // data-remove-params is a comma-delimited attribute
741                     remove = remove.split(',');
742                     $.each(remove, function(i,val) {
743                         val = $.trim(val);
744                         if (currentState[val]) {
745                             delete currentState[val];
746                         }
747                     });
748                 }
749                 // replace means we should replace the state entirely with the new state from the link
750                 if (replace) {
751                     newState = stateToAdd;
752                 } else {
753                     // otherwise we merge the currentState with the stateToAdd
754                     // note that any params in stateToAdd will override those in currentState
755                     newState = $.extend({}, currentState, stateToAdd);
756                 }
757                 // Always push with the 2 argument as newState contains the entire state we want
758                 $.bbq.pushState(newState, 2);
759                 e.preventDefault();
760             });
761             oldState = $.bbq.getState();
762             $(window).bind("hashchange", sakaiWidgetsAPI.handleHashChange);
763             $(window).trigger("hashchange");
764         },
765 
766         /**
767          * This function is used for creating href's to link to a
768          * hash change
769          * @param {Object} paramsObject The object containing key value pairs
770          *                              to add to the URL
771          * @param {String} url          The url you want to add hash parameters
772          *                              to. If not provided, the system will use
773          *                              the current page URL.
774          */
775         createHashURL : function(paramsObject, url) {
776             return $.param.fragment(url || window.location.hash, paramsObject);

779         /**
780          * Handle the hash change and dispatch the hashchange event
781          * to each widget that has the changed parameter
782          * @param {Object} e The hashchange event object
783          */
784         handleHashChange : function(e) {
785             var widgetHashes = {};
786             // get the changed params
787             var currentState = $.bbq.getState();
788 
789             /** 
790              * construct the changedParams object which contains a map like this:
791              * widgetHashes = { "widgetid" : { "changed": {"property": "value"}, "deleted": {}}};
792              */
793             if (!$.isEmptyObject(sakai_widgets_config)) {
794                 $.each(sakai_widgets_config, function(id, obj) {
795                     if (obj.hasOwnProperty('hashParams')) {
796                         // iterate over each of the params that the widet watches
797                         $.each(obj.hashParams, function(i, val) {
798                             // If the current history state has this value
799                             if (currentState.hasOwnProperty(val)) {
800                                 widgetHashes[id] = widgetHashes[id] || {changed:{}, deleted:{}, all: {}};
801                                 // and the oldState value exists and isn't the same as the new value
802                                 // or the oldState didn't have the value
803                                 if ((oldState.hasOwnProperty(val) && oldState[val] !== currentState[val]) ||
804                                     !oldState.hasOwnProperty(val)) {
805 
806                                     widgetHashes[id].changed[val] = currentState[val];
807                                 }
808                                 widgetHashes[id].all[val] = currentState[val];
809 
810                             // Check if the property was in the history state previously,
811                             // indicating that it was deleted from the currentState
812                             } else if (oldState.hasOwnProperty(val)) {
813                                 widgetHashes[id] = widgetHashes[id] || {changed:{}, deleted:{}, all: {}};
814                                 widgetHashes[id].deleted[val] = oldState[val];
815                             }
816                         });
817                     }
818                 });
819                 if (e.currentTarget) {
820                     // Fire an event to each widget that has the hash params in it
821                     $.each(widgetHashes, function(widgetID, hashObj) {
822                         $(window).trigger("hashchanged." + widgetID + ".sakai", [hashObj.changed || {}, hashObj.deleted || {}, hashObj.all || {}, currentState || {}]);
823                     });
824 
825                     // Reset the oldState to the currentState
826                     oldState = currentState;
827                     return true;
828                 } else {
829                     return widgetHashes[e];
830                 }
831             } else {
832                 return null;
833             }
834         },

910         /**
911          * This function will register all widgets that require lazy loading
912          */
913         registerLazyLoading: function() {
914             if (sakai.widgets) {
915                 $.each(sakai.widgets, function(widgetid, widget) {
916                     if (widget.trigger && !widget.trigger.onLoad) {
917     
918                         // Convert all of the properties to an array
919                         if (widget.trigger.events && widget.trigger.events.length) {
920                             if (!$.isArray(widget.trigger.events)) {
921                                 widget.trigger.events = [widget.trigger.events];
922                             }
923                         }
924                         widget.trigger.events = widget.trigger.events || [];
925     
926                         if (widget.trigger.selectors && widget.trigger.selectors.length) {
927                             if (!$.isArray(widget.trigger.selectors)) {
928                                 widget.trigger.selectors = [widget.trigger.selectors];
929                             }
930                         }
931                         widget.trigger.selectors = widget.trigger.selectors || [];
932     
933                         var lazyLoadWidget = function(finishCallBack) {
934                             // Unbind the event
935                             $.each(widget.trigger.events, function(index, eventid) {
936                                 $(document).off(eventid);
937                             });
938                             // Also kill the click events associated to this widget
939                             $.each(widget.trigger.selectors, function(index, selector) {
940                                 $(document).off('click', selector);
941                             });
942     
943                             $('body').prepend('<div id="widget_' + widgetid + '" class="widget_inline"></div>');
944                             sakaiWidgetsAPI.widgetLoader.insertWidgets(null, false, null, null, null, finishCallBack);
945                         };
946     
947                         // Check whether this needs to bind to an event
948                         $.each(widget.trigger.events, function(index, eventid) {
949                             // a, b, c, ..., i, j is a list of possible parameters that can be passed
950                             // in when the event is called. As we have no idea how many will come through,
951                             // we generically catch them and pass them back on when we re-call the event
952                             $(document).on(eventid, function(ev, a, b, c, d, e, f, g, h, i, j) {
953                                 lazyLoadWidget(function() {
954                                     $(document).trigger(eventid, [a, b, c, d, e, f, g, h, i, j]);
955                                 });
956                             });
957                         });
958     
959                         // Check whether this needs to bind to a selector
960                         $.each(widget.trigger.selectors, function(index, selector) {
961                             $(document).on('click', selector, function(event, ui) {
962                                 lazyLoadWidget(function() {
963                                     $(event.target).trigger('click', [event, ui]);
964                                 });
965                             });
966                         });
967     
968                     }
969                 });
970             }
971         },
