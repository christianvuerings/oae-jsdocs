 19 // load the master sakai object to access all Sakai OAE API methods
 20 require(['jquery', 'sakai/sakai.api.core', '/dev/javascript/search_util.js', 'jquery-pager'], function($, sakai) {
 21 
 22     /**
 23      * @name sakai.searchsakai2
 24      *
 25      * @class searchsakai2
 26      *
 27      * @description
 28      * WIDGET DESCRIPTION
 29      *
 30      * @version 0.0.1
 31      * @param {String} tuid Unique id of the widget
 32      * @param {Boolean} showSettings Show the settings of the widget or not
 33      */
 34     sakai_global.searchsakai2 = function (tuid, showSettings) {
 35          
 36         /////////////////////////////
 37         // Configuration variables //
 38         /////////////////////////////
 39 
 40         var resultsToDisplay = 10;
 41         var rootel = $("#" + tuid);
 42 
 43         var search = "#searchsakai2";
 44 
 45         var searchConfig = {
 46             search: "#searchsakai2",
 47             global: {
 48                 button: search + "_button",
 49                 text: search + '_text',
 50                 numberFound: search + '_numberFound',
 51                 pagerClass: ".jq_pager"
 52             },
 53             results: {
 54                 container: search + '_results_container',
 55                 template: search + '_results_template'
 56             },
 57             facetedConfig: {
 58                 title: "Sakai 2 Search",
 59                 value: "Sakai 2",
 60                 facets: {
 61                     "all": {
 62                         "category": "All",
 63                         "searchurl": "",
 64                         "searchurlall": ""
 65                     }
 66                 }
 67             }
 68         };
 69 
 70         ///////////////////////
 71         // Utility functions //
 72         ///////////////////////
 73 
 74         /**
 75          * This function loop through the site and check if search term is in the description or title
 76          * @param {String} searchterm The searchterm you want to search trough.
 77          * @param {string} category The category you want to search in.
 78          * @param {Object} resultJson The search result json object.
 79          */
 80         var filterSearch = function(searchterm, category , resultJson) {
 81             resultJson.sites = resultJson.sites || [];
 82             searchtermlower = searchterm.toLowerCase();
 83             for(var j=0;j<category.sites.length;j++){ 
 84                 var site = category.sites[j];  
 85                 if(searchterm === "*" ) {
 86                     resultJson.sites.push(site);
 87                 } else if(site.title.toLowerCase().search(searchtermlower) > -1 ) {
 88                     resultJson.sites.push(site);
 89                 } else if (site.description && site.description.toLowerCase().search(searchtermlower) > -1 ) {
 90                     resultJson.sites.push(site);
 91                 }
 92              }
 93 
 94              return resultJson;        
 95         };
 96 
 97         /** 
 98          * Update the metadata for the Search Page Facets based on the Sakai 2
 99          * Site categories that come back.
100          */
101         var updateFacets = function(jsondata) {
102             searchConfig.facetedConfig.facets = {};
103             var facets = searchConfig.facetedConfig.facets;
104             if (jsondata.categories) {
105                 for (var i = 0; i < jsondata.categories.length; i++) {
106                     var cat = jsondata.categories[i];
107                     facets[i] = {
108                         category: cat.category,
109                         "searchurl": "",
110                         "searchurlall": ""
111                     };
112                 }
113             }
114         };
115 
116         /**
117          * Get sites list group with categories from back end.
118          */
119         var getCategories = function(callback){
120             var url = "/dev/s23/bundles/sites-categorized.json";
121             if (sakai.config.useLiveSakai2Feeds){
122                 url = "/var/proxy/s23/sitesCategorized.json?categorized=true";
123             }
124             $.ajax({
125                 url: url,
126                 type : "GET",
127                 dataType: "json",
128                 success: function(data){
129                     callback(data);
130                 },
131                 error: function(){
132                     $("#sakai2search_error_container", rootel).show();
133                 }
134             });
135         };
136 
137         /////////////////////////
138         // Functions           //
139         /////////////////////////
140 
141         var pager_click_handler = function(pageclickednumber){
142             $.bbq.pushState({
143                 "q": $(searchConfig.global.text, rootel).val(),
144                 "page": pageclickednumber
145             }, 0);
146         };
147 
148         /* 
149          * Renders the results using the json as structured in 
150          * /var/proxy/s23/sitesCategorized.json?categorized=true
151          */
152         var renderResults = function(jsondata) {
153             var qparams = sakai_global.data.search.getQueryParams();
154             finaljson = {};
155             finaljson.sakai = sakai;
156             var resultstotal;
157 
158             if (searchConfig.facetedConfig.facets[0]) {
159                 $(".searchsakai2_selects", rootel).show();
160                 var categoryName = searchConfig.facetedConfig.facets[0].category;
161                 if (qparams.facet !== undefined) {
162                     categoryName = searchConfig.facetedConfig.facets[qparams.facet].category;
163                 }
164 
165                 var categorydata = {};
166                 for (var i = 0; i < jsondata.categories.length; i++) {
167                     if (categoryName === jsondata.categories[i].category) {
168                         categorydata = jsondata.categories[i];
169                     }
170                 }    
171     
172                 filterSearch(qparams.q,categorydata,finaljson);
173                 finaljson.query = qparams.q;
174                 finaljson.sites.sort(siteTitleSort);
175                 for (var s = 0; s < finaljson.sites.length; s++){
176                     var url = finaljson.sites[s].url;
177                     finaljson.sites[s].url = url.split("/")[url.split("/").length - 1];
178                 }
179     
180                 resultstotal = finaljson.sites.length;
181                 $(searchConfig.global.numberFound, rootel).text("" + resultstotal);
182                 
183                 if (resultstotal > resultsToDisplay && qparams.page) {
184                     var end = qparams.page * resultsToDisplay;
185                     finaljson.sites = finaljson.sites.slice(end-resultsToDisplay,end);
186                 }
187             } else {
188                 finaljson.sites = [];
189                 finaljson.query = "*";
190                 $(".searchsakai2_selects", rootel).hide();
191             }
192 
193             $(searchConfig.results.container, rootel).html(sakai.api.Util.TemplateRenderer(searchConfig.results.template, finaljson));
194             $(searchConfig.results.container, rootel).show();
195 
196 
197             // Putting Pager Reset down here, otherwise I seem to be having
198             // timing issues with different things (facet, pager, etc) getting
199             // rendered.
200             $(searchConfig.global.pagerClass, rootel).pager({
201                 pagenumber: qparams.page,
202                 pagecount: Math.ceil(Math.abs(resultstotal) / resultsToDisplay),
203                 buttonClickCallback: pager_click_handler
204             });
205             if (resultstotal > resultsToDisplay) {
206                 $(searchConfig.global.pagerClass, rootel).show();
207             } else {
208                 $(searchConfig.global.pagerClass, rootel).hide();
209             }
210         };
211 
212         var siteTitleSort = function(a, b){
213             return a.title > b.title;
214         };
215 
216         var doSearch = function(){
217             getCategories(function(data) {renderResults(data);});
218         };
219 
220         ////////////////////
221         // Event Handlers //
222         ////////////////////
223 
224 
225         $(searchConfig.global.text, rootel).live("keydown", function(ev){ 
226             if (ev.keyCode === 13) {
227                 $.bbq.pushState({
228                     "q": $(searchConfig.global.text, rootel).val(),
229                     "page": 0
230                 }, 0);
231             }
232         });
233 
234         $(searchConfig.global.button, rootel).live("click", function(ev){
235             $.bbq.pushState({
236                 "q": $(searchConfig.global.text, rootel).val(),
237                 "page": 0
238             }, 0);
239         });
240 
241 
242         /////////////////////////////
243         // Initialization function //
244         /////////////////////////////
245         
246         $(window).bind("hashchange", function(ev){
247             if ($.bbq.getState("l") === "sakai2sites") {
248                 doSearch();
249             }
250         });
251 
252         /**
253          * Initialization function DOCUMENTATION
254          */
255         $(window).bind("sakai.search.util.finish", function(ev){
256             sakai.api.Widgets.widgetLoader.insertWidgets('searchcontent_widget', false, false, {
257                 "id845719741363": {
258                     "facetedConfig": searchConfig.facetedConfig
259                 }
260             });
261             doSearch();
262         });
263         
264         // run the initialization function when the widget object loads
265         // we're wrapping this up because we need to do an ajax call to Sakai2
266         // to determine the categories for the search facet
267         getCategories(function(data) {
268             updateFacets(data);
269             $(window).trigger("sakai.search.util.init", {"tuid": tuid});
270         });
271         
272     };
273 
274     // inform Sakai OAE that this widget has loaded and is ready to run
275     sakai.api.Widgets.widgetLoader.informOnLoad("searchsakai2");
276 });
