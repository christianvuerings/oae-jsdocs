189         var getRedirectURL = function() {
190             var redirectURL = window.location.pathname + window.location.search + window.location.hash;
191             // Check whether we require a redirect
192             if ($.deparam.querystring().url) {
193                 // Check if the redirect URL is valid
194                 if (checkValidRedirect($.deparam.querystring().url)) {
195                     redirectURL = $.deparam.querystring().url;
196                 } else {
197                     redirectURL = window.location.pathname;
198                 }
199             // Go to You when you're on explore page
200             } else if (window.location.pathname === '/dev/explore.html' || window.location.pathname === '/register' ||
201                 window.location.pathname === '/index' || window.location.pathname === '/' || window.location.pathname === '/dev') {
202                 redirectURL = '/me';
203             // 500 not logged in
204             } else if (sakai_global.nopermissions && sakai.data.me.user.anon && sakai_global.nopermissions.error500) {
205                 redirectURL = '/me';
206             }
207             return redirectURL;
208         };

347         /**
348          * Execute the live search and render the results
349          * @param {Object} templates the available world templates
350          */
351         var doSearch = function(templates) {
352             var searchText = $.trim($('#topnavigation_search_input').val());
353             var filesUrl = sakai.config.URL.SEARCH_ALL_FILES.replace('.json', '.infinity.json');
354             var usersUrl = sakai.config.URL.SEARCH_USERS;
355             var groupsUrl = sakai.config.URL.SEARCH_GROUPS;
356             if (searchText === '*' || searchText === '**') {
357                 filesUrl = sakai.config.URL.SEARCH_ALL_FILES_ALL;
358                 usersUrl = sakai.config.URL.SEARCH_USERS_ALL;
359                 groupsUrl = sakai.config.URL.SEARCH_GROUPS_ALL;
360             }
361 
362             renderObj.query = searchText;
363             searchText = sakai.api.Server.createSearchString(searchText);
364             var requests = [];
365             requests.push({
366                 'url': filesUrl,
367                 'method': 'GET',
368                 'parameters': {
369                     'page': 0,
370                     'items': 3,
371                     'q': searchText
372                 }
373             });
374             requests.push({
375                 'url': usersUrl,
376                 'method': 'GET',
377                 'parameters': {
378                     'page': 0,
379                     'items': 3,
380                     'sortOn': 'lastName',
381                     'sortOrder': 'asc',
382                     'q': searchText
383                 }
384             });
385             for (var c = 0; c < templates.length; c++) {
386                 var category = templates[c];
387                 requests.push({
388                     'url': groupsUrl,
389                     'method': 'GET',
390                     'parameters': {
391                         'page': 0,
392                         'items': 3,
393                         'q': searchText,
394                         'category': category.id
395                     }
396                 });
397             }
398             
399 
400             sakai.api.Server.batch(requests, function(success, data) {
401                 renderContent($.parseJSON(data.results[0].body));
402                 renderPeople($.parseJSON(data.results[1].body));
403                 for (var c = 0; c < templates.length; c++) {
404                     renderGroups($.parseJSON(data.results[2 + c].body), templates[c].id);
405                 }
406             });
407         };

414         /**
415          * Generate a subnavigation item
416          * @param {integer} index Index of the subnavigation item in the array
417          * @param {Array} array Array of subnavigation items
418          */
419         var getNavItem = function(index, array) {
420             var temp = {};
421             var item = array[index];
422             temp.id = item.id;
423             if (temp.id && temp.id == 'subnavigation_hr') {
424                 temp = 'hr';
425             } else {
426                 if (sakai.data.me.user.anon && item.anonUrl) {
427                     temp.url = item.anonUrl;
428                 } else {
429                     temp.url = item.url;
430                     if (item.append) {
431                         temp.append = item.append;
432                     }
433                 }
434                 temp.label = sakai.api.i18n.getValueForKey(item.label);
435                 if (item.cssClass) {
436                     temp.cssClass = item.cssClass;
437                 }
438             }
439             return temp;
440         };

442         /**
443          * Create a list item for the topnavigation menu including the subnavigation
444          * @param {integer} i index of the current item in the loop
445          * @param {Object} templates the available world templates
446          */
447         var createMenuList = function(i, templates) {
448             var temp = getNavItem(i, sakai.config.Navigation);
449             // Add in the template categories
450             if (sakai.config.Navigation[i].id === 'navigation_create_and_add_link') {
451                 for (var c = 0; c < templates.length; c++) {
452                     var category = templates[c];
453                     sakai.config.Navigation[i].subnav.push({
454                         'id': 'subnavigation_' + category.id + '_link',
455                         'label': category.menuLabel || category.title,
456                         'url': '/create#l=' + category.id
457                     });
458                 }
459             } else if (sakai.config.Navigation[i].id === 'navigation_explore_link' || sakai.config.Navigation[i].id === 'navigation_anon_explore_link') {
460                 for (var x = 0; x < templates.length; x++) {
461                     var categoryx = templates[x];
462                     sakai.config.Navigation[i].subnav.push({
463                         'id': 'subnavigation_explore_' + categoryx.id + '_link',
464                         'label': categoryx.titlePlural,
465                         'url': '/search#l=' + categoryx.id
466                     });
467                 }
468             }
469 
470             if (sakai.config.Navigation[i].subnav) {
471                 temp.subnav = [];
472                 for (var ii in sakai.config.Navigation[i].subnav) {
473                     if (sakai.config.Navigation[i].subnav.hasOwnProperty(ii)) {
474                         temp.subnav.push(getNavItem(ii, sakai.config.Navigation[i].subnav));
475                     }
476                 }
477             }
478             return temp;
479         };

481         /**
482          * Initialise the rendering of the topnavigation menu
483          * @param {Object} templates the available world templates
484          */
485         var renderMenu = function(templates) {
486             var obj = {};
487             var leftMenulinks = [];
488             var rightMenuLinks = [];
489 
490             $('#topnavigation_container .s3d-jump-link').each(function() {
491                 if ($($(this).attr('href') + ':visible').length) {
492                     $(this).show();
493                 }
494             });
495             $('#topnavigation_container .s3d-jump-link').on('click', function() {
496                 $($(this).attr('href')).focus();
497                 return false;
498             });
499 
500             for (var i in sakai.config.Navigation) {
501                 if (sakai.config.Navigation.hasOwnProperty(i)) {
502                     var temp = '';
503                     /* Check that the user is anon, the nav link is for anon
504                      * users, and if the link is the account create link,
505                      * that internal account creation is allowed
506                      */
507                     var anonAndAllowed = sakai.data.me.user.anon &&
508                         sakai.config.Navigation[i].anonymous &&
509                         (
510                             sakai.config.Navigation[i].id !== 'navigation_anon_signup_link' ||
511                             (
512                                 sakai.config.Navigation[i].id === 'navigation_anon_signup_link' &&
513                                 sakai.config.Authentication.allowInternalAccountCreation
514                             )
515                         );
516                     var isNotAnon = !sakai.data.me.user.anon &&
517                         !sakai.config.Navigation[i].anonymous;
518                     var shouldPush = anonAndAllowed || isNotAnon;
519                     if (shouldPush) {
520                         temp = createMenuList(i, templates);
521                         if (sakai.config.Navigation[i].rightLink) {
522                             rightMenuLinks.push(temp);
523                         } else {
524                             leftMenulinks.push(temp);
525                         }
526                     }
527                 }
528             }
529             obj.links = leftMenulinks;
530             obj.selectedpage = true;
531             obj.sakai = sakai;
532             // Get navigation and render menu template
533             $(topnavExploreLeft).html(sakai.api.Util.TemplateRenderer(navTemplate, obj));
534 
535             obj.links = rightMenuLinks;
536             $(topnavExploreRight).html(sakai.api.Util.TemplateRenderer(navTemplate, obj));
537             setCountUnreadMessages();
538             addBinding();
539         };

546         var handleArrowKeyInSearch = function(up) {
547             if ($(topnavSearchResultsContainer).find('li').length) {
548                 var currentIndex = 0,
549                     next = 0;
550                 if ($(topnavSearchResultsContainer).find('li.selected').length) {
551                     currentIndex = $(topnavSearchResultsContainer).find('li').index($(topnavSearchResultsContainer).find('li.selected')[0]);
552                     next = up ? (currentIndex - 1 >= 0 ? currentIndex-1 : -1) : (currentIndex + 1 >= $(topnavSearchResultsContainer).find('li').length ? $(topnavSearchResultsContainer).find('li').length-1 : currentIndex +1);
553                     $(topnavSearchResultsContainer).find('li.selected').removeClass('selected');
554                 }
555                 if (next !== 0 && next === currentIndex) {
556                     next = 0;
557                 } else if (next === -1) {
558                     next = $(topnavSearchResultsContainer).find('li').length - 1;
559                 }
560                 if (next !== -1) {
561                     $(topnavSearchResultsContainer).find('li:eq(' + next + ')').addClass('selected');
562                 }
563                 return false;
564             }
565         };

622             if ($topnavUserOptionsLoginForm.length === 0) {

632                         'method': function(value, element) {

635                         'text': sakai.api.i18n.getValueForKey('INVALID_USERNAME_OR_PASSWORD')
636                     },
637                     'failedloginpassword': {
638                         'method': function(value, element) {
639                             return false;
640                         },
641                         'text': ''
642                     }
643                 }
644             };
645             // Initialize the validate plug-in
646             sakai.api.Util.Forms.validate($topnavUserOptionsLoginForm, validateOpts, true);
