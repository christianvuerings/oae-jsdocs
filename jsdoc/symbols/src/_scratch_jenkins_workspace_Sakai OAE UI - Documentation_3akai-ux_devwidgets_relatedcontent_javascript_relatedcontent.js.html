 19 /*
 20  * Dependencies
 21  *
 22  * /dev/lib/misc/trimpath.template.js (TrimpathTemplates)
 23  * /dev/lib/jquery/plugins/jquery.threedots.js (ThreeDots)
 24  */
 25 
 26 /*global $ */
 27 
 28 require(["jquery", "sakai/sakai.api.core"], function($, sakai) {
 29 
 30 
 31     /**
 32      * @name sakai_global.relatedcontent
 33      *
 34      * @class relatedcontent
 35      *
 36      * @description
 37      * Initialize the relatedcontent widget
 38      *
 39      * @version 0.0.1
 40      * @param {String} tuid Unique id of the widget
 41      * @param {Boolean} showSettings Show the settings of the widget or not
 42      */
 43     sakai_global.relatedcontent = function(tuid,showSettings){
 44 
 45         ///////////////////
 46         // CSS Selectors //
 47         ///////////////////
 48 
 49         var relatedcontent = "#relatedcontent";
 50         var relatedcontentContainer = relatedcontent + "_container";
 51         var relatedcontentDefaultTemplate = relatedcontent + "_default_template";
 52         var relatedcontentContent = ".relatedcontent_content";
 53         var relatedcontentFooter = "#relatedcontent_footer";
 54         var relatedcontentShowMore = "#relatedcontent_show_more";
 55         
 56         var contentData = {};
 57         var page = 0;
 58         var numberofitems = 5;
 59 
 60 
 61         //////////////////////
 62         // Render functions //
 63         //////////////////////
 64 
 65         /**
 66          * Render the template
 67          */
 68         var renderTemplate = function(relatedcontentData){
 69             // Render the relatedcontent
 70             relatedcontentData.sakai = sakai;
 71             if (relatedcontentData.hasOwnProperty("relatedContent") && relatedcontentData.relatedContent.hasOwnProperty("results")) {
 72                 for (var item in relatedcontentData.relatedContent.results) {
 73                     if(relatedcontentData.relatedContent.results.hasOwnProperty(item)){
 74                         relatedcontentData.relatedContent.results[item]["sakai:pooled-content-file-name-dotted"] = sakai.api.Util.applyThreeDots(relatedcontentData.relatedContent.results[item]["sakai:pooled-content-file-name"], $(".relatedcontent").width() - 30, {max_rows: 1,whole_word: false}, "s3d-bold");
 75                     }
 76                 }
 77                 sakai.api.Util.TemplateRenderer(relatedcontentDefaultTemplate, relatedcontentData, $(relatedcontentContainer));
 78                 $(relatedcontentContainer).show();
 79             }
 80         };
 81 
 82 
 83         ////////////////////
 84         // Util functions //
 85         ////////////////////
 86 
 87         /**
 88          * Fetches the related content
 89          */
 90         var getRelatedContent = function() {
 91             var managersList = "";
 92             var viewersList = "";
 93             var ajaxSuccess = function(data) {
 94                 var itemsDisplayed = data.items * (page + 1);
 95                 var moreResults = itemsDisplayed < data.total;
 96                 $.each(data.results, function(index, item){
 97                     data.results[index].commentcount = sakai.api.Content.getCommentCount(item);
 98                     var mimeType = sakai.api.Content.getMimeType(data.results[index]);
 99                     var mimeTypeDescription = sakai.api.i18n.getValueForKey(sakai.config.MimeTypes["other"].description);
100                     if (sakai.config.MimeTypes[mimeType]){
101                         mimeTypeDescription = sakai.api.i18n.getValueForKey(sakai.config.MimeTypes[mimeType].description);
102                     }
103                     data.results[index].mimeTypeDescription = mimeTypeDescription;
104                 });
105                 var json = {
106                     "content": contentData,
107                     "relatedContent": data
108                 };
109                 renderTemplate(json);
110                 if (!moreResults) {
111                     $(relatedcontentShowMore).hide();
112                     $('#relatedcontent_footer').addClass('relatedcontent_footer_norelated');
113                 } else {
114                     $(relatedcontentShowMore).show();
115                     $('#relatedcontent_footer').removeClass('relatedcontent_footer_norelated');
116                 }
117             };
118             var ajaxError = function() {
119                 renderTemplate({});
120             };
121 
122             for (var i = 0; i < contentData.members.managers.length; i++) {
123                 if (contentData.members.managers[i]) {
124                     managersList += " " + (contentData.members.managers[i]["rep:userId"] || contentData.members.managers[i]["sakai:group-id"]);
125                 }
126             }
127             for (var j = 0; j < contentData.members.viewers.length; j++) {
128                 if (contentData.members.viewers[j]) {
129                     viewersList += " " + (contentData.members.viewers[j]["rep:userId"] || contentData.members.viewers[j]["sakai:group-id"]);
130                 }
131             }
132             var searchterm = contentData.data["sakai:pooled-content-file-name"].substring(0,400) + " " + managersList + " " + viewersList;
133             if (contentData.data['sakai:tags'] && contentData.data['sakai:tags'].length) {
134                 searchterm = searchterm + ' ' + contentData.data['sakai:tags'].join(' ');
135             }
136             var searchquery = sakai.api.Server.createSearchString(searchterm, false, 'OR');
137 
138             // get related content for contentData
139             // return some search results for now
140             var params = {
141                 'items': numberofitems,
142                 'page': page
143             };
144             var url = sakai.config.URL.SEARCH_ALL_FILES.replace('.json', '.0.json');
145             if (searchquery === '*' || searchquery === '**') {
146                 url = sakai.config.URL.SEARCH_ALL_FILES_ALL;
147             } else {
148                 params["q"] = searchquery;
149             }
150             $.ajax({
151                 url: url,
152                 data: params,
153                 success: ajaxSuccess,
154                 error: ajaxError
155             });
156             
157         };
158 
159         var showMore = function(){
160             page++;
161             getRelatedContent();
162         };
163 
164         //////////////
165         // Bindings //
166         //////////////
167 
168         /**
169          * Bind the widget's links
170          */
171         var addBinding = function(){
172             // bind the more link
173             $(relatedcontentShowMore).die("click", showMore);
174             $(relatedcontentShowMore).live("click", showMore);
175         };
176 
177         ////////////////////
178         // Initialization //
179         ////////////////////
180 
181         /**
182          * Render function
183          */
184        $(window).bind("render.relatedcontent.sakai", function(e, data){
185            page = 0;
186            addBinding();
187            contentData = data;
188            getRelatedContent();
189         });
190 
191         $(relatedcontentContent).live("click", function(){
192             $.bbq.pushState($(this).attr("data-href"));
193         });
194 
195         // Indicate that the widget has finished loading
196         $(window).trigger("ready.relatedcontent.sakai", {});
197         sakai_global.relatedcontent.isReady = true;
198     };
199 
200     sakai.api.Widgets.widgetLoader.informOnLoad("relatedcontent");
201 });
