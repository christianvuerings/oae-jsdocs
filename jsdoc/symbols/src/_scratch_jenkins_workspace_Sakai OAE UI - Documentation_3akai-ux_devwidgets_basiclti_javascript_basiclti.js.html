 90         // see: http://www.ietf.org/rfc/rfc2396.txt Appendix B
 91         var urlRegExp = new RegExp('^(([^:/?#]+):)?(//([^/?#]*))?([^?#]*)(\\?([^#]*))?(#(.*))?');

 98         /**
 99          * Check if the value is a decimal or not
100          * @param {Object} value Value that needs to be checked
101          * @return {Boolean}
102          *     true: is a decimal
103          *     false: is not a decimal
104          */
105         var isDecimal = function(value){
106             return (/^\d+$/).test(value);
107         };
108 
109         /**
110          * Check if the input url is in fact an url or not
111          * @param {String} url Url that needs to be tested
112          * @return {Boolean}
113          *     true: is an url
114          *     false: is not an url
115          */
116         var isUrl = function(url) {
117             var matches = urlRegExp.exec(url);
118             // e.g. if('http:' && 'localhost')
119             if (matches[1] && matches[4]) {
120                 return true;
121             } else {
122                 return false;
123             }
124         };
125 
126         /**
127          * Check to see if both URLs are in the same origin. See: http://en.wikipedia.org/wiki/Same_origin_policy.
128          * @param {String} url1
129          * @param {String} url2
130          * @return {Boolean}
131          *     true: in the same origin policy
132          *     false: NOT in the same origin policy
133          */
134         var isSameOriginPolicy = function(url1, url2) {
135             if (url1 == url2) {
136                 return true;
137             }
138             // i.e. protocol, domain (and optional port numbers) must match
139             if ((urlRegExp.exec(url1)[2] == urlRegExp.exec(url2)[2]) &&
140                (urlRegExp.exec(url1)[4] == urlRegExp.exec(url2)[4])) {
141                 return true;
142             } else {
143                 return false;
144             }
145         };

148          * Called when the data has been saved to the JCR.

150         var savedDataToJCR = function() {
151             displayRemoteContent(json);

153         };

160         /**
161          * Render the iframe for the widget in settings mode
162          * @param {Boolean} complete Render the preview completely or only adjust values
163          */
164         var renderIframeSettings = function(complete) {
165             if (complete) { 
166                 json.launchDataUrl = sakai.config.URL.SDATA_FETCH_URL.replace(/__PLACEMENT__/, sakai.site.currentsite.id + '/_widgets').replace(/__TUID__/, tuid).replace(/__NAME__/, 'basiclti') + '.launch.html';               
167                 $(basicltiSettingsPreview, $rootel).html(sakai.api.Util.TemplateRenderer($basicltiSettingsPreviewTemplate, json));
168             } else {
169                 $(basicltiSettingsPreviewFrame, $rootel).attr('style', 'border: ' + json.border_size + 'px #' + json.border_color + ' solid');
170             }
171         };
172 
173         /**
174          * Render the iframe for the widget
175          */
176         var renderIframe = function() {
177             if (json) {
178                 json.tuidFrame = basicltiSettingsPreviewId;
179                 $(basicltiMainContainer, $rootel).html(sakai.api.Util.TemplateRenderer($basicltiSettingsPreviewTemplate, json));
180                 json.launchDataUrl = sakai.api.Widgets.getWidgetDataStorageURL(tuid) + '.launch.html';
181                 $('#' + json.tuidFrame, $rootel).attr('src', json.launchDataUrl);
182 
183                 // resize the iframe to match inner body height if in the same origin (i.e. same protocol/domain/port)
184                 if (isSameOriginPolicy(window.location.href, json.ltiurl)) {
185                     $(basicltiSettingsPreviewFrame, $rootel).load(function() {
186                         $(this).height($(this).contents().find('body').height() + 15); // add 10px for IE and 5px more for Gradebook weirdness
187                     });
188                 }
189 
190                 // SAKIII-314 We need to show the container, otherwise the second item won't be shown.
191                 $(basicltiMainContainer, $rootel).show();
192             }
193         };
194

196          * Render the html of the basicltisettings
197          */
198         var renderRemoteContentSettings = function() {
199             if (json) {
200                 $(basicltiSettings, $rootel).html(sakai.api.Util.TemplateRenderer($basicltiSettingsTemplate, json));
201             }
202         };
203 
204         /**
205          * Render the color container
206          */
207         var renderColorContainer = function() {
208             if (json) {
209                 $(basicltiSettingsColorContainer, $rootel).html(sakai.api.Util.TemplateRenderer($basicltiSettingsColorContainerTemplate, json));
210             }
211         };

218         /**
219          * Display the iframe in normal mode
220          * @param {Object} parameters JSON object that contains the necessary information for the iframe
221          */
222         var displayRemoteContent = function(parameters) {
223             // default to some reasonable vaules if the settings node does not have them (maybe a legacy node)
224             if (parameters.border_size === null) {
225                 parameters.border_size = 0;
226             }
227             if (parameters.border_color === null) {
228                 parameters.border_color = 'cccccc';
229             }
230             if (parameters.width === null) {
231                 parameters.width = defaultWidth;
232             }
233             if (parameters.width_unit === null) {
234                 parameters.width_unit = defaultWidthUnit;
235             }
236             if (parameters.frame_height === null) {
237                 parameters.frame_height = defaultHeight;
238             }
239             json = parameters;
240             renderIframe();
241         };
242 
243         var saveContentAjax = function(savedata) {
244             $.ajax({
245                 type: 'POST',
246                 url: sakai.api.Widgets.getWidgetDataStorageURL(tuid),
247                 data: savedata,
248                 success: function(data) {
249                     savedDataToJCR(true, data);
250                 }
251             });
252         };
253 
254         /**
255          * Save the basiclti to the jcr
256          */
257         var saveRemoteContent = function(){
258             var savejson = {
259                 ':operation': 'basiclti',
260                 ':contentType': 'json',
261                 ':replace': true,
262                 ':replaceProperties': true,
263                 '_charset_': 'utf-8'
264             };
265 
266             var dbg = $(basicltiSettingsDebug + ':checked', $rootel).val();
267             var savejson_content = {
268                 'sling:resourceType': 'sakai/basiclti',
269                 'ltiurl': $(basicltiSettingsLtiUrl, $rootel).val() || '',
270                 'ltikey': $(basicltiSettingsLtiKey, $rootel).val() || '',
271                 'ltisecret': $(basicltiSettingsLtiSecret, $rootel).val() || '',
272                 'debug@TypeHint': 'Boolean',
273                 'debug': dbg !== undefined  && dbg !== null,
274                 'release_names@TypeHint': 'Boolean',
275                 'release_names': $('#basiclti_settings_release_names:checked', $rootel).val() !== null,
276                 'release_principal_name@TypeHint': 'Boolean',
277                 'release_principal_name': $('#basiclti_settings_release_principal_name:checked', $rootel).val() !== null,
278                 'release_email@TypeHint': 'Boolean',
279                 'release_email': $('#basiclti_settings_release_email:checked', $rootel).val() !== null,
280                 'launchDataUrl': '', // does not need to be persisted
281                 'tuidFrame': '', // does not need to be persisted
282                 'defined': '' // what the heck is this? Where does it come from?
283             };
284             savejson_content = $.extend({}, json, savejson_content);
285             json = savejson_content;
286             savejson[':content'] = JSON.stringify(savejson_content);
287             saveContentAjax(savejson);
288         };

290         /**
291          * Change the direction (up/down) of the arrow for the advanced settings
292          */
293         var changeAdvancedSettingsArrow = function() {
294             if (isAdvancedSettingsVisible) {
295                 $(basicltiSettingsAdvancedDown, $rootel).hide();
296                 $(basicltiSettingsAdvancedUp, $rootel).show();
297             } else {
298                 $(basicltiSettingsAdvancedUp, $rootel).hide();
299                 $(basicltiSettingsAdvancedDown, $rootel).show();
300             }
301         };

303         /**
304          * Add the validate form handling
305          */
306         var validateForm = function() {
307             var validateOpts = {
308                 submitHandler: function(form, validator){
309                     saveRemoteContent();
310                     return false;
311                 }
312             };
313             sakai.api.Util.Forms.validate($basicltiSettingsForm, validateOpts, true);
314         };

320         /*
321          * Add binding to all the elements
322          */
323         var addBinding = function() {
324             $basicltiSettingsForm = $($basicltiSettingsForm.selector);
325             // Change the url for the iFrame
326             $(basicltiSettingsLtiUrl, $rootel).on('change', function(){
327                 var urlValue = $(this).val();
328                 if (urlValue !== '') {
329                     // Check if someone already wrote http inside the url
330                     if (!isUrl(urlValue)) {
331                         urlValue = 'http://' + urlValue;
332                     }
333                     json.ltiurl = urlValue;
334                     //renderIframeSettings(true); // LDS disabled preview
335                 }
336             });
337 
338             // Change the iframe width
339             $(basicltiSettingsWidth, $rootel).on('change', function(){
340                 var widthValue = $(basicltiSettingsWidth, $rootel).val();
341 
342                 if (isDecimal(widthValue)) {
343                     json.width = widthValue;
344                 }
345                 renderIframeSettings(false);
346             });
347 
348             // Change the iframe height
349             $(basicltiSettingsHeight, $rootel).on('change', function(){
350                 var heightValue = $(basicltiSettingsHeight, $rootel).val();
351                 if (isDecimal(heightValue)) {
352                     json.frame_height = heightValue;
353                 }
354                 renderIframeSettings(false);
355             });
356 
357             // Change the border width
358             $(basicltiSettingsBorders, $rootel).on('change', function() {
359                 var borderValue = $(basicltiSettingsBorders, $rootel).val();
360                 if (isDecimal(borderValue)) {
361                     json.border_size = borderValue;
362                     renderIframeSettings(false);
363                 }
364             });
365 
366             // Toggle the advanced view
367             $(basicltiSettingsAdvancedToggleSettings, $rootel).on('click', function() {
368                 $('#basiclti_settings_advanced', $rootel).toggle();
369                 isAdvancedSettingsVisible = !isAdvancedSettingsVisible;
370                 changeAdvancedSettingsArrow();
371             });
372 
373             // When you click on one of the width units (px or percentage)
374             $(basicltiSettingsWidthUnitClass, $rootel).on('click', function() {
375                 var widthUnitValue = $(this).attr('id').split('_')[$(this).attr('id').split('_').length - 1];
376                 if (widthUnitValue === 'px') {
377                     json.width_unit = widthUnitValue;
378                 } else {
379                     json.width_unit = '%';
380                 }
381                 $(basicltiSettingsWidthUnitClass, $rootel).removeClass(basicltiSettingsWidthUnitSelectedClass);
382                 $(this).addClass(basicltiSettingsWidthUnitSelectedClass);
383                 renderIframeSettings(false);
384             });
385 
386             // Add the validate handler
387             validateForm();
388 
389             // Cancel the settings
390             $(basicltiSettingsCancel, $rootel).on('click', function() {
391                 sakai.api.Widgets.Container.informCancel(tuid, 'basiclti');
392             });
393 
394             $('.basiclti_settings_color', $rootel).on('click', function() {
395                 json.border_color = $(this).attr('id').split('_')[$(this).attr('id').split('_').length - 1];
396                 renderIframeSettings(false);
397                 renderColorContainer();
398             });
399         };

406         /**
407          * Function that fills in the input fields in the settings tab.
408          * @param {Object} parameters A JSON object that contains the necessary information.
409          * @param {Boolean} exists Does there exist a previous basiclti
410          */
411         var displaySettings = function(parameters, exists) {
412             if (exists && parameters.ltiurl) {
413                 if (parameters[':content']){
414                     json = $.parseJSON(parameters[':content']);
415                 } else {
416                     json = parameters;
417                 }
418             } else { // use default values
419                 json = {
420                     ltiurl: '',
421                     ltikey: '',
422                     ltisecret: '',
423                     release_names: true,
424                     release_principal_name: true,
425                     release_email: true,
426                     border_size: 0,
427                     border_color: 'ccc',
428                     frame_height: defaultHeight,
429                     width: defaultWidth,
430                     width_unit: defaultWidthUnit,
431                     debug: false
432                 };
433             }
434             renderRemoteContentSettings();
435             renderColorContainer();
436             addBinding();
437             changeAdvancedSettingsArrow();
438             $(basicltiSettings, $rootel).show();
439         };
440 
441         /**
442          * Will fetch the URL and other parameters from the JCR and according to which
443          * view we are in, fill in the settings or display an iframe.
444          */
445         var getRemoteContent = function() {
446             sakai.api.Widgets.loadWidgetData(tuid, function(success,data){
447                 if (success) {
448                     if (showSettings) {
449                         displaySettings(data,true);
450                     } else {
451                         displayRemoteContent(data);
452                     }
453                 } else {
454                     displaySettings(null, false);
455                 }
456             }, false);
457         };
458 
459         /*
460          * Is the widget in settings mode or not
461          */
462         if (showSettings) {
463             $(basicltiMainContainer, $rootel).hide();
464             $(basicltiSettings, $rootel).show();
465         } else {
466             $(basicltiSettings, $rootel).hide();
467             $(basicltiMainContainer, $rootel).show();
468         }
469 
470         getRemoteContent();
471     };
