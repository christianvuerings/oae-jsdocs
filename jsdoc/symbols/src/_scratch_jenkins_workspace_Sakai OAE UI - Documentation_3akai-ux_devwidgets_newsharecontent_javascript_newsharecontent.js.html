 19 require(['jquery', 'sakai/sakai.api.core', 'underscore'], function($, sakai, _) {
 20 
 21     /**
 22      * @name sakai_global.newsharecontent
 23      *
 24      * @class newsharecontent
 25      *
 26      * @description
 27      * Content Share widget
 28      *
 29      * @version 0.0.1
 30      * @param {String} tuid Unique id of the widget
 31      * @param {Boolean} showSettings Show the settings of the widget or not
 32      */
 33     sakai_global.newsharecontent = function(tuid, showSettings){
 34 
 35         /////////////////////////////
 36         // CONFIGURATION VARIABLES //
 37         /////////////////////////////
 38 
 39         // Containers
 40         var $newsharecontentContainer = $("#newsharecontent_widget");
 41         var $newsharecontentCanShareContainer = $('#newsharecontent_canshare_container');
 42         var $newsharecontentCantShareContainer = $('#newsharecontent_cantshare_container');
 43         var $newsharecontentMessageContainer = $("#newsharecontent_message_container");
 44 
 45         // Elements
 46         var $newsharecontentLinkURL = $("#newsharecontent_linkurl");
 47         var $newsharecontentSharelist = $("#newsharecontent_sharelist");
 48         var $newsharecontentMessage = $("#newsharecontent_message");
 49         var $newsharecontentSendButton = $("#sharecontent_send_button");
 50         var newsharecontentListItem = ".as-selection-item";
 51         var newsharecontentShareListContainer = "#newsharecontent_sharelist_container";
 52         var $newsharecontentMessageToggle = $('label.toggletext',$newsharecontentContainer);
 53         var $newsharecontentMessageArrow = $('#newsharecontent_message_arrow');
 54         var $newsharecontentHeading = $('#newsharecontent_heading');
 55         var $newsharecontentAnon = $('.newsharecontent_anon_function');
 56         var $newsharecontentUser = $('.newsharecontent_user_function');
 57         var $newsharecontent_form = $("#newsharecontent_form");
 58         var $newsharecontentTitle = $('#newsharecontent_title');
 59 
 60         // Templates
 61         var $newsharecontentCantShareTemplate = $('#newsharecontent_cantshare_template');
 62 
 63         // Classes
 64         var newsharecontentRequiredClass = "newsharecontent_required";
 65 
 66         // Content object
 67         var contentObj = {};
 68 
 69         
 70         ///////////////
 71         // RENDERING //
 72         ///////////////
 73 
 74 
 75         /**
 76          * Render the list of files that we can't share
 77          * @param {Object} cantShareFiles A list of all the files that we can't share
 78          */
 79         var renderCantShare = function(cantShareFiles) {
 80             $newsharecontentCantShareContainer.html(
 81                 sakai.api.Util.TemplateRenderer($newsharecontentCantShareTemplate, {
 82                     'files': cantShareFiles
 83                 })
 84             );
 85         };
 86 
 87         /**
 88          * Get all the files you can share with other people
 89          * @param {Object} files A list of all the files
 90          * @return {Object} A list of the files you can share with other people
 91          */
 92         var getCanShareFiles = function(files) {
 93             return _.filter(files, function(file) {
 94                 return sakai.api.Content.canCurrentUserShareContent(file.body);
 95             });
 96         };
 97 
 98         /**
 99          * Add validation
100          */
101         var addShareValidation = function() {
102             var validateOpts = {
103                 'methods': {
104                     'requiredsuggest': {
105                         'method': function(value, element) {
106                             return $.trim($(element).next('input.as-values').val()).replace(/,/g, '') !== '';
107                         },
108                         'text': sakai.api.i18n.getValueForKey('AUTOSUGGEST_REQUIRED_ERROR')
109                     }
110                 },
111                 submitHandler: doShare
112             };
113             sakai.api.Util.Forms.validate($newsharecontent_form, validateOpts, true);
114         };
115 
116         var fillShareData = function(hash){
117             addShareValidation();
118 
119             $newsharecontentLinkURL.val(contentObj.shareUrl);
120 
121             var cantShareFiles = _.filter(contentObj.data, function(file) {
122                 return !sakai.api.Content.canCurrentUserShareContent(file.body);
123             });
124             var shareFiles = getCanShareFiles(contentObj.data);
125             var filenames = sakai.api.Util.TemplateRenderer('newsharecontent_filenames_template', {
126                 'files': shareFiles
127             });
128             var shareURLs = sakai.api.Util.TemplateRenderer('newsharecontent_fileURLs_template', {
129                 'files': shareFiles,
130                 'sakai': sakai
131             });
132             var shareData = {
133                 'filename': filenames,
134                 'data': shareFiles,
135                 'path': shareURLs,
136                 'user': sakai.api.User.getFirstName(sakai.data.me.profile)
137             };
138             $newsharecontentMessage.html(sakai.api.Util.TemplateRenderer("newsharecontent_share_message_template", shareData));
139 
140             renderCantShare(cantShareFiles);
141 
142             if (shareFiles.length) {
143                 $newsharecontentCanShareContainer.show();
144                 $newsharecontentTitle.show();
145             } else {
146                 $newsharecontentCanShareContainer.hide();
147                 $newsharecontentTitle.hide();
148             }
149 
150             if (hash) {
151                 hash.w.show();
152             }
153             var tbx = $('#toolbox');
154             if (tbx.find("a").length) {
155                 tbx.find("a").remove();
156             }
157             var svcs = {facebook: 'Facebook', twitter: 'Twitter', delicious:'Delicious', stumbleupon: 'StumbleUpon', blogger:'Blogger', wordpress:'Wordpress', google:'Google', expanded: 'More'};
158             var addThisTitle ="";
159             for (var s in svcs) {
160                 if (s==='twitter'){
161                     addThisTitle = sakai.api.i18n.getValueForKey("SHARE_EXT_MSG1",'newsharecontent')+shareData.filename.replace(/"/gi,'')+' '+sakai.api.i18n.getValueForKey("SHARE_EXT_MSG2",'newsharecontent')+' ' +sakai.api.i18n.getValueForKey("TITLE_PLAIN");
162                 }
163                 else{
164                     addThisTitle =  shareData.filename.replace(/"/gi,'')+' '+sakai.api.i18n.getValueForKey("SHARE_EXT_MSG2",'newsharecontent')+' ' + sakai.api.i18n.getValueForKey("TITLE_PLAIN");
165                 }
166                 tbx.append('<a class="addthis_button_'+s+'" addthis:url="'+shareData.path+'" addthis:title="'+addThisTitle+'"></a>');
167             }
168             addthis.toolbox("#toolbox");
169         };
170 
171         var resetWidget = function(hash){
172             $newsharecontentMessageContainer.hide();
173             $newsharecontentMessageArrow.removeClass('arrow_down');
174             $newsharecontentMessage.removeClass(newsharecontentRequiredClass);
175             $(newsharecontentShareListContainer).removeClass(newsharecontentRequiredClass);
176             sakai.api.Util.AutoSuggest.reset($newsharecontentSharelist);
177             $(window).trigger("hiding.newsharecontent.sakai");
178             hash.w.hide();
179         };
180 
181         ///////////
182         // SHARE //
183         ///////////
184 
185         var getSelectedList = function() {
186             var list = $("#as-values-" + tuid).val();
187             // this value is a comma-delimited list
188             // split it and get rid of any empty values in the array
189             list = list.split(",");
190             var removed = 0;
191             $(list).each(function(i, val) {
192 
193                if (val === "") {
194                    list.splice(i - removed, 1);
195                    removed += 1;
196                }
197             });
198 
199             // Create list to show in the notification
200             var toAddNames = [];
201             $("#newsharecontent_container .as-selection-item").each(function(){
202                 // In IE 7 </A> is returned and in firefox </a>
203                 toAddNames.push($(this).html().split(/<\/[aA]?>/g)[1]);
204             });
205             var returnValue = {"list":list, "toAddNames":toAddNames};
206 
207             return returnValue;
208         };
209 
210         var createActivity = function(activityMessage, canShareFiles) {
211             var activityData = {
212                 "sakai:activityMessage": activityMessage
213             };
214             $.each(canShareFiles, function(i, content){
215                 sakai.api.Activity.createActivity("/p/" + content.body["_path"], "content", "default", activityData);
216             });
217             $(window).trigger("load.content_profile.sakai", function(){
218                 $(window).trigger("render.entity.sakai", ["content", contentObj]);
219             });
220         };
221 
222         var doShare = function(event, userlist, message, contentobj, role) {
223             var userList = userlist || getSelectedList();
224             var messageText = message || $.trim($newsharecontentMessage.val());
225             contentObj = contentobj || contentObj;
226             var canShareFiles = getCanShareFiles(contentObj.data);
227             $newsharecontentMessage.removeClass(newsharecontentRequiredClass);
228             $(newsharecontentShareListContainer).removeClass(newsharecontentRequiredClass);
229             if (userList && userList.list && userList.list.length && messageText && canShareFiles) {
230                 var toAddList = userList.list.slice();
231                 userList.list = toAddList;
232                 if (toAddList.length) {
233                     sakai.api.Communication.sendMessage(userList.list,
234                         sakai.data.me,
235                         sakai.api.i18n.getValueForKey('I_WANT_TO_SHARE', 'newsharecontent') + sakai.api.Util.TemplateRenderer('newsharecontent_filenames_template', {
236                             'files': canShareFiles
237                         }), messageText, 'message', false, false, true, 'shared_content'
238                     );
239                     $.each(canShareFiles, function(i, content){
240                         if (sakai.api.Content.Collections.isCollection(content.body)){
241                             sakai.api.Content.Collections.shareCollection(content.body['_path'], toAddList, role, function() {
242                                 createActivity('ADDED_A_MEMBER', canShareFiles);
243                             });
244                         } else {
245                             sakai.api.Content.addToLibrary(content.body['_path'], toAddList, role, function() {
246                                 createActivity('ADDED_A_MEMBER', canShareFiles);
247                             });
248                         }
249                     });
250                     sakai.api.Util.notification.show(false, $("#newsharecontent_users_added_text").text() + " " + userList.toAddNames.join(", "), "");
251                     $newsharecontentContainer.jqmHide();
252                 }
253             } else {
254                 if (!messageText) {
255                     $newsharecontentMessage.addClass(newsharecontentRequiredClass);
256                     sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("NO_MESSAGE_PROVIDED", "newsharecontent"), sakai.api.i18n.getValueForKey("A_MESSAGE_SHOULD_BE_PROVIDED_TO_SHARE", "newsharecontent"));
257                 }
258                 if (!userList.list.length) {
259                     $(newsharecontentShareListContainer).addClass(newsharecontentRequiredClass);
260                     sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("NO_USERS_SELECTED", "newsharecontent"), sakai.api.i18n.getValueForKey("NO_USERS_TO_SHARE_FILE_WITH", "newsharecontent"));
261                 }
262                 if (!contentObj || !canShareFiles) {
263                     $(newsharecontentShareListContainer).addClass(newsharecontentRequiredClass);
264                     sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("SHARING_FAILED", "newsharecontent"), sakai.api.i18n.getValueForKey("SHARING_FAILED_FULL_MESSAGE", "newsharecontent"));
265                 }
266             }
267         };
268 
269         //////////////
270         // BINDINGS //
271         //////////////
272 
273         var addBinding = function(){
274             $newsharecontentContainer.jqm({
275                 modal: false,
276                 overlay: 0,
277                 toTop: true,
278                 zIndex: 3000,
279                 onShow: fillShareData,
280                 onHide: resetWidget
281             });
282 
283             $(document).on('click', '.share_trigger_click', function() {
284                 if ($newsharecontentContainer.is(':visible')) {
285                     $newsharecontentContainer.jqmHide();
286                 }
287                 sakai.api.Util.Forms.clearValidation($newsharecontent_form);
288                 var idArr = $(this).attr("data-entityid");
289                 if (idArr && idArr.length) {
290                     if (!$.isArray(idArr)) {
291                         idArr = idArr.split(',');
292                     }
293                     var $this = $(this);
294                     $newsharecontentContainer.css({
295                         'top':$this.offset().top + $this.height(),
296                         'left':$this.offset().left + $this.width() / 2 - 119
297                     });
298                     // Fetch data for content items
299                     var batchRequests = [];
300                     $.each(idArr, function(i, id) {
301                         batchRequests.push({
302                             'url': '/p/' + id + '.json',
303                             'method': 'GET'
304                         });
305                     });
306                     sakai.api.Server.batch(batchRequests, function(success, data) {
307                         if (success && data) {
308                             if (data.results) {
309                                 $.each(data.results, function(i, result){
310                                     data.results[i].body = $.parseJSON(data.results[i].body);
311                                 });
312                                 contentObj = {
313                                     data: data.results,
314                                     shareUrl: sakai.api.Content.createContentURL(data.results[0].body)
315                                 };
316                             } else if (data.url) {
317                                 contentObj = {
318                                     data: [data],
319                                     shareUrl:  sakai.api.Content.createContentURL(data)
320                                 };
321                             }
322                             require(['//s7.addthis.com/js/250/addthis_widget.js?%23pubid=' + sakai.widgets.newsharecontent.defaultConfiguration.newsharecontent.addThisAccountId + '&domready=1'], function() {
323                                 $newsharecontentContainer.jqmShow();
324                             });
325                         }
326                     });
327                 }
328             });
329         };
330 
331         $newsharecontentMessageToggle.add($newsharecontentMessageArrow).bind('click',function(){
332             $newsharecontentMessageArrow.toggleClass('arrow_down');
333             $newsharecontentMessageContainer.stop(true, true).slideToggle();
334         });
335 
336         sakai.api.Util.hideOnClickOut(".newsharecontent_dialog", ".share_trigger_click", function(){
337             $newsharecontentContainer.jqmHide();
338         });
339 
340         $(window).on("finished.sharecontent.sakai", doShare);
341 
342         ////////////////////
343         // INITIALIZATION //
344         ////////////////////
345 
346         var init = function(){
347             if (!sakai.data.me.user.anon) {
348                 $newsharecontentAnon.hide();
349                 $newsharecontentUser.show();
350             } else {
351                 $newsharecontentContainer.addClass('anon');
352             }
353             addBinding();
354             sakai.api.Util.AutoSuggest.setup( $newsharecontentSharelist, {
355                 asHtmlID: tuid,
356                 scrollHeight: 120
357             });
358             $("label#newsharecontent_autosuggest_for").attr("for", tuid);
359         };
360 
361         init();
362     };
363 
364     sakai.api.Widgets.widgetLoader.informOnLoad("newsharecontent");
365 });
