 19 require(['jquery', 'sakai/sakai.api.core'], function($, sakai) {
 20 
 21     /**
 22      * @name sakai_global.branding
 23      *
 24      * @param {String} tuid Unique id of the widget
 25      * @param {Boolean} showSettings Show the settings of the widget or not
 26      */
 27     sakai_global.branding = function(tuid, showSettings) {
 28 
 29         var $rootel = $('#' + tuid);
 30         var $brandingWidget = $('.branding_widget', $rootel);
 31 
 32         var doInit = function() {
 33             if (sakai.config.enableBranding) {
 34                 $brandingWidget.show();
 35             }
 36         };
 37 
 38         doInit();
 39 
 40     };
 41 
 42     sakai.api.Widgets.widgetLoader.informOnLoad('branding');
 43 });
