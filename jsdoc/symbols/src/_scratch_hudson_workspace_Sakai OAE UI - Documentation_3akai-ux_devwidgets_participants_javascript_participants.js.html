 19 // load the master sakai object to access all Sakai OAE API methods
 20 require(["jquery", "sakai/sakai.api.core"], function($, sakai) {
 21 
 22     /**
 23      * @name sakai_global.participants
 24      *
 25      * @class participants
 26      *
 27      * @version 0.0.1
 28      * @param {String} tuid Unique id of the widget
 29      * @param {Boolean} showSettings Show the settings of the widget or not
 30      */
 31     sakai_global.participants = function (tuid, showSettings, widgetData) {
 32 
 33         var rootel = $("#" + tuid);
 34 
 35         /////////////////////////////
 36         // Configuration variables //
 37         /////////////////////////////
 38         var infinityScroll = false;
 39 
 40         // Containers
 41         var $participantsListContainer = $("#participants_list_container_list", rootel);
 42 
 43         // Templates
 44         var participantsListTemplate = "participants_list_template";
 45         var participantsListTemplateEmpty = "participants_list_empty_template";
 46 
 47         // Elements
 48         var $participantsSearchField = $("#participants_search_field", rootel);
 49         var participantsListParticipantRequestConnection = ".participants_list_participant_request_connection";
 50         var $participantsSelectAll = $("#participants_select_all", rootel);
 51         var participantsListParticipantCheckbox = ".participants_list_participant_checkbox input:checkbox";
 52         var $participantsSendSelectedMessage = $("#participants_send_selected_message", rootel);
 53         var $participantsAddPeopleButton = $("#participants_addpeople_button", rootel);
 54         var participantsListParticipantName = ".participants_list_participant_name";
 55         var $participants_sort_by = $("#participants_sort_by", rootel);
 56         var participantsShowGrid = $(".s3d-listview-grid", rootel);
 57         var participantsShowList = $(".s3d-listview-list", rootel);
 58         widgetData.listStyle = "list";
 59         widgetData.query = "";
 60         widgetData.sortby = "asc";
 61 
 62         var newlyAdded = [],
 63             roles = false;
 64 
 65         ///////////////////////
 66         // Utility functions //
 67         ///////////////////////
 68 
 69         var enableDisableButtons = function(){
 70             if($(participantsListParticipantCheckbox + ":checked", rootel).length){
 71                 $participantsSendSelectedMessage.removeAttr("disabled");
 72                 $participantsAddPeopleButton.removeAttr("disabled");
 73             } else {
 74                 $participantsSendSelectedMessage.attr("disabled", "disabled");
 75                 $participantsSelectAll.removeAttr("checked");
 76                 $participantsAddPeopleButton.attr("disabled", "disabled");
 77             }
 78         };
 79 
 80         /**
 81          * Set the attributes needed by the sendmessage widget to send a message to all selected users
 82          */
 83         var setSendSelectedMessageAttributes = function(){
 84             var userArr = [];
 85             var userIDArr = [];
 86             $.each($(participantsListParticipantCheckbox + ":checked", rootel), function(index, item){
 87                 userIDArr.push($(item).attr("data-entityid"));
 88                 userArr.push($(item).attr("data-entityname"));
 89             });
 90             $participantsSendSelectedMessage.attr("sakai-entitytype", "user");
 91             $participantsSendSelectedMessage.attr("sakai-entityname", userArr);
 92             $participantsSendSelectedMessage.attr("sakai-entityid", userIDArr);
 93             $participantsAddPeopleButton.attr("data-entityname", userArr);
 94             $participantsAddPeopleButton.attr("data-entityid", userIDArr);
 95             enableDisableButtons();
 96         };
 97 
 98         /**
 99          * Check/Uncheck all items in the members list and enable/disable buttons
100          */
101         var checkAll = function(){
102             if($(this).is(":checked")){
103                 $(participantsListParticipantCheckbox, rootel).attr("checked","checked");
104                 setSendSelectedMessageAttributes();
105             }else{
106                 $(participantsListParticipantCheckbox, rootel).removeAttr("checked");
107                 enableDisableButtons();
108             }
109         };
110 
111 
112         var getRoleTitle = function(result) {
113             var ret = "";
114             if (result.role && result.role.title) {
115                 ret = result.role.title;
116             } else if (result.role === undefined && newlyAdded && newlyAdded.length && roles && roles.length) {
117                 $.each(newlyAdded, function(i, member) {
118                     if (member.user === result["rep:userId"] || member.user === result["sakai:group-id"]) {
119                         $.each(roles, function(i, role) {
120                             if (role.id === member.permission) {
121                                 ret = role.title;
122                             }
123                         });
124                     }
125                 });
126             }
127             return ret;
128         };
129 
130         //////////////////////
131         // Render functions //
132         //////////////////////
133 
134         var processParticipants = function (results, callback){
135             var participantsArr = [];
136             if (results && results.length) {
137                 $.each(results, function(i, result) {
138                     if(result['sakai:group-id'] || result['rep:userId']) {
139                         var contentCount = 0;
140                         var contactsCount = 0;
141                         var membershipsCount = 0;
142                         if (result.counts) {
143                             contentCount = result.counts.contentCount;
144                             contactsCount = result.counts.contactsCount;
145                             membershipsCount = result.counts.membershipsCount;
146                         }
147                         var picture = false;
148                         var roleTitle = getRoleTitle(result);
149                         if (result['sakai:group-id']) {
150                             if (result.basic.elements.picture && result.basic.elements.picture.value) {
151                                 picture = sakai.api.Groups.getProfilePicture(result);
152                             }
153                             participantsArr.push({
154                                 'name': result['sakai:group-title'],
155                                 'id': result['sakai:group-id'],
156                                 'title': roleTitle,
157                                 'type': 'group',
158                                 'connected': false,
159                                 'content': contentCount,
160                                 'contacts': contactsCount,
161                                 'memberships': membershipsCount,
162                                 'profilePicture': picture,
163                                 'membersCount': result.counts.membersCount
164                             });
165                         } else {
166                             // Check if this user is a friend of us already.
167                             var connected = false;
168                             var invited = false;
169                             var pending = false;
170                             var none = false;
171                             if (result['sakai:state'] === 'ACCEPTED') {
172                                 connected = true;
173                             } else if (result['sakai:state'] === 'INVITED') {
174                                 connected = true;
175                                 invited = true;
176                             } else if (result['sakai:state'] === 'PENDING') {
177                                 connected = true;
178                                 pending = true;
179                             } else if (result['sakai:state'] === 'NONE') {
180                                 none = true;
181                             }
182 
183                             if (result.basic.elements.picture && result.basic.elements.picture.value) {
184                                 picture = sakai.api.User.getProfilePicture(result);
185                             }
186                             participantsArr.push({
187                                 'name': sakai.api.User.getDisplayName(result),
188                                 'id': result['rep:userId'],
189                                 'title': roleTitle,
190                                 'type': 'user',
191                                 'content': contentCount,
192                                 'contacts': contactsCount,
193                                 'memberships': membershipsCount,
194                                 'connected': connected,
195                                 'invited': invited,
196                                 'pending': pending,
197                                 'none': none,
198                                 'profilePicture': picture
199                             });
200                         }
201                     }
202                 });
203             }
204             callback(participantsArr);
205         };
206 
207         ////////////////////
208         // Init functions //
209         ////////////////////
210 
211         var loadParticipants = function(cache){
212             // Disable the previous infinite scroll
213             if (infinityScroll){
214                 infinityScroll.kill();
215             }
216             // Set up the infinite scroll for the list of items in the library
217             infinityScroll = $participantsListContainer.infinitescroll(function(parameters, callback){
218                 sakai.api.Groups.searchMembers(widgetData.participants.groupid, widgetData.query, parameters.items, parameters.page, parameters.sortBy, parameters.sortOrder, function(success, data){
219                     callback(true, data);
220                 }, cache);
221             }, {
222                 "query": widgetData.query,
223                 "sortBy": "lastName",
224                 "sortOrder": widgetData.sortby
225             }, function(items, total){
226                 if (sakai.data.me.user.anon){
227                     $(".s3d-page-header-top-row", rootel).show();
228                 } else {
229                     $(".s3d-page-header-top-row", rootel).show();
230                     $(".s3d-page-header-bottom-row", rootel).show();
231                 }
232                 $participantsSelectAll.removeAttr("checked");
233                 setSendSelectedMessageAttributes();
234                 return sakai.api.Util.TemplateRenderer(participantsListTemplate, {
235                     "participants": items,
236                     "sakai": sakai
237                 });
238             }, function(){
239                 $participantsListContainer.html(sakai.api.Util.TemplateRenderer(participantsListTemplateEmpty, {}));
240             }, sakai.config.URL.INFINITE_LOADING_ICON, processParticipants);
241         };
242 
243         var handleHashChange = function(){
244             $(".s3d-listview-options", rootel).find("div").removeClass("selected");
245             var ls = $.bbq.getState("ls") || widgetData.listStyle;
246             if (ls === "list"){
247                 $("#participants_list_container_list", rootel).removeClass("s3d-search-results-grid");
248                 $(participantsShowList, rootel).addClass("selected");
249                 $(participantsShowList, rootel).children().addClass("selected");
250             } else if (ls === "grid"){
251                 $(participantsShowGrid, rootel).addClass("selected");
252                 $(participantsShowGrid, rootel).children().addClass("selected");
253                 $("#participants_list_container_list", rootel).addClass("s3d-search-results-grid");
254             }
255             widgetData.query = $.bbq.getState("pq") || "";
256             $participantsSearchField.val(widgetData.query);
257             widgetData.sortby = $.bbq.getState("psb") || "asc";
258             $participants_sort_by.val(widgetData.sortby);
259             loadParticipants();
260         };
261 
262         var addBinding = function(){
263             $(window).bind("hashchanged.participants.sakai", handleHashChange);
264 
265             $(".participants_widget .s3d-search-button").unbind("click").bind("click", function(){
266                 currentPage = 1;
267                 $.bbq.pushState({"pq": $.trim($participantsSearchField.val())});
268             });
269             $participantsSearchField.unbind("keyup").bind("keyup", function(ev) {
270                 if (ev.keyCode === 13) {
271                     $.bbq.pushState({"pq": $.trim($participantsSearchField.val())});
272                 }
273             });
274             $participants_sort_by.unbind("change").bind("change", function(){
275                 $.bbq.pushState({"psb": $participants_sort_by.val()});
276             });
277             $participantsSelectAll.unbind("click").bind("click", checkAll);
278             $(participantsListParticipantCheckbox, rootel).live("click", setSendSelectedMessageAttributes);
279 
280             $(".participants_accept_invitation").live("click", function(ev){
281                 var userid = $(this).attr("sakai-entityid");
282                 sakai.api.User.acceptContactInvite(userid, function(){
283                     $('.participants_accept_invitation').each(function(index) {
284                         if ($(this).attr("sakai-entityid") === userid){
285                             $(this).hide();
286                         }
287                     });
288                 });
289             });
290 
291             $(window).bind("sakai.addToContacts.requested", function(ev, userToAdd){
292                 $('.sakai_addtocontacts_overlay').each(function(index) {
293                     if ($(this).attr("sakai-entityid") === userToAdd.uuid){
294                         $(this).hide();
295                     }
296                 });
297             });
298 
299             $(participantsShowList, rootel).click(function(){
300                 $.bbq.pushState({"ls": "list"});
301             });
302 
303             $(participantsShowGrid, rootel).click(function(){
304                 $.bbq.pushState({"ls": "grid"});
305             });
306 
307             $('.addpeople_init', rootel).on('click', function() {
308                 $(document).trigger('init.addpeople.sakai', {
309                     editingGroup: true
310                 });
311             });
312         };
313 
314         var init = function(){
315             var groupData = $.extend(true, {}, sakai_global.group.groupData);
316             groupData.roles = groupData["sakai:roles"];
317             roles = sakai.api.Groups.getRoles(groupData);
318             if (sakai.api.Groups.isCurrentUserAManager(sakai_global.group.groupId, sakai.data.me, groupData)){
319                 $("#participants_manage_participants").show();
320             }
321             addBinding();
322             handleHashChange();
323         };
324 
325         $(window).bind("usersselected.addpeople.sakai", function(e, _newlyAdded){
326             newlyAdded = _newlyAdded;
327             setTimeout(function() {
328                 loadParticipants(false);
329             }, 1000);
330         });
331 
332         init();
333 
334     };
335 
336     // inform Sakai OAE that this widget has loaded and is ready to run
337     sakai.api.Widgets.widgetLoader.informOnLoad("participants");
338 });
