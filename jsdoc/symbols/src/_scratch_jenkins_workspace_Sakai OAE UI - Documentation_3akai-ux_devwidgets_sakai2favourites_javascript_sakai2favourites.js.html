 26 /*global $ */
 27 
 28 require(["jquery", "sakai/sakai.api.core"], function($, sakai) {
 29 
 30     /**
 31      * @name sakai_global.sakai2favourites
 32      *
 33      * @class sakai2favourites
 34      *
 35      * @description
 36      * Creategroup widget
 37      *
 38      * @version 0.0.1
 39      * @param {String} tuid Unique id of the widget
 40      * @param {Boolean} showSettings Show the settings of the widget or not
 41      */
 42     sakai_global.sakai2favourites = function(tuid, showSettings){
 43 
 44         var sakai2CategoryList = "#sakai2_category_list";
 45         var sakai2CategoryListTemplate = "#sakai2_category_list_template";
 46         var siteListsjson = false;
 47         var selectedSiteJson = false;
 48 
 49         ///////////////////////
 50         // Utility functions //
 51         ///////////////////////
 52 
 53         /**
 54          * Public function that can be called from elsewhere
 55          * (e.g. sites widget)
 56          * It initializes the creategroup widget and shows the jqmodal (ligthbox)
 57          */
 58         $(window).bind("init.sakai2favourites.sakai", function(){
 59             // render Category List
 60             loadSakai2SiteList();
 61             // render all sites list
 62             renderSiteList();
 63             // render selected site List
 64             renderSelectedList();
 65             // show dialog
 66             $("#sakai2favourites_container").jqmShow();
 67             // bind events
 68             bindEvents();
 69         });
 70 
 71         /**
 72          * This method render Category List.
 73          */
 74         var loadSakai2SiteList = function(){
 75             siteListsjson.categories = siteListsjson.categories || [];
 76             $(sakai2CategoryList).html(sakai.api.Util.TemplateRenderer(sakai2CategoryListTemplate.replace(/#/, ''), siteListsjson));
 77         };
 78 
 79        /**
 80          * This method render all sites or sites related to the categoryname.
 81          * @param {string} categoryname The category(all:render all sites, categoryname:render sites related to the category)
 82          */
 83         var renderSiteList = function(categoryname){
 84             var siteListJson = {};
 85             siteListJson.sites = [];
 86 
 87             //loop through each category and get the related sites
 88             for(var i in siteListsjson.categories){
 89                 if (siteListsjson.categories.hasOwnProperty(i)) {
 90                     var category = siteListsjson.categories[i];
 91                     // if category name is all
 92                     if (!categoryname) {
 93                         // get all the unique site lists
 94                         siteListJson = getAllSites(category, siteListJson);
 95                     // if category name is equal to the category.categoryname
 96                     // for example categoryname = i18n_moresite_01_all_sites
 97                     }
 98                     else
 99                         if (category.category.replace(/ /g, "_") === categoryname) {
100                             // get the site lists in certain category
101                             // for example get site list in i18n_moresite_01_all_sites
102                             siteListJson.sites = siteListsjson.categories[i].sites;
103                             break;
104                         }
105                 }
106             }
107             // render the sites
108             $("#sakai2_site_list").html(sakai.api.Util.TemplateRenderer("#sakai2_site_list_template".replace(/#/, ''),siteListJson));
109             // select related checkboxes based on the selectedlistjson
110             setSite();
111         };
112 
113        /**
114          * This method render all sites selected to display in my sakai2 favourites
115          */
116         var renderSelectedList = function(){
117             $("#sakai2favourites_selected_site_list").html(sakai.api.Util.TemplateRenderer("#sakai2_selected_site_list_template".replace(/#/, ''),sakai.data.me.sakai2List));
118         };
119 
120 
121         var myClose = function(hash) {
122             hash.o.remove();
123             hash.w.hide();
124         };
125 
126         ////////////////////
127         // Event Handlers //
128         ////////////////////
129 
130 
131         /*
132          * Add jqModal functionality to the container.
133          * This makes use of the jqModal (jQuery Modal) plugin that provides support
134          * for lightboxes
135          */
136         $("#sakai2favourites_container").jqm({
137             modal: true,
138             overlay: 20,
139             toTop: true,
140             onHide: myClose
141         });
142 
143         /**
144          * Add event handling
145          */
146         var bindEvents = function(){
147             $(".sakai2_category_title").unbind("click");
148             $(".sakai2_category_title").click(function(ev){
149                 if($(".selected_category")){
150                     $(".selected_category").removeClass("selected_category");
151                     $(".selected").removeClass("selected");
152                 }
153 
154                 $(ev.currentTarget).parent().addClass("selected");
155                 $(ev.currentTarget).addClass("selected_category");
156                 var category = ev.currentTarget.id;
157                 renderSiteList(category);
158                 bindEvents();
159             });
160             $(".sakai2_site_title input").unbind("click");
161             $(".sakai2_site_title input").click(function(ev){
162                 var siteId = ev.currentTarget.id;
163                 if($(ev.currentTarget).attr("checked")){
164                     var site = getObject(siteId);
165                     sakai.data.me.sakai2List.sites.push(site);
166                 }else {
167                     var ind = getToRemoveIndex(siteId);
168                     sakai.data.me.sakai2List.sites.splice(ind,1);
169                 }
170                 renderSiteList($(".selected_category").attr("id"));
171                 renderSelectedList();
172                 // bind events
173                 bindEvents();
174             });
175             $(".sakai2_selected_site").unbind("click");
176             $(".sakai2_selected_site").click(function(ev){
177                 var siteId = ev.currentTarget.id;
178                 var ind = getToRemoveIndex(siteId);
179                 sakai.data.me.sakai2List.sites.splice(ind,1);
180                 renderSiteList($(".selected_category").attr("id"));
181                 renderSelectedList();
182                 // bind events
183                 bindEvents();
184             });
185         };
186 
187         /**
188          *  This function return the list of unique sites.
189          * @param {Object} category The category lists json object
190          * @param {siteJson} siteJson The site list object.
191          */
192         var getAllSites = function(category,siteJson){
193             // loop through each sites in the category and add to the siteJson
194             // if it is not already in the siteJson
195             for(var i in category.sites){
196                 // check if the site already exists in the the siteJson
197                 if(!isItemExists(siteJson.sites, category.sites[i])){
198                     // add to the siteJson object
199                     siteJson.sites.push(category.sites[i]);
200                 }
201             }
202             return siteJson;
203         };
204 
205         /**
206          *  This function check if item is already exists in the site list.
207          *  It check whether site is already in the sites.
208          * @param {String} sites The site lists json object
209          * @param {string} site The site object.
210          */
211         var isItemExists = function(sites, site){
212             var checking = false;
213             // loop through all sites and check if site already exists in the sites.
214             for(var i in sites) {
215                 // if site id is same
216                 if(sites[i].id == site.id) {
217                     checking = true;
218                 }
219             }
220             return checking;
221         };
222 
223         /**
224          *  This function checked the checkboxes for the selected sites.
225          */
226         var setSite = function(){
227             // loop through the list of sites to be displayed in mysakai2 favourites list
228             for(var i in sakai.data.me.sakai2List.sites){
229                 if (sakai.data.me.sakai2List.sites.hasOwnProperty(i)) {
230                     // checked the related checkbox
231                     $("input[id='" + sakai.data.me.sakai2List.sites[i].id + "']").attr('checked', true);
232                 }
233             }
234         };
235 
236         /**
237          * This function get the site object based on the id passed.
238          * @param {String} id The site id
239          */
240         var getObject = function(id){
241             // loop through category list first
242             for(var i in siteListsjson.categories){
243                 if (siteListsjson.categories.hasOwnProperty(i)) {
244                     // loop through site list inside each category
245                     for (var j in siteListsjson.categories[i].sites) {
246                         // if site id is ame return the site object.
247                         if (siteListsjson.categories[i].sites[j].id === id) {
248                             return siteListsjson.categories[i].sites[j];
249                         }
250                     }
251                 }
252             }
253         };
254 
255         /**
256          * This function return the index of site in the sakai.data.me.sakai2List(myskai2 favourites site list)based on the id
257          * @param {String} id The site id
258          */
259         var getToRemoveIndex = function(id){
260             // loop through mysakai2 favourites selected sites
261             for(var i in sakai.data.me.sakai2List.sites){
262                 // if id is same return the index.
263                 if(sakai.data.me.sakai2List.sites[i].id === id){
264                     return i;
265                 }
266             }
267         };
268 
269         /**
270          *  This function get the list of sites list group by category.
271          */
272         var getSiteList = function(){
273             var url = "/dev/s23/bundles/sites-categorized.json";
274             if (sakai.config.useLiveSakai2Feeds){
275                 url = "/var/proxy/s23/sitesCategorized.json?categorized=true";
276             }
277             $.ajax({
278                 url: url,
279                 type : "GET",
280                 dataType: "json",
281                 success: function(data){
282                     siteListsjson = data;
283                 },
284                 error: function(){
285                 }
286             });
287         };
288 
289         /**
290          * Execute this function when the widget get launched
291          */
292         var doInit = function(){
293             // get list of all sites
294             getSiteList();
295             // get lists of site selected to display in my sakai2 favourites
296             //getSelectedSiteList();
297 
298             $("#sakai2favourites_add_save").click(function(ev){
299                 var toSave = {};
300                 toSave.id = [];
301                 for (var i = 0; i < sakai.data.me.sakai2List.sites.length; i++){
302                     toSave.id.push(sakai.data.me.sakai2List.sites[i].id);
303                 }
304                 sakai.api.Server.saveJSON("/~" + sakai.api.Util.safeURL(sakai.data.me.user.userid) + "/private/sakai2favouriteList",toSave, function(success,data){
305                     $("#sakai2favourites_container").jqmHide();
306                     $(window).trigger("sakai2-favourites-selected");
307                 }, true);
308             });
309         };
310         doInit();
311 
312     };
313     sakai.api.Widgets.widgetLoader.informOnLoad("sakai2favourites");
314 });
