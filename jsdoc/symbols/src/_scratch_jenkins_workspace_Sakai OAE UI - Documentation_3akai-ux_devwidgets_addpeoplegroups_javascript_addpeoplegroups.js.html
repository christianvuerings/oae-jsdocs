 25 require(["jquery", "sakai/sakai.api.core"], function($, sakai) {
 26 
 27     /**
 28      * @name sakai_global.addpeoplegroups
 29      *
 30      * @class addpeoplegroups
 31      *
 32      * @description
 33      * addpeoplegroups dialog box
 34      *
 35      * @version 0.0.1
 36      * @param {String} tuid Unique id of the widget
 37      * @param {Boolean} showSettings Show the settings of the widget or not
 38      */
 39     sakai_global.addpeoplegroups = function(tuid, showSettings) {
 40 
 41         var $rootel = $("#" + tuid);
 42         var $addpeoplegroupsWidget = $("#addpeoplegroups_widget", $rootel);
 43         var addpeoplegroupsClose = ".addpeoplegroups_close";
 44         var addpeoplegroupsTrigger = ".addpeoplegroups_trigger";
 45         var addpeoplegroupsSave = "#addpeoplegroups_save";
 46         var targetSelectGroup = "addpeoplegroups_checkbox";
 47         var renderObj = {};
 48         var membershipFetched = 0;
 49         var selectedTitles = [];
 50         var selectedIDs = [];
 51 
 52         var getSelected = function(){
 53             var selected = [];
 54             if(selectedTitles.length > 1 && !$.isArray(selectedTitles)){
 55                 selectedTitles = selectedTitles.split(",");
 56                 selectedIDs = selectedIDs.split(",");
 57             }
 58             $.each(selectedTitles, function(i, title) {
 59                 selected.push({
 60                     id: selectedIDs[i],
 61                     title: title
 62                 });
 63             });
 64             return selected;
 65         };
 66 
 67         var getSelectedIDs = function(){
 68             var selected = [];
 69             $.each(selectedIDs, function(i, id){
 70                 if (id !== sakai.data.me.user.userid) {
 71                     selected.push(id);
 72                 }
 73             });
 74             return selected;
 75         };
 76 
 77         var renderTemplate = function(){
 78             $("#addpeoplegroups_container").html(sakai.api.Util.TemplateRenderer("addpeoplegroups_template", renderObj));
 79             $addpeoplegroupsWidget.toggle();
 80         };
 81 
 82         /**
 83          * Determines if the selected groups are a part of any groups
 84          */
 85         var selectAlreadyGroupMember = function(){
 86             $.each(renderObj.memberOfGroups.entry, function(j, memberOfGroup){
 87                 var alreadyMember = true;
 88                 $.each(getSelected(), function(i, selectedAuthorizable){
 89                     if ($.inArray(selectedAuthorizable.id, memberOfGroup.members) === -1 && selectedAuthorizable.id !== memberOfGroup["sakai:group-id"]){
 90                         alreadyMember = false;
 91                     }
 92                 });
 93                 memberOfGroup.alreadyMember = alreadyMember;
 94             });
 95             renderTemplate();
 96         };
 97 
 98         /**
 99          * Gets memberships for all groups you're a member of to be able to match them to the selected groups
100          */
101         var getMemberships = function(){
102             var groupsToFetch = [];
103             var membershipsManage = [];
104             $.each(renderObj.memberOfGroups.entry, function(index, item){
105                 groupsToFetch.push(item["sakai:group-id"]);
106             });
107             if(renderObj.memberOfGroups.entry.length){
108                 // First fetch the group info so we can tell whether the current user can manage any of the groups
109                 sakai.api.Groups.getGroupAuthorizableData(groupsToFetch, function(success, groupData){
110                     $.each(groupData, function(index, group){
111                         $.each(renderObj.memberOfGroups.entry, function(m, membership){
112                             if (membership["sakai:group-id"] === group.properties["sakai:group-id"]){
113                                 membership.canManage = false;
114                                 var roles = sakai.api.Groups.getRoles({"roles": $.parseJSON(group.properties["sakai:roles"])}, true);
115                                 $.each(roles, function(r, role){
116                                     if (role.isManagerRole){
117                                         var isMember = sakai.api.Groups.isCurrentUserAMember(membership["sakai:group-id"] + "-" + role.id, sakai.data.me);
118                                         if (isMember){
119                                             membership.canManage = true;
120                                         }
121                                     }
122                                 });
123                                 if (membership.canManage){
124                                     membershipsManage.push(membership);
125                                 }
126                             }
127                         });
128                     });
129                     // Now fetch all members of all groups
130                     renderObj.memberOfGroups.entry = membershipsManage;
131                     sakai.api.Groups.getMembers(groupsToFetch, function(success, data){
132                         $.each(renderObj.memberOfGroups.entry, function(g, membership){
133                             membership.members = [];
134                             var groupRoles = data[membership["sakai:group-id"]] || {};
135                             $.each(groupRoles, function(r, role){
136                                 $.each(role.results, function(u, user){
137                                     membership.members.push(user["rep:userId"] || user.groupid);
138                                 });
139                             });
140                         });
141                         selectAlreadyGroupMember();
142                     });
143                 });
144             } else {
145                 renderTemplate();
146             }
147         };
148 
149         var toggleVisibility = function(){
150             sakai.api.Util.getTemplates(function(success, templates) {
151                 if (success) {
152                     // Fill up initial values in object to send to renderer
153                     renderObj = {
154                         api: sakai.api,
155                         groups: getSelected(),
156                         memberOfGroups: sakai.api.Groups.getMemberships(sakai.data.me.groups),
157                         worlds: templates
158                     };
159                     // Check if groups are part of my library
160                     if (!$addpeoplegroupsWidget.is(':visible')) {
161                         getMemberships();
162                     } else {
163                         $addpeoplegroupsWidget.toggle();
164                     }
165                 } else {
166                     debug.error('Could not get the group templates');
167                 }
168             });
169         };
170 
171         var saveMemberships = function(){
172             $(addpeoplegroupsSave, $rootel).attr("disabled", true);
173             var $addPeopleGroupsSelect = $("#addpeoplegroups_select");
174             if(!$addPeopleGroupsSelect.children("option:selected").data("redirect") === true){
175                 var groupsToAdd = [];
176                 $.each(getSelected(), function(i, selectedGroup){
177                     sakai.api.Groups.getGroupAuthorizableData($addPeopleGroupsSelect.val(), function(success, data){
178                         data = data[$addPeopleGroupsSelect.val()];
179                         groupsToAdd.push({
180                             user: selectedGroup.id,
181                             permission: data.properties["sakai:joinRole"]
182                         });
183                     });
184                 });
185                 sakai.api.Groups.addUsersToGroup($addPeopleGroupsSelect.val(), groupsToAdd, sakai.data.me);
186                 $(addpeoplegroupsSave, $rootel).removeAttr("disabled");
187                 toggleVisibility();
188                 sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("SUCCESSFULLY_ADDED", "addpeoplegroups"), sakai.api.Util.TemplateRenderer("addpeoplegroups_added_template", {
189                     groupsToAdd: getSelected(),
190                     groupToAddTo: $("#addpeoplegroups_select option[value='" + $addPeopleGroupsSelect.val() + "']").text()
191                 }));
192             } else {
193                 document.location = "/create#l=" + $addPeopleGroupsSelect.val() + "&members=" + getSelectedIDs().toString();
194             }
195         };
196 
197         var addBinding = function(){
198             sakai.api.Util.hideOnClickOut($addpeoplegroupsWidget, addpeoplegroupsTrigger + "," + addpeoplegroupsClose);
199             $(addpeoplegroupsClose, $rootel).die("click", toggleVisibility);
200             $(addpeoplegroupsClose, $rootel).live("click", toggleVisibility);
201             $(addpeoplegroupsSave, $rootel).die("click", saveMemberships);
202             $(addpeoplegroupsSave, $rootel).live("click", saveMemberships);
203         };
204 
205         var doInit = function(el){
206             toggleVisibility();
207             $addpeoplegroupsWidget.css("top", $(el).position().top + 24);
208             $addpeoplegroupsWidget.css("left", $(el).position().left - ($addpeoplegroupsWidget.width() / 2) + ($(el).width() / 2 + 10) );
209         };
210 
211         $(document).on('click', '.addpeoplegroups_trigger', function() {
212             selectedTitles = $('.addpeoplegroups_trigger:visible').data('entityname');
213             selectedIDs = $('.addpeoplegroups_trigger:visible').data('entityid');
214             if (!$addpeoplegroupsWidget.is(':visible')) {
215                 addBinding();
216                 doInit(this);
217             } else {
218                 toggleVisibility();
219             }
220         });
221     };
222 
223     sakai.api.Widgets.widgetLoader.informOnLoad("addpeoplegroups");
224 });
