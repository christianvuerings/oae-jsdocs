 19 /*
 20  * Dependencies
 21  *
 22  * /dev/lib/misc/trimpath.template.js (TrimpathTemplates)
 23  */
 24 
 25 require(["jquery", "sakai/sakai.api.core"], function($, sakai) {
 26 
 27     /**
 28      * @name sakai_global.mycontacts
 29      *
 30      * @class mycontacts
 31      *
 32      * @description
 33      * Initialize the mycontacts widget
 34      *
 35      * @version 0.0.1
 36      * @param {String} tuid Unique id of the widget
 37      * @param {Boolean} showSettings Show the settings of the widget or not
 38      */
 39     sakai_global.mycontacts = function(tuid,showSettings){
 40 
 41         /////////////////////////////
 42         // Configuration variables //
 43         /////////////////////////////
 44 
 45         var rootel = $("#" + tuid);
 46         var numberFriends = 8; // The number of contacts that will be shown
 47 
 48         // - ID
 49         var mycontacts = "#mycontacts";
 50 
 51         // Contact request
 52         var mycontactsRequests = mycontacts + "_requests";
 53 
 54         // Error
 55         var mycontactsError = mycontacts + "_error";
 56         var mycontactsErrorContactserver = mycontactsError + "_contactserver";
 57 
 58         // List
 59         var mycontactsList = mycontacts + "_list";
 60 
 61         // Templates
 62         var mycontactsListTemplate = "mycontacts_list_template";
 63         var mycontactsRequestsTemplate = "mycontacts_requests_template";
 64 
 65 
 66         ///////////////////////
 67         // Utility functions //
 68         ///////////////////////
 69 
 70         /**
 71          * Parse the name for a user
 72          * @param {String} uuid Uuid of the user
 73          * @param {String} firstName Firstname of the user
 74          * @param {String} lastName Lastname of the user
 75          */
 76         var parseName = function(uuid, profile){
 77             var displayName = sakai.api.User.getDisplayName(profile);
 78             if (displayName) {
 79                 return displayName;
 80             }
 81             else {
 82                 return uuid;
 83             }
 84         };
 85 
 86         /**
 87          * Parse the picture for a user
 88          * @param {Object} profile The users profile
 89          */
 90         var parsePicture = function(profile){
 91             var picture = sakai.api.Util.constructProfilePicture(profile);
 92             if (picture) {
 93                 return picture;
 94             } else {
 95                 return sakai.config.URL.USER_DEFAULT_ICON_URL;
 96             }
 97         };
 98 
 99 
100         ///////////////////////////
101         // Get & process contacts //
102         ///////////////////////////
103 
104         /**
105          * Process the information for each friend
106          * @param {Object} contacts JSON object containing all the contacts the current user
107          */
108         var doProcessing = function(contacts){
109             var jsonFriends = {};
110 
111             // Array that will contain a specified number of contacts of the current user
112             jsonFriends.items = [];
113 
114             if (contacts.results) {
115                 // Run process each friend
116                 for (var i = 0, j = contacts.results.length; i < j; i++) {
117                     if (i < numberFriends) {
118                         var friend = contacts.results[i];
119 
120                         // Set the id of the friend
121                         friend.id = friend.target;
122 
123                         // Parse the name of the friend
124                         friend.name = parseName(friend.target, friend.profile);
125 
126                         // Parse the picture of the friend
127                         friend.photo = parsePicture(friend.profile);
128                         
129                         // Contact type
130                         friend.type = friend.details["sakai:types"];
131 
132                         // Add the friend to the array
133                         jsonFriends.items.push(friend);
134                     }
135                 }
136             }
137             jsonFriends.anon = sakai.data.me.user.anon;
138             jsonFriends.sakai = sakai;
139             // Render the template with the contacts
140             $(mycontactsList).html(sakai.api.Util.TemplateRenderer(mycontactsListTemplate, jsonFriends));
141         };
142 
143 
144         /**
145          * Get all the contacts for the current user.
146          * It only gets the contacts that have an accepted status
147          * and the request is ordered by the first and last name of the contacts
148          */
149         var getFriends = function(){
150             $.ajax({
151                 url: sakai.config.URL.CONTACTS_FIND_STATE + "?state=ACCEPTED&page=0&items=" + numberFriends,
152                 cache: false,
153                 success: function(data){
154 
155                     // Process the contacts: username, picture, ...
156                     doProcessing(data);
157                 },
158                 error: function(xhr, textStatus, thrownError) {
159 
160                     // Show the contact error
161                     $(mycontactsErrorContactserver, rootel).show();
162                 }
163             });
164         };
165 
166 
167         //////////////////////
168         // Contact requests //
169         //////////////////////
170 
171         /**
172          * Get all the contact request for the current user and show
173          * them on the page.
174          */
175         var getContactRequests = function(){
176             $(mycontactsRequests).html(sakai.api.Util.TemplateRenderer(mycontactsRequestsTemplate, sakai.data.me.contacts));
177         };
178 
179 
180         /////////////////////////////
181         // Initialisation function //
182         /////////////////////////////
183 
184         var doInit = function() {
185 
186             // Get the firends for the current user
187             getFriends();
188 
189             // Get the contacts requests for the current user
190             getContactRequests();
191         };
192 
193         doInit();
194 
195     };
196 
197     sakai.api.Widgets.widgetLoader.informOnLoad("mycontacts");
198  
199 });
