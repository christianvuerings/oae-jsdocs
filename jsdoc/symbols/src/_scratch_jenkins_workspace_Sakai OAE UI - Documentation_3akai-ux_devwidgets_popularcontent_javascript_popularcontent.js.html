 27 require(["jquery", "sakai/sakai.api.core"], function($, sakai) {
 28 
 29     /**
 30      * @name sakai_global.popularcontent
 31      *
 32      * @class popularcontent
 33      *
 34      * @description
 35      * Initialize the popularcontent widget
 36      *
 37      * @version 0.0.1
 38      * @param {String} tuid Unique id of the widget
 39      * @param {Boolean} showSettings Show the settings of the widget or not
 40      */
 41     sakai_global.popularcontent = function(tuid, showSettings) {
 42 
 43         var $rootel = $("#" + tuid);
 44         var $popularcontent_main = $("#popularcontent_main", $rootel);
 45         var $popularcontent_main_template = $("#popularcontent_main_template", $rootel);
 46 
 47         var popularcontentMainLoadingProgress = "#popularcontent_main_progress_loading";
 48         var popularcontentMainProgress = "popularcontent_main_progress";
 49 
 50         var contentData = {};
 51         var currentSearch = "";
 52 
 53         var popularContentEllipsisContainer = ".popularcontent_ellipsis_container";
 54 
 55         var renderPopularContent = function(){
 56             $popularcontent_main.html(sakai.api.Util.TemplateRenderer($popularcontent_main_template, {
 57                 data: contentData
 58             })).show();
 59         };
 60 
 61         var handleHashChange = function(e, node) {
 62             var selected = node || $.bbq.getState("location");
 63             if (selected && selected !== currentSearch) {
 64                 loadDataDirectory(selected, renderPopularContent);
 65                 currentSearch = selected;
 66             }
 67         };
 68 
 69         var loadDataDirectory = function(selected, callback){
 70             $popularcontent_main.html("");
 71             $(popularcontentMainLoadingProgress).addClass(popularcontentMainProgress);
 72             var params = {
 73                 page: 0,
 74                 items: 10,
 75                 sortOrder: "desc"
 76             };
 77 
 78             var url = sakai.config.URL.SEARCH_ALL_FILES;
 79             if (selected === '*' || selected === '**') {
 80                 url = sakai.config.URL.SEARCH_ALL_FILES_ALL;
 81             } else {
 82                 params['q'] = selected;
 83             }
 84             $.ajax({
 85                 url: url,
 86                 data: params,
 87                 success: function(data){
 88                     $(popularcontentMainLoadingProgress).removeClass(popularcontentMainProgress);
 89                     contentData = {"results":[], "items": data.items, "total": data.total};
 90                     var content = [];
 91                     for (var i = 0; i < data.results.length; i++){
 92                         var item = {};
 93                         item["id"] = data.results[i]["_path"];
 94                         item["filename"] = data.results[i]["sakai:pooled-content-file-name"];
 95                         item["name"] = sakai.api.Util.applyThreeDots(data.results[i]["sakai:pooled-content-file-name"], $(".popularcontent_widget").width() - 80, {max_rows: 1,whole_word: false}, "s3d-bold");
 96                         content.push(item);
 97                     }
 98                     contentData.results[0] = {"content": content};
 99                     contentData.moreLink = "/search#l=content&q=" + selected;
100                     callback();
101                 }
102             });
103         };
104 
105         var loadData = function(callback){
106             $.ajax({
107                 url: "/var/search/public/mostactivecontent.json?page=0&items=5",
108                 cache: false,
109                 success: function(data){
110                     contentData = data;
111                     callback();
112                 }
113             });
114         };
115 
116         $(window).bind("hashchange nohash.browsedirectory.sakai", handleHashChange);
117 
118         var doInit = function(){
119             if (! sakai.api.Widgets.isOnDashboard(tuid)){
120                 $(".popularcontent-widget-border").show();
121                 $("#popularcontent_widget").addClass("fl-widget s3d-widget");
122             }
123 
124             // If the widget is initialized on the directory page then listen to the event to catch specified tag results
125             if (!(sakai_global.directory && sakai_global.directory.getIsDirectory())) {
126                loadData(renderPopularContent);
127                $("#popularcontent_title_popular").show();
128             } else {
129                 handleHashChange();
130                 $("#popularcontent_title_recent").show();
131             }
132         };
133 
134         doInit();
135     };
136 
137     sakai.api.Widgets.widgetLoader.informOnLoad("popularcontent");
138 
139 });
