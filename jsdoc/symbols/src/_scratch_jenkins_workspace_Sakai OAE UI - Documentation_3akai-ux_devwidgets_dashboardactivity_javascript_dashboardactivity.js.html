 19 /*globals sakai_global */
 20 // load the master sakai object to access all Sakai OAE API methods
 21 require(['jquery', 'sakai/sakai.api.core'], function($, sakai) {
 22 
 23     'use strict';
 24 
 25     /**
 26      * @name sakai_global.dashboardactivity
 27      *
 28      * @class dashboardactivity
 29      *
 30      * @description
 31      * Shows a stream of activity related to the user viewing it
 32      *
 33      * @version 0.0.1
 34      * @param {String} tuid Unique id of the widget
 35      * @param {Boolean} showSettings Show the settings of the widget or not
 36      */
 37     sakai_global.dashboardactivity = function (tuid, showSettings) {
 38 
 39 
 40         /////////////////////////////
 41         // Configuration variables //
 42         /////////////////////////////
 43 
 44         var $rootel = $('#' + tuid);
 45         var $dashboardactivityWidget = $('#dashboardactivity_widget', $rootel);
 46 
 47         // Containers
 48         var $dashboardactivityContainer = $('#dashboardactivity_container', $rootel);
 49         var $dashboardactivityActivityContainer = false;
 50         var dashboardactivityFilterContainer = '.dashboardactivity_filter';
 51 
 52         // Templates
 53         var dashboardactivityActivityTemplate = 'dashboardactivity_activity_template';
 54         var dashboardactivityActivityBadrequestTemplate = 'dashboardactivity_activity_badrequest_template';
 55         var dashboardactivityNoActivityTemplate = 'dashboardactivity_no_activity_template';
 56         var dashboardactivityContentCommentsTemplate = 'dashboardactivity_content_comments_template';
 57 
 58         // Widget variables
 59         var filter = 'all';
 60         var filterMap = {
 61             'ADDED_COMMENT': 'comments',
 62             'CREATED_FILE': 'updates',
 63             'SHARED_CONTENT': 'sharing',
 64             'UPDATED_CONTENT': 'updates'
 65         };
 66         var context = {
 67             'url': '',
 68             'id': ''
 69         };
 70 
 71 
 72         ////////////////////
 73         // Render & Parse //
 74         ////////////////////
 75 
 76         /**
 77          * Render the dashboard activity widget
 78          * @param {String|Object} template Template that needs to be rendered
 79          * @param {Object} data The JSON you want to pass through the dashboard widget
 80          */
 81         var renderActivity = function(template, data) {
 82             $dashboardactivityContainer.html(
 83                 sakai.api.Util.TemplateRenderer(template, data || {})).show();
 84         };
 85 
 86         /**
 87          * Parse the activity data
 88          * @param {Object|Boolean} data The JSON activity data or `false` when there was no data
 89          */
 90         var parseActivityData = function(data) {
 91             // If the request wasn't successful, show it to the user
 92             if (!data) {
 93                 renderActivity(dashboardactivityActivityBadrequestTemplate);
 94                 return;
 95             }
 96 
 97             if (!data.results.length) {
 98                 renderActivity(dashboardactivityNoActivityTemplate);
 99             } else {
100                 var filteredData = [];
101                 $.each(data.results, function(index, item){
102                     item.translatedActivityMessage = sakai.api.i18n.getValueForKey(
103                                                         item['sakai:activityMessage'], 'dashboardactivity');
104                     item.translatedActivityMessageAction = '';
105                     if (item['sakai:activityMessageAction']) {
106                         item.translatedActivityMessageAction = sakai.api.i18n.getValueForKey(
107                            item['sakai:activityMessageAction'] , 'dashboardactivity');
108                     }
109                     // Filter based on the selected tab
110                     if (filter !== 'all') {
111                         if (filterMap[item['sakai:activityMessage']] === filter) {
112                             filteredData.push(item);
113                         }
114                     } else {
115                         filteredData.push(item);
116                     }
117                 });
118 
119                 if (!filteredData.length) {
120                     renderActivity(dashboardactivityNoActivityTemplate);
121                 } else {
122                     renderActivity(dashboardactivityActivityTemplate, {
123                         data: filteredData,
124                         sakai: sakai
125                     });
126                 }
127                 $dashboardactivityActivityContainer = $($dashboardactivityActivityContainer);
128             }
129         };
130 
131         /**
132          * Get the activity data
133          */
134         var getActivityData = function() {
135             $.ajax({
136                 url: context.url,
137                 data: {
138                     items: 1000
139                 },
140                 success: function(data) {
141                     parseActivityData(data);
142                 },
143                 error: function() {
144                     parseActivityData(false);
145                 }
146             });
147         };
148 
149 
150         ///////////
151         // UTILS //
152         ///////////
153 
154         var switchActivityFilter = function() {
155             if (filter !== $(this).data('filter')) {
156                 filter = $(this).data('filter');
157                 $(dashboardactivityFilterContainer + ' button').removeClass('selected');
158                 $(this).addClass('selected');
159 
160                 if (filter === 'comments' && context.id === 'content') {
161                     renderActivity(dashboardactivityContentCommentsTemplate);
162                     sakai.api.Widgets.widgetLoader.insertWidgets(tuid);
163                 } else {
164                     getActivityData();
165                 }
166             }
167         };
168 
169 
170         ////////////////////
171         // Initialization //
172         ////////////////////
173 
174         /**
175          * Add binding to the elements
176          */
177         var addBinding = function() {
178             $(dashboardactivityFilterContainer + ' button').on('click', switchActivityFilter);
179         };
180 
181         var doInit = function() {
182             sakai.api.Widgets.loadWidgetData(tuid, function(success, data) {
183                 if (data.groupid) {
184                     // Load activity for group
185                     context = {
186                         'url': '/devwidgets/dashboardactivity/dummy/groupdummy.json',
187                         'id': data.groupid
188                     };
189                 } else if (sakai_global.content_profile && sakai_global.content_profile.content_data) {
190                     // Load activity for content profile
191                     context = {
192                         'url': '/devwidgets/dashboardactivity/dummy/contentdummy.json',
193                         'id': 'content'
194                     };
195                     $dashboardactivityWidget.css('width', '100%');
196                 } else {
197                     // Load activity for a user
198                     context = {
199                         'url': '/devwidgets/dashboardactivity/dummy/mydummy.json',
200                         'id': 'user'
201                     };
202                 }
203                 addBinding();
204                 getActivityData();
205             });
206         };
207 
208         doInit();
209     };
210 
211     // inform Sakai OAE that this widget has loaded and is ready to run
212     sakai.api.Widgets.widgetLoader.informOnLoad('dashboardactivity');
213 });
