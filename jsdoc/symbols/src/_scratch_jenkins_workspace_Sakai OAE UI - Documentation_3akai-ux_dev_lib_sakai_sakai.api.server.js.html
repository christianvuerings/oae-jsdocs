 38         /**
 39          * Perform a batch request to the server
 40          *
 41          * @param {Array} requests The JSON object of requests
 42          * @param {Function} callback Callback function, passes ({Boolean} success, {Object} data)
 43          * @param {Boolean} forcePOST if we need to force a POST
 44          * @param {Boolean} async If we should do an async request or not
 45          */
 46         batch : function(_requests, _callback, _forcePOST, _async) {
 47             var cache = true;
 48             var method = _forcePOST === true ? "POST" : "GET";
 49             var async = _async === false ? false : true;
 50 
 51             // Append a charset to each request
 52             $.each(_requests, function(i,req) {
 53                 if (!req["_charset_"]) {
 54                     req["_charset_"] = "utf-8";
 55                 }
 56                 if (req["parameters"] && !req["parameters"].hasOwnProperty("_charset_")) {
 57                     req["parameters"]["_charset_"] = "utf-8";
 58                 }
 59                 if (req.hasOwnProperty("cache") && req["cache"] === false) {
 60                     cache = false;
 61                 }
 62             });
 63             // Don't submit a request when the batch is empty
 64             if (_requests.length === 0) {
 65                 if ($.isFunction(_callback)) {
 66                     _callback(true, {"results": []});
 67                 }
 68             }
 69             // Don't issue a batch request for a single, cacheable request
 70             else if (_requests.length === 1) {
 71                 $.ajax({
 72                     url: _requests[0].url,
 73                     cache: cache,
 74                     type: _requests[0].method || "GET",
 75                     dataType: "text",
 76                     data: _requests[0].parameters || "",
 77                     success: function(data) {
 78                         var retObj = {
 79                             "results": [
 80                                 {
 81                                     "url": _requests[0].url,
 82                                     "success": true,
 83                                     "body": data
 84                                 }
 85                             ]
 86                         };
 87                         if ($.isFunction(_callback)) {
 88                             _callback(true, retObj);
 89                         }
 90                     },
 91                     error: function(status){
 92                         if ($.isFunction(_callback)) {
 93                             _callback(false, {"results": [{
 94                                 "url": _requests[0].url,
 95                                 "success": false,
 96                                 "body": "{}"
 97                             }]});
 98                         }
 99                     }
100                 });
101 
102             } else {
103 
104                 // if any request contains a POST, we should be POSTing so the request isn't cached
105                 // maybe just GET with no cache? not sure
106                 for (var i=0; i<_requests.length; i++) {
107                     if (_requests[i].method === "POST") {
108                         method = "POST";
109                         break;
110                     }
111                 }
112                 var requestString = JSON.stringify(_requests);
113                 if (requestString.length > 2000) {
114                     method = "POST";
115                 }
116                 $.ajax({
117                     url: sakai_conf.URL.BATCH,
118                     type: method,
119                     cache: cache,
120                     async: async,
121                     data: {
122                         "_charset_":"utf-8",
123                         requests: requestString
124                     },
125                     success: function(data) {
126                         if ($.isFunction(_callback)) {
127                             _callback(true, data);
128                         }
129                     },
130                     error: function(xhr) {
131                         if ($.isFunction(_callback)) {
132                             _callback(false);
133                         }
134                     }
135                 });
136             }
137         },

205          * Copy one node to another node

216             if (!config.source || !config.destination) {
217

245                 'url': config.source,
246                 'method': 'POST',

249                     ':dest': config.destination,
250                     ':replace': true
251                 }
252             });
253 
254             // Execute the batch operation
255             sakaiServerAPI.batch(batchRequests, callback, true);

259         /**
260          * Saves a specified JSON object to a specified URL in JCR. The structure of JSON data will be re-created in JCR as a node hierarchy.
261          *
262          * @param {String} i_url The path to the preference where it needs to be
263          * saved
264          * @param {Object} i_data A JSON object which we would like to save
265          * (max 200 child object of each object)
266          * @param {Function} callback A callback function which is executed at the
267          * end of the operation
268          * @param {Boolean} removeTree If we should replace the entire tree of saved data or just update it
269          * @param {Array} indexFields Fields to index in the data (used for widgets, and is optional)
270          * @param {Boolean} merge If false, it will truly replace the content during an import - the default is true
271          *
272          * @returns {Void}
273          */
274         saveJSON : function(i_url, i_data, callback, removeTree, indexFields, merge) {
275 
276             // Argument check
277             if (!i_url || !i_data) {
278 
279                 // Log the error message
280                 debug.warn('sakai.api.Server.saveJSON: Not enough or empty arguments!');
281 
282                 // Still invoke the callback function
283                 if ($.isFunction(callback)) {
284                     callback(false, 'The supplied arguments were incorrect.');
285                 }
286 
287                 // Make sure none of the other code in this function is executed
288                 return;
289             }
290 
291             /**
292              * @param {Array} path The path to the data object, split out
293              * @param {Object} obj The data object to add the field to
294              */
295             var addIndexedFields = function(path, obj) {
296                 if (path.length > 1) {
297                     obj = obj[path.shift()];
298                     addIndexedFields(path, obj);
299                 } else {
300                     if (obj['sakai:indexed-fields']) {
301                         obj['sakai:indexed-fields'] = obj['sakai:indexed-fields'].split(',');
302                         if ($.inArray(path[0], obj['sakai:indexed-fields']) === -1) {
303                             obj['sakai:indexed-fields'].push(path[0]);
304                         }
305                         obj['sakai:indexed-fields'] = obj['sakai:indexed-fields'].join(',');
306                     } else {
307                         obj['sakai:indexed-fields'] = path[0];
308                     }
309                     if (!obj['sling:resourceType']) {
310                         obj['sling:resourceType'] = 'sakai/widget-data';
311                     }
312                 }
313             };
314 
315             /**
316              * <p>Convert all the arrays in an object to an object with a unique key.<br />
317              * Mixed arrays (arrays with multiple types) are not supported.
318              * </p>
319              * <code>
320              * {
321              *     'boolean': true,
322              *     'array_object': [{ 'key1': 'value1', 'key2': 'value2'}, { 'key1': 'value1', 'key2': 'value2'}]
323              * }
324              * </code>
325              * to
326              * <code>
327              * {
328              *     'boolean': true,
329              *     'array_object': {
330              *         '__array__0__': { 'key1': 'value1', 'key2': 'value2'},
331              *         '__array__1__': { 'key1': 'value1', 'key2': 'value2'}
332              *     }
333              * }
334              * </code>
335              * @param {Object} obj The Object that you want to use to convert all the arrays to objects
336              * @return {Object} An object where all the arrays are converted into objects
337              */
338             var convertArrayToObject = function(obj) {
339 
340                 // If the current object is an array with elements, we convert it into
341                 // an object
342                 if ($.isArray(obj) && obj.length > 0) {
343 
344                     // Deep copy the array
345                     var arrayCopy = $.extend(true, [], obj);
346 
347                     // Set the original array to an empty object
348                     obj = {};
349 
350                     // Add all the elements that were in the original array to the object with a unique id
351                     for (var j = 0, jl = arrayCopy.length; j < jl; j++) {
352 
353                         // Copy each object from the array and add it to the object
354                         obj['__array__' + j + '__'] = arrayCopy[j];
355 
356                         // Run recursively
357                         convertArrayToObject(arrayCopy[j]);
358                     }
359 
360                 // If the current object is an empty array, we convert it into an empty
361                 // string
362                 } else if ($.isArray(obj) && obj.length === 0) {
363                     obj = '';
364 
365                 // If the current object is a real object, we loop over its children and
366                 // check for additional arrays
367                 } else if ($.isPlainObject(obj)) {
368                     for (var k in obj) {
369                         if (obj.hasOwnProperty(k)){
370                             obj[k] = convertArrayToObject(obj[k]);
371                         }
372                     }
373                 }
374 
375                 return obj;
376             };
377 
378             // Convert the array of objects to only objects
379             // We also need to deep copy the object so we don't modify the input parameter
380             if ($.isArray(i_data)) {
381                 i_data = convertArrayToObject($.extend(true, [], i_data));
382             } else {
383                 i_data = convertArrayToObject($.extend(true, {}, i_data));
384             }
385             var postData = {
386                 ':operation': 'import',
387                 ':contentType': 'json',
388                 ':merge': (merge === false) ? false : true,
389                 ':replace': true,
390                 ':replaceProperties': true,
391                 '_charset_':'utf-8'
392             };
393             if (removeTree) {
394                 postData[':removeTree'] = removeTree;
395             }
396             if (indexFields) {
397                 $.each(indexFields, function(i,val) {
398                     addIndexedFields(val.split('/'), i_data);
399                 });
400             }
401             sakaiServerAPI.removeServerCreatedObjects(i_data, ['_']);
402             postData[':content'] = JSON.stringify(i_data);
403             // Send request
404             $.ajax({
405                 url: i_url,
406                 type: 'POST',
407                 data: postData,
408                 dataType: 'text',
409                 success: function(data) {
410                     // If a callback function is specified in argument, call it
411                     if ($.isFunction(callback)) {
412                         callback(true, data);
413                     }
414                 },
415                 error: function(xhr, status, e) {
416                     // Log error
417                     debug.error('sakai.api.Server.saveJSON: There was an error saving JSON data to: ' + this.url);
418                     // If a callback function is specified in argument, call it
419                     if ($.isFunction(callback)) {
420                         callback(false, xhr);
421                     }
422                 }
423             });
424 
425         },

427         /**
428          * Removes any objects with a given namespace
429          *
430          * @param {Object} the object to clean
431          * @param {Array}  an array containing a string for each namespace to move
432          */
433         removeServerCreatedObjects : function(obj, namespace, notToRemove) {
434             var newobj = false;
435             if ($.isPlainObject(obj)) {
436                 newobj = $.extend(true, {}, obj);
437                 notToRemove = notToRemove || [];
438                 $.each(newobj, function(key,val) {
439                     for (var ns = 0; ns < namespace.length; ns++) {
440                         if (key && key.indexOf && key.indexOf(namespace[ns]) === 0) {
441                             if (notToRemove.indexOf(key) === -1) {
442                                 delete newobj[key];
443                                 break;
444                             }
445                         }
446                     }
447                     if ($.isPlainObject(newobj[key]) || $.isArray(newobj[key])) {
448                         newobj[key] = sakaiServerAPI.removeServerCreatedObjects(newobj[key], namespace, notToRemove);
449                     }
450                 });
451             } else if ($.isArray(obj)) {
452                 newobj = $.merge([], obj);
453                 $.each(newobj, function(key, val) {
454                     if ($.isPlainObject(newobj[key]) || $.isArray(newobj[key])) {
455                         newobj[key] = sakaiServerAPI.removeServerCreatedObjects(newobj[key], namespace, notToRemove);
456                     }
457                 });
458             }
459             return newobj;

474             pagestructure = sakaiServerAPI.removeServerCreatedObjects(pagestructure, toFilter, toExclude);

566         /**
567          * <p>Convert all the objects with format __array__?__ in an object to an array</p>
568          * <code>
569          * {
570          *     "boolean": true,
571          *     "array_object": {
572          *         "__array__0__": {
573          *             "key1": "value1",
574          *             "key2": "value2"
575          *         }
576          *     }
577          * }
578          * </code>
579          * to
580          * <code>
581          * {
582          *     "boolean": true,
583          *     "array_object": [
584          *         {
585          *             "key1": "value1",
586          *             "key2": "value2"
587          *        }
588          *     ]
589          * }
590          * </code>
591          * @param {Object} specficObj The Object that you want to use to convert all the objects with the special format to arrays
592          * @param {Object} [globalObj] The parent object, we need this to run over the elements recursively
593          * @param {Object} [objIndex] The index of the parent object
594          * @return {Object} An object where all the objects with the special format are converted into arrays
595          */
596         convertObjectToArray : function(specficObj, globalObj, objIndex) {
597 
598             var i,j,k,kl;
599             // Run over all the items in the object
600             for (i in specficObj) {
601 
602                 // If exists and it's an object recurse
603                 if (specficObj.hasOwnProperty(i)) {
604 
605                     // If it's a non-empty array-object it will have a first element with the key "__array__0__"
606                     if (i === "__array__0__") {
607 
608                         // We need to get the number of items in the object
609                         var arr = [];
610                         var count = 0;
611                         for (j in specficObj) {
612                             if (specficObj.hasOwnProperty(j) && j.indexOf("__array__") > -1) {
613                                 count++;
614                             }
615                         }
616 
617                         // Construct array of objects
618                         for(k = 0, kl = count; k < kl; k ++) {
619                             arr.push(specficObj["__array__"+k+"__"]);
620                         }
621 
622                         globalObj[objIndex] = arr;
623                     }
624 
625                     if ($.isPlainObject(specficObj[i])) {
626                         this.convertObjectToArray(specficObj[i], specficObj, i);
627                     }
628                 }
629 
630             }
631             return specficObj;
632         },

709         /**
710          * Create a search string for the server
711          * This method exists to transform a user's search string which
712          * they type in into the string we should pass to the server
713          *
714          * Strings with AND, OR, '"', '-', '_' are treated as advanced search queries
715          * and left alone. Those without are transformed into term* AND term2*
716          *
717          * @param {String} searchString The user's search
718          * @param {Boolean} handlePhrases If we should split on ,\s instead of \s to
719          *                      better handle phrases
720          * @param {String} joinOn String to join keywords on, defaults to AND
721          * @return {String} The string to send to the server
722          */
723         createSearchString : function(searchString, handlePhrases, joinOn) {
724             if (!joinOn) {
725                 joinOn = 'AND';
726             }
727             var ret = "";
728             var advancedSearchRegex = new RegExp("(AND|OR|\"|-|_)", "g");
729             var removeArray = ['AND', 'OR'];
730             var truncateLength = 1500;
731 
732             ret = $.trim(searchString);
733 
734             // Remove the forward slashes
735             ret = ret.replace(/\//g, '');
736 
737             // Replace multiple spaces with 1 space
738             ret = ret.replace(/(\s)+/g, ' ');
739 
740             // We only join every single word with "AND" when
741             // we are sure there it isn't an advanced search query
742             if (!advancedSearchRegex.test(searchString)) {
743                 if (handlePhrases) {
744                     ret = '"' + ret.split(', ').join('" ' + joinOn + ' "') + '"';
745                 } else {
746                     ret = ret.split(' ').join(' ' + joinOn + ' ');
747                 }
748             }
749 
750             if (ret.length > truncateLength) {
751                 // Truncate the string just until the maximum length allowed
752                 ret = ret.substring(0, truncateLength);
753                 // Go back to the end of the previous word, so we don't
754                 // truncate in the middle of a word
755                 ret = ret.replace(/\w+$/, '');
756             }
757 
758             // We need to remove AND & OR if they are the first or last words
759             // of the querystring. Otherwise we get a 500 exception
760             ret = $.trim(ret);
761             for (var i = 0, j = removeArray.length; i < j; i++) {
762                 var startItem = removeArray[i];
763                 if (ret.substr(0, startItem.length) === startItem) {
764                     ret = ret.substring(startItem.length, ret.length);
765                 }
766                 var endItem = removeArray[i];
767                 if (ret.substr(-endItem.length) === endItem) {
768                     ret = ret.substring(0, ret.length - endItem.length);
769                 }
770             }
771 
772             ret = $.trim(ret);
773             if (ret.length === 0) {
774                 ret = '*';
775             }
776 
777             return ret;
778         }
779     };
780 
781     return sakaiServerAPI;
