 18 // load the master sakai object to access all Sakai OAE API methods
 19 require(['jquery', 'underscore', 'sakai/sakai.api.core'], function($, _, sakai) {
 20 
 21     /**
 22      * @name sakai.templategenerator
 23      *
 24      * @class templategenerator
 25      *
 26      * @description
 27      * WIDGET DESCRIPTION
 28      *
 29      * @version 0.0.1
 30      * @param {String} tuid Unique id of the widget
 31      * @param {Boolean} showSettings Show the settings of the widget or not
 32      */
 33     sakai_global.templategenerator = function(tuid, showSettings) {
 34 
 35         var $rootel = $('#' + tuid);
 36         var $templategeneratorContainer = $('#templategenerator_container', $rootel);
 37         var $templategeneratorDialog = $('.templategenerator_dialog', $rootel);
 38         var $templategeneratorForm = $('#templategenerator_form', $rootel);
 39         var $templategeneratorExportButton = $('#templategenerator_export_button');
 40         var $templategeneratorTitle = $('#templategenerator_title');
 41         var $templategeneratorUsedFor = $('#templategenerator_used_for');
 42         var $templategeneratorExport = $('#templategenerator_export');
 43 
 44         // Main data storage, used to store the generated or collected data within the widget
 45         var templategeneratorData = {
 46             roles : {},
 47             docstructure : {},
 48             pages : [],
 49             output : '',
 50             generatingTemplate : false
 51         };
 52 
 53         // Main template structure
 54         var templategeneratorDataTemplate = {
 55             id : '',
 56             title : '',
 57             img : '',
 58             fullImg : '',
 59             perfectFor : '',
 60             roles : [],
 61             docs : {},
 62             structure : [],
 63             joinRole : '',
 64             creatorRole : '',
 65             defaultaccess: 'public',
 66             defaultjoin: 'yes'
 67         };
 68 
 69         ////////////////////
 70         // Event Handlers //
 71         ////////////////////
 72 
 73         var handleSentMessage = function(success) {
 74             if (success) {
 75                 sakai.api.Util.notification.show('', sakai.api.i18n.getValueForKey('TEMPLATEGENERATOR_EXPORT_SUCCES', 'templategenerator'), sakai.api.Util.notification.type.INFORMATION);
 76                 templategeneratorData.generatingTemplate = false;
 77             } else {
 78                 sakai.api.Util.notification.show('', sakai.api.i18n.getValueForKey('TEMPLATEGENERATOR_EXPORT_ERROR', 'templategenerator'), sakai.api.Util.notification.type.INFORMATION);
 79             }
 80             sakai.api.Util.Modal.close($templategeneratorDialog);
 81         };
 82 
 83         var bindEvents = function() {
 84             sakai.api.Util.Modal.setup($templategeneratorDialog, {
 85                 modal : true,
 86                 overlay : 20,
 87                 toTop : true
 88             });
 89 
 90             var validateOpts = {
 91                 submitHandler: function() {
 92                     $templategeneratorExportButton.attr( 'disabled', 'disabled' );
 93                     generateTemplateFromData();
 94                 }
 95             };
 96             sakai.api.Util.Forms.validate( $templategeneratorForm, validateOpts, true );
 97         };
 98 
 99         var generateTemplateFromData = function() {
100             if (templategeneratorData.templatesLoaded && !templategeneratorData.generatingTemplate) {
101                 templategeneratorData.generatingTemplate = true;
102                 templategeneratorData.exportData = $.extend({}, templategeneratorDataTemplate);
103                 // 1. Heading
104                 templategeneratorData.exportData.title = $templategeneratorTitle.val();
105                 templategeneratorData.exportData.perfectFor = $templategeneratorUsedFor.val();
106                 templategeneratorData.exportData.id = templategeneratorData.exportData.title.toLowerCase().replace(/ /g, '-').replace(/[^\w-]+/g, '');
107 
108                 // 2. Roles
109                 templategeneratorData.exportData.roles = templategeneratorData.roles.roleData;
110                 templategeneratorData.exportData.joinRole = templategeneratorData.roles.joinRole;
111                 templategeneratorData.exportData.creatorRole = templategeneratorData.roles.creatorRole;
112                 templategeneratorData.exportData.structure = templategeneratorData.docstructure.structure;
113 
114                 var refIndex = 0;
115 
116                 // 3. Docs
117                 $.each(templategeneratorData.pageStructures, function(index, item) {
118                     var pidIndex = '${pid}' + index;
119                     // Get the first page
120                     var firstPage = _.keys(item)[0];
121 
122                     // Set the docref and delete the pid if is hanging around
123                     templategeneratorData.exportData.structure[firstPage]._docref = pidIndex;
124                     delete templategeneratorData.exportData.structure[firstPage]._pid;
125 
126                     // Add the documents for the page
127                     templategeneratorData.exportData.docs['${pid}' + index] = {
128                         'excludeSearch': true,
129                         'structure0': {}
130                     };
131                     // Add structure0 inside of that page
132                     var pageRefs = [];
133                     $.each(templategeneratorData.pages[index].structure, function(pageid, pagedata) {
134                         templategeneratorData.exportData.docs[pidIndex].structure0[pageid] = {
135                             '_ref': '${refid}' + refIndex,
136                             '_title': pagedata._title,
137                             '_order': pagedata._order,
138                             '_nonEditable': pagedata._nonEditable || false,
139                             'main': {
140                                 '_ref': '${refid}' + refIndex,
141                                 '_order': pagedata.main._order,
142                                 '_title': pagedata.main._title,
143                                 '_nonEditable': pagedata.main._nonEditable || false
144                             }
145                         };
146                         pageRefs.push({ pageRef: pagedata._ref, index: refIndex });
147                         if (_.keys(templategeneratorData.pages[index].structure).length > 1) {
148                             refIndex += 1;
149                         }
150                     });
151 
152                     $.each(pageRefs, function(i, pageObj) {
153                         var pageRef = pageObj.pageRef;
154                         var pageRefIndex = '${refid}' + pageObj.index;
155                         // Add the content for the page (rows, cells,...)
156                         templategeneratorData.exportData.docs[pidIndex][pageRefIndex] = {
157                             'rows': []
158                         };
159                         // Rows on the page
160                         $.each(templategeneratorData.pages[index].pageData[pageRef].rows, function(rowIndex, row) {
161                             if (_.isString(row)) {
162                                 try {
163                                     row = $.parseJSON(row);
164                                 } catch (e) {
165                                     // This string was not JSON, continue
166                                     return true;
167                                 }
168                             }
169                             if ($.isPlainObject(row)) {
170                                 refIndex += 1;
171                                 var rowRef = '${refid}' + refIndex;
172                                 templategeneratorData.exportData.docs[pidIndex][pageRefIndex].rows[rowIndex] = {
173                                     'id': rowRef,
174                                     'columns': []
175                                 };
176                                 // Columns in the rows
177                                 $.each(row.columns, function(columnIndex, column) {
178                                     if ($.isPlainObject(column)) {
179                                         templategeneratorData.exportData.docs[pidIndex][pageRefIndex].rows[rowIndex].columns[columnIndex] = {
180                                             'width': column.width,
181                                             'elements': []
182                                         };
183                                         // Cells in the column
184                                         $.each(column.elements, function(cellIndex, cell) {
185                                             if ($.isPlainObject(cell)) {
186                                                 refIndex += 1;
187                                                 var cellRef = '${refid}' + refIndex;
188                                                 templategeneratorData.exportData.docs[pidIndex][pageRefIndex].rows[rowIndex].columns[columnIndex].elements[cellIndex] = {
189                                                     'id': cellRef,
190                                                     'type': cell.type
191                                                 };
192                                                 if (templategeneratorData.pages[index].pageData[pageRef][cell.id]) {
193                                                     templategeneratorData.exportData.docs[pidIndex][pageRefIndex][cellRef] = {};
194                                                     templategeneratorData.exportData.docs[pidIndex][pageRefIndex][cellRef][cell.type] = {};
195                                                     var thisWidgetData = sakai.api.Server.removeServerCreatedObjects(templategeneratorData.pages[index].pageData[pageRef][cell.id][cell.type], '_');
196                                                     $.extend(templategeneratorData.exportData.docs[pidIndex][pageRefIndex][cellRef][cell.type], thisWidgetData);
197                                                 }
198                                             }
199                                         });
200                                     }
201                                 });
202                             }
203                         });
204 
205                         refIndex += 1;
206                     });
207                 });
208 
209                 templategeneratorData.output = JSON.stringify(templategeneratorData.exportData, null, 4);
210                 templategeneratorData.output = templategeneratorData.output.replace(/\\/g, '');
211 
212                 createTemplateFile();
213             }
214         };
215 
216         var createTemplateFile = function() {
217             var body = '--AAAAA\r\n';
218             body = body + 'Content-Disposition: form-data; name=\'*\'; filename="' + templategeneratorData.exportData.id + '.txt" \r\n';
219             body = body + 'Content-Type: text/plain \r\n';
220             body = body + 'Content-Transfer-Encoding: binary\r\n\r\n';
221             body = body + templategeneratorData.output + '\r\n';
222             body = body + '--AAAAA--\r\n';
223 
224             $.ajax({
225                 url : '/system/pool/createfile',
226                 data : body,
227                 type : 'POST',
228                 beforeSend : function(xmlReq) {
229                     xmlReq.setRequestHeader('Content-type', 'multipart/form-data; boundary=AAAAA');
230                 },
231                 success : function(data) {
232                     var fileData = $.parseJSON(data);
233                     fileData = fileData[templategeneratorData.exportData.id + '.txt'];
234 
235                     // Set the permissions of the file to private
236                     sakai.api.Content.setFilePermissions([{
237                         'hashpath' : fileData.poolId,
238                         'permissions' : 'private'
239                     }], function() {
240                         // Remove the file from the user's library and move it to the admin user
241                         $.ajax({
242                             url : '/p/' + fileData.poolId + '.members.json',
243                             type : 'POST',
244                             data : {
245                                 ':manager' : sakai.widgets.templategenerator.defaultConfiguration.templategenerator.targetUser
246                             },
247                             success : function() {
248                                 $.ajax({
249                                     url : '/p/' + fileData.poolId + '.members.json',
250                                     type : 'POST',
251                                     data : {
252                                         ':manager@Delete' : sakai.data.me.user.userid
253                                     },
254                                     success : function() {
255 
256                                         // Get the link to our generated file
257                                         var filePath = 'http://' + window.location.host + '/p/' + fileData.poolId + '/' + fileData.item['sakai:pooled-content-file-name'];
258 
259                                         // Sends a link with the template file to the admin user
260                                         var userDisplayName = sakai.api.User.getDisplayName(sakai.data.me.profile);
261                                         var msg = sakai.api.Util.TemplateRenderer('templategenerator_admin_message_template', {
262                                             'system': sakai.api.i18n.getValueForKey('SAKAI'),
263                                             'user': userDisplayName
264                                         });
265                                         sakai.api.Communication.sendMessage(sakai.widgets.templategenerator.defaultConfiguration.templategenerator.targetUser,
266                                             sakai.data.me,
267                                             userDisplayName + ' ' + sakai.api.i18n.getValueForKey('TEMPLATEGENERATOR_ADMIN_MESSAGE_SUBJECT', 'templategenerator'), msg + '\n\n' + filePath, 'message', false, handleSentMessage, true, 'new_message');
268 
269                                         // Sends a message to the user that created the template
270                                         sakai.api.Communication.sendMessage(sakai.data.me.user.userid, sakai.data.me, sakai.api.i18n.getValueForKey('TEMPLATEGENERATOR_USER_MESSAGE_SUBJECT', 'templategenerator'), sakai.api.i18n.getValueForKey('TEMPLATEGENERATOR_USER_MESSAGE', 'templategenerator'), 'message', false, null, true, 'new_message');
271                                     },
272                                     error : function() {
273                                         sakai.api.Util.notification.show('', sakai.api.i18n.getValueForKey("TEMPLATEGENERATOR_DELETE_FILE_ERROR", "templategenerator"), sakai.api.Util.notification.type.INFORMATION);
274                                         sakai.api.Util.Modal.close($templategeneratorDialog);
275                                     }
276                                 });
277                             },
278                             error : function() {
279                                 sakai.api.Util.notification.show('', sakai.api.i18n.getValueForKey("TEMPLATEGENERATOR_PERMISSION_ERROR", "templategenerator"), sakai.api.Util.notification.type.INFORMATION);
280                                 sakai.api.Util.Modal.close($templategeneratorDialog);
281                             }
282                         });
283                     });
284                 },
285                 error : function(error) {
286                     sakai.api.Util.notification.show('', sakai.api.i18n.getValueForKey("TEMPLATEGENERATOR_FILE_ERROR", "templategenerator"), sakai.api.Util.notification.type.INFORMATION);
287                     sakai.api.Util.Modal.close($templategeneratorDialog);
288                 }
289             });
290         };
291 
292         /**
293          * Retrieve all the data from the server and prepare for the templateGenerating process
294          */
295         var getTemplateData = function(callback) {
296             templategeneratorData.templateName = sakai_global.group.groupData.name;
297             templategeneratorData.docstructureUrl = '~' + templategeneratorData.templateName + '/docstructure.infinity.json';
298             templategeneratorData.rolesUrl = '/system/userManager/group/' + templategeneratorData.templateName + '.infinity.json';
299             $templategeneratorTitle.val(templategeneratorData.templateName);
300 
301             var batchRequests = [];
302             batchRequests.push({
303                 'url' : templategeneratorData.docstructureUrl,
304                 'method' : 'GET'
305             });
306             batchRequests.push({
307                 'url' : templategeneratorData.rolesUrl,
308                 'method' : 'GET'
309             });
310             sakai.api.Server.batch(batchRequests, function(success, data) {
311 
312                 if (success && data.results && data.results[0] && data.results[0].body && data.results[1] && data.results[1].body) {
313                     templategeneratorData.docstructure.docstructureData = $.parseJSON(data.results[0].body);
314                     templategeneratorData.docstructure.structure = $.parseJSON(templategeneratorData.docstructure.docstructureData.structure0);
315 
316                     // Get the urls for each page in the docstructure and convert the elements within the structure to json
317                     templategeneratorData.pageUrls = [];
318                     $.each(templategeneratorData.docstructure.structure, function(docstructureIndex, docstructureElement) {
319                         docstructureElement._view = $.parseJSON(docstructureElement._view);
320                         docstructureElement._edit = $.parseJSON(docstructureElement._edit);
321                         templategeneratorData.pageUrls.push({
322                             'url' : '/p/' + docstructureElement._pid + '.infinity.json',
323                             'method' : 'GET'
324                         });
325 
326                     });
327                     templategeneratorData.pages = [];
328                     templategeneratorData.pageStructures = [];
329 
330                     sakai.api.Server.batch(templategeneratorData.pageUrls, function(success, data) {
331                         if (success) {
332                             $.each(data.results, function(pageIndex, pageElement) {
333                                 var page = {};
334                                 page.pageData = $.parseJSON(pageElement.body);
335                                 page.structure = $.parseJSON(page.pageData.structure0);
336 
337                                 templategeneratorData.pages.push(sakai.api.Server.cleanUpSakaiDocObject(page));
338                                 templategeneratorData.pageStructures.push(page.structure);
339                                 if ( $.isFunction( callback ) ) {
340                                     callback( true );
341                                 }
342                             });
343                         } else {
344                             if ( $.isFunction( callback ) ) {
345                                 callback( false );
346                             }
347                         }
348                     }, false);
349                     // 2. Roles
350                     var roleData = $.parseJSON(data.results[1].body);
351                     templategeneratorData.roles.roleData = $.parseJSON(roleData.properties['sakai:roles']);
352                     templategeneratorData.roles.joinRole = roleData.properties['sakai:joinRole'];
353                     templategeneratorData.roles.creatorRole = roleData.properties['sakai:creatorRole'];
354 
355                 } else {
356                     if ( $.isFunction( callback ) ) {
357                         callback( false );
358                     }
359                     sakai.api.Util.notification.show('', sakai.api.i18n.getValueForKey('TEMPLATEGENERATOR_LOAD_ERROR', 'templategenerator'), sakai.api.Util.notification.type.INFORMATION);
360                 }
361             });
362         };
363 
364         /**
365          * Initializes the template generator
366          */
367         var init = function() {
368             $templategeneratorExportButton.removeAttr('disabled');
369             templategeneratorData.templatesLoaded = false;
370             getTemplateData(function(success) {
371                 templategeneratorData.templatesLoaded = true;
372             });
373             sakai.api.Util.Modal.open($templategeneratorDialog);
374         };
375         bindEvents();
376 
377         $(window).bind('init.templategenerator.sakai', function(e) {
378             init();
379         });
380     };
381 
382     sakai.api.Widgets.widgetLoader.informOnLoad('templategenerator');
