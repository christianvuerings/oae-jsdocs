 18 /*global $ */
 19 
 20 require(["jquery", "sakai/sakai.api.core", "underscore"], function($, sakai, _) {
 21 
 22     /**
 23      * @name sakai_global.joinrequestbuttons
 24      *
 25      * @class joinrequestbuttons
 26      *
 27      * @description
 28      * Renders join request buttons (join, leave, pending)
 29      *
 30      * @version 0.0.1
 31      * @param {String} tuid Unique id of the widget
 32      * @param {Boolean} showSettings Show the settings of the widget or not
 33      */
 34     sakai_global.joinrequestbuttons = function (tuid, showSettings) {
 35 
 36         /////////////////////////////
 37         // Configuration variables //
 38         /////////////////////////////
 39 
 40         var joinrequestbuttons = {  // global widget data
 41             groupid: false,
 42             data: false,
 43             joinCallback: false,
 44             leaveCallback: false
 45         };
 46 
 47         // DOM elements
 48         var $rootel = $("#" + tuid);
 49         var $joinrequestbuttons_widget = $("#joinrequestbuttons_widget", $rootel);
 50         var $joinrequestbuttons_template = $("#joinrequestbuttons_template", $rootel);
 51         var $joinrequestbuttons_request = $(".joinrequestbuttons_request");
 52         var $joinrequestbuttons_join = $(".joinrequestbuttons_join");
 53         var $joinrequestbuttons_leave = $(".joinrequestbuttons_leave");
 54 
 55         // messages
 56         var $joinrequestbuttons_group_membership = $("#joinrequestbuttons_group_membership", $rootel);
 57         var $joinrequestbuttons_group_problem_removing = $("#joinrequestbuttons_group_problem_removing", $rootel);
 58         var $joinrequestbuttons_group_removal_successful = $("#joinrequestbuttons_group_removal_successful", $rootel);
 59         var $joinrequestbuttons_group_request_sent = $("#joinrequestbuttons_group_request_sent", $rootel);
 60         var $joinrequestbuttons_group_problem_with_request = $("#joinrequestbuttons_group_problem_with_request", $rootel);
 61         var $joinrequestbuttons_group_problem_adding = $("#joinrequestbuttons_group_problem_adding", $rootel);
 62         var $joinrequestbuttons_group_adding_successful = $("#joinrequestbuttons_group_adding_successful", $rootel);
 63 
 64         /**
 65          * Displays the specific type of group membership button
 66          * @param {String} type specifies which button to show. Options are:
 67          *  - join: "Join group" button
 68          *  - leave: "Leave group" button
 69          *  - request: "Request to join group" button
 70          *  - pending: "Join request pending" button
 71          * @return None
 72          */
 73         var showButton = function (type) {
 74             hideButtons();
 75             if (_.isString(type)) {
 76                 switch(type) {
 77                     case "join":
 78                         $(".joinrequestbuttons_join").show();
 79                         break;
 80                     case "leave":
 81                         // don't display this button on the entity widget
 82                         if (!$rootel.parents("#entity_container")){
 83                             $(".joinrequestbuttons_leave").show();
 84                         }
 85                         break;
 86                     case "request":
 87                         $(".joinrequestbuttons_request").show();
 88                         break;
 89                     case "pending":
 90                         $(".joinrequestbuttons_pending").show();
 91                         break;
 92                 }
 93             }
 94         };
 95 
 96         /**
 97          * Hides any showing group membership button
 98          */
 99         var hideButtons = function () {
100             $(".joinrequestbuttons_join").hide();
101             $(".joinrequestbuttons_leave").hide();
102             $(".joinrequestbuttons_request").hide();
103             $(".joinrequestbuttons_pending").hide();
104         };
105 
106         /**
107          * Renders the correct join request button
108          */
109         var render = function () {
110             // render the template
111             $joinrequestbuttons_widget.html(sakai.api.Util.TemplateRenderer($joinrequestbuttons_template, {
112                 id:joinrequestbuttons.groupid,
113                 buttonStyle:joinrequestbuttons.buttonStyle
114             }));
115 
116             // determine which button to show
117             var isMember = sakai.api.Groups.isCurrentUserAMember(joinrequestbuttons.groupid, sakai.data.me);
118             var isManager = sakai.api.Groups.isCurrentUserAManager(joinrequestbuttons.groupid, sakai.data.me);
119             var isAnon = sakai.data.me.user.userid ? false : true;
120 
121             if ((isManager || isMember) && joinrequestbuttons.groupData.leaveAllowed) {
122                 // we have either a group member or manager, but not the last group manager
123                 showButton("leave");
124             }
125             else if ((isManager && joinrequestbuttons.managerCount === 1) ||
126                 (!isMember && joinrequestbuttons.joinability === sakai.config.Permissions.Groups.joinable.manager_add) || isAnon) {
127                 // we have either the last group manager or a non-member with
128                 // joinability set to 'only managers can add' or an anonymous user
129                 hideButtons();
130             }
131             else if (!isMember && !isAnon && joinrequestbuttons.joinability === sakai.config.Permissions.Groups.joinable.user_direct) {
132                 // we have a non-member with joinability set to 'users can join directly'
133                 showButton("join");
134             }
135             else if (!isMember && !isAnon && joinrequestbuttons.joinability === sakai.config.Permissions.Groups.joinable.user_request) {
136                 // we have a non-member with joinability set to 'users must request to join'
137 
138                 /**
139                  * Function to loop through join requests in search of a request
140                  * from the current user
141                  */
142                 var searchForJoinRequest = function (reqs) {
143                     var foundRequest = false;
144                     $.each(reqs, function (i, req) {
145                         if (req.userid === sakai.data.me.user.userid) {
146                             foundRequest = true;
147                             return false;  // will break the $.each loop
148                         }
149                     });
150                     if (foundRequest) {
151                         // user has a pending join request
152                         showButton("pending");
153                     } else {
154                         // user has not requested to join
155                         showButton("request");
156                     }
157                 };
158 
159                 // has this user already requested to join the group? Search the list of join requests
160                 if (joinrequestbuttons.joinrequests) {
161                     if (joinrequestbuttons.joinrequests.length) {
162                         searchForJoinRequest(joinrequestbuttons.joinrequests);
163                     } else {
164                         // no requests
165                         showButton("request");
166                     }
167                 } else {
168                     var joinGroup = joinrequestbuttons.groupid+"-"+joinrequestbuttons.groupData.groupProfile["sakai:joinRole"];
169                     sakai.api.Groups.getJoinRequests(joinGroup,
170                     function (success, data) {
171                         if (success) {
172                             if (data.results && data.results.length) {
173                                 searchForJoinRequest(data.results);
174                             } else {
175                                 // no requests
176                                 showButton("request");
177                             }
178                         } else {
179                             // not sure if this user has requested, show request button
180                             showButton("request");
181                             debug.warn("Could not get join requests for group id: " + joinrequestbuttons.groupid);
182                         }
183                     },
184                     false);  // this is an non-async call
185                 }
186             }
187             else {
188                 // unrecognized combination of user and joinability setting
189                 hideButtons();
190             }
191 
192             if ($.isFunction(joinrequestbuttons.onShow)) {
193                 joinrequestbuttons.onShow($("#joinrequestbuttons_widget"));
194             }
195         };
196 
197 
198         /////////////////////////////
199         // Event Bindings          //
200         /////////////////////////////
201 
202         $joinrequestbuttons_request.die("click");
203         $joinrequestbuttons_request.live("click", function (ev) {
204             var groupid = this.id.split("joinrequestbuttons_request_")[1];
205             if (!groupid || $.trim(groupid) === "") {
206                 debug.error("No group id found");
207                 return false;
208             }
209             if (!sakai.data.me.user.userid) {
210                 debug.error("Anonymous user tried to request group (id: " + groupid +
211                     ") membership");
212                 return false;
213             }
214             sakai.api.Groups.addJoinRequest(groupid, function (success) {
215                 if (success) {
216                     // show a notification and change the button
217                     sakai.api.Util.notification.show($joinrequestbuttons_group_membership.text(),
218                         $joinrequestbuttons_group_request_sent.text(),
219                         sakai.api.Util.notification.type.INFORMATION);
220                     showButton("pending");
221                 } else {
222                     sakai.api.Util.notification.show($joinrequestbuttons_group_membership.text(),
223                         $joinrequestbuttons_group_problem_with_request.text(),
224                         sakai.api.Util.notification.type.ERROR);
225                 }
226                 // call callback
227                 if (joinrequestbuttons.requestCallback && $.isFunction(joinrequestbuttons.requestCallback)) {
228                     joinrequestbuttons.requestCallback(success, groupid);
229                 }
230             });
231             return true;
232         });
233 
234         $joinrequestbuttons_join.die("click");
235         $joinrequestbuttons_join.live("click", function (ev) {
236             hideButtons();
237             var groupid = this.id.split("joinrequestbuttons_join_")[1];
238             if (!groupid || $.trim(groupid) === "") {
239                 debug.error("No group id found");
240                 return false;
241             }
242             if (!sakai.data.me.user.userid) {
243                 debug.error("Anonymous user tried to join group id: " + groupid);
244                 return false;
245             }
246             sakai.api.Groups.addJoinRequest(joinrequestbuttons.groupid, function (success) {
247                 if (success) {
248                     sakai.api.Util.notification.show($joinrequestbuttons_group_membership.text(), $joinrequestbuttons_group_adding_successful.text(), sakai.api.Util.notification.type.INFORMATION);
249                     showButton("leave");
250                     $(window).trigger("usersselected.addpeople.sakai");
251                     $(window).trigger("updateParticipantCount.entity.sakai", 1);
252                 } else {
253                     debug.error("Could not add member: " + sakai.data.me.user.userid + " to pseduoGroup: " + pseduoGroup);
254                     sakai.api.Util.notification.show($joinrequestbuttons_group_membership.text(), $joinrequestbuttons_group_problem_adding.text(), sakai.api.Util.notification.type.ERROR);
255                 }
256                 // call callback
257                 if (joinrequestbuttons.joinCallback &&
258                     $.isFunction(joinrequestbuttons.joinCallback)) {
259                     joinrequestbuttons.joinCallback(success, groupid);
260                 }
261             });
262             return true;
263         });
264 
265         $joinrequestbuttons_leave.die("click");
266         $joinrequestbuttons_leave.live("click", function (ev) {
267             var groupid = this.id.split("joinrequestbuttons_leave_")[1];
268             if (!groupid || $.trim(groupid) === "") {
269                 debug.error("No group id found");
270                 return false;
271             }
272             if (!sakai.data.me.user.userid) {
273                 debug.error("Anonymous user tried to leave group id: " + groupid);
274                 return false;
275             }
276 
277             sakai.api.Groups.getRole(sakai.data.me.user.userid, groupid, function(success, myRole){
278                 if (success){
279                     sakai.api.Groups.leave(groupid, myRole, sakai.api.User.data.me, function (success) {
280                         if (success) {
281                             $(window).trigger("updated.counts.lhnav.sakai");
282                             sakai.api.Util.notification.show($joinrequestbuttons_group_membership.text(),
283                                 $joinrequestbuttons_group_removal_successful.text(),
284                                 sakai.api.Util.notification.type.INFORMATION);
285                             // re-render to determine which button to now show
286                             render();
287                             $(window).trigger("usersselected.addpeople.sakai");
288                             $(window).trigger("updateParticipantCount.entity.sakai", -1);
289                         } else {
290                             sakai.api.Util.notification.show($joinrequestbuttons_group_membership.text(),
291                                 $joinrequestbuttons_group_problem_removing.text(),
292                                 sakai.api.Util.notification.type.ERROR);
293                         }
294                         // call callback
295                         if (joinrequestbuttons.leaveCallback &&
296                             $.isFunction(joinrequestbuttons.leaveCallback)) {
297                             joinrequestbuttons.leaveCallback(success, groupid);
298                         }
299                     });
300                 }
301             });
302             return true;
303         });
304 
305         /////////////////////////////
306         // Initialization          //
307         /////////////////////////////
308 
309         /**
310          * Initializes and renders the joinrequest buttons.  The following params
311          * should be sent while triggering the event:
312          *  {Object} groupData    the profile of the group for which buttons need to be rendered
313          *  {String} groupid      the id of the group for which buttons need to be rendered
314          *  {String} joinability  the group's joinability setting
315          *  {int} managerCount    optional number of managers in the group, defaults to 1
316          *  {Function} onShow     optional callback function called when the buttons are rendered
317          *  {Function} requestCallback  optional callback function called when the 'Request to join' button is clicked
318          *  {Function} joinCallback     optional callback function called when the 'Join' button is clicked
319          *  {Function} leaveCallback    optional callback function called when the 'Leave' button is clicked
320          *  {Object} joinrequests  optional joinrequest data from the server. If not provided, this
321          *      widget will issue a server request to get the data if needed.
322          */
323         $(document).on('init.joinrequestbuttons.sakai', function (ev, groupData, groupid, joinability, managerCount, buttonStyle, onShow, requestCallback, joinCallback, leaveCallback, joinrequests) {
324             if (!groupid || !joinability) {
325                 return;
326             }
327             joinrequestbuttons.groupData = groupData;
328             joinrequestbuttons.groupid = groupid;
329             joinrequestbuttons.joinability = joinability;
330             joinrequestbuttons.managerCount = managerCount || 1;
331             joinrequestbuttons.buttonStyle = buttonStyle || false;
332             joinrequestbuttons.onShow = onShow || false;
333             joinrequestbuttons.joinrequests = joinrequests || false;
334             joinrequestbuttons.requestCallback = requestCallback || false;
335             joinrequestbuttons.joinCallback = joinCallback || false;
336             joinrequestbuttons.leaveCallback = leaveCallback || false;
337             render();
338         });
339         $(document).trigger('ready.joinrequestbuttons.sakai');
340     };
341 
342     sakai.api.Widgets.widgetLoader.informOnLoad("joinrequestbuttons");
343 });
