302         /**
303          * Update group profile
304          *
305          * @param {String} id The id of the group to update
306          * @param {Object} profile The group's profile
307          * @param {Array} tags The group's tags
308          * @param {Object} groupData The group's authprofile data - need this for role extraction
309          * @param {Function} callback Callback function, passes (success, updated)
310          */
311         updateGroupProfile : function(id, profile, tags, groupData, callback) {
312             var groupProfileURL = "/~" + id + "/public/authprofile";
313             var groupProfileSaveURL = groupProfileURL + ".profile.json";
314             var batch = [];
315             var doProfilePost = false,
316                 doTagsPost = false,
317                 doPermissionPost = false;
318 
319             var updatePermissions = function(_callback) {
320                 var roles = $.parseJSON(groupData["sakai:roles"]);
321                 sakaiGroupsAPI.setPermissions(id, profile[ "sakai:group-joinable" ], profile[ "sakai:group-visible" ], roles, function( success, data ) {
322                     if ( $.isFunction( _callback ) ) {
323                         _callback( success );
324                     }
325                 });
326             };
327 
328             var updateProfile = function(_callback) {
329                 sakai_serv.batch( batch, function( success, data ) {
330                     if ( $.isFunction( _callback ) ) {
331                         _callback( success );
332                     }
333                 });
334             };
335 
336             // Get the difference of the tags arrays. If there is one, then we should update it
337             groupData[ "sakai:tags" ] = groupData[ "sakai:tags" ] || [];
338             var merged = _.uniq( $.merge( $.merge( [], tags ), groupData[ "sakai:tags" ] ) );
339             if ( merged.length !== tags.length || merged.length !== groupData[ "sakai:tags" ].length ) {
340                 doTagsPost = true;
341             }
342 
343             if ( groupData[ "sakai:group-joinable" ] !== profile[ "sakai:group-joinable" ] || groupData[ "sakai:group-visible" ] !== profile[ "sakai:group-visible" ] ) {
344                 doPermissionPost = true;
345             }
346 
347             $.each( profile, function(i, data) {
348                 if ( groupData[i] !== data ) {
349                     doProfilePost = true;
350                     // Update the group data immediately
351                     groupData[i] = data;
352                 }
353             });
354 
355             if ( doProfilePost || doTagsPost || doPermissionPost ) {
356                 if ( doProfilePost ) {
357                     batch.push({
358                         "url": groupProfileSaveURL,
359                         "method": "POST",
360                         "parameters": profile
361                     });
362                     // Also update the pseudo-groups sakai:parent-group-title property
363                     var roles = $.parseJSON(groupData["sakai:roles"]);
364                     $.each(roles, function(i, role) {
365                         batch.push({
366                             "url": "/system/userManager/group/" + id + "-" + role.id + ".update.json",
367                             "method": "POST",
368                             "parameters": {
369                                 "sakai:parent-group-title": profile["sakai:group-title"]
370                             }
371                         });
372                     });
373                 }
374 
375                 // Always call tagEntity, it has it's own internal 'no POSTing if no changes' mechanism
376                 sakai_util.tagEntity( groupProfileURL, tags, groupData[ "sakai:tags" ], function( success, newTags ) {
377                     groupData[ "sakai:tags" ] = newTags;
378 
379                     if ( doProfilePost ) {
380                         updateProfile(function(success, data) {
381                             if ( doPermissionPost ) {
382                                 updatePermissions(function(success, data) {
383                                     if ( $.isFunction( callback ) ) {
384                                         callback( success );
385                                     }
386                                 });
387                             } else {
388                                 if ( $.isFunction( callback ) ) {
389                                     callback( success );
390                                 }
391                             }
392                         });
393                     } else if ( doPermissionPost ) {
394                         updatePermissions(function(success, data) {
395                             if ( $.isFunction( callback ) ) {
396                                 callback( success );
397                             }
398                         });
399                     } else {
400                         if ( $.isFunction( callback ) ) {
401                             callback( success );
402                         }
403                     }
404                 });
405             } else {
406                 callback( true );
407             }

411          * Checks whether the given value is valid as defined by the given

437         /**
438          * Public function used to set joinability and visibility permissions for a
439          * group with groupid.
440          *
441          * @param {String} groupid The id of the group that needs permissions set
442          * @param {String} joinable The joinable state for the group (from sakai.config.Permissions.Groups)
443          * @param {String} visible The visibile state for the group (from sakai.config.Permissions.Groups)
444          * @param {Array} roles The roles for this group, from the "sakai:roles" property of the group
445          * @param {Function} callback Function to be called on complete - callback
446          *   args: (success)
447          * @return None
448          */
449         setPermissions : function(groupid, joinable, visible, roles, callback) {
450             if ( groupid && _.isString(groupid) &&
451                this.isValidPermissionsProperty(sakai_conf.Permissions.Groups.joinable, joinable) &&
452                this.isValidPermissionsProperty(sakai_conf.Permissions.Groups.visible, visible)) {
453 
454                 // issue a BATCH POST to update Jackrabbit group & Home Folder group
455                 var batchRequests = [];
456                 // add in the main group, we need to modify their permissions too
457                 roles.push({id:groupid});
458                 $.each(roles, function(i, role) {
459                     var groupURL = groupid;
460                     if (role.id !== groupid) {
461                         groupURL += "-" + role.id;
462                     }
463                     var groupUpdateURL = "/system/userManager/group/" + groupURL + ".update.html";
464 
465                     // determine visibility state
466                     if (visible === sakai_conf.Permissions.Groups.visible.members) {
467                         // visible to members only, so remove everyone & anonymous, as they're not a member
468                         batchRequests.push({
469                             "url": groupUpdateURL,
470                             "method": "POST",
471                             "parameters": {
472                                 ":viewer@Delete":
473                                 [
474                                     "everyone",
475                                     "anonymous"
476                                 ],
477                                 "sakai:group-visible": visible,
478                                 "sakai:group-joinable": joinable
479                             }
480                         });
481                     } else if (visible === sakai_conf.Permissions.Groups.visible.allusers) {
482                         // visible to all logged in users
483                         // remove anonymous, as this is only for logged in users
484                         batchRequests.push({
485                             "url": groupUpdateURL,
486                             "method": "POST",
487                             "parameters": {
488                                 ":viewer": "everyone",
489                                 ":viewer@Delete": "anonymous",
490                                 "sakai:group-visible": visible,
491                                 "sakai:group-joinable": joinable
492                             }
493                         });
494                     } else {
495                         // visible to the public
496                         // all logged in users 'everyone'
497                         // all non-logged in users 'anonymous'
498                         batchRequests.push({
499                             "url": groupUpdateURL,
500                             "method": "POST",
501                             "parameters": {
502                                 ":viewer":
503                                 [
504                                     "everyone",
505                                     "anonymous"
506                                 ],
507                                 "sakai:group-visible": visible,
508                                 "sakai:group-joinable": joinable
509                             }
510                         });
511                     }
512                 });
513 
514 
515                 // issue the BATCH POST
516                 sakai_serv.batch(batchRequests, function(success, data) {
517                     if (success) {
518                         // update group context and call callback
519                         if(sakai_global.currentgroup && sakai_global.currentgroup.data && sakai_global.currentgroup.data.authprofile) {
520                             sakai_global.currentgroup.data.authprofile["sakai:group-joinable"] = joinable;
521                             sakai_global.currentgroup.data.authprofile["sakai:group-visible"] = visible;
522                         }
523                         if ($.isFunction(callback)) {
524                             callback(true);
525                         }
526                     } else {
527                         // Log an error message
528                         debug.error("Setting permissions on the group failed");
529                         if ($.isFunction(callback)) {
530                             callback(false);
531                         }
532                     }
533                 });
534             } else {
535                 debug.warn("Invalid arguments sent to sakai.api.Groups.setPermissions");
536                 if ($.isFunction(callback)) {
537                     callback(false);
538                 }
539             }

543         /**
544          * Determines whether the current user is a manager of the given group.
545          *
546          * @param groupid {String} id of the group to check
547          * @param {Object} meData the data from sakai.api.User.data.me
548          * @return true if the current user is a manager, false otherwise
549          */
550         isCurrentUserAManager : function(groupid, meData, groupinfo) {
551             if (groupinfo) {
552                 var managementRoles = [];
553                 var roles = $.parseJSON(groupinfo["sakai:roles"]);
554                 for (var r = 0; r < roles.length; r++) {
555                     if (roles[r].isManagerRole) {
556                         managementRoles.push(roles[r].id);
557                     }
558                 }
559                 var canManage = false;
560                 for (var i = 0; i < meData.groups.length; i++) {
561                     for (var mr = 0; mr < managementRoles.length; mr++) {
562                         if (meData.groups[i]["sakai:group-id"] === groupinfo["sakai:group-id"] + "-" + managementRoles[mr]) {
563                             canManage = true;
564                         }
565                     }
566                 }
567                 return canManage;
568             } else {
569                 if (!groupid || typeof(groupid) !== "string") {
570                     return false;
571                 }
572                 var managersGroupId = groupid + "-manager";
573                 return $.inArray(managersGroupId, meData.user.subjects) !== -1;
574             }
575         },

602         isAllowedToLeave : function(groupids, meData, callback) {
603             sakaiGroupsAPI.getGroupAuthorizableData(groupids, function(groupSuccess, groupData){
604                 var toReturn = {};
605                 var groupDataCache = {};
606                 var toCheck = [];
607                 $.each(groupData, function(groupid, group){
608                     if (sakaiGroupsAPI.isCurrentUserAManager(groupid, meData, group.properties)) {
609                         // Check if there is more then one manager in the group
610                         toCheck.push(groupid);
611                         groupDataCache[groupid] = group.properties;
612                     } else {
613                         // Members are always allowed to leave the group, managers should always be present and cannot leave when they are the last one in the group
614                         toReturn[groupid] = true;
615                     }
616                 });
617                 sakaiGroupsAPI.getMembers(toCheck, function(membersSuccess, memberData){
618                     $.each(memberData, function(groupid, members){
619                         var numManagers = sakaiGroupsAPI.getManagerCount(groupDataCache[groupid], members);
620                         toReturn[groupid] = numManagers > 1;
621                     });
622                     if ($.isFunction(callback)) {
623                         callback(toReturn);
624                     }
625                 });
626             });

629         /**
630          * Get the number of managers in the group
631          *
632          * @param {Object} groupdata The data from the group's authprofile
633          * @param {Object} members The result of sakai.api.Groups.getMembers()
634          */
635         getManagerCount : function(groupdata, members) {
636             var managers = 0;
637             if (groupdata["sakai:roles"]) {
638                 var roles = [],
639                     managerRoles = [];
640                 if (_.isString(groupdata["sakai:roles"])) {
641                     roles = $.parseJSON(groupdata["sakai:roles"]);
642                 }
643                 $.each(roles, function(i, role) {
644                     if (role.isManagerRole) {
645                         managerRoles.push(role.id);
646                     }
647                 });
648                 $.each(members, function(i, member) {
649                     member = member.results ? member.results : member;
650                     if ($.inArray(i, managerRoles) > -1 && member.length) {
651                         managers += member.length;
652                     }
653                 });
654             }
655             return managers;
