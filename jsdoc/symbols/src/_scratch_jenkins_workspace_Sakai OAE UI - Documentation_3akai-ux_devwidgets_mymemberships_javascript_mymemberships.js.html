 19 // load the master sakai object to access all Sakai OAE API methods
 20 require(['jquery', 'sakai/sakai.api.core'], function($, sakai) {
 21 
 22     /**
 23      * @name sakai_global.mymemberships
 24      *
 25      * @class mymemberships
 26      *
 27      * @description
 28      * My Memberships lists the groups and worlds that a member is affiliated with
 29      *
 30      * @version 0.0.1
 31      * @param {String} tuid Unique id of the widget
 32      * @param {Boolean} showSettings Show the settings of the widget or not
 33      */
 34     sakai_global.mymemberships = function (tuid, showSettings) {
 35 
 36         /////////////////////////////
 37         // Configuration variables //
 38         /////////////////////////////
 39 
 40         var mymemberships = {  // global widget data
 41             isOwnerViewing: false,
 42             sortOrder: 'modified',
 43             listStyle: 'list'
 44         };
 45 
 46         // DOM jQuery Objects
 47         var $rootel = $('#' + tuid);  // unique container for each widget instance
 48         var $mymemberships_items = $('#mymemberships_items', $rootel);
 49         var $mymemberships_nodata = $('#mymemberships_nodata', $rootel);
 50         var $mymemberships_nogroups = $('#mymemberships_nogroups', $rootel);
 51         var $mymemberships_actionbar = $('#mymemberships_actionbar', $rootel);
 52         var $mymemberships_sortby = $('#mymemberships_sortby', $rootel);
 53         var $mymemberships_item = $('.mymemberships_item', $rootel);
 54         var $mymemberships_show_grid = $('.s3d-listview-grid', $rootel);
 55         var $mymemberships_show_list = $('.s3d-listview-list', $rootel);
 56         var $mymemberships_nosearchresults = $('#mymemberships_nosearchresults');
 57         var $mymemberships_result_count = $('.s3d-search-result-count', $rootel);
 58 
 59         var currentQuery = '';
 60 
 61         ///////////////////////
 62         // Utility Functions //
 63         ///////////////////////
 64 
 65         /**
 66          * Compare the titles of 2 group objects
 67          *
 68          * @param {Object} a
 69          * @param {Object} b
 70          * @return 1, 0 or -1
 71          */
 72         var groupSortName = function (a, b) {
 73             if (a['sakai:group-title'].toLowerCase() > b['sakai:group-title'].toLowerCase()) {
 74                 return 1;
 75             } else {
 76                 if (a['sakai:group-title'].toLowerCase() === b['sakai:group-title'].toLowerCase()) {
 77                     return 0;
 78                 } else {
 79                     return -1;
 80                 }
 81             }
 82         };
 83 
 84         /**
 85          * Compare the modification date of 2 group objects
 86          *
 87          * @param {Object} a
 88          * @param {Object} b
 89          * @return 1, 0 or -1
 90          */
 91         var groupSortModified = function (a, b) {
 92             if (a['lastModified'] > b['lastModified']) {
 93                 return 1;
 94             } else {
 95                 if (a['lastModified'] === b['lastModified']) {
 96                     return 0;
 97                 } else {
 98                     return -1;
 99                 }
100             }
101         };
102 
103         ////////////////////
104         // Event Handlers //
105         ////////////////////
106 
107         $mymemberships_sortby.change(function () {
108             var sortSelection = this.options[this.selectedIndex].value;
109             if (sortSelection === 'desc') {
110                 mymemberships.sortOrder = 'desc';
111                 $.bbq.pushState({'mso': 'desc'});
112             } else if (sortSelection === 'asc') {
113                 mymemberships.sortOrder = 'asc';
114                 $.bbq.pushState({'mso': 'asc'});
115             } else {
116                 mymemberships.sortOrder = 'modified';
117                 $.bbq.pushState({'mso': 'modified'});
118             }
119             doInit();
120         });
121 
122         ////////////////////////////////////////////
123         // Data retrieval and rendering functions //
124         ////////////////////////////////////////////
125 
126         /**
127          * Renders the given groups
128          *
129          * @param {Object} groups  JSON containing group data
130          */
131         var render = function (groups) {
132             if (!groups.entry.length) {
133                 $mymemberships_items.hide();
134                 sakai.api.Util.TemplateRenderer('mymemberships_nogroups_template', {isMe: mymemberships.isOwnerViewing}, $mymemberships_nogroups);
135                 $mymemberships_nogroups.show();
136                 $('.s3d-page-header-top-row', $rootel).hide();
137                 $('.s3d-page-header-bottom-row', $rootel).hide();
138                 return;
139             } else {
140                 if (sakai.data.me.user.anon) {
141                     $('.s3d-page-header-bottom-row', $rootel).hide();
142                 } else {
143                     $('.s3d-page-header-top-row', $rootel).show();
144                     $('.s3d-page-header-bottom-row', $rootel).show();
145                 }
146                 if (mymemberships.sortOrder === 'modified') {
147                     groups.entry = groups.entry.sort(groupSortModified);
148                     groups.entry.reverse();
149                 } else {
150                     groups.entry = groups.entry.sort(groupSortName);
151                     if (mymemberships.sortOrder === 'desc') {
152                         groups.entry.reverse();
153                     }
154                 }
155                 var groupData = [];
156                 var tempGroupData = sakai.api.Groups.prepareGroupsForRender(groups.entry, sakai.data.me);
157                 $.each(tempGroupData, function (i, group) {
158                     var titleMatch = group['sakai:group-title'] && group['sakai:group-title'].toLowerCase().indexOf(currentQuery.toLowerCase()) >= 0;
159                     var descriptionMatch = group['sakai:group-description'] && group['sakai:group-description'].toLowerCase().indexOf(currentQuery.toLowerCase()) >= 0;
160                     var idMatch = group.groupid.toLowerCase().indexOf(currentQuery.toLowerCase()) >= 0;
161                     if (titleMatch || descriptionMatch || idMatch) {
162                         var groupType = sakai.api.i18n.getValueForKey('OTHER');
163                         if (group['sakai:category']) {
164                             sakai.api.Util.getTemplates(function(success, templates) {
165                                 if (success) {
166                                     for (var c = 0; c < templates.length; c++) {
167                                         if (templates[c].id === group['sakai:category']) {
168                                             groupType = sakai.api.i18n.getValueForKey(templates[c].title);
169                                         }
170                                     }
171                                 } else {
172                                     debug.error('Could not get the group templates');
173                                 }
174                             });
175                         }
176 
177                         groupData.push({
178                             id: group.groupid,
179                             url: '/~' + sakai.api.Util.makeSafeURL(group.groupid),
180                             picPath: group.picPath,
181                             picPathLarge: group.picPathLarge,
182                             edit_url: '/dev/group_edit2.html?id=' + group.groupid,
183                             title: group['sakai:group-title'],
184                             titleShort: group['sakai:group-title-short'],
185                             titleShorter: group['sakai:group-title-shorter'],
186                             descShort: group['sakai:group-description-short'],
187                             descShorter: group['sakai:group-description-shorter'],
188                             type: groupType,
189                             lastModified: group.lastModified,
190                             contentCount: group.counts.contentCount,
191                             membersCount: group.counts.membersCount,
192                             tags: group.tagsProcessed,
193                             userMember: sakai.api.Groups.isCurrentUserAMember(group.groupid,sakai.data.me),
194                             joinable: group['sakai:group-joinable']
195                         });
196                     }
197                 });
198                 var json = {
199                     groups: groupData,
200                     isOwnerViewing: mymemberships.isOwnerViewing,
201                     user_manages: function (group) {
202                         if (!group) { return false; }
203                         return sakai.api.Groups.isCurrentUserAManager(group.id, sakai.data.me);
204                     },
205                     sakai: sakai
206                 };
207                 $mymemberships_nodata.hide();
208                 $mymemberships_nogroups.hide();
209                 // Show message that no search results where returned.
210                 if (!json.groups.length) {
211                     $mymemberships_nosearchresults.show();
212                     $mymemberships_items.hide();
213                 } else {
214                     $mymemberships_nosearchresults.hide();
215                     $mymemberships_items.show();
216                     $('#mymemberships_sortarea', $rootel).show();
217                     $('#mymemberships_items', $rootel).html(sakai.api.Util.TemplateRenderer(
218                         $('#mymemberships_items_template', $rootel), json));
219                 }
220 
221                 // display search results count
222                 if (currentQuery && groupData.length) {
223                     $mymemberships_result_count.show();
224                     var resultLabel = sakai.api.i18n.getValueForKey('RESULTS');
225                     if (groupData.length === 1) {
226                         resultLabel = sakai.api.i18n.getValueForKey('RESULT');
227                     }
228                     $mymemberships_result_count.children('.s3d-search-result-count-label').text(resultLabel);
229                     $mymemberships_result_count.children('.s3d-search-result-count-count').text(groupData.length);
230                 } else {
231                     $mymemberships_result_count.hide();
232                 }
233 
234                 // display functions available to logged in users
235                 if (!sakai.data.me.user.anon) {
236                     $('.mymemberships_item_anonuser').hide();
237                     $('.mymemberships_item_user_functions').show();
238                 }
239 
240                 if (mymemberships.isOwnerViewing) {
241                     // disable remove membership button if not allowed to leave
242                     var groupsToCheck = [];
243                     $.each(groups.entry, function(i, group) {
244                         groupsToCheck.push(group.groupid);
245                     });
246                     sakai.api.Groups.isAllowedToLeave(groupsToCheck, sakai.data.me, function(leaveAllowed) {
247                         $.each(leaveAllowed, function(groupid, canLeave) {
248                             if (!canLeave) {
249                                 $('.mymemberships_leave[data-sakai-entityid="' + groupid + '"]').addClass('mymemberhips_disabled_leave');
250                             }
251                         });
252                     });
253                 }
254             }
255         };
256 
257         var checkAddingEnabled = function() {
258             if ($('.mymemberships_select_group_checkbox:checked')[0]) {
259                 $('#mymemberships_addpeople_button').removeAttr('disabled');
260                 $('#mymemberships_message_button').removeAttr('disabled');
261             } else {
262                 $('#mymemberships_addpeople_button').attr('disabled', true);
263                 $('#mymemberships_message_button').attr('disabled', true);
264                 $('#mymemberships_select_checkbox').removeAttr('checked');
265             }
266         };
267 
268         var updateMessageAndAddToData = function() {
269             var idArr = [];
270             var titleArr = [];
271             $.each($('.mymemberships_select_group_checkbox:checked'), function(i, group) {
272                 idArr.push($(group).data('groupid'));
273                 titleArr.push($(group).data('grouptitle'));
274             });
275             $('#mymemberships_message_button').attr('sakai-entityid', idArr);
276             $('#mymemberships_message_button').attr('sakai-entityname', titleArr);
277             $('#mymemberships_addpeople_button').data('entityid', idArr);
278             $('#mymemberships_addpeople_button').data('entityname', titleArr);
279         };
280 
281         var removeMembership = function(groupid,groupname) {
282             sakai.api.Groups.getRole(sakai.data.me.user.userid,groupid,function(success,role) {
283                 sakai.api.Groups.leave(groupid,role,sakai.data.me,function(success) {
284                     if (success) {
285                         $(window).trigger('lhnav.updateCount', ['memberships', -1]);
286                         sakai.api.Util.Modal.close('#mymemberships_delete_membership_dialog');
287                         $('#mymemberships_item_'+groupid).fadeOut(false, function() {
288                             // Show the default message if I have no remaining memberships
289                             if ($('#mymemberships_items li:visible').length === 0) {
290                                 render({
291                                     entry: []
292                                 });
293                             }
294                         });
295                         sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey('MY_MEMBERSHIPS','mymemberships'),
296                             sakai.api.i18n.getValueForKey('YOU_HAVE_LEFT_GROUP','mymemberships').replace('${groupname}',groupname),
297                             sakai.api.Util.notification.type.INFORMATION);
298                     } else {
299                         sakai.api.Util.Modal.close('#mymemberships_delete_membership_dialog');
300                         sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey('MY_MEMBERSHIPS','mymemberships'),
301                             sakai.api.i18n.getValueForKey('ERROR_LEAVING_GROUP','mymemberships').replace('${groupname}',groupname),
302                             sakai.api.Util.notification.type.ERROR);
303                     }
304                 });
305             });
306         };
307 
308         var uncheckAll = function() {
309             $('#mymemberships_select_checkbox').removeAttr('checked');
310         };
311 
312 
313         /////////////////////////////
314         // Initialization function //
315         /////////////////////////////
316 
317         var handleHashChange = function() {
318             if (sakai_global.profile.main.data.userid === sakai.data.me.user.userid) {
319                 mymemberships.isOwnerViewing = true;
320                 render(sakai.api.Groups.getMemberships(sakai.data.me.groups));
321             } else {
322                 sakai.api.Server.loadJSON('/system/me', function(success, data) {
323                     mymemberships.isOwnerViewing = false;
324                     render(sakai.api.Groups.getMemberships(data.groups));
325                 }, { uid: sakai_global.profile.main.data.userid });
326             }
327             sakai.api.Util.TemplateRenderer('mymemberships_title_template', {
328                 isMe: mymemberships.isOwnerViewing,
329                 user: sakai_global.profile.main.data.basic.elements.firstName.value
330             }, $('#mymemberships_title_container', $rootel));
331 
332             uncheckAll();
333 
334             $('.s3d-listview-options', $rootel).find('div').removeClass('selected');
335             mymemberships.listStyle = $.bbq.getState('ls') || 'list';
336             if (mymemberships.listStyle === 'list') {
337                 $('#mymemberships_items', $rootel).removeClass('s3d-search-results-grid');
338                 $mymemberships_show_list.addClass('selected');
339                 $mymemberships_show_list.children().addClass('selected');
340             } else {
341                 $('#mymemberships_items', $rootel).addClass('s3d-search-results-grid');
342                 $mymemberships_show_grid.addClass('selected');
343                 $mymemberships_show_grid.children().addClass('selected');
344             }
345         };
346 
347         var addBinding = function() {
348             $(window).bind('hashchanged.mymemberships.sakai', handleHashChange);
349 
350             $('#mymemberships_search_button').click(function() {
351                 var q = $.trim($('#mymemberships_livefilter').val());
352                 if (q !== currentQuery) {
353                     $.bbq.pushState({'mq': q, 'mp': 1});
354                     currentQuery = q;
355                 }
356             });
357 
358             $mymemberships_show_list.click(function() {
359                 $.bbq.pushState({'ls': 'list'});
360             });
361 
362             $mymemberships_show_grid.click(function() {
363                 $.bbq.pushState({'ls': 'grid'});
364             });
365 
366             $('#mymemberships_livefilter').keyup(function(ev) {
367                 var q = $.trim($('#mymemberships_livefilter').val());
368                 if (q !== currentQuery && ev.keyCode === 13) {
369                     $.bbq.pushState({'mq': q, 'mp': 1});
370                     currentQuery = q;
371                 }
372                 return false;
373             });
374 
375             $('#mymemberships_select_checkbox').change(function() {
376                 if ($(this).is(':checked')) {
377                     $('#mymemberships_addpeople_button').removeAttr('disabled');
378                     $('.mymemberships_select_group_checkbox').attr('checked', true);
379                 } else {
380                     $('#mymemberships_addpeople_button').attr('disabled', true);
381                     $('.mymemberships_select_group_checkbox').removeAttr('checked');
382                 }
383                 checkAddingEnabled();
384                 updateMessageAndAddToData();
385             });
386 
387             $('.mymemberships_select_group_checkbox').live('change', function() {
388                 checkAddingEnabled();
389                 updateMessageAndAddToData();
390             });
391 
392             sakai.api.Util.Modal.setup('#mymemberships_delete_membership_dialog', {
393                 modal: true,
394                 overlay: 20,
395                 toTop: true
396             });
397 
398             $('.s3d-actions-delete', $rootel).live('click', function() {
399                 if ($(this).hasClass('mymemberhips_disabled_leave')) {
400                     sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey('GROUP_MEMBERSHIP'),
401                         sakai.api.i18n.getValueForKey('UNABLE_TO_LEAVE', 'mymemberships').replace('${groupname}', $(this).attr('data-sakai-entityname')),
402                         sakai.api.Util.notification.type.ERROR);
403                 } else {
404                     var msg = sakai.api.i18n.getValueForKey('ARE_YOU_SURE_YOU_WANT_TO_LEAVE', 'mymemberships').replace('${groupname}', '<span class="s3d-bold">' + $(this).data('sakai-entityname') + '</span>');
405                     $('#mymemberships_are_you_sure').html(msg);
406                     $('#mymemberships_delete_membership_confirm').attr('data-sakai-entityid', $(this).attr('data-sakai-entityid'));
407                     $('#mymemberships_delete_membership_confirm').attr('data-sakai-entityname', $(this).attr('data-sakai-entityname'));
408                     sakai.api.Util.Modal.open('#mymemberships_delete_membership_dialog');
409                 }
410             });
411 
412             $('#mymemberships_delete_membership_confirm').live('click', function() {
413                 removeMembership($(this).attr('data-sakai-entityid'), $(this).attr('data-sakai-entityname'));
414                 updateMessageAndAddToData();
415             });
416 
417             if (sakai_global.profile.main.data.userid !== sakai.data.me.user.userid) {
418                     $('.searchgroups_result_plus',$rootel).live('click', function(ev) {
419                     var joinable = $(this).data('group-joinable');
420                     var groupid = $(this).data('groupid');
421                     var itemdiv = $(this);
422                     sakai.api.Groups.addJoinRequest(groupid, function (success) {
423                         if (success) {
424                             var notimsg = '';
425                             if (joinable === 'withauth') {
426                                 // Don't add green tick yet because they need to be approved.
427                                 notimsg = sakai.api.i18n.getValueForKey('YOUR_REQUEST_HAS_BEEN_SENT');
428                             }
429                             else  { // Everything else should be regular success
430                                 $('#searchgroups_memberimage_'+groupid).show();
431                                 notimsg = sakai.api.i18n.getValueForKey('SUCCESSFULLY_ADDED_TO_GROUP');
432                             }
433                             sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey('GROUP_MEMBERSHIP'),
434                                 notimsg, sakai.api.Util.notification.type.INFORMATION);
435                             itemdiv.removeClass('s3d-action-icon s3d-actions-addtolibrary searchgroups_result_plus');
436                             $('#searchgroups_memberimage_' + groupid).show();
437                         } else {
438                             sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey('GROUP_MEMBERSHIP'),
439                                 sakai.api.i18n.getValueForKey('PROBLEM_ADDING_TO_GROUP'),
440                                 sakai.api.Util.notification.type.ERROR);
441                         }
442                     });
443                 });
444             }
445         };
446 
447         /**
448          * Initialization function that is run when the widget is loaded. Determines
449          * which mode the widget is in (settings or main), loads the necessary data
450          * and shows the correct view.
451          */
452         var doInit = function () {
453             addBinding();
454             currentQuery = $.bbq.getState('mq') || '';
455             $('#mymemberships_livefilter').val(currentQuery);
456             mymemberships.sortOrder = $.bbq.getState('mso') || 'modified';
457             $('#mymemberships_sortby').val(mymemberships.sortOrder);
458             mymemberships.listStyle = $.bbq.getState('ls') || 'list';
459 
460             handleHashChange();
461         };
462 
463         // run the initialization function when the widget object loads
464         doInit();
465 
466     };
467 
468     // inform Sakai OAE that this widget has loaded and is ready to run
469     sakai.api.Widgets.widgetLoader.informOnLoad('mymemberships');
470 });
