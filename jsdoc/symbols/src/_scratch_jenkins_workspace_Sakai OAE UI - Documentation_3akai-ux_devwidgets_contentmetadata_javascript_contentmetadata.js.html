 60         var contentmetadataInputEdit = ".contentmetadata_edit_input";

 74         var contentmetadataCopyrightTemplate = "contentmetadata_copyright_template";
 75         var contentmetadataDetailsTemplate = "contentmetadata_details_template";

 92         /**
 93          * Add binding to the input elements that allow editting
 94          * @param {String|Boolean} mode Can be false or 'edit' depending on the mode you want to be in
 95          */
 96         var addEditBinding = function( mode, tags ) {
 97             if (mode === "edit") {
 98                 if ($(".contentmetadata_edit_input").length) {
 99                     $(".contentmetadata_edit_input").focus();
100                 }
101                 if ( !tags ) {
102                     $(contentmetadataInputEdit).blur( editInputBlur );
103                 } else {
104                     sakai.api.Util.hideOnClickOut( $( ".autosuggest_wrapper", $contentmetadataTagsContainer ) , "#assignlocation_container, " + $contentmetadataTagsContainer.selector + " .autosuggest_wrapper", function() {
105                       editInputBlur(false, "tags");
106                     });
107                 }
108             }
109         };
110 
111         /**
112          * Render the Description template
113          * @param {String|Boolean} mode Can be false or 'edit' depending on the mode you want to be in
114          */
115         var renderDescription = function(mode){
116             sakai_global.content_profile.content_data.mode = mode;
117             var json = {
118                 data: sakai_global.content_profile.content_data,
119                 sakai: sakai
120             };
121             $contentmetadataDescriptionContainer.toggleClass("contentmetadata_editing", mode === "edit");
122             $contentmetadataDescriptionContainer.html(sakai.api.Util.TemplateRenderer(contentmetadataDescriptionTemplate, json));
123             addEditBinding(mode);
124         };
125 
126         /**
127          * Render the URL template
128          * @param {String|Boolean} mode Can be false or 'edit' depending on the mode you want to be in
129          */
130         var renderUrl = function(mode){
131             sakai_global.content_profile.content_data.mode = mode;
132             contentType = sakai.api.Content.getMimeType(sakai_global.content_profile.content_data.data);
133             if(contentType === "x-sakai/link") {
134                 var json = {
135                     data: sakai_global.content_profile.content_data,
136                     sakai: sakai
137                 };
138                 sakai.api.Util.TemplateRenderer(contentmetadataUrlTemplate, json, $contentmetadataUrlContainer);
139                 $contentmetadataUrlContainer.show();
140                 $contentmetadataTagsContainer.removeClass("last");
141             } else {
142                 $contentmetadataUrlContainer.hide();
143             }
144             addEditBinding(mode);
145         };
146 
147         /**
148          * Render the Tags template
149          * @param {String|Boolean} mode Can be false or 'edit' depending on the mode you want to be in
150          */
151         var renderTags = function(mode){
152             sakai_global.content_profile.content_data.mode = mode;
153             var json = {
154                 data: sakai_global.content_profile.content_data,
155                 sakai: sakai,
156                 tags: sakai.api.Util.formatTags(sakai_global.content_profile.content_data.data["sakai:tags"])
157             };
158             $contentmetadataTagsContainer.html(sakai.api.Util.TemplateRenderer(contentmetadataTagsTemplate, json));
159             $contentmetadataTagsContainer.toggleClass("contentmetadata_editing", mode === "edit");
160             $contentmetadataTagsContainer.toggleClass("contentmetadata_editable", mode !== "edit");
161             if (mode === "edit") {
162                 $contentmetadataAutosuggestElt = $( "#contentmetadata_tags_tags" );
163                 sakai.api.Util.AutoSuggest.setupTagAndCategoryAutosuggest($contentmetadataAutosuggestElt , null, $( '.list_categories', $contentmetadataTagsContainer ), sakai_global.content_profile.content_data.data['sakai:tags'], function(){
164                     $('.as-selections', $contentmetadataTagsContainer ).addClass('contentmetadata_edit_input');
165                     $contentmetadataAutosuggestElt.focus();
166                 });
167             }
168             addEditBinding( mode, true );
169         };
170 
171         /**
172          * Render the Copyright template
173          * @param {String|Boolean} mode Can be false or 'edit' depending on the mode you want to be in
174          */
175         var renderCopyright = function(mode){
176             sakai_global.content_profile.content_data.mode = mode;
177             var json = {
178                 data: sakai_global.content_profile.content_data,
179                 sakai: sakai
180             };
181             $contentmetadataCopyrightContainer.toggleClass("contentmetadata_editing", mode === "edit");
182             $contentmetadataCopyrightContainer.html(sakai.api.Util.TemplateRenderer(contentmetadataCopyrightTemplate, json));
183             addEditBinding(mode);
184         };

186         /**
187          * Render the Details template
188          * @param {String|Boolean} mode Can be false or 'edit' depending on the mode you want to be in
189          */
190         var renderDetails = function(mode){
191             sakai_global.content_profile.content_data.mode = mode;
192             sakai.api.Content.getCreatorProfile(sakai_global.content_profile.content_data.data, function(success, profile) {
193                 var json = {
194                     data: sakai_global.content_profile.content_data,
195                     sakai: sakai,
196                     creator: profile
197                 };
198                 sakai.api.Util.TemplateRenderer(contentmetadataDetailsTemplate, json, $contentmetadataDetailsContainer);
199                 addEditBinding(mode);
200             });
201         };
202 
203         var createActivity = function(activityMessage){
204             var activityData = {
205                 "sakai:activityMessage": activityMessage
206             };
207             sakai.api.Activity.createActivity("/p/" + sakai_global.content_profile.content_data.data["_path"], "content", "default", activityData, function(responseData, success){
208                 if (success) {
209                     // update the entity widget with the new activity
210                     $(window).trigger("updateContentActivity.entity.sakai", activityMessage);
211                 }
212             });
213         };

219         var updateTags = function(){
220             var tags = sakai.api.Util.AutoSuggest.getTagsAndCategories($contentmetadataAutosuggestElt, true);
221             var path = sakai_global.content_profile.content_data.data["_path"];
222             sakai.api.Util.tagEntity("/p/" + path, tags, sakai_global.content_profile.content_data.data["sakai:tags"], function(success, newTags){
223                 // We need this check because it's possible that this is called after new content is shown.
224                 // If that is the case, we still need to send the update tags activity, but not render the actual tags.
225                 if(path === sakai_global.content_profile.content_data.data["_path"]){
226                     sakai_global.content_profile.content_data.data["sakai:tags"] = newTags;
227                     renderTags(false);
228                 }
229                 if (success) {
230                     // Create an activity
231                     createActivity("UPDATED_TAGS");
232                 }
233             });
234         };
235 
236         /**
237          * Updates the url
238          * @param {String} updateUrl The new URL to update with
239          */
240         var updateUrl = function(url) {
241             var preview = sakai.api.Content.getPreviewUrl(url);
242             sakai_global.content_profile.content_data.data["sakai:pooled-content-url"] = url;
243             sakai_global.content_profile.content_data.data["sakai:pooled-content-revurl"] = url;
244             sakai_global.content_profile.content_data.data["sakai:preview-url"] = preview.url;
245             sakai_global.content_profile.content_data.data["sakai:preview-type"] = preview.type;
246             sakai_global.content_profile.content_data.data["sakai:preview-avatar"] = preview.avatar;
247             sakai_global.content_profile.content_data.data["length"] = url.length;
248             $.ajax({
249                 url: "/p/" + sakai_global.content_profile.content_data.data["_path"] + ".html",
250                 type: "POST",
251                 cache: false,
252                 data: {
253                     "sakai:pooled-content-url": url,
254                     "sakai:pooled-content-revurl": url,
255                     "sakai:preview-url": preview.url,
256                     "sakai:preview-type": preview.type,
257                     "sakai:preview-avatar": preview.avatar,
258                     "length": url.length
259                 },
260                 success: function(){
261                     createActivity("UPDATED_URL");
262                     $(window).trigger("updated.version.content.sakai");
263                 }
264             });
265         };
266 
267         /**
268          * Trigger the template to render the edit mode
269          * @param {Object} ev Trigger event
270          */
271         var editData = function(ev){
272             if ( !$( ev.target ).is( "a, select, option, textarea" ) ) {
273                 $target = $( ev.target ).closest( ".contentmetadata_editable" );
274                 if ( $target.length ) {
275                     // Need to clear out any active editingElements before creating a new one
276                     if (editingElement !== "") {
277                         editInputBlur(false, editingElement);
278                     }
279                     editingElement = $target.attr( "data-edit-field" );
280                     if (editingElement === 'tags') {
281                         renderTags('edit');
282                     }
283                 }
284             }
285         };
286 
287         /**
288          * Handle losing of focus on an input element
289          * @param {Object} e Element that lost the focus
290          * @param {String} forceElt Force the editingElement here, to indicate that only that should be blurred
291          */
292         var editInputBlur = function( e, forceElt ) {
293             if ( !e && forceElt !== editingElement ) {
294                 return;
295             }
296             if (editingElement === 'tags') {
297                 updateTags();
298             }
299             editingElement = "";
300         };
301 
302         /**
303          * Updates data for a specific field
304          * @param {String} field The field to update
305          * @param {String} value The new field value
306          */
307         var updateData = function(field, value) {
308             if (value !== sakai_global.content_profile.content_data.data['sakai:' + field]) {
309                 switch (field) {
310                     case 'description':
311                         saveData({'sakai:description': value}, function(success) {
312                             if (success) {
313                                 sakai_global.content_profile.content_data.data['sakai:description'] = value;
314                                 createActivity('UPDATED_DESCRIPTION');
315                             }
316                         });
317                         break;
318                     case 'copyright':
319                         saveData({'sakai:copyright': value}, function(success) {
320                             if (success) {
321                                 sakai_global.content_profile.content_data.data['sakai:copyright'] = value;
322                                 createActivity('UPDATED_COPYRIGHT');
323                             }
324                         });
325                         break;
326                     case 'url':
327                         updateUrl(value);
328                         break;
329                 }
330             }
331         };

333         /**
334          * Saves the content metadata
335          * @param {Object} dataToSave Object containing the field and data to save
336          * @param {Function} callback Function to call when the request is completed
337          */
338         var saveData = function(dataToSave, callback) {
339             var url = '/p/' + sakai_global.content_profile.content_data.data['_path'] + '.json';
340             sakai.api.Server.saveJSON(url, dataToSave, function(success, data) {
341                 if ($.isFunction(callback)) {
342                     callback(success);
343                 }
344             });
345         };

351         /**
352          * Animate the hidden or shown data containers
353          */
354         var animateData = function(){
355             $collapsibleContainers.animate({
356                 'margin-bottom': 'toggle',
357                 height: 'toggle',
358                 opacity: 'toggle',
359                 'padding-top': 'toggle',
360                 'padding-bottom': 'toggle'
361             }, 400);
362             $("#contentmetadata_show_more > div").toggle();
363             if (contentType === "x-sakai/link") {
364                 $contentmetadataUrlContainer.toggleClass("last");
365             } else {
366                 $contentmetadataTagsContainer.toggleClass("last");
367             }
368         };

370         /**
371          * Add binding/events to the elements in the widget
372          */
373         var addBinding = function(){
374             $(".contentmetadata_editable_for_maintainers").toggleClass("contentmetadata_editable",
375                 (sakai_global.content_profile.content_data.isManager || sakai_global.content_profile.content_data.isEditor));
376 
377             $contentmetadataShowMore.die("click").live("click", animateData);
378 
379             $(".contentmetadata_editable").die("click").live("click", editData);
380 
381             $(contentmetadataViewRevisions).die("click").live("click", function() {
382                 $(window).trigger("initialize.filerevisions.sakai", sakai_global.content_profile.content_data);
383             });
384 
385             // jeditable bindings
386             $('.contentmetadata_jedit.contentmetadata_editable').click(function(e) {
387                 if (!$('.contentmetadata_edit_input', $(this)).find('textarea').length &&
388                     !$('.contentmetadata_edit_area_select', $(this)).find('select').length) {
389                     if ($(this).attr('data-edit-field') === 'url') {
390                         // if the url hyperlink was clicked, we don't wait to edit the field
391                         if ($(e.target).is('a')) {
392                             return;
393                         }
394                         var $input = $(this).find('.contentmetadata_edit_input');
395                         $input.html($.trim($input.text()));
396                     }
397                     $(this).addClass('contentmetadata_editing');
398                     $('.contentmetadata_edit_input, .contentmetadata_edit_area_select', $(this))
399                     .trigger(contentmetadataJEditTrigger);
400                 }
401             });
402 
403             // setup jeditable for the description textarea
404             var placeholderStart = '<span class="contentmetadata_placeholder">';
405             var placeholderEnd = '</span>';
406             var placeholder = sakai.api.i18n.getValueForKey('CLICK_TO_EDIT_DESCRIPTION', 'contentmetadata');
407             var tooltip = sakai.api.i18n.getValueForKey('CLICK_TO_EDIT', 'contentmetadata');
408             var jeditableUpdate = function(value, settings) {
409                 var $editingContainer = $(this).parents('.contentmetadata_jedit');
410                 if (!$editingContainer.length) {
411                     $editingContainer = $(this);
412                 }
413                 $editingContainer.removeClass('contentmetadata_editing');
414                 var field = $editingContainer.attr('data-edit-field');
415                 updateData(field, $.trim(value));
416                 return value;
417             };
418             $('.contentmetadata_description_textarea').editable(jeditableUpdate, {
419                 type: 'textarea',
420                 onblur: 'submit',
421                 event: contentmetadataJEditTrigger,
422                 tooltip: tooltip,
423                 placeholder: placeholderStart + placeholder + placeholderEnd
424             });
425 
426             // setup jeditable for the copyright selection
427             var copyrightData = {};
428             var selected = false;
429             $.each(sakai.config.Permissions.Copyright.types, function(key, val) {
430                 if (sakai_global.content_profile.content_data.data['sakai:copyright'] === key) {
431                     selected = key;
432                 }
433                 copyrightData[key] = sakai.api.i18n.getValueForKey(val.title, 'contentmetadata');
434             });
435             if (selected) {
436                 copyrightData['selected'] = selected;
437             }
438             var copyrightCallback = function(value, settings) {
439                 $(this).html(settings.data[value]);
440                 copyrightData['selected'] = value;
441             };
442             $('.contentmetadata_edit_area_select').editable(jeditableUpdate, {
443                 data: copyrightData,
444                 type: 'select',
445                 onblur: 'submit',
446                 event: contentmetadataJEditTrigger,
447                 callback: copyrightCallback,
448                 tooltip: tooltip
449             });
450 
451             // setup jeditable for the url textarea
452             var urlPlaceholder = sakai.api.i18n.getValueForKey('CLICK_TO_EDIT_URL', 'contentmetadata');
453             var urlCallback = function(value, settings) {
454                 // add a hyperlink that can be clicked
455                 if ($(this).text()) {
456                     var link = '<a class="s3d-action" target="_blank" href="' +
457                     $(this).text() + '">' + $(this).text() + '</a>';
458                     $(this).html(link);
459                 }
460             };
461             $('.contentmetadata_url_textarea').editable(jeditableUpdate, {
462                 type: 'textarea',
463                 onblur: 'submit',
464                 event: contentmetadataJEditTrigger,
465                 callback: urlCallback,
466                 placeholder: placeholderStart + urlPlaceholder + placeholderEnd
467             });
468         };

500     };
