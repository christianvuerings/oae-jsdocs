 50         /**
 51          * This changes properties file into a json object
 52          */
 53         changeToJSON : function(input){
 54             var json = {};
 55             var inputLine = input.split(/\n/);
 56             var i;
 57             for (i in inputLine) {
 58                 // IE 8 i has indexof as well which breaks the page.
 59                 if (inputLine.hasOwnProperty(i)) {
 60                     var keyValuePair = inputLine[i].split(/\=/);
 61                     var key = $.trim(keyValuePair.shift());
 62                     var value = $.trim(keyValuePair.join("="));
 63                     json[key] = value;
 64                 }
 65             }
 66             return json;
 67         },

 69         /**
 70          * <p>Main i18n process</p>
 71          * <p>This function will be executed once the entire DOM has been loaded. The function will take
 72          * the HTML that's inside the body of the page, and load the default and user specific language
 73          * bundle. We then look for i18n keys inside the HTML, and replace with the values from the
 74          * language bundles. We always first check whether the key is available in the user specific
 75          * bundle and only if that one doesn't exist, we will take the value out of the default bundle.</p>
 76          *
 77          * @param {Object} meData the data from sakai.api.User.data.me
 78          */
 79         init : function(meData){
 80             ////////////////////
 81             // HELP VARIABLES //
 82             ////////////////////
 83 
 84             sakaii18nAPI.data.meData = meData;
 85 
 86             /*
 87              * Cache the jQuery i18nable element. This makes sure that only pages with
 88              * the class i18nable on the body element get i18n translations and won't do
 89              * it on other pages.
 90              * <body class="i18nable">
 91              */
 92             var $i18nable = $("body.i18nable");
 93 
 94             /*
 95              * We take the HTML that is inside the body tag. We will use this to find string we want to
 96              * translate into the language chosen by the user
 97              */
 98             var tostring = $i18nable.html();
 99 
100             ////////////////////
101             // I18N FUNCTIONS //
102             ////////////////////
103 
104             /**
105              * Once all of the i18n strings have been replaced, we will finish the i18n step.
106              * The content of the body tag is hidden by default, in
107              * order not to show the non-translated string before they are translated. When i18n has
108              * finished, we can show the body again.
109              * We then tell the container that pre-loading of the page has finished and that widgets are
110              * now ready to be loaded. This will mostly mean that now the general page/container code
111              * will be executed.
112              * Finally, we will look for the definition of widgets inside the HTML code, which should look
113              * like this:
114              *  - Single instance: <div id="widget_WIDGETNAME" class="widget_inline"></div>
115              *  - Multiple instance support: <div id="widget_WIDGETNAME_UID_PLACEMENT" class="widget_inline"></div>
116              * and load them into the document
117              */
118             var finishI18N = function(){
119                 var currentPage = window.location.pathname;
120                 if (!sakai_config.anonAllowed) {
121                     sakai_config.requireUser = sakai_config.requireUser.concat(sakai_config.requireUserAnonNotAllowed);
122                     sakai_config.requireAnonymous = sakai_config.requireAnonymous.concat(sakai_config.requireAnonymousAnonNotAllowed);
123                 }
124                 if (meData && meData.user && meData.user.anon) {
125                     if ($.inArray(currentPage, sakai_config.requireUser) > -1){
126                         // This is not great, but our util.Templating code needs to call i18n at the moment. TODO
127                         require("sakai/sakai.api.util").Security.sendToLogin();
128                         return false;
129                     }
130                 } else {
131                     if ($.inArray(currentPage, sakai_config.requireAnonymous) > -1){
132                         document.location = sakai_config.URL.MY_DASHBOARD_URL;
133                         return false;
134                     }
135                 }
136 
137                 if ($.inArray(currentPage, sakai_config.requireProcessing) === -1 && window.location.pathname.substring(0, 2) !== "/~"){
138                     // This is not great, but our util.Templating code needs to call i18n at the moment. TODO
139                     require("sakai/sakai.api.util").Security.showPage();
140                 }
141                 translateJqueryPlugins();
142                 translateDirectory(sakai_config.Directory);
143                 require("sakai/sakai.api.widgets").initialLoad();
144                 sakaii18nAPI.done = true;
145                 $(window).trigger("done.i18n.sakai");
146                 return true;
147             };
148 
149             /**
150              * Function that will internationalize the different jquery plugins
151              * we use. For example, we want to make sure that the previous and next
152              * buttons in the pager plugin are properly using the current user's
153              * locale settings
154              */
155             var translateJqueryPlugins = function(){
156                 // Translate the jquery.timeago.js plugin
157                 $.timeago.settings.strings = {
158                     prefixAgo: sakaii18nAPI.getValueForKey("JQUERY_TIMEAGO_PREFIXAGO"),
159                     prefixFromNow: sakaii18nAPI.getValueForKey("JQUERY_TIMEAGO_PREFIXFROMNOW"),
160                     suffixAgo: sakaii18nAPI.getValueForKey("JQUERY_TIMEAGO_SUFFIXAGO"),
161                     suffixFromNow: sakaii18nAPI.getValueForKey("JQUERY_TIMEAGO_SUFFIXFROMNOW"),
162                     seconds: sakaii18nAPI.getValueForKey("JQUERY_TIMEAGO_SECONDS"),
163                     minute: sakaii18nAPI.getValueForKey("JQUERY_TIMEAGO_MINUTE"),
164                     minutes: sakaii18nAPI.getValueForKey("JQUERY_TIMEAGO_MINUTES"),
165                     hour: sakaii18nAPI.getValueForKey("JQUERY_TIMEAGO_HOUR"),
166                     hours: sakaii18nAPI.getValueForKey("JQUERY_TIMEAGO_HOURS"),
167                     day: sakaii18nAPI.getValueForKey("JQUERY_TIMEAGO_DAY"),
168                     days: sakaii18nAPI.getValueForKey("JQUERY_TIMEAGO_DAYS"),
169                     month: sakaii18nAPI.getValueForKey("JQUERY_TIMEAGO_MONTH"),
170                     months: sakaii18nAPI.getValueForKey("JQUERY_TIMEAGO_MONTHS"),
171                     year: sakaii18nAPI.getValueForKey("JQUERY_TIMEAGO_YEAR"),
172                     years: sakaii18nAPI.getValueForKey("JQUERY_TIMEAGO_YEARS")
173                 };
174             };
175 
176             /**
177              * Translate all of the elements inside of the directory into the current's
178              * language
179              */
180             var translateDirectory = function(directory){
181                 for (var dir in directory){
182                     if (directory.hasOwnProperty(dir)){
183                         directory[dir].title = sakaii18nAPI.General.process(directory[dir].title);
184                         if (directory[dir].children){
185                             translateDirectory(directory[dir].children);
186                         }
187                     }
188                 }
189             };
190 
191             /**
192              * This will give the body's HTML string, the local bundle (if present) and the default bundle to the
193              * general i18n function. This will come back with an HTML string where all of the i18n strings will
194              * be replaced. We then change the HTML of the body tag to this new HTML string.
195              * @param {Object} localjson
196              *  JSON object where the keys are the keys we expect in the HTML and the values are the translated strings
197              * @param {Object} defaultjson
198              *  JSON object where the keys are the keys we expect in the HTML and the values are the translated strings
199              *  in the default language
200              */
201             var doI18N = function() {
202                 var newstring = sakaii18nAPI.General.process(tostring);
203                 // We actually use the old innerHTML function here because the $.html() function will
204                 // try to reload all of the JavaScript files declared in the HTML, which we don't want as they
205                 // will already be loaded
206                 if($i18nable.length > 0){
207                     $i18nable[0].innerHTML = newstring;
208                 }
209                 finishI18N();
210             };
211 
212             /**
213              * This will load the default language bundle and will store it in a global variable. This default bundle
214              * will be saved in a file called bundle/default.properties.
215              * This function will load the general language bundle specific to the language chosen by
216              * the user and will store it in a global variable. This language will either be the prefered
217              * user language or the prefered server language. The language will be available in the me feed
218              * and we'll use the global sakai.api.User.data.me object to extract it from. If there is no prefered langauge,
219              * we'll use the default bundle to translate everything.
220              */
221             var loadLanguageBundles = function() {
222                 var langCode = '';
223                 var langBundle = '';
224                 var i10nCode = '';
225                 var loadDefaultBundleRequest = {};
226                 var loadCustomBundleRequest = {};
227                 var loadLocalBundleRequest = {};
228 
229                 if (meData && meData.user && meData.user.locale && meData.user.locale.country) {
230                     langCode = meData.user.locale.language + "_" + meData.user.locale.country.replace("_", "-");
231                     // Set the path for the language file
232                     // SAKIII-5891 Hashed default bundles not loaded
233                     $.each(sakai_config.Languages, function(index, lang) {
234                         if (lang.country === meData.user.locale.country) {
235                             langBundle = lang.bundle;
236                             return false;
237                         }
238                     });
239                 } else {
240                     langCode = sakai_config.defaultLanguage;
241                     langBundle = sakai_config.defaultLanguageBundle;
242                 }
243                 i10nCode = langCode.replace("_", "-");
244 
245                 // set the language attribute for the html tag
246                 $("html").attr("lang", langCode.substr(0, 2));
247 
248                 if (Globalize.cultures && Globalize.cultures[i10nCode]) {
249                     Globalize.culture(i10nCode);
250                 } else {
251                     $.getScript(sakai_config.URL.I10N_BUNDLE_URL.replace("__CODE__", i10nCode), function(success, textStatus) {
252                         Globalize.culture(i10nCode);
253                     });
254                 }
255 
256                 loadDefaultBundleRequest = sakai_config.URL.I18N_DEFAULT_BUNDLE;
257                 loadCustomBundleRequest = sakai_config.URL.I18N_CUSTOM_BUNDLE;
258                 loadLocalBundleRequest = langBundle;
259 
260                 // callback function for response from batch request
261                 var bundleReqFunction = function(success, reqData){
262                     if (success){
263                         var loadDefaultBundleSuccess = reqData.results[0].success;
264                         var loadDefaultBundleData = reqData.results[0].body;
265                         var loadLocalBundleSuccess;
266                         var loadLocalBundleData;
267                         var loadCustomBundleSuccess;
268                         var loadCustomBundleData;
269 
270                         // Custom bundle
271                         if (reqData.results[1]) {
272                             loadCustomBundleSuccess = reqData.results[1].success;
273                             loadCustomBundleData = reqData.results[1].body;
274                         }
275 
276                         // Local bundle
277                         if (reqData.results[2]) {
278                             loadLocalBundleSuccess = reqData.results[2].success;
279                             loadLocalBundleData = reqData.results[2].body;
280                         }
281 
282                         // process the responses
283                         if (loadCustomBundleSuccess) {
284                             loadCustomBundleData = sakaii18nAPI.changeToJSON(loadCustomBundleData);
285                             sakaii18nAPI.data.customBundle = loadCustomBundleData;
286                         }
287 
288                         if (loadLocalBundleSuccess) {
289                             loadLocalBundleData = sakaii18nAPI.changeToJSON(loadLocalBundleData);
290                             sakaii18nAPI.data.localBundle = loadLocalBundleData;
291                         }
292 
293                         if (loadDefaultBundleSuccess) {
294                             loadDefaultBundleData = sakaii18nAPI.changeToJSON(loadDefaultBundleData);
295                             sakaii18nAPI.data.defaultBundle = loadDefaultBundleData;
296                         }
297 
298                         doI18N();
299                     }
300                 };
301 
302                 var batchRequest = [loadDefaultBundleRequest, loadCustomBundleRequest, loadLocalBundleRequest];
303                 sakai_serv.staticBatch(batchRequest, bundleReqFunction);
304             };
305

308             // INITIALIZATION FUNCTION //
309             /////////////////////////////
310

325             /**
326              * General process functions that will replace all the messages in a string with their corresponding translation.
327              * @example sakai.api.i18n.General.process(
328              *     "<h1>__MSG__CHANGE_LAYOUT__</h1>",
329              *     {"__MSG__CHANGE_LAYOUT__" : "verander layout"},
330              *     {"__MSG__CHANGE_LAYOUT__" : "change layout"}
331              * );
332              * @param {String} toprocess
333              *  HTML string in which we want to replace messages. Messages have the following
334              *  format: __MSG__KEY__
335              *  in the default language
336              * @param {String} widget optional widget name. This will cause the widget language
337              *                 bundles to be checked for a translation first
338              * @return {String} A processed string where all the messages are replaced with values from the language bundles
339              */
340 
341             process : function(toprocess, widget) {
342 
343                 if(!toprocess){
344                     return "";
345                 }
346 
347                 var expression = new RegExp("__MSG__(.*?)__", "gm"), processed = "", lastend = 0;
348 
349                 while(expression.test(toprocess)) {
350                     var replace = RegExp.lastMatch;
351                     var lastParen = RegExp.lastParen;
352                     var quotes = "";
353 
354                     // need to add quotations marks if key is adjacent to an equals sign which means its probably missing quotes - IE
355                     if (replace.substr(0,2) !== "__"){
356                         if (replace.substr(0,1) === "="){
357                             quotes = '"';
358                         }
359                         replace = replace.substr(1, replace.length);
360                     }
361                     var toreplace;
362                     // check for i18n debug
363                     if (sakai_config.displayDebugInfo === true && sakaii18nAPI.data.meData.user && sakaii18nAPI.data.meData.user.locale && sakaii18nAPI.data.meData.user.locale.language === "lu" && sakaii18nAPI.data.meData.user.locale.country === "GB"){
364                         toreplace = quotes + replace.substr(7, replace.length - 9) + quotes;
365                         processed += toprocess.substring(lastend, expression.lastIndex - replace.length) + toreplace;
366                         lastend = expression.lastIndex;
367                     } else {
368                         toreplace = quotes + sakaii18nAPI.getValueForKey(lastParen, widget) + quotes;
369                         processed += toprocess.substring(lastend, expression.lastIndex - replace.length) + toreplace;
370                         lastend = expression.lastIndex;
371                     }
372                 }
373                 processed += toprocess.substring(lastend);
374                 return processed;
375 
376             }
377         },

379         /**
380          * Get the internationalised value for a specific key.
381          * We expose this function so you can do internationalisation within JavaScript.
382          * If the key isn't found in a translation bundle, the key will be returned unmodified
383          *
384          * @example sakai.api.i18n.getValueForKey("CHANGE_LAYOUT", ["widgetid"]);
385          * @param {String} key The key that will be used to get the translation
386          * @param {String} optional widget name. This will cause the widget language
387          *                 bundles to be checked for a translation first
388          * @return {String} The translated value for the provided key
389          */
390         getValueForKey: function(key, widgetname) {
391             // Get the user's current locale from the me object
392             var locale = sakaii18nAPI.getUserLocale();
393             // Check for i18n debug language
394             //   Because the debug language has to be a valid Java locale, 
395             //   we are currently using lu_GB to identify the debug language
396             if (locale === "lu_GB") {
397                 return key;
398             } else {
399                 // First check if the key can be found in the custom bundle,
400                 // so that those values override everything
401                 if (sakaii18nAPI.data.customBundle && _.isString(sakaii18nAPI.data.customBundle[key])) {
402                     return sakaii18nAPI.processUTF16ToText(sakaii18nAPI.data.customBundle[key]);
403                 }
404                 // Check the bundle for the widget, if provided
405                 if (widgetname) {
406                     if (typeof sakaii18nAPI.data.widgets[widgetname]) {
407                         // First check if the key can be found in the widget's locale bundle
408                         if ($.isPlainObject(sakaii18nAPI.data.widgets[widgetname][locale]) && _.isString(sakaii18nAPI.data.widgets[widgetname][locale][key])) {
409                             return sakaii18nAPI.processUTF16ToText(sakaii18nAPI.data.widgets[widgetname][locale][key]);
410                         }
411                         // If the key wasn't found in the widget's locale bundle, search in the widget's default bundle
412                         else if ($.isPlainObject(sakaii18nAPI.data.widgets[widgetname]["default"]) && _.isString(sakaii18nAPI.data.widgets[widgetname]["default"][key])) {
413                             return sakaii18nAPI.processUTF16ToText(sakaii18nAPI.data.widgets[widgetname]["default"][key]);
414                         }
415                     }
416                 }
417 
418                 // Check if the key can be found in the general locale bundle
419                 if (sakaii18nAPI.data.localBundle && _.isString(sakaii18nAPI.data.localBundle[key])) {
420                     return sakaii18nAPI.processUTF16ToText(sakaii18nAPI.data.localBundle[key]);
421                 }
422                 // If the key wasn't found in the general locale bundle, search in the general default bundle
423                 else if (sakaii18nAPI.data.defaultBundle && _.isString(sakaii18nAPI.data.defaultBundle[key])) {
424                     return sakaii18nAPI.processUTF16ToText(sakaii18nAPI.data.defaultBundle[key]);
425                 } 
426                 // If none of the about found something, log an error message
427                 else {
428                     debug.warn("sakai.api.i18n.getValueForKey: Not found in any bundles. Key: " + key);
429                     return key;
430                 }
431             }
432         },

434         /**
435          * Utility regular expression that is used to find
436          * escaped unicode characters in translation string
437          */
438         UnicodeExpression: new RegExp("[\\][u][A-F0-9]{4}", "g"),
439 
440         /**
441          * Utility function that will take a translation string
442          * and replace escaped unicode characters with the actual unicode
443          * character
444          * @param {String} translation   Translation key that we want to scan for escaped unicode
445          * @return {String} Translation key where all escaped unicode characters
446          *                  have been replaced by the actual unicode character
447          */
448         processUTF16ToText: function(translation){
449             var matches =  translation.match(sakaii18nAPI.UnicodeExpression);
450             if (matches) {
451                 for (var r = 0; r < matches.length; r++) {
452                     var replace = matches[r];
453                     translation = translation.replace("\\" + replace, String.fromCharCode(parseInt(replace.substring(1), 16)));
454                 }
455             }
456             return $.trim(translation);
457         },

459         /**
460          * Get the language for the editor for the current user
461          * If the editor language doesn't exist, we default to English
462          * @return {String} The language for the current user (e.g. 'nl')
463          */
464         getEditorLanguage: function() {
465             var language = sakaii18nAPI.getUserLocale().split('_')[0];
466 
467             if ($.inArray(language, sakai_config.Editor.languagePacks) === -1) {
468                 language = 'en';
469             }
470 
471             return language;
472         },

474         /**
475          * Function that will return the current user's locale
476          *     Example: en_GB
477          */
478         getUserLocale: function(){
479             var locale = false;
480             if (sakaii18nAPI.data.meData.user && sakaii18nAPI.data.meData.user.locale) {
481                 locale = sakaii18nAPI.data.meData.user.locale.language + "_" + sakaii18nAPI.data.meData.user.locale.country;
482             } else {
483                 locale = sakai_config.defaultLanguage;
484             }
485             return locale;
486         }
