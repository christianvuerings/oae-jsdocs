 18 // load the master sakai object to access all Sakai OAE API methods
 19 require(["jquery", "underscore", "sakai/sakai.api.core"], function($, _, sakai){
 20 
 21     /**
 22      * @name sakai_global.versions
 23      *
 24      * @class versions
 25      *
 26      * @version 0.0.1
 27      * @param {String} tuid Unique id of the widget
 28      * @param {Boolean} showSettings Show the settings of the widget or not
 29      */
 30     sakai_global.versions = function(tuid, showSettings){
 31 
 32 
 33         ///////////////
 34         // VARIABLES //
 35         ///////////////
 36 
 37         var $rootel = $("#" + tuid);
 38 
 39         // Vars
 40         var contentPath = "";
 41         var currentPageShown = "";
 42         var versions = [];
 43         var itemsBeforeScroll = 0;
 44 
 45         // Containers
 46         var versionsContainer = "#versions_container";
 47 
 48         // Templates
 49         var versionsTemplate = "versions_template";
 50 
 51         // Elements
 52         var versionsVersionItem = ".versions_version_item";
 53         var versionsRestoreVersion = ".versions_restore_version";
 54 
 55 
 56         var carouselBinding = function(carousel){
 57             $("#versions_newer", $rootel).live("click",function(){
 58                 carousel.prev();
 59             });
 60 
 61             $("#versions_older", $rootel).live("click",function(){
 62                 if (carousel.last !== carousel.size()){
 63                     carousel.next();
 64                 }
 65             });
 66 
 67             $("#versions_oldest", $rootel).live("click",function(){
 68                 carousel.scroll(carousel.size() || 0);
 69             });
 70 
 71             $("#versions_newest", $rootel).live("click",function(){
 72                 carousel.scroll(0);
 73             });
 74         };
 75 
 76 
 77         ///////////////
 78         // RENDERING //
 79         ///////////////
 80 
 81         var renderVersions = function(users) {
 82             $(versionsContainer, $rootel).html(sakai.api.Util.TemplateRenderer(versionsTemplate, {
 83                 'itemsBeforeScroll': itemsBeforeScroll,
 84                 'users': users,
 85                 "data": versions,
 86                 "sakai": sakai,
 87                 "currentPage": currentPageShown
 88             }));
 89             if (versions.length) {
 90                 $('#versions_carousel_container', $rootel).jcarousel({
 91                     animation: 'slow',
 92                     easing: 'swing',
 93                     scroll: 4,
 94                     start: 0,
 95                     initCallback: carouselBinding,
 96                     itemFallbackDimension: 123
 97                 });
 98             }
 99         };
100 
101         var parseVersions = function(success, data){
102             $.each(data.versions, function(index, version){
103                 version.versionId = index;
104                 versions.push(version);
105             });
106             renderVersions(data.users);
107         };
108 
109 
110         //////////
111         // UTIL //
112         //////////
113 
114         var getVersions = function(){
115             sakai.api.Server.loadJSON(currentPageShown.pageSavePath + "/" + currentPageShown.saveRef + ".versions.json", parseVersions);
116         };
117 
118         var getContext = function(){
119             contentPath = $.bbq.getState("content_path");
120         };
121 
122         var getVersionContent = function(versionId, callback) {
123             var version = versions[versionId];
124             if (!version.version) {
125                 var vurl = currentPageShown.pageSavePath + '/' + currentPageShown.saveRef + '.version.,' + version.versionId + ',.json';
126                 sakai.api.Server.loadJSON(vurl, function(success, data) {
127                     if (success) {
128                         if (!data.version) {
129                             version.version = {
130                                 'rows': [{
131                                     'id': sakai.api.Util.generateWidgetId(),
132                                     'columns': [{
133                                         'width': 1,
134                                         'elements': []
135                                     }]
136                                 }]
137                             };
138                         } else if (_.isString(data.version)) {
139                             version.version = $.parseJSON(data.version);
140                         }
141                         callback(version);
142                     }
143                 });
144             } else {
145                 callback(version);
146             }
147         };
148 
149         var previewVersion = function(event){
150             event.stopPropagation();
151             if (!sakai_global.content_profile || sakai_global.content_profile.content_data.data.mimeType == "x-sakai/document") {
152                 $(".versions_selected", $rootel).removeClass("versions_selected");
153                 $("#" + currentPageShown.ref).remove();
154                 $(this).addClass("versions_selected");
155                 getVersionContent($(this).attr('data-versionId'), showPageVersion);
156             } else{
157                 window.open(currentPageShown.pageSavePath + ".version.," + $(this).attr("data-version") + ",/" + $(this).attr("data-pooleditemname"), "_blank");
158             }
159         };
160 
161         var showPageVersion = function(version) {
162             var newPageShown = $.extend(true, {}, currentPageShown);
163             newPageShown.content = version.version;
164             newPageShown.isVersionHistory = true;
165             newPageShown.ref = currentPageShown.ref + '_previewversion';
166             $(window).trigger('showpage.contentauthoring.sakai', newPageShown);
167         };
168 
169         var restoreVersion = function(e) {
170             getVersionContent($(this).parent().attr('data-versionId'), saveRestoredVersion);
171         };
172 
173         var addIgnores = function(version) {
174             for (var i in version) {
175                 if (version.hasOwnProperty(i)) {
176                     if (version[i].comments) {
177                         delete version[i].comments.message.inbox;
178                         version[i].comments.message['inbox@Ignore'] = true;
179                     } else if (version[i].discussion) {
180                         delete version[i].discussion.message.inbox;
181                         version[i].discussion.message['inbox@Ignore'] = true;
182                     }
183                 }
184             }
185             return version;
186         };
187 
188         var saveRestoredVersion = function(version) {
189             var toStore = $.extend({}, version.version);
190             currentPageShown.content = toStore;
191             toStore.version = addIgnores(version.version);
192             toStore.version = JSON.stringify(toStore.version);
193             sakai.api.Server.saveJSON(currentPageShown.pageSavePath + "/" + currentPageShown.saveRef, toStore, function(success) {
194                 $.ajax({
195                     url: currentPageShown.pageSavePath + "/" + currentPageShown.saveRef + ".save.json",
196                     type: "POST",
197                     success: function(){
198                         $(window).trigger("update.versions.sakai", currentPageShown);
199                     }
200                 });
201             }, false, false, false);
202         };
203 
204 
205         /////////////
206         // BINDING //
207         /////////////
208 
209         var addBinding = function(){
210             if (!sakai_global.content_profile || sakai_global.content_profile.content_data.data.mimeType == "x-sakai/document") {
211                 $(versionsVersionItem, $rootel).die("click", previewVersion);
212                 $(versionsVersionItem, $rootel).live("click", previewVersion);
213             }
214 
215             $(versionsRestoreVersion, $rootel).die("click", restoreVersion);
216             $(versionsRestoreVersion, $rootel).live("click", restoreVersion);
217         };
218 
219 
220         ////////////////////
221         // INITIALIZATION //
222         ////////////////////
223 
224         var doInit = function(){
225             versions = [];
226             addBinding();
227             getContext();
228             getVersions();
229         };
230 
231         $(window).bind("init.versions.sakai", function(ev, cps){
232             if ($('.s3d-page-column-left').is(':visible')) {
233                 // There is a left hand navigation visible, versions widget will be smaller
234                 $(versionsContainer, $rootel).removeClass("versions_without_left_hand_nav");
235                 itemsBeforeScroll = 6;
236             } else {
237                 // No left hand navigation visible, versions widget will be wider
238                 $(versionsContainer, $rootel).addClass("versions_without_left_hand_nav");
239                 itemsBeforeScroll = 7;
240             }
241             currentPageShown = cps;
242             $('.versions_widget', $rootel).show();
243             doInit();
244         });
245         $(window).bind("close.versions.sakai", function(ev, cps) {
246             $('.versions_widget', $rootel).hide();
247             $(window).trigger('showpage.contentauthoring.sakai', currentPageShown);
248         });
249 
250         $(window).bind("update.versions.sakai", function(ev, cps) {
251             if ($('.versions_widget', $rootel).is(":visible")) {
252                 currentPageShown = cps;
253                 doInit();
254             }
255         });
256 
257 
258     };
259 
260     sakai.api.Widgets.widgetLoader.informOnLoad("versions");
261 });
