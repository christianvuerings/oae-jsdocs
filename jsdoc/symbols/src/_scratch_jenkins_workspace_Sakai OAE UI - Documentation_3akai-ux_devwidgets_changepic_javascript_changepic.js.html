101          /**
102          * Hides and reset image select area
103          */
104         var hideSelectArea = function(){
105             if (imageareaobject) {
106                 imageareaobject.setOptions({
107                     hide: true,
108                     disable: true
109                 });
110                 imageareaobject.update();
111             }
112 
113             $(selectContentArea).hide();
114             $(uploadNewCancel).show();
115         };

117          /**
118          * Shows image select area
119          */
120         var showSelectArea = function(){
121             $(uploadNewCancel).hide();
122             $(selectContentArea).show();
123         };

125         /**
126          * When the user has drawn a square this function will be called by imgAreaSelect.
127          * This will draw the thumbnail by modifying it's css values.
128          * @param {Object} img    The thumbnail
129          * @param {Object} selection The selection object from imgAreaSelect
130          */
131         var preview = function(img, selection){
132             // Save the user his selection in a global variable.
133             userSelection = selection;
134 
135             // How much has the user scaled down the image?
136             var scaleX = thumbnailWidth / (selection.width || 1);
137             var scaleY = thumbnailHeight / (selection.height || 1);
138 
139             // Change the thumbnail according to the user his selection via CSS.
140             $(thumbnail).css({
141                 width: Math.round(scaleX * img.width) + 'px',
142                 height: Math.round(scaleY * img.height) + 'px',
143                 marginLeft: '-' + Math.round(scaleX * selection.x1) + 'px',
144                 marginTop: '-' + Math.round(scaleY * selection.y1) + 'px'
145             });
146         };

149          * Shows file input error

151         var showInputError = function(){
152             $(picInputError).show();
153             $(picInput).addClass(picInputErrorClass);
154             if ($(selectContentArea + ":visible") && imageareaobject){
155                 imageareaobject.update();
156             }
157         };

159         /**
160          * Hides file input error
161          */
162         var hideInputError = function(){
163             $(picInputError).hide();
164             $(picInput).removeClass(picInputErrorClass);
165             if ($(selectContentArea + ":visible") && imageareaobject){
166                 imageareaobject.update();
167             }

171          * Empty upload field by resetting the form

187         /**
188          * On changepic form submit, check that a file has been selected
189          * and submit the form.
190          */
191         $("#profile_upload").unbind("click").bind("click", function(){
192             // validate args
193             // file extension allow for image
194             var extensionArray = [".png", ".jpg", ".jpeg",".gif"];
195             // get file name
196             fileName = $(picInput).val();
197             // get extension from the file name.
198             var extension = fileName.slice(fileName.lastIndexOf(".")).toLowerCase();
199             var allowSubmit = false;
200 
201             for (var i = 0; i < extensionArray.length; i++) {
202                  // extension is acceptable image format
203                  if (extensionArray[i] === extension) {
204                     allowSubmit = true;
205                     break;
206                  }
207             }
208             // if image format is acceptable
209             if(allowSubmit) {
210                 hideInputError();
211                 $(uploadNewButtons).hide();
212                 $(uploadProcessing).show();
213                 fileName = "tmp" + new Date().getTime() + ".jpg";
214                 $(picInput).attr("name",fileName);
215                 hideSelectArea();
216                 $(picForm).ajaxForm({
217                     success: function(data){
218                         doInit(true);
219                     },
220                     error: function(){
221                         showInputError();
222                         return false;
223                     }
224                 });
225                 $(picForm).submit();
226             } else {
227                 // no input, show error
228                 showInputError();
229                 return false;
230             }
231         });

233         /**
234          * Initilise function
235          * @param {boolean} newpic True if a new picture has just been uploaded
236          */
237         var doInit = function(newpic){
238             hideSelectArea();
239 
240             if (!id) {
241                 id = sakai.data.me.user.userid;
242                 mode = "user";
243             }
244 
245             var showPicture = true;
246             var json;
247 
248             if (mode === "group") {
249                 // fetch group data to check if it has a picture
250                 $.ajax({
251                     url: "/~" + id + "/public.infinity.json",
252                     async: false,
253                     success: function(data){
254                         json = data.authprofile;
255                     }
256                 });
257             } else {
258                 // Check whether there is a base picture at all
259                 me = sakai.data.me;
260                 json = me.profile;
261             }
262             // If the image is freshly uploaded then reset the imageareaobject to reset all values on init
263             if (newpic) {
264                 imageareaobject = null;
265                 picture = {
266                     "_name": fileName,
267                     "selectedx1":0,
268                     "selectedy1":0,
269                     "selectedx2":64,
270                     "selectedy2":64
271                 };
272             }
273             else if (json.picture) {
274                 picture = $.parseJSON(json.picture);
275             } else {
276                 showPicture = false;
277             }
278 
279             $(picForm).attr("action", "/~" + sakai.api.Util.safeURL(id) + "/public/profile");
280 
281             // Get the preferred size for the thumbnail.
282             var prefThumbWidth = parseInt($(thumbnailContainer).css("width").replace(/px/gi,""), 10);
283             var prefThumbHeight = parseInt($(thumbnailContainer).css("height").replace(/px/gi,""), 10);
284 
285             // Make sure we don't have 0
286             thumbnailWidth  = (prefThumbWidth > 0) ? prefThumbWidth : thumbnailWidth;
287             thumbnailHeight  = (prefThumbHeight > 0) ? prefThumbHeight : thumbnailHeight;
288 
289             if (showPicture && picture && picture._name) {
290                 // The user has already uploaded a picture.
291                 // Show the image select area
292                 existingPicture = true;
293 
294                 // Set the unvisible image to the full blown image. (make sure to filter the # out)
295                 $(pictureMeasurer).html(sakai.api.Security.saneHTML("<img src='" + "/~" + sakai.api.Util.safeURL(id) + "/public/profile/" + picture._name + "?sid=" + Math.random() + "' id='" + pictureMeasurerImage.replace(/#/gi, '') + "' />"));
296 
297                 // Check the current picture's size
298                 $(pictureMeasurerImage).bind("load", function(ev){
299                     resetUploadField();
300 
301                     // save the image size in global var.
302                     realw = $(pictureMeasurerImage).width();
303                     realh = $(pictureMeasurerImage).height();
304 
305                     // Set the images
306                     $(fullPictureSpan).html('<img alt="' + $("#changepic_fullpicture_alt").html() + '" id="changepic_fullpicture_img" src="/~' + id + "/public/profile/" + picture._name + "?sid=" + Math.random() + '" />');
307                     $(thumbnailSpan).html('<img alt="' + $("#thumbnail_alt").html() + '" id="thumbnail_img" src="/~' + id + "/public/profile/" + picture._name + "?sid=" + Math.random() + '" />');
308 
309                     // Reset ratio
310                     ratio = 1;
311 
312                     // fullPicWidth (500) and fullPicHeight (300) set in config variables
313                     // Width < 500 ; Height < 300 => set the original height and width
314                     if (realw < fullPicWidth && realh < fullPicHeight){
315                         $(fullPicture).width(realw);
316                         $(fullPicture).height(realh);
317 
318                     // Width > 500 ; Height < 300 => Width = 500
319                     } else if (realw > fullPicWidth && (realh / (realw / fullPicWidth) < fullPicHeight)){
320                         ratio = realw / fullPicWidth;
321                         $(fullPicture).width(fullPicWidth);
322                         $(fullPicture).height(Math.floor(realh / ratio));
323 
324                     // Width < 500 ; Height > 300 => Height = 300
325                     } else if (realh > fullPicHeight && (realw / (realh / fullPicHeight) < fullPicWidth)) {
326                         ratio = realh / fullPicHeight;
327                         $(fullPicture).height(fullPicHeight);
328                         $(fullPicture).width(Math.floor(realw / ratio));
329 
330                     // Width > 500 ; Height > 300
331                     } else if (realh > fullPicHeight && (realw / (realh / fullPicHeight) > fullPicWidth)) {
332 
333                         var heightonchangedwidth = realh / (realw / fullPicWidth);
334                         if (heightonchangedwidth > fullPicHeight){
335                             ratio = realh / fullPicHeight;
336                             $(fullPicture).height(fullPicHeight);
337                         } else {
338                             ratio = realw / fullPicWidth;
339                             $(fullPicture).width(fullPicWidth);
340                         }
341                     }
342 
343                     var selectionObj = {
344                         width : picture.selectedx2 - picture.selectedx1,
345                         height :picture.selectedy2 - picture.selectedy1,
346                         x1 : picture.selectedx1,
347                         y1 : picture.selectedy1,
348                         x2 : picture.selectedx2,
349                         y2 : picture.selectedy2,
350                         picture : picture._name
351                     };
352                     if (!newpic){
353                         originalPic = selectionObj;
354                     }
355 
356                     // Set the imgAreaSelect to a function so we can access it later on
357                     imageareaobject = $(fullPicture).imgAreaSelect({
358                         aspectRatio: "1:1",
359                         enable: true,
360                         show: true,
361                         instance: true,
362                         onInit: function(){
363                             // If the image gets loaded, make a first selection
364                             imageareaobject.setSelection(picture.selectedx1, picture.selectedy1, picture.selectedx2, picture.selectedy2);
365                             imageareaobject.setOptions({show: true, enable: true});
366                             imageareaobject.update();
367                             preview($("img" + fullPicture)[0], selectionObj);
368                         },
369                         onSelectChange: preview
370                     });
371                     showSelectArea();
372                 });
373 
374                 // if there is upload error show the error message
375                 $(pictureMeasurerImage).bind("error", function(){
376                     showInputError();
377                 });
378             }
379         };
380 
381         // Remove error notification when a new file is chosen
382         $(picInput).bind("change", function(){
383             hideInputError();
384             $('#profile_upload').removeAttr('disabled');
385         });
386 
387         // This is the function that will be called when a user has cut out a selection
388         // and saves it.
389         $(saveNewSelection).click(function(ev){
390             if (!userSelection) {
391                 userSelection = imageareaobject.getSelection();
392                 savePicture();
393             } else if (originalPic &&
394                 (userSelection.x1 === originalPic.x1 &&
395                 userSelection.x2 === originalPic.x2 &&
396                 userSelection.y1 === originalPic.y1 &&
397                 userSelection.y2 === originalPic.y2 &&
398                 userSelection.picture === originalPic.picture)){
399                 // no need to save if picture hasn't changed, so just close the dialog
400                 // Hide the layover.
401                 sakai.api.Util.Modal.close(container);
402             } else {
403                 savePicture();
404             }
405         });

407         /**
408          * savePicture
409          */
410         var savePicture = function(){
411             // The parameters for the cropit service.
412             var data = {
413                 img: "/~" + id + "/public/profile/" + picture._name,
414                 save: "/~" + id + "/public/profile",
415                 x: Math.floor(userSelection.x1 * ratio),
416                 y: Math.floor(userSelection.y1 * ratio),
417                 width: Math.floor(userSelection.width * ratio),
418                 height:Math.floor(userSelection.height * ratio),
419                 dimensions: "256x256",
420                 "_charset_":"utf-8"
421             };
422 
423             if(data.width === 0 || data.height === 0){
424                 data.width = $(fullPicture).width();
425                 data.height = $(fullPicture).height();
426                 data.x = 0;
427                 data.y = 0;
428             }
429 
430             // Post all of this to the server
431             $.ajax({
432                 url: sakai.config.URL.IMAGE_SERVICE,
433                 type: "POST",
434                 data: data,
435                 success: function(data){
436 
437                     var tosave = {
438                         "name": "256x256_" + picture._name,
439                         "_name": picture._name,
440                         "_charset_":"utf-8",
441                         "selectedx1" : userSelection.x1,
442                         "selectedy1" : userSelection.y1,
443                         "selectedx2" : userSelection.width + userSelection.x1,
444                         "selectedy2" : userSelection.height + userSelection.y1
445                     };
446 
447                     var stringtosave = JSON.stringify(tosave);
448 
449                     sakai.data.me.profile.picture = stringtosave;
450 
451                     // Do a patch request to the profile info so that it gets updated with the new information.
452                     $.ajax({
453                         url: "/~" + sakai.api.Util.safeURL(id) + "/public/authprofile.profile.json",
454                         type : "POST",
455                         data : {
456                             "picture" : JSON.stringify(tosave),
457                             "_charset_":"utf-8"
458                         },
459                         success : function(data) {
460                             // Change the picture in the page. (This is for my_sakai.html)
461                             // Math.random is for cache issues.
462                             for (var i = 0; i < imagesToChange.length;i++) {
463                                 $(imagesToChange[i]).attr("src", "/~" + id + "/public/profile/" + tosave.name + "?sid=" + Math.random());
464                             }
465 
466                             // Hide the layover.
467                             sakai.api.Util.Modal.close(container);
468 
469                             if (sakai.currentgroup && sakai.currentgroup.data && sakai.currentgroup.data.authprofile) {
470                                 sakai.currentgroup.data.authprofile.picture = JSON.stringify(tosave);
471                             }
472                         },
473                         error: function(xhr, textStatus, thrownError) {
474                             sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("AN_ERROR_HAS_OCCURRED"),"",sakai.api.Util.notification.type.ERROR);
475                         }
476                     });
477 
478                 },
479                 error: function(xhr, textStatus, thrownError) {
480                     sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("AN_ERROR_HAS_OCCURRED"),"",sakai.api.Util.notification.type.ERROR);
481                 }
482             });
483 
484         };

491         /**
492          * Hide the layover
493          * @param {Object} hash the object that represents the layover
494          */
495         var hideArea = function(hash){
496             // Remove the selecting of an area on an image.
497             if (imageareaobject) {
498                 imageareaobject.setOptions({
499                     hide: true,
500                     disable: true
501                 });
502                 imageareaobject.update();
503             }
504 
505             hash.w.hide();
506             hash.o.remove();
507         };
508 
509         /**
510          * Show the layover.
511          * @param {Object} hash
512          */
513         var showArea = function(hash){
514             doInit();
515             hash.w.show();
516         };
517

519             modal: true,
520             overlay: 20,
521             toTop: true,
522             onHide: hideArea,
523             onShow: showArea
524         });

528             id = _id;
