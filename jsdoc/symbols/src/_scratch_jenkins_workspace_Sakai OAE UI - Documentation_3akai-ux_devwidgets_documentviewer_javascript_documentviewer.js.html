 27 require(['jquery', 'sakai/sakai.api.core', '/devwidgets/documentviewer/lib/document-viewer/assets/viewer.js', '/dev/lib/misc/jwplayer/swfobject.js'], function($, sakai) {
 28 
 29     /**
 30      * @name sakai.documentviewer
 31      *
 32      * @class documentviewer
 33      *
 34      * @description
 35      * Initialize the documentviewer widget
 36      *
 37      * @version 0.0.1
 38      * @param {String} tuid Unique id of the widget
 39      * @param {Boolean} showSettings Show the settings of the widget or not
 40      */
 41     sakai_global.documentviewer = function(tuid,showSettings,widgetData){
 42 
 43         var $rootel = $("#" + tuid);
 44         var $documentviewerPreview = $("#documentviewer_preview", $rootel);
 45         var documentviewerPreviewSelector = "#" + tuid + " .documentviewer_preview";
 46         var templateObject = {};
 47         var docType = false;
 48         var data = false;
 49 
 50         var getPath = function(data) {
 51             return "/p/" + data["_path"];
 52         };
 53 
 54         var renderDocumentPreview = function(data){
 55             var url = window.location.protocol + '//' + window.location.host + getPath(data);
 56             var pdfDoc = {
 57                 id: data['_path'],
 58                 title: data['sakai:pooled-content-file-name'],
 59                 pages: data['sakai:pagecount'],
 60                 resources: {
 61                     pdf: url,
 62                     page: {
 63                         image: url + "/page{page}.{size}.jpg"
 64                     }
 65                 }
 66             };
 67             var container = documentviewerPreviewSelector;
 68             DV.load(pdfDoc, {
 69                 container: container,
 70                 width: "100%",
 71                 height: 500,
 72                 sidebar: false,
 73                 text: false
 74             });
 75         };
 76 
 77         /**
 78          * Get the extension for the image preview according to the mimeType
 79          * @param {String} mimeType The mimeType of the image preview
 80          */
 81         var getImagePreviewExtension = function(mimeType) {
 82             if (mimeType === 'image/png' || mimeType === 'image/gif') {
 83                 return '.' + mimeType.split('/')[1];
 84             } else {
 85                 return '.jpg';
 86             }
 87         };
 88 
 89         /**
 90          * Render the preview of an image
 91          * @param {String} url The URL for the image
 92          * @param {Object} data The data for the image (will be undefined for flickr / 3th party images)
 93          */
 94         var renderImagePreview = function(url, data){
 95             $documentviewerPreview.html("");
 96 
 97             if (data && data['sakai:hasPreview'] === 'true' && data['sakai:needsprocessing'] === 'false' && data.page1) {
 98                 templateObject.contentURL = getPath(data.page1) + '.normal' + getImagePreviewExtension(data.page1._mimeType);
 99             } else {
100                 templateObject.contentURL = url;
101             }
102             var date = new Date();
103             if (date){
104                 templateObject.contentURL += "?_=" + date.getTime();
105             }
106 
107             $documentviewerPreview.html(
108                 sakai.api.Util.TemplateRenderer('documentviewer_image_template', templateObject)
109             );
110         };
111 
112         var renderEmbedPreview = function(data){
113             $documentviewerPreview.html(data);
114         };
115 
116         var renderHTMLPreview = function(data){
117             sakai.api.Util.TemplateRenderer("documentviewer_html_template", templateObject, $documentviewerPreview);
118             $("#documentviewer_html_iframe", $rootel).attr("src", getPath(data));
119             $("#documentviewer_html_iframe", $rootel).attr("frameborder", "0");
120         };
121 
122         var renderExternalHTMLPreview = function(url){
123             sakai.api.Util.TemplateRenderer("documentviewer_externalhtml_template", templateObject, $documentviewerPreview);
124             $("#documentviewer_externalhtml_iframe", $rootel).attr("src", url);
125             $("#documentviewer_externalhtml_iframe", $rootel).attr("frameborder", "0");
126         };
127 
128          var renderKalturaPlayer = function(data){
129             var html5FlashCompatibilityURL = sakai.config.kaltura.serverURL +"/p/"+sakai.config.kaltura.partnerId+"/sp/"+sakai.config.kaltura.partnerId+"00/embedIframeJs/uiconf_id/"+sakai.config.kaltura.playerId+"/partner_id/"+sakai.config.kaltura.partnerId;
130             $.getScript(html5FlashCompatibilityURL, function() {
131                 var kaltura_id = data["kaltura-id"];
132                 var url = sakai.config.kaltura.serverURL + "/kwidget/wid/_"+sakai.config.kaltura.partnerId+"?ui_conf_id="+sakai.config.kaltura.playerId;
133                 var so = createSWFObject(url, {}, {});
134                 so.addVariable('stretching','uniform');
135                 so.addVariable('image', data["kaltura-thumbnail"]);
136                 so.addVariable('entryId',kaltura_id);
137                 so.write("documentviewer_video_" + tuid);
138             });
139         };
140 
141         var renderVideoPlayer = function(url, preview_avatar){
142             var so = createSWFObject(false, {}, {});
143             so.addVariable('file', url);
144             if (url.indexOf("youtube") !== -1){
145                 so.addVariable('provider', 'youtube');
146             } else {
147                 so.addVariable('provider', 'video');
148             }
149             if (preview_avatar) {
150                 so.addVariable('image', preview_avatar);
151             }
152             so.addVariable('stretching','uniform');
153             so.write("documentviewer_video_" + tuid);
154         };
155 
156         var renderAudioPlayer = function(data){
157             var so = createSWFObject(false, {"height":"25px"}, {});
158             so.addVariable('file', getPath(data));
159             so.addVariable('provider', 'sound');
160             so.addVariable('stretching','fill');
161             so.write("documentviewer_video_" + tuid);
162         };
163 
164         var renderFlashPlayer = function(data){
165             var so = createSWFObject(getPath(data), {'allowscriptaccess':'never'}, {});
166             so.addParam('scale','exactfit');
167             so.write("documentviewer_video_" + tuid);
168         };
169 
170         var renderGoogleMap = function(url) {
171             var callback = "sakai_global.documentviewer.googlemaps." + tuid;
172             sakai_global.documentviewer.googlemaps[tuid].url=url;
173             if (window["google"]) {
174                 debug.info("Already have google maps api calling the callback ourselves");
175                 sakai_global.documentviewer.googlemaps[tuid]();
176             } else {
177                 debug.info("Getting google maps api");
178                 require(["//maps.google.com/maps/api/js?sensor=false&callback="+callback]);
179             }
180         };
181 
182         var codeAddress = function (map){
183             var geocoder = new google.maps.Geocoder();
184             var address = uri.queryKey.q;
185             geocoder.geocode({
186                 'address': address
187             }, function(results, status){
188                 if (status == google.maps.GeocoderStatus.OK) {
189                     map.setCenter(results[0].geometry.location);
190                     var marker = new google.maps.Marker({
191                         map: map,
192                         position: results[0].geometry.location
193                     });
194                 } else {
195                     error.log("Geocode was not successful for the following reason: " + status);
196                 }
197             });
198         };
199 
200         // Callback for googlemaps api
201         sakai_global.documentviewer.googlemaps = sakai_global.documentviewer.googlemaps || {};
202         sakai_global.documentviewer.googlemaps[tuid] = function() {
203             var uri = parseUri(sakai_global.documentviewer.googlemaps[tuid].url);
204             // maybe we could use browser geolocation here (if ll isn't defined)
205             var lat = 0;
206             var lng = 0;
207             var zoom = uri.queryKey.z || "8";
208             zoom = parseInt(zoom, 0);
209             var type = google.maps.MapTypeId.ROADMAP;
210 
211             if (uri.queryKey.t) {
212                 switch (uri.queryKey.t) {
213                     case "h":
214                         type = google.maps.MapTypeId.HYBRID;
215                         break;
216                     case "p":
217                         type = google.maps.MapTypeId.TERRAIN;
218                         break;
219                     case "k":
220                         type = google.maps.MapTypeId.SATTELITE;
221                         break;
222                 }
223             }
224             if (uri.queryKey.ll){
225                 var ll = uri.queryKey.ll.split(',');
226                 lat = ll[0];
227                 lng = ll[1];
228             }
229             var latlng = new google.maps.LatLng(lat, lng);
230             var myOptions = {
231                 "zoom": zoom,
232                 "center": latlng,
233                 "mapTypeId": type
234             };
235             $documentviewerPreview.css({width:"100%", height:"500px"});
236             var elm = $documentviewerPreview[0];
237             var map = new google.maps.Map(elm, myOptions);
238             if (uri.queryKey.q) {
239                 codeAddress(map);
240             }
241         };
242 
243         var createSWFObject = function(url, params, flashvars){
244             if (!url){
245                 url = '/dev/lib/misc/jwplayer/player.swf';
246             }
247             var so = new SWFObject(url,'ply', '100%', params.height || '100%','9','#000000');
248             so.addParam('allowfullscreen','true');
249             if (params.allowscriptaccess) {
250                 so.addParam('allowscriptaccess', params.allowscriptaccess);
251             } else {
252                 so.addParam('allowscriptaccess', 'always');
253             }
254             so.addParam('wmode','opaque');
255             sakai.api.Util.TemplateRenderer("documentviewer_video_template", {"tuid":tuid}, $documentviewerPreview);
256             return so;
257         };
258 
259         var handleShown = function(e, showing) {
260             if (showing && docType === 'document') {
261                 // For some reason, the document view doesn't handle hide/show very well
262                 renderDocumentPreview(data);
263             }
264         };
265 
266         if (sakai.api.Content.hasPreview(widgetData.data)){
267             data = widgetData.data;
268             var mimeType = sakai.api.Content.getMimeType(widgetData.data);
269 
270             if (sakai.api.Content.isKalturaPlayerSupported(mimeType)) {
271                 renderKalturaPlayer(data);
272             } else if (sakai.api.Content.isJwPlayerSupportedVideo(mimeType)){            
273                 renderVideoPlayer(getPath(data));
274             } else if (sakai.api.Content.isJwPlayerSupportedAudio(mimeType)) {
275                 renderAudioPlayer(data);
276             } else if (mimeType === "application/x-shockwave-flash") {
277                 renderFlashPlayer(data);
278             } else if (mimeType === "text/html" || mimeType === "text/plain" || mimeType ==="text/tab-separated-values" ) {
279                 renderHTMLPreview(data);
280             } else if (mimeType === "x-sakai/link"){
281                 var pUrl = data["sakai:preview-url"];
282                 var pType = data["sakai:preview-type"];
283 
284                 if (pUrl && pType === "iframe") {
285                     renderExternalHTMLPreview(pUrl);
286                 } else if (pUrl && pType === "video") {
287                     var avatar = data["sakai:preview-avatar"];
288                     renderVideoPlayer(pUrl, avatar);
289                 } else if (pUrl && pType === "image") {
290                     renderImagePreview(pUrl);
291                 } else if (pUrl && pType === "embed") {
292                     renderEmbedPreview(pUrl);
293                 } else if (pUrl && pType ==="googlemap") {
294                     renderGoogleMap(pUrl);
295                 } else {
296                     pUrl = widgetData["sakai:pooled-content-url"];
297                     renderExternalHTMLPreview(pUrl);
298                 }
299             } else  if (mimeType.substring(0, 6) === "image/") {
300                 renderImagePreview(getPath(data), data);
301             } else if (data["sakai:pagecount"]){
302                 docType = 'document';
303                 renderDocumentPreview(data);
304             }
305         }
306 
307         // Indicate that the widget has finished loading
308         sakai_global.documentviewer.isReady = true;
309         $(window).trigger("ready.documentviewer.sakai", {});
310         $(window).bind(tuid + '.shown.sakai', handleShown);
311 
312     };
313     sakai.api.Widgets.widgetLoader.informOnLoad("documentviewer");
314 });
