 28 // Namespaces
 29 require(["jquery", "sakai/sakai.api.core"], function($, sakai) {
 30 
 31     /**
 32      * @name sakai_global.pickeradvanced
 33      *
 34      * @class pickeradvanced
 35      *
 36      * @description
 37      * Advanced Picker widget<br />
 38      * This is a general widget which aims to display an arbitriary number of
 39      * items, loading dynamically if the list is very long and return the
 40      * selected users in an object.
 41      *
 42      * @version 0.0.1
 43      * @param {String} tuid Unique id of the widget
 44      * @param {Boolean} showSettings Show the settings of the widget or not
 45      */
 46     sakai_global.pickeradvanced = function(tuid, showSettings) {
 47         var $rootel = $("#" + tuid);
 48 
 49         var $pickeradvanced_container = $("#pickeradvanced_container", $rootel);
 50         var $pickeradvanced_content_list = $("#pickeradvanced_content_list", $rootel);
 51         var $pickeradvanced_search_query = $("#pickeradvanced_search_query", $rootel);
 52         var $pickeradvanced_search_button = $("#pickeradvanced_search_button", $rootel);
 53         var $pickeradvanced_close_button = $("#pickeradvanced_close_button", $rootel);
 54         var $pickeradvanced_select_all_button = $("#pickeradvanced_select_all_button", $rootel);
 55         var $pickeradvanced_content_search_form = $("#pickeradvanced_content_search_form", $rootel);
 56         var $pickeradvanced_add_button = $("#pickeradvanced_add_button", $rootel);
 57         var $pickeradvanced_sort_on = $("#pickeradvanced_sort_on", $rootel);
 58         var $pickeradvanced_close_dialog = $(".pickeradvanced_close_dialog", $rootel);
 59         var $pickeradvanced_search_filter = $(".pickeradvanced_search_filter");
 60 
 61         var $pickeradvanced_error_template = $("#pickeradvanced_error_template", $rootel);
 62         var $pickeradvanced_content_search_pagetemplate = $("#pickeradvanced_content_search_pagetemplate", $rootel);
 63         var $pickeradvanced_content_search_listtemplate = $("#pickeradvanced_content_search_listtemplate", $rootel);
 64 
 65         var $pickeradvanced_search_titles = $(".pickeradvanced_search_titles", $rootel);
 66         var $pickeradvanced_search_files = $("#pickeradvanced_search_files", $rootel);
 67 
 68         var pickeradvanced_page = ".pickeradvanced_page";
 69         var pickeradvanced_content_search = "#pickeradvanced_content_search";
 70         
 71         var pickerlist = false;
 72         var firstTime = true;
 73 
 74         var pickerData = {
 75           "selected": {},
 76           "searchIn": sakai.config.URL.POOLED_CONTENT_SPECIFIC_USER,
 77           "currentElementCount": 0,
 78           "selectCount": 0,
 79           "mode": "search",
 80           "type": "people",
 81           "items": 50,
 82           "selectable": true,
 83           "sortOn": "lastName",
 84           "sortOrder": "asc"
 85         };
 86 
 87         /**
 88          * Reset
 89          * Resets the people picker to a default state
 90          * @returns void
 91          */
 92         var reset = function() {
 93             $(pickeradvanced_content_search).html("");
 94             $(pickeradvanced_content_search).unbind("scroll");
 95             pickerData.selected = {};
 96             pickerData.currentElementCount = 0;
 97             pickerData.selectCount = 0;
 98         };
 99 
100         /**
101          * Render
102          * Renders the people picker
103          * @param iConfig {String} Config element for the widget
104          * @returns void
105          */
106         var render = function(iConfig) {
107             // Merge user defined config with default
108             for (var element in iConfig) {
109                 if (iConfig.hasOwnProperty(element) && pickerData.hasOwnProperty(element)) {
110                     pickerData[element] = iConfig[element];
111                 }
112             }
113 
114             // display the groups list, bind elements and submit a search
115             $pickeradvanced_search_titles.hide();
116             $(".pickeradvanced_selected_list").removeClass("pickeradvanced_selected_list"); // deselect anything that was selected
117             $("#pickeradvanced_search_files_mylibrary").parent("li").addClass("pickeradvanced_selected_list");
118             $pickeradvanced_sort_on.hide();
119             $pickeradvanced_search_files.show();
120             $("ul.pickeradvanced_search_" + pickerData["type"]).show();
121             $pickeradvanced_search_query.focus();
122             $pickeradvanced_search_button.click(submitSearch);
123             $pickeradvanced_content_search_form.submit(submitSearch);
124             $pickeradvanced_add_button.unbind("click");
125             $pickeradvanced_add_button.bind("click", function(){
126                 addPeople();
127             });
128             submitSearch();
129         };
130 
131         /**
132          * RenderSearch
133          * Renders the people picker with a specified set of data. The function uses
134          * a search query initially, then does the paginating and subsequent requests
135          * for data automatically
136          * @param iSearchQuery {String} A Sakai search query
137          * @returns void
138          */
139         var submitSearch = function() {
140             reset();
141             var parameters = {
142                 "q": sakai.api.Server.createSearchString($pickeradvanced_search_query.val()) || "*",
143                 "page": 0,
144                 "items": pickerData["items"],
145                 "userid": sakai.data.me.user.userid,
146                 "sortOn": pickerData["sortOn"],
147                 "sortOrder": pickerData["sortOrder"]
148             };
149             var searchURL = pickerData['searchIn'];
150             addPage(parameters, searchURL);
151             return false;
152         };
153 
154         /**
155          * addPage
156          * Adds another page of search result to the picker's result list
157          * @pageNumber {Int} The page we want to load
158          * @url {String}     An object containing the search query elements
159          */
160         var addPage = function(parameters, url) {
161             // Create new container for the bit we load. This is then appended to the
162             // main container
163             var $pl_pageContainer = $("<ul id=\"pickeradvanced_page_" + parameters.page + "\" class=\"pickeradvanced_page pickeradvanced_page_list loadinganim\"></ul>");
164 
165             // Display empty new container with loading anim
166             $(pickeradvanced_content_search).append($pl_pageContainer);
167 
168             // Make the request
169             $.ajax({
170                 url: url,
171                 data: parameters,
172                 type: "GET",
173                 dataType: "json",
174                 success: function(rawData) {
175                     if (!rawData.results) {
176                         rawData.results = rawData;
177                     }
178                     rawData.total = rawData.results.length;
179                     rawData.sakai = sakai;
180 
181                     // Render the results data template
182                     var pageHTML = sakai.api.Util.TemplateRenderer($pickeradvanced_content_search_pagetemplate, rawData);
183 
184                     // Remove loading animation
185                     $pl_pageContainer = $(".pickeradvanced_page.pickeradvanced_page_list.loadinganim");
186                     $pl_pageContainer.removeClass("loadinganim");
187 
188                     // Inject results into DOM
189                     $pl_pageContainer.html(pageHTML);
190                     // Wire loading the next page when user scrolls to the bottom of the list
191                     if ((rawData.total > parameters.items) || (rawData.total === -1)) {
192 
193                         $(pickeradvanced_content_search).bind("scroll", function(e){
194                             if ((e.target.scrollHeight - e.target.scrollTop - $(e.target).height() ) === 0) {
195                                 $(pickeradvanced_content_search).unbind("scroll");
196                                 parameters.page++;
197                                 addPage(parameters, url);
198                             }
199                         });
200 
201                     }
202 
203                     // Wire item selection
204                     if (pickerData.selectable) {
205                         $pickeradvanced_select_all_button.unbind("click");
206                         $pickeradvanced_select_all_button.click(function(){
207                             pickerData.selectCount = 0;
208                             $('#pickeradvanced_content_search ul li').each(function(i) {
209                                 $(this).addClass("pickeradvanced_selected_user");
210                                 $(this).children("input").attr('checked', true);
211                                 pickerData.selectCount += 1;
212                                 pickerData["selected"][$(this).attr("id")] = rawData.results[i];
213                                 if ($pickeradvanced_add_button.is(":disabled")) {
214                                     $pickeradvanced_add_button.removeAttr("disabled");
215                                 }
216                             });
217                         });
218                         $("#pickeradvanced_page_" + parameters.page + " li").unbind("click");
219                         $("#pickeradvanced_page_" + parameters.page + " li").bind("click", function(e){
220                             // Check if user click on top of a link
221                             if (e.target.tagName.toLowerCase() !== "a") {
222                                 // Remove from selected list
223                                 if ($(this).hasClass("pickeradvanced_selected_user")) {
224                                     $(this).removeClass("pickeradvanced_selected_user");
225                                     $(this).children("input").removeAttr('checked');
226                                     delete pickerData["selected"][$(this).attr("id")];
227                                     pickerData.selectCount -= 1;
228                                     if (pickerData.selectCount < 1) {
229                                         $pickeradvanced_add_button.attr("disabled", "disabled");
230                                     }
231                                 } else {
232                                     // Add to selected list
233                                     $(this).addClass("pickeradvanced_selected_user");
234                                     $(this).children("input").attr('checked', true);
235                                     for (var j = 0; j < rawData.results.length; j++) {
236                                         if (rawData.results[j]['_path'] && rawData.results[j]['_path'] == [$(this).attr("id")]) {
237                                             pickerData.selectCount += 1;
238                                             pickerData["selected"][$(this).attr("id")] = rawData.results[j];
239                                             pickerData["selected"][$(this).attr("id")].entityType = "file";
240                                         }
241                                     }
242                                     if ($pickeradvanced_add_button.is(":disabled")) {
243                                         $pickeradvanced_add_button.removeAttr("disabled");
244                                     }
245                                 }
246                             }
247                         });
248                     }
249 
250                     // Wire sorting select dropdown
251                     $pickeradvanced_sort_on.unbind("change");
252                     $pickeradvanced_sort_on.bind("change", function(e){
253                         // Reset everything
254                         reset();
255 
256                         // Set config to new sort key
257                         // Start from scratch
258                         pickerData["sortOn"] = $(this).val().split("_")[0];
259                         pickerData["sortOrder"] = $(this).val().split("_")[1];
260                         submitSearch();
261 
262                     });
263                 },
264                 error: function(xhr, status, thrown) {
265 
266                     // If it's likely to be a genuine server error
267                     if ($pl_pageContainer.length === 0) {
268                         $pickeradvanced_content_search.html($pickeradvanced_error_template.html());
269                     } else {
270                         // Probably it's the last page of the result set
271                         $pl_pageContainer.last().remove();
272                     }
273                 }
274             });
275         };
276 
277         sakai.api.Util.Modal.setup($pickeradvanced_container, {
278             modal: true,
279             overlay: 20,
280             zIndex: 5000,
281             toTop: true
282         });
283 
284         var addPeople = function() {
285             // this value is a comma-delimited list
286             // split it and get rid of any empty values in the array
287             sakai.api.Util.Modal.close($pickeradvanced_container);
288             $(window).trigger("finished.pickeradvanced.sakai", {"toAdd":pickerData["selected"]});
289         };
290 
291 
292         ////////////
293         // Events //
294         ////////////
295 
296         $(window).unbind("init.pickeradvanced.sakai");
297         $(window).bind("init.pickeradvanced.sakai", function(e, config) {
298             firstTime = true;
299             render(config.config);
300             sakai.api.Util.Modal.open($pickeradvanced_container);
301             pickerlist = config.list;
302         });
303 
304         $pickeradvanced_close_dialog.unbind("click");
305         $pickeradvanced_close_dialog.bind("click", function() {
306             sakai.api.Util.Modal.close($pickeradvanced_container);
307         });
308 
309         $pickeradvanced_close_button.unbind("click");
310         $pickeradvanced_close_button.bind("click", function() {
311             sakai.api.Util.Modal.close($pickeradvanced_container);
312         });
313 
314         $pickeradvanced_search_filter.die("click");
315         $pickeradvanced_search_filter.live("click", function() {
316            $(".pickeradvanced_selected_list").removeClass("pickeradvanced_selected_list");
317            $(this).parent("li").addClass("pickeradvanced_selected_list");
318            if ( $( this ).attr( "data-search" ) === "all" ) {
319                pickerData.searchIn = sakai.config.URL.SEARCH_ALL_FILES;
320            } else if ($( this ).attr( "data-search" ) === "my" ) {
321                pickerData.searchIn = sakai.config.URL.POOLED_CONTENT_SPECIFIC_USER;
322            }
323            $pickeradvanced_sort_on.hide();
324            submitSearch();
325         });
326 
327         // Reset to defaults
328         reset();
329 
330         // Send out an event that says the widget is ready to
331         // accept a search query to process and display. This event can be picked up
332         // in a page JS code
333         $(window).trigger("ready.pickeradvanced.sakai");
334 
335     };
336 
337     sakai.api.Widgets.widgetLoader.informOnLoad("pickeradvanced");
338 });
