 54         /**
 55          * Filters out pseudogroups and adds the parent group to the list to be displayed
 56          * @param {Array} data required array of user and group objects to filter
 57          * @param {Boolean} setCount required Set to true if the context is content and the counts should be updated (Filtered pseudogroups don't count)
 58          * @param {Object} context not required if setCount is false, provides the context of the entity widget and holds the counts
 59          * @Return {Object} parentGroups Object containing the parent groups to display
 60          */
 61         var getParentGroups = function(data, setCount, context){
 62             var parentGroups = {};
 63             if (setCount) {
 64                 context.data.members.counts.groups = 0;
 65                 context.data.members.counts.collections = 0;
 66             }
 67             $.each(data, function(index, group){
 68                 // Check for pseudogroups, if a pseudogroup filter out the parent
 69                 if (group.pseudoGroup) {
 70                     // Only groups should be added to the object
 71                     if (!parentGroups.hasOwnProperty(group.parent["sakai:group-id"]) && group.parent["sakai:group-id"]) {
 72                         if (setCount) {
 73                             context.data.members.counts.groups++;
 74                         }
 75                         // Discard pseudogroup but store parent group
 76                         parentGroups[group.parent["sakai:group-id"]] = {
 77                             "sakai:group-id": group.parent["sakai:group-id"],
 78                             "sakai:group-title": group.parent["sakai:group-title"]
 79                         };
 80                     }
 81                 // If no pseudogroup store the group as it is
 82                 } else if (!parentGroups.hasOwnProperty(group["sakai:group-id"]) && group["sakai:group-id"]) {
 83                     if (setCount) {
 84                         if (sakai.api.Content.Collections.isCollection(group)){
 85                             context.data.members.counts.collections++;
 86                         } else {
 87                             context.data.members.counts.groups++;
 88                         }
 89                     }
 90                     parentGroups[group["sakai:group-id"]] = group;
 91                 }
 92             });
 93             return parentGroups;
 94         };

 96         /**
 97          * Saves the content/collection name
 98          * @param {String} newTitle The new content/collection title to save

100         var saveName = function(newTitle) {
101             var oldTitle = $.trim($('#entity_name').attr('data-original-title'));
102             $('#entity_name').attr('data-original-title', newTitle);
103             if (newTitle && newTitle !== oldTitle) {
104                 $.ajax({
105                     url: '/p/' + sakai_global.content_profile.content_data.data['_path'] + '.html',
106                     type: 'POST',
107                     cache: false,
108                     data: {
109                         'sakai:pooled-content-file-name': newTitle
110                     },
111                     success: function() {
112                         var contentData = sakai_global.content_profile.content_data.data;
113                         if (sakai.api.Content.Collections.isCollection(contentData)) {
114                             // Change the group title as well
115                             var groupId = sakai.api.Content.Collections.getCollectionGroupId(contentData);
116                             $.ajax({
117                                 'url': '/system/userManager/group/' + groupId + '.update.json',
118                                 'type': 'POST',
119                                 'data': {
120                                     'sakai:group-title': newTitle
121                                 },
122                                 'success': function() {
123                                     // Update the me object
124                                     var memberships = sakai.api.Groups.getMemberships(sakai.data.me.groups, true);
125                                     $.each(memberships.entry, function(index, membership) {
126                                         if (membership['sakai:group-id'] === groupId) {
127                                             membership['sakai:group-title'] = newTitle;
128                                         }
129                                     });
130                                     finishChangeTitle(newTitle);
131                                 }
132                             });
133                         } else {
134                             finishChangeTitle(newTitle);
135                         }
136                     }
137                 });
138             }
139         };

141         var finishChangeTitle = function(newTitle) {

145             // Export as IMS Package
146             if (sakai.api.Content.getMimeType(sakai_global.content_profile.content_data.data) === 'x-sakai/document') {
147                 link = '/imscp/' + sakai_global.content_profile.content_data.data['_path'] + '/' +
148                 sakai.api.Util.safeURL(sakai_global.content_profile.content_data.data['sakai:pooled-content-file-name']) + '.zip';
149                 $('#contentpreview_download_button').attr('href', link);
150             // Download as a normal file
151             } else {
152                 link = sakai_global.content_profile.content_data.smallPath + '/' +
153                 sakai.api.Util.safeURL(sakai_global.content_profile.content_data.data['sakai:pooled-content-file-name']);
154                 $('#contentpreview_download_button').attr('href', link);
155             }
156             document.title = sakai.api.i18n.getValueForKey(sakai.config.PageTitles.prefix) + " " + title;
157         };

417         var prepareRenderContext = function(context) {
418             if (context.context === "content") {
419                 if ($.isArray(sakai_global.content_profile.content_data.members.managers)) {
420                     getParentGroups(getUserList, true, context);
421                 }
422 
423                 // Collaborators are managers & editors
424                 var collaborators = sakai_global.content_profile.content_data.members.managers.concat(
425                     sakai_global.content_profile.content_data.members.editors
426                 );
427 
428                 sakai_global.content_profile.content_data.members.counts.collaboratorgroups = 0;
429                 sakai_global.content_profile.content_data.members.counts.collaboratorusers = 0;
430                 $.each(collaborators, function(i, collaborator) {
431                     if(collaborator['sakai:group-id']){
432                         sakai_global.content_profile.content_data.members.counts.collaboratorgroups++;
433                     } else {
434                         sakai_global.content_profile.content_data.members.counts.collaboratorusers++;
435                     }
436                 });
437 
438                 sakai_global.content_profile.content_data.members.counts.viewergroups = 0;
439                 sakai_global.content_profile.content_data.members.counts.viewerusers = 0;
440                 sakai_global.content_profile.content_data.members.counts.viewercollections = 0;
441                 $.each(sakai_global.content_profile.content_data.members.viewers, function(i, viewer){
442                     if(viewer["sakai:group-id"] && sakai.api.Content.Collections.isCollection(viewer)){
443                         sakai_global.content_profile.content_data.members.counts.viewercollections++;
444                     } else if(viewer["sakai:group-id"]){
445                         sakai_global.content_profile.content_data.members.counts.viewergroups++;
446                     } else {
447                         sakai_global.content_profile.content_data.members.counts.viewerusers++;
448                     }
449                 });
450             }
451             context.sakai = sakai;
452             context.entitymacros = sakai.api.Util.processLocalMacros($("#entity_macros_template"));
453         };
