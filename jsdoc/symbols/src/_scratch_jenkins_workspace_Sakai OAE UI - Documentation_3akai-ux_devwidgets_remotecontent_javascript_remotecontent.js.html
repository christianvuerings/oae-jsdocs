 18 /*
 19  * Dependencies
 20  *
 21  * /dev/lib/misc/trimpath.template.js (TrimpathTemplates)
 22  */
 23 /*global $, get_cookie, Config */
 24 
 25 require(["jquery", "sakai/sakai.api.core"], function($, sakai) {
 26 
 27     /**
 28      * @name sakai_global.remotecontent
 29      *
 30      * @class remotecontent
 31      *
 32      * @description
 33      * Initialize the remotecontent widget
 34      *
 35      * @version 0.0.1
 36      * @param {String} tuid Unique id of the widget
 37      * @param {Boolean} showSettings Show the settings of the widget or not
 38      */
 39     sakai_global.remotecontent = function(tuid, showSettings, widgetData){
 40 
 41 
 42         /////////////////////////////
 43         // Configuration variables //
 44         /////////////////////////////
 45 
 46         var rootel = $("#" + tuid);
 47         var json = false;
 48         var isAdvancedSettingsVisible = false;
 49 
 50         // Default values
 51         var defaultWidth = 100;
 52         var defaultWidthUnit = "%";
 53         var defaultHeight = 200;
 54         var clickSubmit = false;
 55 
 56         // Links and labels
 57         var remotecontent = "#remotecontent";
 58         var remotecontentTitle = remotecontent + "_title";
 59         var remotecontentSettings = remotecontent + "_settings";
 60         var remotecontentSettingsAdvanced = remotecontentSettings + "_advanced";
 61         var remotecontentSettingsAdvancedDown = remotecontentSettingsAdvanced + "_down";
 62         var remotecontentSettingsAdvancedToggleSettings = remotecontentSettingsAdvanced + "_toggle_settings";
 63         var remotecontentSettingsAdvancedUp = remotecontentSettingsAdvanced + "_up";
 64         var remotecontentSettingsBorders = remotecontentSettings + "_borders";
 65         var remotecontentSettingsCancel = remotecontentSettings + "_cancel";
 66         var remotecontentSettingsColorContainer = remotecontentSettings + "_color_container";
 67         var remotecontentSettingsHeight = remotecontentSettings + "_height";
 68         var remotecontentSettingsInsert = remotecontentSettings + "_insert";
 69         var remotecontentSettingsPreview = remotecontentSettings + "_preview";
 70         var remotecontentSettingsPreviewFrame = remotecontentSettingsPreview + "_frame";
 71         var remotecontentSettingsUrl = remotecontentSettings + "_url";
 72         var remotecontentSettingsUrlError = remotecontentSettingsUrl + "_error";
 73         var remotecontentSettingsUrlBlank = remotecontentSettingsUrl + "_blank";
 74         var remotecontentSettingsWidth = remotecontentSettings + "_width";
 75         var remotecontentSettingsResetSize = remotecontentSettings + "_reset_size";
 76         var remotecontentSettingsWidthUnitPercent = remotecontentSettings + "_width_unit_percent";
 77 
 78         // Containers
 79         var remotecontentMainContainer = remotecontent + "_main_container";
 80 
 81         // Classes
 82         var remotecontentSettingsWidthUnitClass = ".remotecontent_settings_width_unit";
 83         var remotecontentSettingsWidthUnitSelectedClass = "remotecontent_settings_width_unit_selected";
 84 
 85         // Templates
 86         var $remotecontentSettingsColorContainerTemplate = $("#remotecontent_settings_color_container_template", rootel);
 87         var $remotecontentSettingsTemplate = $("#remotecontent_settings_template", rootel);
 88         var $remotecontentSettingsPreviewTemplate = $("#remotecontent_settings_preview_template", rootel);
 89         var $noRemoteContentSet = $("#remotecontent_no_content_set", rootel);
 90 
 91 
 92         ///////////////////////
 93         // Utility functions //
 94         ///////////////////////
 95 
 96         /**
 97          * Check if the value is a decimal or not
 98          * @param {Object} value Value that needs to be checked
 99          * @return {Boolean}
100          *     true: is a decimal
101          *     false: is not a decimal
102          */
103         var isDecimal = function(value){
104             return (/^\d+$/).test(value);
105         };
106 
107         /**
108          * Check if the input url is missing http/s or not
109          * @param {String} url Url that needs to be tested
110          * @return {Boolean}
111          *     true: is an url
112          *     false: is not an url
113          */
114         var isUrl = function(url){
115             return (/^http\:\/\/|^https\:\/\//i).test(url);
116         };
117 
118         /**
119          * Check if the input url is in fact a complete url or not
120          * @param {String} url Url that needs to be tested
121          * @return {Boolean}
122          *     true: is an url
123          *     false: is not an url
124          */
125         var isCompleteUrl = function(url){
126             var regEx = new RegExp(/^(?:ftp|https?):\/\/(?:(?:[\w\.\-\+%!$&'\(\)*\+,;=]+:)*[\w\.\-\+%!$&'\(\)*\+,;=]+@)?(?:[a-z0-9\-\.%]+)(?::[0-9]+)?(?:[\/|\?][\w#!:\.\?\+=&%@!$'~*,;\/\(\)\[\]\-]*)?$/);
127             return regEx.test(url);
128         };
129 
130         /**
131          * Called when the data has been saved to the JCR.
132          */
133         var savedDataToJCR = function(){
134             sakai.api.Widgets.Container.informFinish(tuid, "remotecontent");
135         };
136 
137 
138         //////////////////////
139         // Render functions //
140         //////////////////////
141 
142         /**
143          * Render the iframe for the widget in settings mode
144          * @param {Boolean} complete Render the preview completely or only adjust values
145          */
146         var renderIframeSettings = function(complete) {
147             if (complete) {
148                 $(remotecontentSettingsPreview, rootel).html(sakai.api.Util.TemplateRenderer($remotecontentSettingsPreviewTemplate, json, null, false));
149             } else {
150                 $(remotecontentSettingsPreviewFrame, rootel).attr('style', 'border: ' + json.border_size + 'px #' + json.border_color + ' solid');
151             }
152         };
153 
154         /**
155          * Render the iframe for the widget
156          */
157         var renderIframe = function(){
158             if (json) {
159                 $(remotecontentMainContainer, rootel).html(sakai.api.Util.TemplateRenderer($remotecontentSettingsPreviewTemplate, json, null, false));
160 
161                 // SAKIII-314 We need to show the container, otherwise the second item won't be shown.
162                 $(remotecontentMainContainer, rootel).show();
163             }
164         };
165 
166         /**
167          * Render the html of the remotecontentsettings
168          */
169         var renderRemoteContentSettings = function(){
170             if (json) {
171                 $(remotecontentSettings, rootel).html(sakai.api.Util.TemplateRenderer($remotecontentSettingsTemplate, json));
172             }
173         };
174 
175         /**
176          * Render the color container
177          */
178         var renderColorContainer = function(){
179             if (json) {
180                 $(remotecontentSettingsColorContainer, rootel).html(sakai.api.Util.TemplateRenderer($remotecontentSettingsColorContainerTemplate, json));
181             }
182         };
183 
184 
185         //////////////////////
186         // Global functions //
187         //////////////////////
188 
189         /**
190          * Display the iframe in normal mode
191          * @param {Object} parameters JSON object that contains the necessary information for the iframe
192          */
193         var displayRemoteContent = function(parameters){
194             json = parameters;
195             renderIframe();
196         };
197 
198         /**
199          * Save the remotecontent to the jcr
200          */
201         var saveRemoteContent = function(){
202             clickSubmit = true;
203             if ($("#remotecontent_form", rootel).valid()) {
204                 $(remotecontentSettingsPreview, rootel).html("");
205                 sakai.api.Widgets.saveWidgetData(tuid, json, savedDataToJCR);
206             }
207         };
208 
209         /**
210          * Change the direction (up/down) of the arrow for the advanced settings
211          */
212         var changeAdvancedSettingsArrow = function(){
213             if (isAdvancedSettingsVisible) {
214                 $(remotecontentSettingsAdvancedDown, rootel).hide();
215                 $(remotecontentSettingsAdvancedUp, rootel).show();
216             }
217             else {
218                 $(remotecontentSettingsAdvancedUp, rootel).hide();
219                 $(remotecontentSettingsAdvancedDown, rootel).show();
220             }
221         };
222 
223 
224         //////////////
225         // Bindings //
226         //////////////
227 
228         /*
229          * Add binding to the color boxes
230          */
231         var addColorBinding = function(){
232             $(".remotecontent_settings_color", rootel).click(function(){
233                 json.border_color = $(this).attr("id").split("_")[$(this).attr("id").split("_").length - 1];
234                 renderIframeSettings(false);
235                 renderColorContainer();
236                 addColorBinding();
237             });
238         };
239 
240         /*
241          * Add binding to all the elements
242          */
243         var addBinding = function(){
244             // FORM VALIDATION
245 
246             var validateOpts = {
247                 onclick: true,
248                 onkeyup: true,
249                 onfocusout: true,
250                 messages: {
251                     required: $(remotecontentSettingsUrlBlank).html(),
252                     url: $(remotecontentSettingsUrlError).html()
253                 },
254                 methods: {
255                     'appendhttp': {
256                         'method': function(value, element) {
257                             if (value.substring(0,7) !== 'http://' &&
258                                 value.substring(0,6) !== 'ftp://' &&
259                                 value.substring(0,8) !== 'https://' &&
260                                 $.trim(value) !== '') {
261                                 $(element).val('http://' + value);
262                                 json.url = 'http://' + value;
263                             } else {
264                                 json.url = value;
265                             }
266                             return true;
267                         },
268                         'text': 'No error message, this is just an appender'
269                     }
270                 },
271                 success: function() {
272                     renderIframeSettings(true);
273                 },
274                 invalidCallback: function() {
275                     renderIframeSettings(false);
276                 }
277             };
278 
279             // Initialize the validate plug-in
280             sakai.api.Util.Forms.validate($("#remotecontent_form", rootel), validateOpts, true);
281 
282             // Change the iframe width
283             $(remotecontentSettingsWidth, rootel).change(function(){
284                 var widthValue = $(remotecontentSettingsWidth, rootel).val();
285 
286                 if (isDecimal(widthValue)) {
287                     json.width = widthValue;
288                 }
289                 renderIframeSettings(false);
290             });
291 
292             // Change the iframe height
293             $(remotecontentSettingsHeight, rootel).change(function(){
294                 var heightValue = $(remotecontentSettingsHeight, rootel).val();
295 
296                 if (isDecimal(heightValue)) {
297                     json.height = heightValue;
298                 }
299                 renderIframeSettings(false);
300             });
301 
302             // Change the border width
303             $(remotecontentSettingsBorders, rootel).on('click', function() {
304                 if ($(remotecontentSettingsBorders, rootel).is(':checked')) {
305                     json.border_size = 2;
306                 } else {
307                     json.border_size = 0;
308                 }
309                 renderIframeSettings(false);
310             });
311 
312             // Toggle the advanced view
313             $(remotecontentSettingsAdvancedToggleSettings, rootel).click(function(){
314                 $("#remotecontent_settings_advanced", rootel).toggle();
315                 isAdvancedSettingsVisible = !isAdvancedSettingsVisible;
316                 changeAdvancedSettingsArrow();
317             });
318 
319             // Reset size to default values
320             $(remotecontentSettingsResetSize, rootel).click(function(){
321                 $(remotecontentSettingsWidth, rootel).val(defaultWidth);
322                 $(remotecontentSettingsHeight, rootel).val(defaultHeight);
323                 $(remotecontentSettingsWidth, rootel).change();
324                 $(remotecontentSettingsHeight, rootel).change();
325                 $(remotecontentSettingsWidthUnitPercent, rootel).click();
326             });
327 
328             // When you click on one of the width units (px or percentage)
329             $(remotecontentSettingsWidthUnitClass, rootel).on('change', function() {
330                 var widthUnitValue = $(this).val();
331                 if (widthUnitValue === "px") {
332                     json.width_unit = widthUnitValue;
333                 }
334                 else {
335                     json.width_unit = "%";
336                 }
337                 renderIframeSettings(false);
338             });
339 
340             // When you push the save button..
341             $(remotecontentSettingsInsert, rootel).click(function(){
342                 saveRemoteContent();
343             });
344 
345             // Cancel it
346             $(remotecontentSettingsCancel, rootel).click(function(){
347                 sakai.api.Widgets.Container.informCancel(tuid, "remotecontent");
348             });
349 
350             addColorBinding();
351         };
352 
353 
354         ///////////////////////
355         // Initial functions //
356         ///////////////////////
357 
358         /**
359          * Function that fills in the input fields in the settings tab.
360          * @param {Object} parameters A JSON object that contains the necessary information.
361          * @param {Boolean} exists Does there exist a previous remotecontent
362          */
363         var displaySettings = function(parameters, exists){
364             /**
365              * We also blank the URL field if it matches the default URL in widget config,
366              * i.e. if it's the default path to user instructions.
367              */
368             if (exists && parameters.url && parameters.url !== sakai.widgets.remotecontent.defaultConfiguration.remotecontent.url) {               
369                 json = parameters;
370             }
371             else {
372                 json = {
373                     border_size: 0,
374                     border_color: "cccccc",
375                     height: defaultHeight,
376                     url: "",
377                     width: defaultWidth,
378                     width_unit: defaultWidthUnit
379                 };
380             }
381             renderRemoteContentSettings();
382             renderIframeSettings(true);
383             renderColorContainer();
384             addBinding(); // Add binding to the various elements
385             changeAdvancedSettingsArrow();
386             $(remotecontentSettings, rootel).show(); // Show the remotecontent settings
387         };
388 
389         /*
390          * Is the widget in settings mode or not
391          */
392         if (showSettings) {
393             $(remotecontentMainContainer, rootel).hide();
394             $(remotecontentSettings, rootel).show();
395         }
396         else {
397             $(remotecontentSettings, rootel).hide();
398             $(remotecontentMainContainer, rootel).show();
399         }
400 
401         /**
402          * Will fetch the URL and other parameters from the JCR and according to which
403          * view we are in, fill in the settings or display an iframe.
404          */
405         var getRemoteContent = function(){
406             if (widgetData.remotecontent){
407                 processRemoteContent(true, widgetData.remotecontent);
408             } else {
409                 sakai.api.Widgets.loadWidgetData(tuid, processRemoteContent);
410             }
411         };
412 
413         var processRemoteContent = function(success, data){
414             if (success) {
415                 // Get a JSON string that contains the necessary information.
416                 var parameters = data;
417 
418                 if (showSettings) {
419                     displaySettings(parameters, true); // Fill in the settings page.
420                 } else {
421                     displayRemoteContent(parameters); // Show the frame
422                 }
423             } else {
424                 // When the request isn't successful, it means that  there was no existing remotecontent
425                 // so we show the basic settings.
426                 if (showSettings) {
427                     displaySettings(null, false);
428                 } else {
429                     $(remotecontentMainContainer, rootel).html(sakai.api.Util.TemplateRenderer($noRemoteContentSet, {}));
430                     $(remotecontentMainContainer, rootel).show();
431                 }
432             }
433         };
434 
435         getRemoteContent();
436 
437     };
438 
439     sakai.api.Widgets.widgetLoader.informOnLoad("remotecontent");
440 });
