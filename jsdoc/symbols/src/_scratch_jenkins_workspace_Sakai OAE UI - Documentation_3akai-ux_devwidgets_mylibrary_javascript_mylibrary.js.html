 18 // load the master sakai object to access all Sakai OAE API methods
 19 require(['jquery', 'sakai/sakai.api.core'], function($, sakai) {

 21     /**
 22      * @name sakai_global.mylibrary
 23      *
 24      * @class mylibrary
 25      *
 26      * @description
 27      * Widget that lists all of the content items that live inside of a given
 28      * library
 29      *
 30      * @param {String} tuid Unique id of the widget
 31      * @param {Boolean} showSettings Show the settings of the widget or not
 32      */
 33     sakai_global.mylibrary = function (tuid, showSettings, widgetData, state) {

 35         /////////////////////////////

 37         /////////////////////////////

 39         var mylibrary = {  // global data for mylibrary widget
 40             sortBy: '_lastModified',
 41             sortOrder: 'desc',
 42             isOwnerViewing: false,
 43             default_search_text: '',
 44             userArray: [],
 45             contextId: false,
 46             infinityScroll: false,
 47             widgetShown: true,
 48             listStyle: 'list'
 49         };
 50 
 51         // DOM jQuery Objects
 52         var $rootel = $('#' + tuid);  // unique container for each widget instance
 53         var $mylibrary_check = $('.mylibrary_check', $rootel);
 54         var $mylibrary_items = $('#mylibrary_items', $rootel);
 55         var $mylibrary_check_all = $('#mylibrary_select_checkbox', $rootel);
 56         var $mylibrary_remove = $('#mylibrary_remove', $rootel);
 57         var $mylibrary_addto = $('#mylibrary_addpeople_button', $rootel);
 58         var $mylibrary_share = $('#mylibrary_content_share', $rootel);
 59         var $mylibrary_sortby = $('#mylibrary_sortby', $rootel);
 60         var $mylibrary_livefilter = $('#mylibrary_livefilter', $rootel);
 61         var $mylibrary_livefilter_container = $('#mylibrary_livefilter_container', $rootel);
 62         var $mylibrary_sortarea = $('#mylibrary_sortarea', $rootel);
 63         var $mylibrary_empty = $('#mylibrary_empty', $rootel);
 64         var $mylibrary_admin_actions = $('#mylibrary_admin_actions', $rootel);
 65         var $mylibrary_remove_icon = $('.mylibrary_remove_icon', $rootel);
 66         var $mylibrary_search_button = $('#mylibrary_search_button', $rootel);
 67         var $mylibrary_result_count = $('.s3d-search-result-count', $rootel);
 68         var $mylibrary_show_grid = $('.s3d-listview-grid', $rootel);
 69         var $mylibrary_show_list = $('.s3d-listview-list', $rootel);
 70         var $mylibraryAddContentOverlay = $('.sakai_add_content_overlay', $rootel);

 72         var currentGroup = false;

 74         ///////////////////////
 75         // Utility functions //
 76         ///////////////////////
 77 
 78         /**
 79          * Get personalized text for the given message bundle key based on
 80          * whether this library is owned by the viewer, or belongs to someone else.
 81          * The message should contain a '${firstname}' variable to replace with
 82          * and be located in this widget's properties files.
 83          *
 84          * @param {String} bundleKey The message bundle key
 85          */
 86         var getPersonalizedText = function (bundleKey) {
 87             if(currentGroup) {
 88                 return sakai.api.i18n.getValueForKey(
 89                     bundleKey, 'mylibrary').replace(/\$\{firstname\}/gi,
 90                         currentGroup.properties['sakai:group-title']);
 91             } else if (mylibrary.isOwnerViewing) {
 92                 return sakai.api.i18n.getValueForKey(
 93                     bundleKey, 'mylibrary').replace(/\$\{firstname\}/gi,
 94                         sakai.api.i18n.getValueForKey('YOUR').toLowerCase());
 95             } else {
 96                 return sakai.api.i18n.getValueForKey(bundleKey, 'mylibrary').replace(/\$\{firstname\}/gi, sakai.api.User.getFirstName(sakai_global.profile.main.data) + '\'s');
 97             }
 98         };
 99 
100         /**
101          * Bring all of the topbar items (search, checkbox, etc.) back into its original state
102          */
103         var resetView = function() {
104             $mylibrary_result_count.hide();
105             $mylibrary_check_all.removeAttr('checked');
106             $mylibrary_remove.attr('disabled', 'disabled');
107             $mylibrary_addto.attr('disabled', 'disabled');
108             $mylibrary_share.attr('disabled', 'disabled');
109             $('#mylibrary_title_bar').show();
110         };
111 
112         /**
113          * Show or hide the controls on the top of the library (search box, sort, etc.)
114          * @param {Object} show    True when you want to show the controls, false when
115          *                         you want to hide the controls
116          */
117         var showHideTopControls = function(show) {
118             if (show) {
119                 if (mylibrary.isOwnerViewing) {
120                     $mylibrary_remove.show();
121                     $mylibrary_admin_actions.show();
122                     $mylibraryAddContentOverlay.show();
123                 }
124                 $('.s3d-page-header-top-row', $rootel).show();
125                 $mylibrary_livefilter_container.show();
126                 $mylibrary_sortarea.show();
127             } else {
128                 $('.s3d-page-header-top-row', $rootel).hide();
129                 $mylibrary_admin_actions.hide();
130                 $mylibrary_livefilter_container.hide();
131                 $mylibrary_sortarea.hide();
132             }
133         };
134 
135         /**
136          * Renders library title
137          * @param {String} contextName The name to render
138          * @param {Boolean} isGroup Flag if this is a groups library or not
139          */
140         var renderLibraryTitle = function(contextName, isGroup) {
141             sakai.api.Util.TemplateRenderer('mylibrary_title_template', {
142                 isMe: mylibrary.isOwnerViewing,
143                 isGroup: isGroup,
144                 user: sakai.api.Util.Security.safeOutput(contextName)
145             }, $('#mylibrary_title_container', $rootel));
146         };
147 
148         /////////////////////////////
149         // Deal with empty library //
150         /////////////////////////////
151 
152         /**
153          * Function that manages how a library with no content items is handled. In this case,
154          * the items container will be hidden and a 'no items' container will be shown depending
155          * on what the current context of the library is
156          */
157         var handleEmptyLibrary = function() {
158             resetView();
159             $mylibrary_items.hide();
160             var query = $mylibrary_livefilter.val();
161             // We only show the controls when there is a search query. 
162             // All other scenarios with no items don't show the top controls
163             if (!query) {
164                 showHideTopControls(false);
165             } else {
166                 showHideTopControls(true);
167             }
168             // Determine the state of the current user in the current library
169             var mode = 'user_me';
170             if (sakai_global.profile && mylibrary.contextId !== sakai.data.me.user.userid) {
171                 mode = 'user_other';
172             } else if (!sakai_global.profile && (mylibrary.isOwnerViewing || mylibrary.isMemberViewing)) {
173                 mode = 'group_manager_member';
174             } else if (!sakai_global.profile) {
175                 mode = 'group';
176             }
177             $mylibrary_empty.html(sakai.api.Util.TemplateRenderer('mylibrary_empty_template', {
178                 mode: mode,
179                 query: query
180             }));
181 
182             $mylibrary_empty.show();
183         };
184 
185         ////////////////////
186         // Load a library //
187         ////////////////////
188 
189         /**
190          * Load the items of the current library and start an infinite scroll on
191          * them
192          */
193         var showLibraryContent = function () {
194             resetView();
195             var query = $mylibrary_livefilter.val();
196             // Disable the previous infinite scroll
197             if (mylibrary.infinityScroll) {
198                 mylibrary.infinityScroll.kill();
199             }
200             // Set up the infinite scroll for the list of items in the library
201             var sortOrder = mylibrary.sortOrder;
202             if (mylibrary.sortOrder === 'modified') {
203                 sortOrder = 'desc';
204             }
205             mylibrary.infinityScroll = $mylibrary_items.infinitescroll(sakai.config.URL.POOLED_CONTENT_SPECIFIC_USER, {
206                 userid: mylibrary.contextId,
207                 sortOn: mylibrary.sortBy,
208                 sortOrder: sortOrder,
209                 q: query
210             }, function(items, total) {
211                 if (total && query && query !== '*') {
212                     $mylibrary_result_count.show();
213                     var resultLabel = sakai.api.i18n.getValueForKey('RESULTS');
214                     if (total === 1) {
215                         resultLabel = sakai.api.i18n.getValueForKey('RESULT');
216                     }
217                     $mylibrary_result_count.children('.s3d-search-result-count-label').text(resultLabel);
218                     $mylibrary_result_count.children('.s3d-search-result-count-count').text(total);
219                 }
220                 if (!sakai.data.me.user.anon) {
221                     if (total !== 0) {
222                         $('.s3d-page-header-top-row', $rootel).show();
223                         $('.s3d-page-header-bottom-row', $rootel).show();
224                     } else {
225                         $('.s3d-page-header-bottom-row', $rootel).hide();
226                     }
227                 } else {
228                     if (total !== 0) {
229                         $('.s3d-page-header-top-row', $rootel).show();
230                     }
231                 }
232                 return sakai.api.Util.TemplateRenderer('mylibrary_items_template', {
233                     'items': items,
234                     'sakai': sakai,
235                     'isMe': mylibrary.isOwnerViewing
236                 });
237             }, handleEmptyLibrary, sakai.config.URL.INFINITE_LOADING_ICON, handleLibraryItems, function() {
238                 // Initialize content draggable
239                 sakai.api.Util.Draggable.setupDraggable({}, $mylibrary_items);
240             }, sakai.api.Content.getNewList(mylibrary.contextId));
241         };

244         // State handling //

247         /**
248          * Deal with changed state in the library. This can be triggered when
249          * a search was initiated, sort was changed, etc.
250          */
251         var handleHashChange = function(e) {
252             var ls = $.bbq.getState('ls') || 'list';
253             $('.s3d-listview-options', $rootel).find('div').removeClass('selected');
254             if (ls === 'list') {
255                 $('#mylibrary_items', $rootel).removeClass('s3d-search-results-grid');
256                 $mylibrary_show_list.addClass('selected');
257                 $mylibrary_show_list.children().addClass('selected');
258                 mylibrary.listStyle = 'list';
259             } else if (ls === 'grid') {
260                 $('#mylibrary_items', $rootel).addClass('s3d-search-results-grid');
261                 $('.s3d-listview-options', $rootel).find('div').removeClass('selected');
262                 $mylibrary_show_grid.addClass('selected');
263                 $mylibrary_show_grid.children().addClass('selected');
264                 mylibrary.listStyle = 'grid';
265             }
266 
267             if (mylibrary.widgetShown) {
268                 // Set the sort states
269                 var parameters = $.bbq.getState();
270                 mylibrary.sortOrder = parameters.lso || 'modified';
271                 mylibrary.sortBy = parameters.lsb || '_lastModified';
272                 $mylibrary_livefilter.val(parameters.lq || '');
273                 $mylibrary_sortby.val(mylibrary.sortOrder);
274                 showLibraryContent();
275             }
276         };

282         /**
283          * Search the current library. This function will extract the query
284          * from the search box directly.
285          */
286         var doSearch = function() {
287             var q = $.trim($mylibrary_livefilter.val());
288             if (q) {
289                 $.bbq.pushState({ 'lq': q });
290             } else {
291                 $.bbq.removeState('lq');
292             }
293         };
294 
295         var updateButtonData = function() {
296             var shareIdArr = [];
297             var addToIdArr = [];
298             var addToTitleArr = [];
299             $.each($('.mylibrary_check:checked:visible', $rootel), function(i, checked) {
300                 addToIdArr.push($(checked).attr('data-entityid'));
301                 addToTitleArr.push($(checked).attr('data-entityname'));
302                 shareIdArr.push($(checked).attr('data-entityid'));
303                 if (!$(checked).attr('data-canshare-error')) {
304                     $(checked).attr('data-canshare-error', 'true');
305                 }
306             });
307             $mylibrary_share.attr('data-entityid', shareIdArr);
308             $mylibrary_addto.attr('data-entityid', addToIdArr);
309             $mylibrary_addto.attr('data-entityname', addToTitleArr);
310             if (!shareIdArr.length) {
311                 $mylibrary_share.attr('disabled', 'disabled');
312             }
313         };

319         var addBinding = function() {
320             /**
321              * Enable/disable the remove selected button depending on whether
322              * any items in the library are checked
323              */
324             $mylibrary_check.live('change', function (ev) {
325                 if ($(this).is(':checked')) {
326                     $mylibrary_remove.removeAttr('disabled');
327                     $mylibrary_addto.removeAttr('disabled');
328                     $mylibrary_share.removeAttr('disabled');
329                 } else if (!$('.mylibrary_check:checked', $rootel).length) {
330                     $mylibrary_remove.attr('disabled', 'disabled');
331                     $mylibrary_addto.attr('disabled', 'disabled');
332                     $mylibrary_share.attr('disabled', 'disabled');
333                 }
334                 updateButtonData();
335             });
336 
337             /**
338              * Check/uncheck all loaded items in the library
339              */
340             $mylibrary_check_all.change(function (ev) {
341                 if ($(this).is(':checked')) {
342                     $('.mylibrary_check').attr('checked', 'checked');
343                     $mylibrary_remove.removeAttr('disabled');
344                     $mylibrary_addto.removeAttr('disabled');
345                     $mylibrary_share.removeAttr('disabled');
346                 } else {
347                     $('.mylibrary_check').removeAttr('checked');
348                     $mylibrary_remove.attr('disabled', 'disabled');
349                     $mylibrary_addto.attr('disabled', 'disabled');
350                     $mylibrary_share.attr('disabled', 'disabled');
351                 }
352                 updateButtonData();
353             });
354 
355             /**
356              * Gather all selected library items and initiate the
357              * deletecontent widget
358              */
359             $mylibrary_remove.click(function (ev) {
360                 var $checked = $('.mylibrary_check:checked:visible', $rootel);
361                 if ($checked.length) {
362                     var paths = [];
363                     var collectionPaths = [];
364                     $checked.each(function () {
365                         paths.push($(this).data('entityid'));
366                         if($checked.attr('data-type') === 'x-sakai/collection') {
367                             collectionPaths.push($(this).data('entityid'));
368                         }
369                     });
370                     $(document).trigger('init.deletecontent.sakai', [{
371                         paths: paths,
372                         context: mylibrary.contextId
373                     }, function (success) {
374                         if (success) {
375                             resetView();
376                             $(window).trigger('lhnav.updateCount', ['library', -(paths.length)]);
377                             mylibrary.infinityScroll.removeItems(paths);
378                             if(collectionPaths.length) {
379                                 $(document).trigger('sakai.mylibrary.deletedCollections', {
380                                     items: collectionPaths
381                                 });
382                             }
383                         }
384                     }]);
385                 }
386             });
387 
388             /**
389              * Called when clicking the remove icon next to an individual content
390              * item
391              */
392             $mylibrary_remove_icon.live('click', function(ev) {
393                 if ($(this).attr('data-entityid')) {
394                     var paths = [];
395                     var collection = false;
396                     if($(this).attr('data-type') === 'x-sakai/collection') {
397                         collection = true;
398                     }
399                     paths.push($(this).attr('data-entityid'));
400                     $(document).trigger('init.deletecontent.sakai', [{
401                         paths: paths,
402                         context: mylibrary.contextId
403                     }, function (success) {
404                         if (success) {
405                             resetView();
406                             $(window).trigger('lhnav.updateCount', ['library', -(paths.length)]);
407                             if(collection) {
408                                 $(document).trigger('sakai.mylibrary.deletedCollections', {
409                                     items: paths
410                                 });
411                             }
412                             mylibrary.infinityScroll.removeItems(paths);
413                         }
414                     }]);
415                 }
416             });
417 
418             /**
419              * Reload the library list based on the newly selected
420              * sort option
421              */
422             $mylibrary_sortby.change(function (ev) {
423                 var query = $.trim($mylibrary_livefilter.val());
424                 var sortSelection = $(this).val();
425                 if (sortSelection === 'desc') {
426                     mylibrary.sortOrder = 'desc';
427                     mylibrary.sortBy = 'filename';
428                 } else if (sortSelection === 'asc') {
429                     mylibrary.sortOrder = 'asc';
430                     mylibrary.sortBy = 'filename';
431                 } else {
432                     mylibrary.sortOrder = 'modified';
433                     mylibrary.sortBy = '_lastModified';
434                 }
435                 $.bbq.pushState({'lsb': mylibrary.sortBy, 'lso': mylibrary.sortOrder, 'lq': query});
436             });
437 
438             /**
439              * Initiate a library search when the enter key is pressed
440              */
441             $mylibrary_livefilter.keyup(function (ev) {
442                 if (ev.keyCode === 13) {
443                     doSearch();
444                 }
445                 return false;
446             });
447 
448             /**
449              * Initiate a search when the search button next to the search
450              * field is pressed
451              */
452             $mylibrary_search_button.click(doSearch);
453 
454             /**
455              * An event to listen from the worldsettings dialog so that we can refresh the title if it's been changed.
456              * @param {String} title     New group name
457              */
458             $(window).on('updatedTitle.worldsettings.sakai', function(e, title) {
459                 renderLibraryTitle(title, true);
460             });
461 
462             /**
463              * Listen for newly the newly added content or newly saved content
464              * @param {Object} data        Object that contains the new library items
465              * @param {Object} library     Context id of the library the content has been added to
466              */
467             $(document).on('done.newaddcontent.sakai', function(e, data, library) {
468                 if (library === mylibrary.contextId || mylibrary.contextId === sakai.data.me.user.userid) {
469                     mylibrary.infinityScroll.prependItems(data);
470                 }
471             });
472 
473             /**
474              * Keep track as to whether the current library widget is visible or not. If the
475              * widget is not visible, it's not necessary to respond to hash change events.
476              */
477             $(window).bind(tuid + '.shown.sakai', function(e, shown) {
478                 mylibrary.widgetShown = shown;
479             });
480             /**
481              * Bind to hash changes in the URL
482              */
483             $(window).bind('hashchange', handleHashChange);
484 
485             $mylibrary_show_list.click(function() {
486                 $.bbq.pushState({'ls': 'list'});
487             });
488 
489             $mylibrary_show_grid.click(function() {
490                 $.bbq.pushState({'ls': 'grid'});
491             });
492         };

494         ////////////////////////////////////////////
495         // Data retrieval and rendering functions //
496         ////////////////////////////////////////////
497 
498         /**
499          * Process library item results from the server
500          */
501         var handleLibraryItems = function (results, callback) {
502             sakai.api.Content.prepareContentForRender(results, sakai.data.me, function(contentResults) {
503                 if (contentResults.length > 0) {
504                     callback(contentResults);
505                     showHideTopControls(true);
506                     $mylibrary_empty.hide();
507                     $mylibrary_items.show();
508                 } else {
509                     callback([]);
510                 }
511             });
512         };
513 
514         /////////////////////////////
515         // Initialization function //
516         /////////////////////////////
517 
518         var initGroupLibrary = function() {
519             sakai.api.Server.loadJSON('/system/userManager/group/' +  mylibrary.contextId + '.json', function(success, data) {
520                 if (success) {
521                     currentGroup = data;
522                     var contextName = currentGroup.properties['sakai:group-title'];
523                     mylibrary.isOwnerViewing = sakai.api.Groups.isCurrentUserAManager(currentGroup.properties['sakai:group-id'], sakai.data.me, currentGroup.properties);
524                     mylibrary.isMemberViewing = sakai.api.Groups.isCurrentUserAMember(currentGroup.properties['sakai:group-id'], sakai.data.me);
525                     finishInit(contextName, true);
526                 }
527             });
528         };
529 
530         var initUserLibrary = function() {
531             mylibrary.contextId = sakai_global.profile.main.data.userid;
532             var contextName = sakai.api.User.getFirstName(sakai_global.profile.main.data);
533             if (mylibrary.contextId === sakai.data.me.user.userid) {
534                 mylibrary.isOwnerViewing = true;
535             }
536             finishInit(contextName, false);
537         };
538 
539         /**
540          * Initialization function that is run when the widget is loaded. Determines
541          * which mode the widget is in (settings or main), loads the necessary data
542          * and shows the correct view.
543          */
544         var doInit = function () {
545             mylibrary.contextId = '';
546 
547             // We embed the deletecontent widget, so make sure it's loaded
548             sakai.api.Widgets.widgetLoader.insertWidgets(tuid, false);
549 
550             if (widgetData && widgetData.mylibrary) {
551                 mylibrary.contextId = widgetData.mylibrary.groupid;
552                 initGroupLibrary();
553             } else if (sakai_global.group && sakai_global.group.groupId) {
554                 mylibrary.contextId = sakai_global.group.groupId;
555                 initGroupLibrary();
556             } else {
557                 initUserLibrary();
558             }
559         };
560 
561         /**
562          * Additional set-up of the widget
563          */
564         var finishInit = function(contextName, isGroup) {
565             if (mylibrary.contextId) {
566                 mylibrary.default_search_text = getPersonalizedText('SEARCH_YOUR_LIBRARY');
567                 mylibrary.currentPagenum = 1;
568                 mylibrary.listStyle = $.bbq.getState('ls') || 'list';
569                 handleHashChange(null, true);
570                 renderLibraryTitle(contextName, isGroup);
571             } else {
572                 debug.warn('No user found for My Library');
573             }
574             addBinding();
575         };
576 
577         // run the initialization function when the widget object loads
578         doInit();
579 
580     };
581 
582     // inform Sakai OAE that this widget has loaded and is ready to run
583     sakai.api.Widgets.widgetLoader.informOnLoad('mylibrary');
584 });
