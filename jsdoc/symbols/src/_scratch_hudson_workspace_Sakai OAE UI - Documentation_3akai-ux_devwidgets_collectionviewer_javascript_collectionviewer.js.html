 75         /**
 76          * Checks the format of a collectionId and returns the pool ID
 77          * @return {String} collectionId returns the collectionID as a String

 79         var getCollectionId = function(collectionId) {

 89          * Adds binding for the carousel actions and buttons

 91         var carouselBinding = function(carousel) {
 92             $('#collectionviewer_newer', $rootel).live('click',function() {
 93                 carousel.prev();
 94                 $(this).focus();
 95             });
 96             $('#collectionviewer_older', $rootel).live('click',function() {
 97                 $(this).focus();
 98                 carousel.next();
 99             });
100             $('#collectionviewer_oldest', $rootel).live('click',function() {
101                 $(this).focus();
102                 carousel.scroll(carousel.size() || 0);
103             });
104             $('#collectionviewer_newest', $rootel).live('click',function() {
105                 $(this).focus();
106                 carousel.scroll(0);
107             });
108             $rootel.keyup(function(e) {
109                 if (e.which === $.ui.keyCode.LEFT) {
110                     carousel.prev();
111                 }else if (e.which === $.ui.keyCode.RIGHT) {
112                     carousel.next();
113                 }
114             });
115             if (carousel.size()) {
116                 var $selectedItem =  $('.collectionviewer_carousel_item[data-item-id="' + $.bbq.getState("item") + '"]', $rootel);
117                 if ($.bbq.getState('item') && $selectedItem.length) {
118                     $selectedItem.click();
119                 } else{
120                     loadCollectionItem($('.collectionviewer_carousel_item:first', $rootel).attr('data-item-id'));
121                 }
122             }
123         };

125         /**
126         * Renders the carousel on the page and initializes it
127         */
128         var renderCarousel = function() {
129             sakai.api.Util.TemplateRenderer('collectionviewer_carousel_template', {
130                 data: collectionData,
131                 sakai: sakai,
132                 collectionName: getCollectionName(),
133                 collectionId: getCollectionId(collectionviewer.contextId),
134                 isEditor: sakai.api.Content.Collections.canCurrentUserEditCollection(collectionviewer.contextId)
135             }, $collectionviewerCarouselContainer);
136             $('#collectionviewer_finish_editing_collection_button', $rootel).hide();
137             if (sakai.api.Content.Collections.canCurrentUserEditCollection(collectionviewer.contextId)) {
138                 $('#collectionviewer_edit_collection_button', $rootel).show();
139             }
140             $collectionviewerCarouselContainer.animate({
141                 height: 'toggle',
142                 opacity: 'toggle'
143             }, 500);
144             $collectionviewerExpandedContentContainer.animate({
145                 height: 'toggle',
146                 opacity: 'toggle'
147             }, 500);
148             $('.collectionviewer_controls', $rootel).hide();
149             if (collectionData.length) {
150                 var totalItems = 0;
151                 $.each(collectionData, function(index, item) {
152                     if (item) {
153                         totalItems += item.length;
154                     }
155                 });
156                 if (totalItems > carouselSize) {
157                     $('.collectionviewer_controls', $rootel).show();
158                 }
159                 $('#collectionviewer_carousel', $rootel).jcarousel({
160                     animation: 'slow',
161                     easing: 'swing',
162                     scroll: carouselSize,
163                     start: 0,
164                     initCallback: carouselBinding,
165                     itemFallbackDimension: 123
166                 });
167 
168                 if (totalItems > carouselSize && $.bbq.getState('item')) {
169                     $('#collectionviewer_carousel', $rootel).data('jcarousel').scroll($('.collectionviewer_carousel_item[data-item-id="' + $.bbq.getState('item') + '"]', $rootel).attr('data-arr-index'));
170                 }
171             }
172         };

174         /**
175          * Renders the items for a selected item in the carousel
176          * @param {String} pageIndex index of the page you're currently on in the collection viewer
177          * @param {String} selectedIndex index of the selected item in the carousel
178          */
179         var renderItemsForSelected = function(pageIndex, selectedIndex) {
180             if (!isNaN(pageIndex)) {
181                 var selectedData = collectionData[pageIndex][selectedIndex];
182                 if (selectedData._mimeType === 'x-sakai/collection') {
183                     getCollectionData('c-' + selectedData._path, false, function(data) {
184                         if (data.results.fetchMultipleUserDataInWidget) {
185                             delete data.results.fetchMultipleUserDataInWidget;
186                         }
187                         selectedData.collectionItems = data.results;
188                         sakai.api.Util.TemplateRenderer('collectionviewer_list_item_template', {
189                             data: selectedData,
190                             sakai: sakai,
191                             collectionName: getCollectionName(),
192                             collectionId: getCollectionId(collectionviewer.contextId),
193                             isEditor: sakai.api.Content.Collections.canCurrentUserEditCollection(collectionviewer.contextId),
194                             pagePreviewDisabled: pagePreviewDisabled
195                         }, $('#collectionviewer_expanded_content_container', $rootel));
196                         if (previewsAllowed) {
197                             sakai.api.Widgets.widgetLoader.insertWidgets(tuid);
198                         }
199                     });
200                 } else {
201                     sakai.api.Util.TemplateRenderer('collectionviewer_list_item_template', {
202                         data: selectedData,
203                         sakai: sakai,
204                         collectionName: getCollectionName(),
205                         collectionId: getCollectionId(collectionviewer.contextId),
206                         isEditor: sakai.api.Content.Collections.canCurrentUserEditCollection(collectionviewer.contextId),
207                         pagePreviewDisabled: pagePreviewDisabled
208                     }, $('#collectionviewer_expanded_content_container', $rootel));
209                     if (previewsAllowed) {
210                         sakai.api.Widgets.widgetLoader.insertWidgets(tuid);
211                     }
212                 }
213             }

241                 items: collectionData[pageNumber],
242                 sakai: sakai,
243                 grid: grid,
244                 editMode: editMode,
245                 collectionName: getCollectionName(),
246                 collectionId: getCollectionId(collectionviewer.contextId),
247                 isEditor: sakai.api.Content.Collections.canCurrentUserEditCollection(collectionviewer.contextId)

252                 $('#collectionviewer_paging', $rootel).show();
253                 $('#collectionviewer_paging', $rootel).pager({
254                     pagenumber: parseInt(collectionviewer.page, 10),
255                     pagecount: Math.ceil(collectionviewer.total / listSize),

259                         $.bbq.pushState({'lp': collectionviewer.page});
260                     }
261                 });
262             } else {
263                 $('#collectionviewer_paging', $rootel).hide();
264             }

310         /**
311          * Gets the profile data for multiple users
312          * @param {Object} data Object containing data for the user profiles to look up
313          * @param {Function} callback Function to be executed on retrieval of the user profiles
314          */
315         var getMultipleUserData = function(data, callback) {
316             var usersToFetch = [];
317             $.each(data.results, function(i, user) {
318                 if (user['sakai:pool-content-created-for']) {
319                     usersToFetch.push(user['sakai:pool-content-created-for']);
320                 }
321             });
322 
323             sakai.api.User.getMultipleUsers(usersToFetch, function(fetchedUsers) {
324                 $.each(data.results, function(index, item) {
325                     var userid = item['sakai:pool-content-created-for'];
326                     if (userid && fetchedUsers[userid]) {
327                         var displayName = sakai.api.User.getDisplayName(fetchedUsers[userid]);
328                         data.results[index].ownerId = userid;
329                         data.results[index].ownerDisplayName = displayName;
330                         data.results[index].ownerDisplayNameShort = sakai.api.Util.applyThreeDots(displayName, 580, {
331                             max_rows: 1,
332                             whole_word: false
333                         }, 's3d-bold', true);
334                         data.results[index].ownerDisplayNameShorter = sakai.api.Util.applyThreeDots(displayName, 180, {
335                             max_rows: 1,
336                             whole_word: false
337                         }, 's3d-bold', true);
338                     }
339                 });
340                 if ($.isFunction(callback)) {
341                     callback();
342                 }
343             }, false);
344         };

346         /**
347          * Retrieves the basic data for items in a collection
348          * @param {String} userid ID of the collection to retrieve data for, if empty cache will be checked for ID
349          * @param {Boolean} refresh Reloads the collection interface if set to true
350          * @param {Function} callback Function executed after the data has been retrieved
351          */
352         var getCollectionData = function(userid, refresh, callback) {
353             toggleButtons(collectionviewer.listStyle);
354             if (refresh) {
355                 collectionviewer.page = $.bbq.getState('lp') || 1;
356                 collectionData = [];
357             }
358             var id;
359             if (userid) {
360                 id = userid;
361             } else if (collectionviewer.contextId.substring(0, 2) === 'c-') {
362                 id = collectionviewer.contextId;
363             } else {
364                 id = 'c-' + collectionviewer.contextId;
365             }
366             var data = {
367                 sortOn: 'filename',
368                 sortOrder: collectionviewer.sortOrder,
369                 userid: id,
370                 items: 15,
371                 page: (collectionviewer.page - 1)
372             };
373             if (collectionviewer.sortOrder === 'modified') {
374                 data.sortOrder = 'desc';
375                 data.sortOn = '_lastModified';
376             }
377             if (collectionviewer.listStyle === 'carousel') {
378                 data.items = 1000;
379                 data.page = 0;
380             }
381             if (!refresh && !$.isFunction(callback)) {
382                 $collectionviewerCarouselLoading.show();
383             }
384             $.ajax({
385                 url: sakai.config.URL.POOLED_CONTENT_SPECIFIC_USER,
386                 data: data,
387                 success: function(data) {
388                     if ($.isFunction(callback)) {
389                         getMultipleUserData(data, function() {
390                             data.results.fetchMultipleUserDataInWidget = true;
391                             sakai.api.Content.prepareContentForRender(data.results, sakai.data.me, function(parsedContent) {
392                                 callback(data);
393                             });
394                         });
395                     } else {
396                         $('#collectionviewer_add_content_button > div', $rootel).text(data.total);
397                         collectionviewer.total = data.total;
398                         if (data.results && data.results.length) {
399                             getMultipleUserData(data, function() {
400                                 data.results.fetchMultipleUserDataInWidget = true;
401                                 sakai.api.Content.prepareContentForRender(data.results, sakai.data.me, function(parsedContent) {
402                                     collectionData[(collectionviewer.page - 1)] = parsedContent;
403                                     showData();
404                                 });
405                             });
406                         } else {
407                             showData();
408                         }
409                     }
410                 },
411                 error: function(xhr, textStatus, thrownError) {
412                     $collectionviewerCarouselLoading.hide();
413                 }
414             });
415         };

507         /**
508          * Refreshes the collection viewer widget
509          */
510         var refreshCollection = function() {
511             var pageNumber = collectionviewer.page - 1;
512             getCollectionData('', true, function(data) {
513                 collectionviewer.listStyle = $.bbq.getState(collectionviewer.tuidls) || 'list';
514                 $('#collectionviewer_add_content_button > div', $rootel).text(data.total);
515                 collectionviewer.total = data.total;
516                 if (data.results.length) {
517                     collectionData[pageNumber] = data.results;
518                 } else {
519                     delete collectionData[pageNumber];
520                 }
521                 renderGridOrList(false, true);
522                 sakai.api.Util.progressIndicator.hideProgressIndicator();
523             });

526         /**
527          * Initializes the widget preview
528          * @param {String} which Can be 'collectioncontentpreview' or 'pageviewer' depending
529          *                 on the context of the preview
530          */
531         var doStart = function(which) {
532             if ($rootel.is(':visible')) {
533                 var arrIndex1 = 0;
534                 if ($('.collectionviewer_carousel_item.selected', $rootel).length) {
535                     arrIndex1 = parseInt($('.collectionviewer_carousel_item.selected', $rootel).attr('data-page-index'), 10);
536                 }
537                 var arrIndex2 = 0;
538                 if ($('.collectionviewer_carousel_item.selected', $rootel).length) {
539                     arrIndex2 = parseInt($('.collectionviewer_carousel_item.selected', $rootel).attr('data-arr-index'), 10);
540                 }
541                 if (which === 'collectioncontentpreview' && collectionviewer.listStyle === 'carousel') {
542                     if (collectionData.length && collectionData[arrIndex1] && collectionData[arrIndex1][arrIndex2]) {
543                         $('.collectionviewer_widget', $rootel).trigger('start.collectioncontentpreview.sakai', collectionData[arrIndex1][arrIndex2]);
544                         $('.collectionviewer_collection_item_preview', $rootel).show();
545                     }
546                 } else if (which === 'pageviewer') {
547                     if (collectionData.length && collectionData[arrIndex1] && collectionData[arrIndex1][arrIndex2]) {
548                         $(window).trigger('start.pageviewer.sakai', collectionData[arrIndex1][arrIndex2]);
549                     }
550                 }
551             }
552         };

568                 $.bbq.pushState(state);
569             });

572             $rootel.on('click', '.collectionviewer_carousel_item', function() {
573                 if ($(this).hasClass('selected')) {
574                     return;
575                 }
576                 if (collectionviewer.listStyle === 'carousel') {
577                     $.bbq.pushState({'item': $(this).attr('data-item-id')});
578                     fetchCollectionData = false;
579                 }
580                 if (initialload) {
581                     initialload = false;
582                     handleHashChange();
583                 }
584             });
585 
586             $rootel.on('click', '.collectionviewer_comments_button', showComments);
587 
588             $rootel.on('ready.collectioncontentpreview.sakai', '.collectionviewer_widget', function() {
589                 doStart('collectioncontentpreview');
590             });
591 
592             $rootel.on('change', '#collectionviewer_sortby', function() {
593                 var sortSelection = $(this).val();
594                 var state = {
595                     item: ''
596                 };
597                 if (sortSelection === 'desc') {
598                     collectionviewer.sortOrder = 'desc';
599                     state[collectionviewer.tuidso] = 'desc';
600                     $.bbq.pushState(state);
601                 } else if (sortSelection === 'asc') {
602                     collectionviewer.sortOrder = 'asc';
603                     state[collectionviewer.tuidso] = 'asc';
604                     $.bbq.pushState(state);
605                 } else {
606                     collectionviewer.sortOrder = 'modified';
607                     state[collectionviewer.tuidso] = 'modified';
608                     $.bbq.pushState(state);
609                 }
610             });
611 
612             $rootel.on('click', '.collectionviewer_collection_item_comments #contentcomments_postComment', function() {
613                 collectionData[parseInt($('.collectionviewer_carousel_item.selected', $rootel).attr('data-page-index'),10)][parseInt($('.collectionviewer_carousel_item.selected', $rootel).attr('data-arr-index'), 10)].numComments++;
614                 $('.collectionviewer_comments_count', $rootel).text(collectionData[parseInt($('.collectionviewer_carousel_item.selected', $rootel).attr('data-page-index'), 10)][parseInt($('.collectionviewer_carousel_item.selected', $rootel).attr('data-arr-index'), 10)].numComments);
615             });
616 
617             $rootel.on('click', '#collectionviewer_finish_editing_collection_button', function() {
618                 $('#collectionviewer_expanded_content_container').empty();
619                 $(this).hide();
620                 $('#collectionviewer_edit_collection_button', $rootel).show();
621                 var state = {};
622                 state[collectionviewer.tuidls] = 'carousel';
623                 $.bbq.pushState(state);
624             });
625

707         /**
708          * Initialize the widget by adding bindings to elements and gathering collection information
709          */
710         var doInit = function() {
711             collectionviewer.listStyle = $.bbq.getState(collectionviewer.tuidls) || 'carousel';
712             collectionviewer.sortOrder = $.bbq.getState(collectionviewer.tuidso) || 'modified';
713             var ref = '';
714             if (widgetData.data && widgetData.data._path) {
715                 collectionviewer.contextId = widgetData.data._path;
716                 if (widgetData.data.structure0) {
717                     ref = $.parseJSON(widgetData.data.structure0).main._ref;
718                 }
719                 decidePagePreviewDisabled(collectionviewer.contextId);
720             } else {
721                 collectionviewer.contextId = widgetData.collectionviewer.groupid;
722                 decidePagePreviewDisabled(collectionviewer.contextId.slice(2, collectionviewer.contextId.length));
723             }
724             if ($rootel.parents('.pageviewer_widget').length) {
725                 previewsAllowed = false;
726             }
727             $('.collectionviewer_widget', $rootel).show();
728             if (sakai.api.Content.Collections.canCurrentUserEditCollection(collectionviewer.contextId)) {
729                 $('#collectionviewer_header_container #collectionviewer_add_content_button', $rootel).show();
730                 $('#collectionviewer_header_container #collectionviewer_edit_collection_button', $rootel).show();
731                 $('#collectionviewer_finish_editing_collection_button', $rootel).hide();
732             }
733 
734             $('#content_profile_sakaidoc_container', $rootel).addClass('collections');
735             $('#collectionviewer_sortby', $rootel).val(collectionviewer.sortOrder);
736 
737             if (sakai_global && sakai_global.content_profile && sakai_global.content_profile.content_data) {
738                 if (widgetData.data && widgetData.data['sakai:pooled-content-file-name']) {
739                     collectionviewer.collectionName = widgetData.data['sakai:pooled-content-file-name'];
740                 } else {
741                     collectionviewer.collectionName = sakai_global.content_profile.content_data.data['sakai:pooled-content-file-name'];
742                 }
743                 switchListView();
744             // Retrieve the name of the collection as we're not in a content profile page
745             } else {
746                 $.ajax({
747                     url: '/p/' + collectionviewer.contextId + '.json',
748                     success: function(data) {
749                         collectionviewer.collectionName = data['sakai:pooled-content-file-name'];
750                         switchListView();
751                     }
752                 });
753             }
754             addBinding();
755         };

757         doInit();
