 25 require(["jquery", "sakai/sakai.api.core"], function($, sakai) {
 26 
 27     /**
 28      * sakai_global.tooltip
 29      * a multi-purpose tooltip dialog box
 30      */
 31     sakai_global.tooltip = function(tuid, showSettings) {
 32         var tooltipOnShow = false,
 33             tooltipAutoClose = null,
 34             tooltipArrow = null,
 35             tooltipTop = null,
 36             tooltipLeft = null;
 37 
 38         var $rootel = $("#" + tuid);
 39         var $tooltip_widget = $("#tooltip_widget", $rootel),
 40             $tooltip_content = $(".tooltip_content", $rootel),
 41             $tooltip_close = $(".tooltip_close", $rootel),
 42             $tooltip_dialog = $(".tooltip_dialog", $rootel),
 43             $tooltip_title = $("#tooltip_title", $rootel),
 44             $tooltip_description = $("#tooltip_description", $rootel),
 45             $tooltip_header_arrow = $(".dialog_tooltip_header_arrow", $rootel),
 46             $tooltip_left_arrow = $(".dialog_tooltip_left_arrow", $rootel),
 47             $tooltip_footer_arrow = $(".dialog_tooltip_footer_arrow", $rootel);
 48 
 49         $tooltip_widget.jqm({
 50             modal: false,
 51             overlay: 0,
 52             zIndex: 6000,
 53             toTop: true
 54         });
 55 
 56         var hideTooltip = function() {
 57             $tooltip_widget.jqmHide();
 58             $(window).trigger("closed.tooltip.sakai");
 59             $(window).unbind("update.tooltip.sakai");
 60             $(document).unbind("click.tooltip_close");
 61         };
 62 
 63         /**
 64          * toggleTooltip sets tooltip configuration and displays the tooltip
 65          */
 66         var toggleTooltip = function() {
 67             // bind auto close of tooltip on outside mouse click
 68             if (tooltipAutoClose) {
 69                 $(document).unbind("click.tooltip_close").bind("click.tooltip_close", function (e) {
 70                     var $clicked = $(e.target);
 71                     // Check if one of the parents is the tooltip
 72                     if (!$clicked.parents().is("#tooltip") && $tooltip_widget.is(":visible")) {
 73                         hideTooltip();
 74                     }
 75                 });
 76             }
 77 
 78             // position tooltip and display directional arrow
 79             var topOffset = tooltipTop - 3;
 80             var leftOffset = tooltipLeft - 42;
 81             $tooltip_header_arrow.hide();
 82             $tooltip_footer_arrow.hide();
 83 
 84             if (tooltipArrow === "bottom"){
 85                 topOffset = "auto";
 86                 $tooltip_footer_arrow.show();
 87             } else if (tooltipArrow === "top"){
 88                 $tooltip_header_arrow.show();
 89             }
 90             $tooltip_widget.css({
 91                 top: topOffset,
 92                 left: leftOffset
 93             });
 94             $tooltip_content.html(tooltipHTML);
 95             $tooltip_widget.jqmShow();
 96             if (tooltipOnShow && typeof(tooltipOnShow) === "function") {
 97                 tooltipOnShow();
 98             }
 99 
100             // bind tooltip movement
101             $(window).unbind("update.tooltip.sakai");
102             $(window).bind("update.tooltip.sakai", function(e, tooltipData) {
103                 hideTooltip();
104                 $(document).trigger('init.tooltip.sakai', tooltipData);
105             });
106             // bind tooltip close
107             $(window).unbind("done.tooltip.sakai");
108             $(window).bind("done.tooltip.sakai", function() {
109                 hideTooltip();
110             });
111             $(".tooltip_close").unbind("click");
112             $(".tooltip_close").bind("click", function () {
113                 $(window).unbind(".tooltip_close");
114                 hideTooltip();
115                 return false;
116             });
117         };
118 
119         $tooltip_close.bind("click", function() {
120             hideTooltip();
121         });
122 
123         /* Renders a tooltip.  The tooltipConfig param should contain:
124          * {String} tooltipHTML  a <div> containing all the HTML you want to display in the tool tip
125          * {Boolean} tooltipAutoClose  If we close the tooltip on an outside click
126          * {String} tooltipArrow  Direction for where the tooltip arrow is placed ("top" or "bottom")
127          * {Integer} tooltipLeft  absolute position of where the tooltip should spawn: X value (jQuery.Event.pageX)
128          * {Integer} tooltipTop   absolute position of where the tooltip should spawn: Y value (jQuery.Event.pageY)
129          * {Function} onShow      callback called when the tooltip is shown
130          */
131         $(document).on('init.tooltip.sakai', function(e, tooltipConfig) {
132             if (tooltipConfig) {
133                 tooltipHTML = tooltipConfig.tooltipHTML || false;
134                 tooltipAutoClose = tooltipConfig.tooltipAutoClose || false;
135                 tooltipArrow = tooltipConfig.tooltipArrow || false;
136                 tooltipTop = tooltipConfig.tooltipTop || false;
137                 tooltipLeft = tooltipConfig.tooltipLeft || false;
138                 tooltipOnShow = tooltipConfig.onShow || false;
139                 toggleTooltip();
140             } else {
141                 debug.error("No tooltip mode specifed");
142             }
143         });
144 
145     };
146 
147     sakai.api.Widgets.widgetLoader.informOnLoad("tooltip");
148 });
