 90         /**
 91          * Updates the global content object with the changed membership data
 92          * @param {String} authId The ID for the user/group that has changed membership
 93          * @param {String} oldPermission The users old permission
 94          * @param {String} newPermission The users new permission (optional)
 95          */
 96         var updateContentPermissionData = function(authId, oldPermission, newPermission) {
 97             var userData = false;
 98             var index = false;
 99             if (oldPermission !== newPermission && sakai_global.content_profile.content_data.members[oldPermission]) {
100                 $.each(sakai_global.content_profile.content_data.members[oldPermission], function(i, userObject) {
101                     if (userObject.userid && userObject.userid === authId) {
102                         index = i;
103                         userData = userObject;
104                     } else if (userObject.groupid && userObject.groupid === authId) {
105                         index = i;
106                         userData = userObject;
107                     }
108                 });
109                 if (userData && sakai_global.content_profile.content_data.members[oldPermission]) {
110                     sakai_global.content_profile.content_data.members[oldPermission].splice(index, 1);
111                     if (newPermission && sakai_global.content_profile.content_data.members[newPermission]) {
112                         sakai_global.content_profile.content_data.members[newPermission].push(userData);
113                     }
114                 }
115             }

118         /**
119          * Saves permissions for each individual member in the list of members
120          */
121         var saveMemberPermissions = function(){
122             var permissionBatch = [];
123             var atLeastOneManager = false;
124             var savePermissions = false;
125             var permissionsToUpdate = [];
126 
127             if (sakai.api.Content.Collections.isCollection(sakai_global.content_profile.content_data.data)){
128                 var groupID = sakai.api.Content.Collections.getCollectionGroupId(sakai_global.content_profile.content_data.data);
129                 $(contentPermissionsEditList).each(function(index, item){
130                     var newPermission = $(item).children(contentpermissionsMemberPermissions).val();
131                     var oldPermission = $(item).attr('data-originalpermission');
132                     var userId = $(item).attr("id").split("_")[1];
133                     if (newPermission === "manager") {
134                         atLeastOneManager = true;
135                     }
136                     var addToCollectionPseudoGroup = 'members';
137                     var removeFromCollectionPseudoGroups = ['managers', 'editors'];
138                     if (newPermission === 'manager') {
139                         addToCollectionPseudoGroup = 'managers';
140                         removeFromCollectionPseudoGroups = ['editors', 'members'];
141                     } else if (newPermission === 'editor') {
142                         addToCollectionPseudoGroup = 'editors';
143                         removeFromCollectionPseudoGroups = ['managers', 'members'];
144                     }
145                     permissionBatch.push({
146                         'url': '/system/userManager/group/' + groupID + '-' + addToCollectionPseudoGroup + '.update.json',
147                         'method': 'POST',
148                         'parameters': {
149                             ':member': userId,
150                             ':viewer': userId
151                         }
152                     });
153                     $.each(removeFromCollectionPseudoGroups, function(idx, group) {
154                         permissionBatch.push({
155                             'url': '/system/userManager/group/' + groupID + '-' + group + '.update.json',
156                             'method': 'POST',
157                             'parameters': {
158                                 ':member@Delete': userId,
159                                 ':viewer@Delete': userId
160                             }
161                         });
162                     });
163                     permissionsToUpdate.push({
164                         'authId': userId,
165                         'oldPermission': oldPermission,
166                         'newPermission': newPermission + 's'
167                     });
168                 });
169             } else {
170                 $(contentPermissionsEditList).each(function(index, item){
171                     var newPermission = $(item).children(contentpermissionsMemberPermissions).val();
172                     var oldPermission = $(item).attr('data-originalpermission');
173                     var userId = $(item).attr("id").split("_")[1];
174                     if (newPermission === "manager") {
175                         atLeastOneManager = true;
176                         permissionBatch.push({
177                             "url": "/p/" + sakai_global.content_profile.content_data.data["_path"] + ".members.json",
178                             "method": "POST",
179                             "parameters": {
180                                 ":manager": userId,
181                                 ':editor@Delete': userId,
182                                 ":viewer@Delete": userId
183                             }
184                         });
185                     } else if (newPermission === 'editor') {
186                         permissionBatch.push({
187                             'url': '/p/' + sakai_global.content_profile.content_data.data['_path'] + '.members.json',
188                             'method': 'POST',
189                             'parameters': {
190                                 ':editor': userId,
191                                 ':viewer@Delete': userId,
192                                 ':manager@Delete': userId
193                             }
194                         });
195                     } else {
196                         permissionBatch.push({
197                             "url": "/p/" + sakai_global.content_profile.content_data.data["_path"] + ".members.json",
198                             "method": "POST",
199                             "parameters": {
200                                 ":viewer": userId,
201                                 ':editor@Delete': userId,
202                                 ":manager@Delete": userId
203                             }
204                         });
205                     }
206                     permissionsToUpdate.push({
207                         'authId': userId,
208                         'oldPermission': oldPermission,
209                         'newPermission': newPermission + 's'
210                     });
211                 });
212             }
213 
214             if (atLeastOneManager) {
215                 // Do the Batch request
216                 sakai.api.Server.batch(permissionBatch, function(success, data){
217                     if(globalPermissionsChanged){
218                         if (sakai.api.Content.Collections.isCollection(sakai_global.content_profile.content_data.data)) {
219                             sakai.api.Content.Collections.setCollectionPermissions(sakai_global.content_profile.content_data.data["_path"], sakai_global.content_profile.content_data.data["sakai:permissions"], finishSavePermissions);
220                         } else {
221                             sakai.api.Content.setFilePermissions([{
222                                 "hashpath": sakai_global.content_profile.content_data.data["_path"],
223                                 "permissions": sakai_global.content_profile.content_data.data["sakai:permissions"]
224                             }], finishSavePermissions);
225                         }
226                     } else {
227                         finishSavePermissions();
228                     }
229                     $.each(permissionsToUpdate, function(i, permissionData) {
230                         updateContentPermissionData(
231                             permissionData.authId,
232                             permissionData.oldPermission,
233                             permissionData.newPermission);
234                     });
235                 });
236             } else {
237                 if (!globalPermissionsChanged) {
238                     closeOverlay();
239                 } else {
240                     sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey('CANNOT_SAVE_SETTINGS', 'contentpermissions'),
241                         sakai.api.i18n.getValueForKey('THERE_SHOULD_BE_AT_LEAST_ONE_MANAGER', 'contentpermissions'));
242                 }
243             }
244         };

278         /**
279          * Deletes a user from the list of members by deleting the user from the content and removing permissions
280          */
281         var doDelete = function(){
282             var userid = $(this).data("sakai-entityid");
283             var originalpermission = $(this).parent().attr('data-originalpermission');
284             var manager = originalpermission === 'managers';
285             var editor = originalpermission === 'editors';
286             var $itemToDelete = $(this).parent();
287             var userToDelete = {};
288 
289             if (sakai_global.content_profile.content_data.members.managers.length > 1 || !manager) {
290 
291                 if (sakai.api.Content.Collections.isCollection(sakai_global.content_profile.content_data.data)) {
292                     var userObj = [
293                         {
294                             'permission': 'managers',
295                             'userid': userid
296                         },
297                         {
298                             'permission': 'editors',
299                             'userid': userid
300                         },
301                         {
302                             'permission': 'members',
303                             'userid': userid
304                         }
305                     ];
306                     sakai.api.Groups.removeUsersFromGroup(sakai.api.Content.Collections.getCollectionGroupId(sakai_global.content_profile.content_data.data), userObj, sakai.data.me, function() {
307                         $itemToDelete.remove();
308                     });
309                 } else {
310                     if (manager) {
311                         userToDelete = {
312                             ':manager@Delete': userid
313                         };
314                     } else if (editor) {
315                         userToDelete = {
316                             ':editor@Delete': userid
317                         };
318                     } else {
319                         userToDelete = {
320                             ':viewer@Delete': userid
321                         };
322                     }
323                 }
324 
325                 // Do the Batch request
326                 $.ajax({
327                     url: "/p/" + sakai_global.content_profile.content_data.data["_path"] + ".members.json",
328                     type: "POST",
329                     data: userToDelete,
330                     success: function(){
331                         $itemToDelete.remove();
332                         updateContentPermissionData(userid, originalpermission);
333                     }
334                 });
335                 changesMade = true;
336             } else {
337                 sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("CANNOT_DELETE_USERS", "contentpermissions"), sakai.api.i18n.getValueForKey("THERE_SHOULD_BE_AT_LEAST_ONE_MANAGER", "contentpermissions"));
338             }
339         };

341         /**
342          * Checks the data for duplicate users/groups and removes the duplicate; gives preference to the user/group with edit permissions
343          * n.b. duplicates were being introduced via the sharing widget which will be addressed, so this sanity check will ultimately be optional
344          * but might be good to have (if it doesn't affect performance) in case any new widgets make the same mistake...
345          */
346         var removeDuplicateUsersGroups = function(data){
347             if(!data || !data.members){
348                 return data;
349             }
350             var tmpManagers = {};
351             var managers = data.members.managers || [];
352             var ml = managers.length;
353             var tmpEditors = {};
354             var editors = data.members.editors || [];
355             var el = editors.length;
356             var tmpViewers = {};
357             var viewers = data.members.viewers || [];
358             var vl = viewers.length;
359             for (var i = 0; i < ml; i++) { //though unlikely, this will remove any duplicates within the manager permission
360                 tmpManagers[managers[i].userid || managers[i].groupid] = managers[i];
361             }
362             for (var k = 0; k < el; k++) { //if the editor is a manager, don't add them. Also removes duplicates within viewer permissions
363                 if (!tmpManagers[editors[k].userid || editors[k].groupid]) {
364                     tmpEditors[editors[k].userid || editors[k].groupid] = editors[k];
365                 }
366             }
367             for (var j = 0; j < vl; j++) { //if the viewer is a manager or editor, don't add them. Also removes duplicates within viewer permissions
368                 if (!tmpManagers[viewers[j].userid || viewers[j].groupid] && !tmpEditors[viewers[j].userid || viewers[j].groupid]) {
369                     tmpViewers[viewers[j].userid || viewers[j].groupid] = viewers[j];
370                 }
371             }
372             var sortById = function(a, b) {
373                 var x = (a.userid || a.groupid).toLowerCase();
374                 var y = (b.userid || b.groupid).toLowerCase();
375                 return ((x < y) ? -1 : ((x > y) ? 1 : 0));
376             };
377 
378             data.members.managers = _.toArray(tmpManagers).sort(sortById);
379             data.members.editors = _.toArray(tmpEditors).sort(sortById);
380             data.members.viewers = _.toArray(tmpViewers).sort(sortById);
381             return data;
382         };

384         /**
385          * Get the list of selected users/groups from the autosuggest plugin
386          * @return {Object} returnValue An object containing a list of displayNames and an Array of userID's to be added to the members list
387          */
388         var getSelectedList = function() {
389             var list = $("#as-values-" + tuid).val();
390             // this value is a comma-delimited list
391             // split it and get rid of any empty values in the array
392             list = list.split(",");
393             var removed = 0;
394             $(list).each(function(i, val) {
395                if (val === "") {
396                    list.splice(i - removed, 1);
397                    removed += 1;
398                }
399             });
400 
401             // Create list to show in the notification
402             var toAddNames = [];
403             $("#contentpermissions_container .as-selection-item").each(function(){
404                 // In IE 7 </A> is returned and in firefox </a>
405                 toAddNames.push($(this).html().split(/<\/[aA]?>/g)[1]);
406             });
407 
408             var returnValue = {"list":list, "toAddNames":toAddNames};
409 
410             return returnValue;
411         };

467         /**
468          * Share the piece of content with a user by adding the user to the list of members (manager or viewer)

471             $(window).unbind("ready.contentprofile.sakai", doInit);
472             $(window).bind("ready.contentprofile.sakai", doInit);
473             var userList = getSelectedList();
474             $(window).trigger("finished.sharecontent.sakai", [
475                 userList, $.trim($("#contentpermissions_members_autosuggest_text").val()), { 
476                     data: [{
477                         body: sakai_global.content_profile.content_data.data
478                     }]
479                 }, $('#contentpermissions_members_autosuggest_permissions').val()
480             ]);
481             changesMade = true;

484         /**
485          * Gets the list of member users and groups from sakai_global and returns an array of userIds/groupIds to pass to autosuggest for filtering out
486          * @return {Array} Array of users and groups to populate the autosuggest field with
487          */     
488         var autosuggestFilterUsersGroups = function(){
489             var filterlist = [];
490             var filterUsersGroups = sakai_global.content_profile.content_data.members.managers
491                 .concat(sakai_global.content_profile.content_data.members.viewers, sakai_global.content_profile.content_data.members.editors);
492             $.each(filterUsersGroups,function(i,val){
493                 filterlist.push(val.userid || val.groupid);
494             });
495             return filterlist;
496         };
