155         /**
156          * Show the comments in a paged state or not
157          */
158         var displayCommentsPagedOrNot = function(){
159             jsonDisplay = {
160                 "comments": [],
161                 "settings": widgetSettings
162             };
163             // Loops through all the comments and does the necessary changes to render the JSON-object
164             for (var i = 0; i < json.results.length; i++) {
165                 jsonDisplay.comments[i] = {};
166                 var comment = json.results[i].post;
167                 // Checks if the date is already parsed to a date object
168                 var tempDate = sakai.api.Util.parseSakaiDate(comment['sakai:created']);
169                 if (isNaN(parseDate(comment['sakai:created']).getTime())) {
170                     tempDate = sakai.api.Util.parseSakaiDate(
171                                    sakai.api.Util.createSakaiDate(new Date(comment['sakai:created'])));
172                 }
173                 try {
174                     // if the date is not a string this should generate en exception
175                     comment.date = parseDate(tempDate);
176                 } catch (ex) {
177                     comment.date = tempDate;
178                 }
179                 comment.timeAgo = $.timeago(tempDate);
180                 // Use the sakai API function to parse the date and convert to the users local time
181                 comment.date = parseDate(tempDate.getTime(), sakai.data.me);
182                 comment.formatDate = sakai.api.l10n.transformDateTimeShort(comment.date);
183                 comment.messageTxt = comment["sakai:body"];
184                 comment.message = tidyInput(comment["sakai:body"]);
185                 // weird json bug.
186                 comment["sakai:deleted"] = (comment["sakai:deleted"] && (comment["sakai:deleted"] === "true" || comment["sakai:deleted"] === true)) ? true : false;
187 
188                 var user = {};
189                 // User
190                 // Puts the userinformation in a better structure for trimpath
191                 if (comment.profile) {
192                     var profile = comment.profile[0];
193                     user.fullName = sakai.api.User.getDisplayName(profile);
194                     user.uid = profile["userid"];
195                     user.pictureUrl = sakai.config.URL.USER_DEFAULT_ICON_URL;
196                     // Check if the user has a picture
197                     var pictureUrl = sakai.api.Util.constructProfilePicture(profile);
198                     if (pictureUrl){
199                         user.pictureUrl = pictureUrl;
200                     }
201                     user.profile = "/~" + sakai.api.Util.safeURL(user.uid);
202                 }
203                 else {
204                     // This is an anonymous user.
205                     comment.profile = {};
206                     comment.profile.fullName = "Anonymous";
207                     comment.profile.email = "noreply@sakaiproject.org";
208                     if (widgetSettings["sakai:forcename"] === true) {
209                         comment.profile.fullName = comment['sakai:name'];
210                     }
211                     if (widgetSettings["sakai:forcemail"] === true) {
212                         comment.profile.email = comment['sakai:email'];
213                     }
214                 }
215 
216                 comment.user = user;
217 
218                 jsonDisplay.comments[i] = comment;
219             }
220             $(commentsShowComments, rootel).html(sakai.api.Util.TemplateRenderer(commentsShowCommentsTemplate, jsonDisplay));
221             // Render Math formulas in the text
222             sakai.api.Util.renderMath(tuid);
223         };

225         /**
226          * Show all the posted comments
227          * This function first retrieves all the users who have posted in this widget and then call the displayCommentsPagedOrNot function
228          */
229         var showComments = function(){
230             // Show the nr of comments we are showing.
231             var showingComments = json.total;
232             if (widgetSettings.perPage < json.total) {
233                 showingComments = widgetSettings.perPage + extraComments;
234             }
235             $(commentsNumCommentsDisplayed, rootel).html(showingComments);
236             // Puts the number of comments on the page
237             $(commentsNumComments, rootel).html(json.total);
238             // Change to "comment" or "comments"
239             if (json.total === 1) {
240                 $(commentsCommentComments, rootel).text(sakai.api.i18n.getValueForKey("COMMENT_LC", "comments"));
241             }
242 
243 
244             // Change the page-number on the display
245             $(commentsPager, rootel).pager({
246                 pagenumber: clickedPage,
247                 pagecount: Math.ceil(json.total / widgetSettings.perPage),
248                 buttonClickCallback: pagerClickHandler
249             });
250 
251             if (json.total > widgetSettings.perPage) {
252                 $(commentsPager, rootel).show();
253             }
254             // Checks if the comments undefined or if it's length is 0
255             displayCommentsPagedOrNot();
256         };

261         var getComments = function(){
262             var sortOn = '_created';
263             var sortOrder = "desc";
264             var items = 10;
265             if (widgetSettings.direction && widgetSettings.direction === "comments_FirstDown") {
266                 sortOrder = "asc";
267             }
268             if (widgetSettings.perPage) {
269                 items = widgetSettings.perPage;
270             }
271 
272             var url = "/var/search/comments/flat.json?sortOn=" + sortOn + "&sortOrder=" + sortOrder + "&page=" + (clickedPage - 1) + "&items=" + items + "&marker=" + tuid + "&path=" + store;
273             $.ajax({
274                 url: url,
275                 cache: false,
276                 success: function(data){
277                     json = $.extend(data, {}, true);
278                     showComments();
279                 },
280                 error: function(xhr, textStatus, thrownError){
281                     sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("COMMENTS_AN_ERROR_OCCURRED") + " (" + xhr.status + ")","",sakai.api.Util.notification.type.ERROR);

306         var postComment = function(){
307             var comment = {
308                 // Replaces the \n (enters) with <br />
309                 "message": $(commentsMessageTxt, rootel).val()
310             };
311             comment["sakai:type"] = "comment";
312 
313             var isLoggedIn = (me.user.anon && me.user.anon === true) ? false : true;
314             var allowPost = true;
315             // If the user is not loggedin but we allow anon comments, we check some extra fields.
316             if (!isLoggedIn && widgetSettings['sakai:allowanonymous'] === true) {
317                 if (!isLoggedIn && widgetSettings['sakai:forcename']) {
318                     comment["sakai:name"] = $(commentsNamePosterTxt, rootel).val();
319                     if (comment["sakai:name"].replace(/\s/g, "") === "") {
320                         allowPost = false;
321                     }
322                 }
323                 if (!isLoggedIn && widgetSettings['sakai:forcemail']) {
324                     comment["sakai:email"] = $(commentsMailPosterTxt, rootel).val();
325                     if (comment["sakai:email"].replace(/\s/g, "") === "") {
326                         allowPost = false;
327                     }
328                 }
329             }
330             if (!isLoggedIn && widgetSettings['sakai:allowanonymous'] === false) {
331                 // This should not even happen.. Somebody is tinkering with the HTML.
332                 allowPost = false;
333                 sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("ANON_NOT_ALLOWED"),"",sakai.api.Util.notification.type.ERROR);
334             }
335 
336             var widgeturl = sakai.api.Widgets.getWidgetDataStorageURL(tuid);
337             var subject = 'Comment';
338             var to = "internal:" + widgeturl + "/message";
339 
340             if (allowPost) {
341                 var body = $(commentsMessageTxt, rootel).val();
342                 var message = {
343                     'sakai:type': 'comment',
344                     'sakai:to': to,
345                     'sakai:marker': tuid,
346                     'sakai:subject': subject,
347                     'sakai:body': body,
348                     'sakai:messagebox': 'pending',
349                     'sakai:sendstate': 'pending',
350                     'sling:resourceType': 'sakai/message',
351                     '_charset_':'utf-8'
352                 };
353 
354 
355                 var url = widgeturl + "/message.create.html";
356                 $.ajax({
357                     url: url,
358                     type: "POST",
359                     cache: false,
360                     success: function(data){
361                         // Hide the form.
362                         $(commentsFillInComment, rootel).hide();
363                         // Clear the textboxes.
364                         $(commentsMessageTxt, rootel).val("");
365                         $(commentsNamePosterTxt, rootel).val("");
366                         $(commentsMailPosterTxt, rootel).val("");
367                         var postData = {
368                             "post": data.message,
369                             "replies": []
370                         };
371                         postData.post["profile"] = [me.profile];
372                         postData.post["_path"] = widgeturl.slice(3, widgeturl.length) + "/message/inbox/" + data.id;
373                         postData.post["canDelete"] = allowedDelete;
374                         postData.post["canEdit"] = allowedEdit;
375                         if (widgetSettings && widgetSettings.direction && widgetSettings.direction === "comments_FirstDown") {
376                             json.results.push(postData);
377                         } else {
378                             json.results.unshift(postData);
379                         }
380                         json.total++;
381                         extraComments++;
382                         // Show the added comment
383                         showComments();
384                     },
385                     error: function(xhr, textStatus, thrownError){
386                         if (xhr.status === 401) {
387                             sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("YOU_NOT_ALLOWED"),"",sakai.api.Util.notification.type.ERROR);
388                         }
389                         else {
390                             sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("FAILED_TO_SAVE"),"",sakai.api.Util.notification.type.ERROR);
391                         }
392                     },
393                     data: message
394                 });
395             }
396             else {
397                 sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("PLEASE_FILL_ALL_FIELDS"),"",sakai.api.Util.notification.type.ERROR);
398             }
399         };

444         /**
445          * fills up the settings JSON-object
446          * @return {Object} the settings JSON-object, returns {Boolean} false if input is invalid
447          */
448         var getCommentsSettings = function(){
449             var comments = {};
450             comments.comments = [];
451 
452             // Checks if there's already some comments placed on the widget
453             comments.comments = json.comments || [];
454 
455             comments.perPage = parseInt($(commentsPageTxt, rootel).val(), 10);
456             if (isNaN(comments.perPage)) {
457                 comments.perPage = defaultPostsPerPage;
458             }
459 
460             if (comments.perPage < 1) {
461                 sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("PLEASE_FILL_POSITIVE_NUM"),"",sakai.api.Util.notification.type.ERROR);
462                 return false;
463             }
464             // Check if a valid number is inserted
465             else
466                 if ($(commentsPageTxt, rootel).val().search(/^\d*$/)) {
467                     sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("PLEASE_FILL_VALID_NUM"),"",sakai.api.Util.notification.type.ERROR);
468                     return false;
469                 }
470 
471 
472             comments.direction = $("input[name=" + commentsDirectionRbt + "]:checked", rootel).val();
473 
474             // These properties are not yet used in the comments-widget, but are saved in JCR
475             comments['sakai:allowanonymous'] = false;
476             if ($("#comments_RequireLogInID", rootel).is(":checked")) {
477                 comments['sakai:allowanonymous'] = false;
478             }
479             comments['sakai:forcename'] = $(commentsNameReqChk, rootel).attr("checked");
480             comments['sakai:forcemail'] = $(commentsEmailReqChk, rootel).attr("checked");
481             comments['sakai:notification'] = $(commentsSendMailChk, rootel).attr("checked");
482             comments['sakai:notificationaddress'] = me.user.userid;
483             comments['sling:resourceType'] = 'sakai/settings';
484             comments['sakai:marker'] = tuid;
485             comments['sakai:type'] = "comment";
486 
487             return comments;
488         };

501         /**
502          * Gets the widget settings and shows the appropriate view.
503          */
504         var getWidgetSettings = function(){
505 
506             sakai.api.Widgets.loadWidgetData(tuid, function(success, data){
507                 if (!sakai.api.User.isAnonymous(sakai.data.me)) {
508                     $(commentsCommentBtn, rootel).show();
509                 }
510 
511                 if (success) {
512                     if (!data.message) {
513                         sakai.api.Widgets.saveWidgetData(tuid, {"message":{"sling:resourceType":"sakai/messagestore"}}, null);
514                     }
515                     widgetSettings = data;
516                     // Clean up some values so that true is really true and not "true" or 1 ...
517                     var keysToClean = ['sakai:forcename', 'sakai:forcemail', 'notification', 'sakai:allowanonymous'];
518                     cleanBooleanSettings(keysToClean);
519 
520                     if (showSettings) {
521                         showSettingScreen(true, data);
522                     } else {
523                         pagerClickHandler(1);
524                     }
525                 } else {
526                     if (showSettings) {
527                         showSettingScreen(false, data);
528                     } else {
529                         saveSettings();
530                         pagerClickHandler(1);
531                     }
532                 }
533             });
534 
535         };

549                     }
550                 });
551 
552             }
553             return false;
554         };

581                         if (json.results[i].post["sakai:id"] === id){
582                             json.results[i].post["sakai:deleted"] = deleteValue;

584                     }
585                     showComments();
586                 },

588                     sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("FAILED_TO_UNDELETE"),"",sakai.api.Util.notification.type.ERROR);

594         var saveCommentEdit = function(form) {
595             var id = $(form).find("button.comments_editComment_save").attr("id").replace(commentsEditSave.replace(/\./g, ""), "");
596             var message = $(commentsEditText + id, rootel).val();
597             var data = {
598                 "sakai:body": message,
599                 "sakai:editedby": me.user.userid
600             };
601             // Do a post to the comment to edit the message.
602             var commentUrl = $(commentsPath+id).val();
603             $.ajax({
604                 url: commentUrl,
605                 cache: false,
606                 type: 'POST',
607                 success: function(data){
608                     // Set the new message

610                     // Hide the form
611                     $(commentsMessageEditContainer + id, rootel).hide();
612                     $(commentsMessage + id, rootel).show();
613                     // update the comment body
614                     for (var i = 0; i < json.results.length; i++) {
615                         if (json.results[i].post["sakai:id"] === id){
616                             json.results[i].post["sakai:body"] = message;
617                             json.results[i].post["sakai:editedby"] = me.user.userid;
618                         }
619                     }
620                     // Render Math formulas in the text

622                 },
623                 error: function(xhr, textStatus, thrownError){
624                     sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("FAILED_TO_EDIT"),"",sakai.api.Util.notification.type.ERROR);
625                 },
626                 data: data
627             });

749         /**
750          * Switch between main and settings page
751          * @param {Boolean} showSettings Show the settings of the widget or not
752          */
753         var doInit = function(){
754             addBindings();
755             var widgeturl = sakai.api.Widgets.getWidgetDataStorageURL(tuid);
756             widgeturl = sakai.api.Widgets.widgetLoader.widgets[tuid] ? sakai.api.Widgets.widgetLoader.widgets[tuid].placement : false;
757             if (widgeturl) {
758                 store = widgeturl + "/message";
759                 $.ajax({
760                     url: widgeturl + ".0.json",
761                     type: "GET",
762                     dataType: "json",
763                     success: function(data){
764                         // no op
765                     },
766                     error: function(xhr, textStatus, thrownError) {
767                         if (xhr.status == 404) {
768                             // we need to create the initial message store
769                             $.post(store, {"sling:resourceType":"sakai/messagestore"} );
770                         }
771                     }
772                 });
773             }
774             if (!showSettings) {
775                 // Show the main view.
776                 $(commentsSettingsContainer, rootel).hide();
777                 $(commentsOutputContainer, rootel).show();
778             }
779             getWidgetSettings();
780 
781             // determine if the edit or delete options should be shown for new posts
782             if (sakai_global.group && sakai.api.Groups.isCurrentUserAManager(sakai_global.group.groupId, me, sakai_global.group.groupData)){
783                 allowedEdit = true;
784                 allowedDelete = true;
785             }
786         };
787         doInit();
788     };
