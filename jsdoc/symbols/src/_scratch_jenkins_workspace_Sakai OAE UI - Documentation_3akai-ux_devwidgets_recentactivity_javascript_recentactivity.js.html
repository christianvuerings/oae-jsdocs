 18 // load the master sakai object to access all Sakai OAE API methods
 19 require(["jquery", "sakai/sakai.api.core"], function($, sakai){
 20 
 21     /**
 22      * @name sakai_global.recentactivity
 23      *
 24      * @class recentactivity
 25      *
 26      * @version 0.0.1
 27      * @param {String} tuid Unique id of the widget
 28      * @param {Boolean} showSettings Show the settings of the widget or not
 29      */
 30     sakai_global.recentactivity = function(tuid, showSettings){
 31 
 32         // Templates
 33         var recentactivityActivityItemTemplate = "recentactivity_activity_item_template";
 34 
 35         // Container
 36         var $recentactivityActivityContainer = $("#recentactivity_activity_container");
 37 
 38         var activityMap = {
 39             CONTENT_ADDED_COMMENT: "comment",
 40             CREATED_FILE: "upload",
 41             GROUP_CREATED: "new",
 42             JOINED_GROUP: "new",
 43             SENT_MESSAGE: "upload",
 44             UPDATED_CONTENT: "new",
 45             UPDATED_DESCRIPTION: "new",
 46             UPLOADED_CONTENT: "upload",
 47             UPDATED_COPYRIGHT: "new",
 48             UPDATED_URL: "new",
 49             UPDATED_TAGS: "new",
 50             USER_CREATED: "new"
 51         };
 52 
 53         var t = "";
 54         var numItems = 0;
 55         var numDiff = 0;
 56         var currentData = '';
 57 
 58         var parseActivity = function(success, data, initialLoad){
 59             if (success) {
 60                 var results = [];
 61                 numDiff = data.total - numItems;
 62                 numItems = data.total;
 63                 var total = 0;
 64                 $.each(data.results, function(index, item){
 65                     if (item["sakai:pooled-content-file-name"] && activityMap[item["sakai:activityMessage"]] && item["who"] && item["who"].basic && total < 5) {
 66                         if (index < numDiff) {
 67                             item.fadeIn = true;
 68                         }
 69                         item.showdetails = true;
 70                         if (item["sakai:activity-type"] && item["sakai:activity-type"] == "message" || item["sakai:activity-type"] == "group") {
 71                             item.showdetails = false;
 72                         }
 73                         item.who.name = sakai.api.User.getDisplayName(item.who);
 74                         item["sakai:activity-appid"] = activityMap[item["sakai:activityMessage"]];
 75                         item["sakai:activityMessage"] = sakai.api.i18n.getValueForKey(item["sakai:activityMessage"], "recentactivity");
 76                         item.who.picture = sakai.api.User.getProfilePicture(item.who);
 77                         item.usedin = sakai.api.Content.getPlaceCount(item);
 78                         results.push(item);
 79                         total++;
 80                     }
 81                 });
 82                 var json = {
 83                     "data": results,
 84                     "sakai": sakai
 85                 };
 86                 sakai.api.Util.TemplateRenderer(recentactivityActivityItemTemplate, json, $recentactivityActivityContainer);
 87                 applyThreeDots();
 88             }
 89 
 90             var $recentactivity_activity_item_container = $(".recentactivity_activity_item_container");
 91             $recentactivity_activity_item_container.filter(":visible").css("opacity", 1);
 92             if (!initialLoad) {
 93                 $recentactivity_activity_item_container.filter(':hidden').fadeTo(1000, 1);
 94             } else {
 95                 $recentactivity_activity_item_container.filter(":hidden").show();
 96                 $recentactivity_activity_item_container.filter(":visible").css("opacity", 1);
 97             }
 98             applyThreeDots();
 99         };
100 
101         var applyThreeDots = function(){
102             $.each($(".recentactivity_activity_item_text"), function(index, element) {
103                 var $el = $(element);
104                 var width = effectiveWidth($el, $el.siblings("a").find("img"));
105                 $el.html(sakai.api.Util.applyThreeDots($el.html(), width, {max_rows: 1}, $el.attr("class"), true));
106             });
107             $.each($(".recentactivity_activity_item_description"), function(index, element) {
108                 var $el = $(element);
109                 var width = effectiveWidth($el, $el.siblings("a").find("img"));
110                 $el.html(sakai.api.Util.applyThreeDots($el.html(), width, undefined, $el.attr("class"), true));
111             });
112         };
113 
114         var effectiveWidth = function(container, floated) {
115             var eWidth = $(container).width();
116             $.each(floated, function(index, floater) {
117                 eWidth -= $(floater).outerWidth(true);
118             });
119             return eWidth;
120         };
121 
122         var fetchActivity = function(initialLoad){
123             sakai.api.Server.loadJSON(sakai.config.URL.SEARCH_ACTIVITY_ALL_URL, function(success, data){
124                 var recentActivityJson = JSON.stringify(data.results);
125                 if (recentActivityJson !== currentData) {
126                     currentData = recentActivityJson;
127                     parseActivity(success, data, initialLoad);
128                 }
129                 t = setTimeout(function(){
130                         fetchActivity(false);
131                     }, 8000);
132             }, {
133                 "items": 12
134             });
135         };
136 
137         var doInit = function(initialLoad){
138             fetchActivity(initialLoad);
139         };
140 
141         doInit(true);
142 
143     };
144 
145     sakai.api.Widgets.widgetLoader.informOnLoad("recentactivity");
146 });
