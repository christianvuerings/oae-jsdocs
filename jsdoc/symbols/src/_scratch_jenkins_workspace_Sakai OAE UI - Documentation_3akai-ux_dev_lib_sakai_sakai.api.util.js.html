123         /**
124          * Parse a JavaScript date object to a JCR date string (2009-10-12T10:25:19)
125          *
126          * <p>
127          *     Accepted values for the format [1-6]:
128          *     <ol>
129          *         <li>Year: YYYY (eg 1997)</li>
130          *         <li>Year and month: YYYY-MM <br /> (eg 1997-07)</li>
131          *         <li>Complete date: YYYY-MM-DD <br /> (eg 1997-07-16)</li>
132          *         <li>Complete date plus hours and minutes: YYYY-MM-DDThh:mmTZD <br /> (eg 1997-07-16T19:20+01:00)</li>
133          *         <li>Complete date plus hours, minutes and seconds: YYYY-MM-DDThh:mm:ssTZD <br /> (eg 1997-07-16T19:20:30+01:00)</li>
134          *         <li>Complete date plus hours, minutes, seconds and a decimal fraction of a second YYYY-MM-DDThh:mm:ss.sTZD <br /> (eg 1997-07-16T19:20:30.45+01:00)</li>
135          *     </ol>
136          * </p>
137          * <p>
138          *     External links:
139          *     <ul>
140          *         <li><a href="http://www.w3.org/TR/NOTE-datetime">W3C datetime documentation</a></li>
141          *         <li><a href="http://delete.me.uk/2005/03/iso8601.html">ISO8601 JavaScript function</a></li>
142          *         <li><a href="http://confluence.sakaiproject.org/display/KERNDOC/KERN-643+Multiple+date+formats+in+the+back-end">Specification</a></li>
143          *     </ul>
144          * </p>
145          * @param {Date} date
146          *     JavaScript date object.
147          *     If not set, the current date is used.
148          * @param {Integer} format
149          *     The format you want to put out
150          * @param {String} offset
151          *     Optional timezone offset +HH:MM or -HH:MM,
152          *     if not set Z(ulu) or UTC is used
153          * @return {String} a JCR date string
154          */
155         createSakaiDate : function(date, format, offset) {
156             if (!format) { format = 5; }
157             if (!date) { date = new Date(); }
158             if (!offset) {
159                 offset = 'Z';
160             } else {
161                 var d = offset.match(/([\-+])([0-9]{2}):([0-9]{2})/);
162                 var offsetnum = (Number(d[2]) * 60) + Number(d[3]);
163                 offsetnum *= ((d[1] === '-') ? -1 : 1);
164                 date = new Date(Number(Number(date) + (offsetnum * 60000)));
165             }
166 
167             var zeropad = function (num) { return ((num < 10) ? '0' : '') + num; };
168 
169             var str = "";
170             str += date.getUTCFullYear();
171             if (format > 1) { str += "-" + zeropad(date.getUTCMonth() + 1); }
172             if (format > 2) { str += "-" + zeropad(date.getUTCDate()); }
173             if (format > 3) {
174                 str += "T" + zeropad(date.getUTCHours()) +
175                     ":" + zeropad(date.getUTCMinutes());
176             }
177             if (format > 4) { str += ":" + zeropad(date.getUTCSeconds()); }
178             if (format > 3) { str += offset; }
179             if (format > 5) {
180                 str = date.getTime();
181             }
182             return str;
183         },

185         /**
186          * Convert a file's size to a human readable size
187          * example: 2301 = 2.301kB
188          *
189          * @param (Integer) filesize The file's size to convert
190          * @return (String) the file's size in human readable format
191          */
192 
193         convertToHumanReadableFileSize : function(filesize) {
194             if (! filesize){
195                 filesize = 0;
196             }
197             var i;
198             if (filesize.indexOf && filesize.indexOf("binary-length:") > -1) {
199                 filesize = filesize.replace("binary-length:", "");
200             }
201             // Divide the length into its largest unit
202             var units = [[1024 * 1024 * 1024, 'GB'], [1024 * 1024, 'MB'], [1024, 'KB'], [1, 'bytes']];
203             var lengthunits;
204             for (i = 0, j=units.length; i < j; i++) {
205 
206                 var unitsize = units[i][0];
207                 var unittext = units[i][1];
208 
209                 if (filesize >= unitsize) {
210                     filesize = filesize / unitsize;
211                     // 1 decimal place
212                     filesize = Math.ceil(filesize * 10) / 10;
213                     lengthunits = unittext;
214                     break;
215                 }
216             }
217             // Return the human readable filesize (and localized)
218             return sakai_l10n.transformDecimal(filesize, 1) + " " + lengthunits;

283         /**
284          * Add and delete tags from an entity
285          * The two arrays, newTags and currentTags, represent the state of tags on the entity
286          * newTags should be the tags that you want on the entity, the whole set
287          * currentTags should be the set of tags the entity had before the user modified it
288          * tagEntity will delete any tags in currentTags but not in newTags, and add any in

299                 // set the tag on the entity
300                 var doSetTags = function(tags, doSetTagsCallback) {

315                                 doSetTagsCallback(true);
316                             }
317                         },
318                         error: function(xhr){
319                             debug.error(tagLocation + " failed to be tagged as " + tagArray);
320                             if ($.isFunction(doSetTagsCallback)) {
321                                 doSetTagsCallback(false);
322                             }
323                         }
324                     });
325                 };
326 
327                 if (tags.length) {
328                     doSetTags(tags, function(_success) {
329                         setTagsCallback(_success);
330                     });
331                 } else {

339             /**
340              * Delete tags on a given node
341              *
342              * @param (String) tagLocation the URL to the tag, ie. (~userid/public/authprofile)
343              * @param (Array) tags Array of tags to be deleted from the entity
344              * @param (Function) callback The callback function
345              */
346             var deleteTags = function(tagLocation, tags, deleteTagsCallback) {
347                 if (tags.length) {
348                     var requests = [];
349                     $(tags).each(function(i,val) {
350                         requests.push({
351                             "url": tagLocation,
352                             "method": "POST",
353                             "parameters": {
354                                 "key": "/tags/" + val,
355                                 ":operation": "deletetag"

357                         });
358                     });
359                     sakai_serv.batch(requests, function(success, data) {
360                         if (!success) {
361                             debug.error(val + " tag failed to be removed from " + tagLocation);
362                         }
363                         if ($.isFunction(deleteTagsCallback)) {
364                             deleteTagsCallback(success);
365                         }
366                     }, true);
367                 } else {
368                     if ($.isFunction(deleteTagsCallback)) {
369                         deleteTagsCallback();
370                     }
371                 }
372             };
373 
374             var tagsToAdd = [];
375             var tagsToDelete = [];
376             // determine which tags to add and which to delete
377             $(newTags).each(function(i,val) {
378                 if (val.indexOf("directory/") !== 0) {
379                     val = newTags[i] = sakai_util.makeSafeTag($.trim(val));
380                 }
381                 if (val && (!currentTags || $.inArray(val,currentTags) === -1)) {
382                     if (val.length) {
383                         if ($.inArray(val, tagsToAdd) < 0) {
384                             tagsToAdd.push(val);
385                         }
386                     }
387                 }
388             });
389             $(currentTags).each(function(i,val) {
390                 if (val.indexOf("directory/") !== 0) {
391                     val = currentTags[i] = sakai_util.makeSafeTag($.trim(val));
392                 }
393                 if (val && $.inArray(val,newTags) == -1) {
394                     if (val.length) {
395                         if ($.inArray(val, tagsToDelete) < 0) {
396                             tagsToDelete.push(val);
397                         }
398                     }
399                 }
400             });
401             currentTags = currentTags || [];
402             // determine the tags the entity has
403             var tags = $.unique($.merge($.merge([], currentTags), tagsToAdd)),
404                 finalTags = [];
405 
406             $(tags).each(function(i,val) {
407                 if (val.indexOf("directory/") !== 0) {
408                     val = sakai_util.makeSafeTag(val);
409                 }
410                 if ($.inArray(val, tagsToDelete) > -1) {
411                     tags.splice(tags.indexOf(val), 1);
412                 } else if (val && $.trim(val) !== ""){
413                     finalTags.push(val);
414                 }
415             });
416             deleteTags(tagLocation, tagsToDelete, function(deleteSuccess) {
417                 setTags(tagLocation, tagsToAdd, function(addSuccess) {
418                     if ($.isFunction(callback)) {
419                         callback(addSuccess || deleteSuccess, finalTags);
420                     }
421                 });
422             });

426         /**
427          * Truncate a string of text using the threedots plugin
428          * @param {String} body String of text to be truncated
429          * @param {int} width Width of the parent element
430          * @param {Object} params Object containing parameters, Threedots plugin specific. The row limit for widget headers should be 4 rows.
431          * @param {String} Optional class(es) to give container div. Used to give specific mark-up to the content to avoid wrong calculations. e.g. s3d-bold
432          */
433         applyThreeDots : function(body, width, params, optClass, alreadySecure){
434             body = sakai_util.Security.safeOutput(body);
435             // IE7 and IE6 have trouble with width
436             if(!jQuery.support.leadingWhitespace || jQuery.browser.webkit){
437                 width = width - 15;
438             } else {
439                 width = width - 5;
440             }
441 
442             // Create elements to apply threedots
443             var $container = $("<div class=\"" + optClass + "\" style=\"width:" + width + "px; ; word-wrap:break-word; visibility:hidden;\"><span style=\"word-wrap:break-word;\" class=\"ellipsis_text\">" + body + "</span></div>");
444             $("body").append($container);
445 
446             // There seems to be a race condition where the
447             // newly-added element returns a height of zero.  This
448             // would cause ThreeDots to truncate the input string to
449             // the first letter.  Try a couple of times for a non-zero
450             // height and then give up.
451             for (var attempt = 0; attempt < 10; attempt++) {
452                 if ($container.height() > 0) {
453                     $container.ThreeDots(params);
454                     break;
455                 }
456             }
457 
458             var dotted = $container.children("span").text();
459             $container.remove();
460             if (!alreadySecure) {
461                 dotted = sakai_util.Security.safeOutput(dotted);
462             }
463             // if params contains middledots = true then the string is threedotted in the middle
464             if(params && params.middledots && body.length > dotted.length){
465                 var maxlength = dotted.length - 3;
466                 if (!alreadySecure) {
467                     body = sakai_util.Security.safeOutput(body);
468                 }
469                 var prepend = body.slice(0, maxlength / 2);
470                 var append = body.slice(body.length - (maxlength / 2), body.length);
471                 return prepend + "..." + append;
472             }
473             return dotted;
474         },

476         /**
477          * Search for and replace parameters in a template (replaces both keys and properties)
478          * primarily used for making unique IDs for the group/course templates in config.js
479          *
480          * @param {Object} variables The variables to replace in the template with, ie. "groupid"
481          * @param {Object} replaceIn The object to modify
482          * @return {Object} the template structure with replaced variables
483          */
484         replaceTemplateParameters : function(variables, replaceIn) {
485             var loopAndReplace = function(structure, variable, replace) {
486                 var toReplace = "${" + variable + "}";
487                 var regex = new RegExp("\\$\\{" + variable + "\\}", 'g');
488                 for (var i in structure) {
489                     if (structure.hasOwnProperty(i)) {
490                         if (_.isString(structure[i]) && structure[i].indexOf(toReplace) !== -1) {
491                             structure[i] = structure[i].replace(regex, replace);
492                         } else if ($.isPlainObject(structure[i])) {
493                             structure[i] = loopAndReplace(structure[i], variable, replace);
494                         } else if ($.isArray(structure[i])) {
495                             for (var j = 0; j < structure[i].length; j++){
496                                 structure[i][j] = loopAndReplace(structure[i][j], variable, replace);
497                             }
498                         }
499                         if (i.indexOf(toReplace) !== -1) {
500                             var newKey = i.replace(regex, replace);
501                             structure[newKey] = structure[i];
502                             delete structure[i];
503                         }
504                     }
505                 }
506                 return structure;
507             };
508 
509             $.each(variables, function(variable,value) {
510                 replaceIn = loopAndReplace(replaceIn, variable, value);
511             });
512 
513             return replaceIn;
514         },

983                  * Author: Jim Palmer (based on chunking idea from Dave Koelle)
984                  * Contributors: Mike Grier (mgrier.com), Clint Priest, Kyle Adams, guillermo
985                  * Released under MIT license.

991                     sre = /(^[ ]*|[ ]*$)/g,
992                     hre = /^0x[0-9a-f]+$/i,
993                     dre = /(^[0-9\-\.\/]{5,}$)|[0-9]+:[0-9]+|( [0-9]{4})/i,
994                     ore = /^0/,
995                     // convert all to strings and trim()
996                     x = a ? a.toString().toLowerCase().replace(sre, '') || '' : '',
997                     y = b ? b.toString().toLowerCase().replace(sre, '') || '' : '',
998                     // chunk/tokenize
999                     xN = x.replace(re, String.fromCharCode(0) + "$1" + String.fromCharCode(0)).replace(/\0$/,'').replace(/^\0/,'').split(String.fromCharCode(0)),
1000                     yN = y.replace(re, String.fromCharCode(0) + "$1" + String.fromCharCode(0)).replace(/\0$/,'').replace(/^\0/,'').split(String.fromCharCode(0)),

1007                     else if ( xD > yD ) {return 1;}
1008                 }

1010                 for(var cLoc=0, numS=Math.max(xN.length, yN.length); cLoc < numS; cLoc++) {
1011                     // find floats not starting with '0', string or 0 if not defined (Clint Priest)
1012                     oFxNcL = !(xN[cLoc] || '').match(ore) && parseFloat(xN[cLoc]) || xN[cLoc] || 0;
1013                     oFyNcL = !(yN[cLoc] || '').match(ore) && parseFloat(yN[cLoc]) || yN[cLoc] || 0;
1014                     // handle numeric vs string comparison - number < string - (Kyle Adams)
1015                     if (isNaN(oFxNcL) !== isNaN(oFyNcL)) {
1016                         return (isNaN(oFxNcL)) ? 1 : -1;
1017                     }
1018                     // rely on string comparison if different types - i.e. '02' < 2 != '02' < '2'
1019                     else if (typeof oFxNcL !== typeof oFyNcL) {
1020                         oFxNcL += '';
1021                         oFyNcL += '';
1022                     }
1023                     if (oFxNcL < oFyNcL) {return -1;}
1024                     if (oFxNcL > oFyNcL) {return 1;}
1025                 }
1026                 if (x === y) {
1027                     if (a < b) {return -1;}
1028                     if (a > b) {return 1;}
1029                 }
1030                 return 0;
1031            }
