232         var parseQuote = function(message){
233             var quote = false;
234             if (message.substring(0,6) === "[quote") {
235                 // Parse the quoted message
236                 quote = message.split("[/quote]")[0];
237                 quote = quote.substring(quote.indexOf("]") + 1, quote.length);
238                 // Parse the original author
239                 var by = message.split("[/quote]")[0];
240                 by = by.substring(by.indexOf("'") + 1, by.indexOf("]") - 1);
241                 return {"quote":quote, "by":by};
242             } else {
243                 return quote;
244             }
245         };

266         var renderPosts = function(arrPosts){
267             // Loop fetched posts and do markup
268             for (var i = 0, j = arrPosts.length; i < j; i++) {
269                 arrPosts[i].post.profile[0].pictureImg = parsePicture(arrPosts[i].post.profile[0]);
270                 var tempPostDate = sakai.api.Util.parseSakaiDate(arrPosts[i].post['sakai:created']).getTime();
271                 if (isNaN(parseDate(arrPosts[i].post['sakai:created']))) {
272                     tempPostDate = sakai.api.Util.parseSakaiDate(
273                                        sakai.api.Util.createSakaiDate(new Date(arrPosts[i].post['sakai:created']))).getTime();
274                 }
275                 arrPosts[i].post["sakai:createdOn"] = sakai.api.l10n.transformDateTimeShort(parseDate(tempPostDate));
276                 if(arrPosts[i].post["sakai:editedOn"]){
277                     arrPosts[i].post["sakai:editedOn"] = sakai.api.l10n.transformDateTimeShort(parseDate(arrPosts[i].post["sakai:editedOn"]));
278                 }
279                 for(var ii = 0, jj = arrPosts[i].replies.length; ii < jj; ii++){
280                     arrPosts[i].replies[ii].post.profile[0].pictureImg = parsePicture(arrPosts[i].replies[ii].post.profile[0]);
281                     var tempReplyDate = sakai.api.Util.parseSakaiDate(arrPosts[i].replies[ii].post['sakai:created']).getTime();
282                     if (isNaN(parseDate(arrPosts[i].replies[ii].post['sakai:created']))) {
283                         tempReplyDate = sakai.api.Util.parseSakaiDate(
284                                             sakai.api.Util.createSakaiDate(new Date(arrPosts[i].replies[ii].post['sakai:created']))).getTime();
285                     }
286                     arrPosts[i].replies[ii].post["sakai:createdOn"] = sakai.api.l10n.transformDateTimeShort(parseDate(tempReplyDate));
287                     if(arrPosts[i].replies[ii].post["sakai:deletedOn"]){
288                         arrPosts[i].replies[ii].post["sakai:deletedOn"] = sakai.api.l10n.transformDateTimeShort(parseDate(arrPosts[i].replies[ii].post["sakai:deletedOn"]));
289                     }
290                     if(arrPosts[i].replies[ii].post["sakai:editedOn"]){
291                         arrPosts[i].replies[ii].post["sakai:editedOn"] = sakai.api.l10n.transformDateTimeShort(parseDate(arrPosts[i].replies[ii].post["sakai:editedOn"]));
292                     }
293                     arrPosts[i].replies[ii].post["sakai:quoted"] = parseQuote(arrPosts[i].replies[ii].post["sakai:body"]);
294                     if(arrPosts[i].replies[ii].post["sakai:body"].split(["[/quote]"])[1]){
295                         arrPosts[i].replies[ii].post["sakai:body"] = arrPosts[i].replies[ii].post["sakai:body"].split(["[/quote]"])[1];
296                     }
297                 }
298                 // Sort replies
299                 arrPosts[i].replies.sort(sortReplies);
300             }
301 
302             // Render formatted posts
303             sakai.api.Util.TemplateRenderer(discussionListTopicsTemplate, {
304                 "postData":arrPosts,
305                 "settings":parsedSettings,
306                 "sakai": sakai
307             }, $(discussionListTopicsContainer, $rootel));
308             sakai.api.Util.renderMath(tuid);
309         };

387         var parseSettings = function(data){
388             var contact = false;
389             var canEditPage = false;
390             // Check if the logged in user manages the discussion page or not (for a content profile or group)
391             if ((sakai_global.group &&
392                 sakai.api.Groups.isCurrentUserAManager(sakai_global.group.groupId, sakai.data.me, sakai_global.group.groupData)) ||
393                 (sakai_global.content_profile &&
394                 sakai.api.Content.isUserAManager(sakai_global.content_profile.content_data.data, sakai.data.me))) {
395                     canEditPage = true;
396             }
397             parsedSettings["ismanager"] = canEditPage;
398             // Anonymous can't do anything
399             if (sakai.api.User.isAnonymous(sakai.data.me)) {
400                 parsedSettings["addtopic"] = false;
401                 parsedSettings["canreply"] = false;
402                 parsedSettings["anon"] = true;
403             } else {
404                 parsedSettings["anon"] = false;
405                 parsedSettings["userid"] = sakai.data.me.user.userid;
406 
407                 // Check for who can add topics
408                 if (data["whocanaddtopic"] === "managers_only") {
409                     if (canEditPage) {
410                         parsedSettings["addtopic"] = true;
411                     } else {
412                         parsedSettings["addtopic"] = false;
413                     }
414                 } else {
415                     parsedSettings["addtopic"] = true;
416                 }
417 
418                 // Check for who can add replies
419                 if (data["whocanreply"] === "anyone") {
420                     parsedSettings["canreply"] = true;
421                 } else {
422                     if (canEditPage) {
423                         parsedSettings["canreply"] = true;
424                     }
425                 }
426             }
427             getPosts();
428         };

540         /**
541          * Reply to a post.
542          * @param {String} id The ID of the topic that's being replied on
543          * @param {String} body The message in the reply
544          * @param {String} $parentDiv the parent div that should be hidden on success
545          */
546         var replyToTopic = function(id, body, $parentDiv, $replyParent){
547             var store = getMessageStoreURL();
548             var object = {
549                 "sakai:body": body,
550                 "sakai:marker": marker,
551                 "sakai:type": "discussion",
552                 "sling:resourceType": "sakai/message",
553                 "sakai:replyon": id,
554                 "sakai:messagebox": "pending",
555                 "sakai:sendstate": "pending",
556                 "sakai:to": "discussion:" + store,
557                 "sakai:deleted": false,
558                 "_charset_": "utf-8"
559             };
560             var url = store + ".create.html";
561             $.ajax({
562                 url: url,
563                 type: "POST",
564                 success: function(data){
565                     $parentDiv.hide();
566                     $parentDiv.parents(discussionTopicContainer).find(discussionReplyTopicBottom).show();
567 
568                     data.message["profile"] = $.extend(data.message["profile"], sakai.data.me.profile);
569                     data.message.profile.pictureImg = parsePicture(data.message.profile);
570                     var tempReplyDate = sakai.api.Util.parseSakaiDate(data.message['sakai:created']).getTime();
571                     if (isNaN(parseDate(data.message['sakai:created']))) {
572                         tempReplyDate = sakai.api.Util.parseSakaiDate(
573                                             sakai.api.Util.createSakaiDate(new Date(data.message['sakai:created']))).getTime();
574                     }
575                     data.message['sakai:created'] = sakai.api.l10n.transformDateTimeShort(parseDate(tempReplyDate));
576                     data.message["sakai:createdOn"] = data.message['sakai:created'];
577 
578                     data.message["sakai:quoted"] = parseQuote(data.message["sakai:body"]);
579                     if (data.message["sakai:body"].split(["[/quote]"])[1]) {
580                         data.message["sakai:body"] = data.message["sakai:body"].split(["[/quote]"])[1];
581                     }
582 
583                     var renderedTemplate = sakai.api.Util.TemplateRenderer(discussionTopicNewlyPostedReplyTemplate, {
584                         "post":data,
585                         "settings": parsedSettings,
586                         sakai: sakai
587                     });
588 
589                     $parentDiv.prevAll(discussionTopicRepliesContainer).append(renderedTemplate);
590 
591                     $parentDiv.parents(discussionTopicContainer).find(discussionNumberOfReplies).text(parseInt($parentDiv.parents(discussionTopicContainer).find(discussionNumberOfReplies).text(), 10) + 1);
592                     $parentDiv.parents(discussionTopicContainer).find(".discussion_show_replies_icon").show();
593                     sakai.api.Util.renderMath(tuid);
594                     disableEnableButtons(true);
595 
596                     var $repliesIcon = $replyParent.find(discussionRepliesIcon);
597                     if ($repliesIcon.hasClass(discussionShowRepliesIcon)) {
598                         // expand topic reply list
599                         $("#discussion_post_" + id + " " + discussionShowTopicReplies, $rootel).click();
600                     }
601                 },
602                 error: function(xhr, textStatus, thrownError){
603                     if (xhr.status === 401) {
604                         $parentDiv.hide();
605                         sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("YOU_CANT_REPLY"), "", sakai.api.Util.notification.type.ERROR);
606                     }
607                     else {

646         /**
647          * Deletes or undeletes the post with the provided id.
648          * @param {String} id The id of the post.
649          * @param {boolean} deleteValue true = delete, false = undelete
650          */
651         var deletePost = function(id, deleteValue, post){
652             var store = getMessageStoreURL();
653             var url = store + "/inbox/" + id;
654             var data = {
655                 "sakai:deleted": deleteValue,
656                 "sakai:deletedBy": sakai.api.User.getDisplayName(sakai.data.me.profile),
657                 "sakai:deletedOn": Date.now()
658             };
659             $.ajax({
660                 url: url,
661                 type: "POST",
662                 data: data,
663                 success: function(){
664                     if (deleteValue) {
665                         // Apply grey class
666                         post.addClass(discussionDeletedReplyClass);
667 
668                         // hide message option links
669                         $('#' + id + ' ' + discussionMessageOptions, $rootel).hide();
670 
671                         // Remove/add links and information
672                         post.find(discussionPostMessage).nextAll().remove();
673                         post.find(discussionPostMessage).after(sakai.api.Util.TemplateRenderer(discussionDeletedPostActionsTemplate, {}));
674                         post.find(discussionPostingDate).after(sakai.api.Util.TemplateRenderer(discussionDeletedPostEntityInfoTemplate, {
675                             "deletedBy": data["sakai:deletedBy"],
676                             "deletedOn": sakai.api.l10n.transformDateTimeShort(parseDate(data["sakai:deletedOn"]))
677                         }));
678                     }else{
679                         // Apply grey class
680                         post.removeClass(discussionDeletedReplyClass);
681 
682                         // hide message option links
683                         $('#' + id + ' ' + discussionMessageOptions, $rootel).hide();
684                         $(discussionDeletedMessage, $rootel).hide();
685 
686                         // Remove links
687                         post.find(discussionPostingDate).next().remove();
688                         post.find(discussionPostMessage).nextAll().remove();
689                         post.find(discussionPostMessage).after(sakai.api.Util.TemplateRenderer(discussionRestoredPostActionsTemplate, {}));
690                     }
691                 },
692                 error: function(xhr, textStatus, thrownError){
693                     sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("FAILED_DELETE_POST"),"",sakai.api.Util.notification.type.ERROR);
694                 }
695             });
696         };

698         var updatePost = function(id, body, quote, quoted, post){
699             var store = getMessageStoreURL();
700             var url = store + "/inbox/" + id;
701             var data = {
702                 "sakai:edited": true,
703                 "sakai:editedBy": sakai.api.User.getDisplayName(sakai.data.me.profile),
704                 "sakai:editedOn": Date.now(),
705                 "sakai:body": body
706             };
707             if(quote){
708                 data["sakai:body"] = "[quote='" + quoted + "']" + quote + "[/quote]" + body;
709             }
710 
711             $.ajax({
712                 url: url,
713                 type: "POST",
714                 data: data,
715                 success: function(){
716                     // remove edit divs
717                     post.find(discussionEditContainer).children().remove();
718 
719                     // Set post data
720                     post.find(discussionPostMessage).text(body);
721                     post.find(discussionPostMessage).attr("data-source-text", body);
722                     post.find(discussionReplyContentsText).text(quote);
723                     post.find(discussionReplyContentsText).attr("data-source-text", quote);
724                     sakai.api.Util.renderMath(tuid);
725 
726                     // Set entity data
727                     post.children(discussionEntityContainer).find(discussionUpdatingDate).children("span").text(sakai.api.User.getDisplayName(sakai.data.me.profile) + " " + sakai.api.l10n.transformDateTimeShort(parseDate(data["sakai:editedOn"])));
728                     post.children(discussionEntityContainer).children(discussionPostingDate).children().show();
729 
730                     // Show all
731                     post.children(".discussion_entity_container, .discussion_reply_contents").show();
732                 }
733             });
