 19 /*
 20  * Dependencies
 21  *
 22  * /dev/lib/misc/trimpath.template.js (TrimpathTemplates)

 26 /*global $, Config, sakai, History, opensocial, Widgets */
 27 require(["jquery", "sakai/sakai.api.core"], function($, sakai, sakai_util) {
 28     /**
 29      * This is a widget that can be placed in other pages and widgets.
 30      * It shows an Add to contacts dialog.
 31      * All you have to do is provide a user object that contains:
 32      *  - first name
 33      *  - last name
 34      *  - uuid of the person you wish to add.
 35      *
 36      *  You can also set the personal note by using the setPersonalNote() method
 37      *  and select the type by using the setTypes().
 38      */
 39     sakai_global.addtocontacts = function(tuid, showSettings){
 40 
 41 
 42         /////////////////////////////
 43         // Configuration variables //
 44         /////////////////////////////
 45 
 46         var $rootel = $("#" + tuid);
 47 
 48         // Help variables
 49         var contactToAdd = false;
 50 
 51         // CSS selectors
 52         var addToContacts = "#addtocontacts";
 53         var addToContactsClass = ".addtocontacts";
 54 
 55         var addToContactsDialog = addToContacts + "_dialog";
 56 
 57         // Form elements
 58         var addToContactsForm = addToContacts + "_form";
 59         var addToContactsFormButtonInvite = addToContactsForm + "_invite";
 60         var addToContactsFormPersonalNote = addToContactsForm + "_personalnote";
 61         var addToContactsFormPersonalNoteTemplate = addToContactsFormPersonalNote + "_template";
 62         var addToContactsFormType = addToContactsForm + "_type";
 63         var addToContactsFormTypeTemplate = addToContactsFormType + "_template";
 64         // Profile info
 65         var addToContactsInfoProfilePicture = addToContacts + "_profilepicture";
 66         var addToContactsInfoTypes = addToContacts + "_types";
 67         var addToContactsInfoDisplayName = addToContactsClass + "_displayname";
 68 
 69         // Error messages
 70         var addToContactsError = addToContacts + "_error";
 71         var addToContactsErrorMessage = addToContactsError + "_message";
 72         var addToContactsErrorRequest = addToContactsError + "_request";
 73         var addToContactsErrorNoTypeSelected = addToContactsError + "_noTypeSelected";
 74 
 75         var addToContactsResponse = addToContacts + "_response";
 76 
 77         ///////////////////
 78         // Functionality //
 79         ///////////////////
 80 
 81         /**
 82          * Disables or enables the invite button on the widget
 83          * @param {Boolean} disable Flag to disable or enable the button
 84          */
 85         var enableDisableInviteButton = function(disable){
 86             if(disable){
 87                 $(addToContactsFormButtonInvite).attr("disabled","disabled");
 88             }else{
 89                 $(addToContactsFormButtonInvite).removeAttr("disabled");
 90             }
 91         };
 92 
 93         /**
 94          * Render the templates that are needed for the add contacts widget.
 95          * It renders the contacts types and the personal note
 96          */
 97         var renderTemplates = function(){
 98             sakai.api.Util.TemplateRenderer(addToContactsFormTypeTemplate.replace(/#/gi, ""), {
 99                 "relationships": sakai.config.Relationships,
100                 "sakai": sakai
101             }, $(addToContactsInfoTypes));
102             var json = {
103                 sakai: sakai,
104                 me: sakai.data.me
105             };
106             sakai.api.Util.TemplateRenderer(addToContactsFormPersonalNoteTemplate.replace(/#/gi, ""), json, $(addToContactsFormPersonalNote));
107         };
108 
109         /**
110          * This method will fill in the info for the user.
111          * @param {Object} user The JSON object containing the user info. This follows the /rest/me format.
112          */
113         var fillInUserInfo = function(user){
114             if (user) {
115                 $(addToContactsInfoDisplayName, $rootel).text(user.displayName);
116                 if (!user.pictureLink) {
117                     user.pictureLink = sakai.api.Util.constructProfilePicture(user);
118                 }
119                 // Check for picture
120                 if (user.pictureLink) {
121                     $(addToContactsInfoProfilePicture).html('<img alt="' + $("#addtocontacts_profilepicture_alt").html() + '" src="' + user.pictureLink + '" class="s3d-icon-50" />');
122                 } else {
123                     $(addToContactsInfoProfilePicture).html('<img alt="' + $("#addtocontacts_profilepicture_alt").html() + '" src="' + sakai.config.URL.USER_DEFAULT_ICON_URL + '" class="s3d-icon-50" />');
124                 }
125             }
126         };
127 
128         /**
129          * This function looks up and retrieves relationship information from a set of pre-defined relationships
130          * @param {String} relationshipName
131          */
132         var getDefinedRelationship = function(relationshipName){
133             for (var i = 0, j = sakai.config.Relationships.contacts.length; i < j; i++) {
134                 var definedRelationship = sakai.config.Relationships.contacts[i];
135                 if (definedRelationship.name === relationshipName) {
136                     return definedRelationship;
137                 }
138             }
139             return null;
140         };
141 
142         /**
143          * Does the invitation stuff. Will send a request for an invitation and a message to the user.
144          * @param {String} userid
145          */
146         var doInvite = function(userid){
147             enableDisableInviteButton(true);
148             var formValues = $(addToContactsForm).serializeObject();
149             var types = formValues[addToContactsFormType.replace(/#/gi, "")];
150             if (!$.isArray(types)) {
151                 types = [types];
152             }
153             $(addToContactsResponse).text("");
154             if (types.length) {
155                 var fromRelationshipsToSend = [];
156                 var toRelationshipsToSend = [];
157                 for (var i = 0, j = types.length; i < j; i++) {
158                     var type = types[i];
159                     fromRelationshipsToSend.push(type);
160                     var definedRelationshipToSend = getDefinedRelationship(type);
161                     if (definedRelationshipToSend && definedRelationshipToSend.inverse) {
162                         toRelationshipsToSend.push(definedRelationshipToSend.inverse);
163                     }
164                     else {
165                         toRelationshipsToSend.push(type);
166                     }
167                 }
168 
169                 var personalnote = $.trim(formValues[addToContactsFormPersonalNote.replace(/#/gi, '')]);
170 
171                 // send message to other person
172                 var userstring = $.trim(sakai.api.User.getDisplayName(sakai.data.me.profile));
173 
174                 var title = $.trim($("#addtocontacts_invitation_title_key").text().replace(/\$\{user\}/g, userstring));
175                 var message = $.trim($("#addtocontacts_invitation_body_key").text().replace(/\$\{user\}/g, userstring).replace(/\$\{comment\}/g, personalnote).replace(/\$\{br\}/g,"\n"));
176 
177                 // Do the invite and send a message
178                 $.ajax({
179                     url: "/~" + sakai.api.Util.safeURL(sakai.data.me.user.userid) + "/contacts.invite.html",
180                     type: "POST",
181                     traditional: true,
182                     data: {
183                         "fromRelationships": fromRelationshipsToSend,
184                         "toRelationships": toRelationshipsToSend,
185                         "targetUserId": userid
186                     },
187                     success: function(data){
188                         enableDisableInviteButton(false);
189                         sakai.api.Util.Modal.close(addToContactsDialog);
190                         sakai.api.Communication.sendMessage(userid, sakai.data.me, title, message, "invitation", false,false,true,"contact_invitation");
191                         $(window).trigger("sakai.addToContacts.requested", [contactToAdd]);
192                         //reset the form to set original note
193                         $(addToContactsForm)[0].reset();
194                         var notificationMessage = contactToAdd.displayName + ' ' + sakai.api.i18n.getValueForKey('HAS_BEEN_ADDED_TO_YOUR_CONTACTS_LIST', 'addtocontacts');
195                         sakai.api.Util.notification.show('', notificationMessage);
196                     },
197                     error: function(xhr, textStatus, thrownError){
198                         enableDisableInviteButton(false);
199                         $(addToContactsResponse).text(sakai.api.Security.saneHTML($(addToContactsErrorRequest).text()));
200                     }
201                 });
202 
203             }
204             else {
205                 enableDisableInviteButton(false);
206                 $(addToContactsResponse).text(sakai.api.Security.saneHTML($(addToContactsErrorNoTypeSelected).text()));
207             }
208         };
209 
210         ///////////////////////
211         // jqModal functions //
212         ///////////////////////
213 
214         /**
215          * This will load the overlay to add a new contact.
216          * This method will fill in all the user info.
217          * @param {Object} hash The layover object we get from jqModal
218          */
219         var loadDialog = function(hash){
220             $("#addtocontacts_dialog_title").html($("#addtocontacts_dialog_title_template").html().replace("${user}", sakai.api.Security.safeOutput(contactToAdd.displayName)));
221             hash.w.show();
222         };
223 
224         /////////////////////////
225         // Initialise function //
226         /////////////////////////
227 
228         /**
229          * People should call this function if they want to initiate the widget
230          * @param {Object} user The userid or the /rest/me info for this user.
231          * @param {Function} callback The callback function that will be executed after the request.
232          */
233         var initialize = function(user){
234             user.userid = user.userid || user.uuid;
235             contactToAdd = user;
236             fillInUserInfo(contactToAdd);
237 
238             // Render the templates
239             renderTemplates();
240 
241             // Show the layover
242             sakai.api.Util.Modal.open(addToContactsDialog);
243 
244         };
245 
246         $(document).on('initialize.addToContacts.sakai', function(e, userObj) {
247             initialize(userObj);
248         });
249         $(document).on('click', '.sakai_addtocontacts_overlay', function(ev, ui) {
250             var $el = $(this);
251             if ($el.attr('sakai-entityid') && $el.attr('sakai-entityname')) {
252                 initialize({
253                     'uuid': $el.attr('sakai-entityid'),
254                     'displayName': $el.attr('sakai-entityname'),
255                     'pictureLink': $el.attr('sakai-entitypicture') || false
256                 });
257             }
258         });
259 
260         /////////////////////
261         // Event listeners //
262         /////////////////////
263 
264         // Bind the invite button
265         $(addToContactsFormButtonInvite).bind("click", function(){
266             // Invite this person.
267             doInvite(contactToAdd.userid);
268             return false;
269         });
270 
271         // Bind the jqModal
272         sakai.api.Util.Modal.setup(addToContactsDialog, {
273             modal: true,
274             overlay: 20,
275             toTop: true,
276             onShow: loadDialog
277         });
278     };
279 
280     sakai.api.Widgets.widgetLoader.informOnLoad("addtocontacts");
281 });
