348         /**
349          * Returns an array of relative widths for all of the columns in a given row
350          * @param {jQuery} $row     jQuery element representing row for which to get the widths
351          *                          of its columns
352          */
353         var getColumnWidths = function($row) {
354             var totalWidth = $('#contentauthoring_widget_container', $pageRootEl).width();
355             var $cells = $('.contentauthoring_cell', $row);
356             var widths = [];
357             // Variable to track the remaining width
358             var lastWidth = 1;
359             for (var i = 0; i < $cells.length; i++) {
360                 // We give the last column the remaining width
361                 if (i === $cells.length - 1) {
362                     widths.push(lastWidth); 
363                 // We give each column a relative width rather than
364                 // an absolute one
365                 } else {
366                     lastWidth -= $($cells[i]).width() / totalWidth;
367                     widths.push($($cells[i]).width() / totalWidth);
368                 }
369             }
370             return widths;
371         };

402         /**
403          * Recalculate the widths of all the columns in a row after having resized a column. This
404          * makes sure that each of the columns has sufficient width, ratios between the non-dragged
405          * columns are preserved, etc.
406          * @param {Object} ui             jQuery ui object
407          * @param {jQuery} $row           jQuery element representing the row we're calculating
408          *                                widths for
409          * @param {jQuery} $resizedCell   jQuery element representing the column that's being resized
410          * @param {Array} currentSizes    The arrray containing the current column widths, used
411          *                                to preserve the column width ratios
412          */
413         var recalculateColumnWidths = function(ui, $row, $resizedCell, currentSizes) {
414             var totalRowWidth = $('#contentauthoring_widget_container', $pageRootEl).width();
415             var newColumnWidth = (ui.size.width + 12) / totalRowWidth;
416             var oldColumnWidth = ui.originalSize.width / totalRowWidth;
417             
418             var rowId = $row.attr('data-row-id');
419             var $cells = $('.contentauthoring_cell', $row);
420 
421             var hasFoundResizedCell = false;
422 
423             var totalWidth = 0;
424             var numberOfColumns = $cells.length;
425             for (var i = 0; i < $cells.length; i++) {
426                 var currentColumnWidth = 0;
427                 if ($($cells[i]).is($resizedCell)) {
428                     // New percentage based width
429                     if (newColumnWidth < MINIMUM_COLUMN_SIZE) {
430                         currentColumnWidth = MINIMUM_COLUMN_SIZE;
431                     } else if (totalWidth + newColumnWidth + ((numberOfColumns - i - 1) * MINIMUM_COLUMN_SIZE) > 1) {
432                         currentColumnWidth = 1 - totalWidth - ((numberOfColumns - i - 1) * MINIMUM_COLUMN_SIZE);
433                     } else {
434                         currentColumnWidth = newColumnWidth;
435                     }
436                     $($cells[i]).css('width', currentColumnWidth * 100 + '%');
437                     hasFoundResizedCell = true;
438                 } else if (hasFoundResizedCell) {
439                     // New percentage based width
440                     if (numberOfColumns - i === 1) {
441                         // This is the final cell, fill it up
442                         currentColumnWidth = 1 - totalWidth;
443                     } else {
444                         // There are 2 more cells
445                         // Does the 2nd have enough space after pulling in the 1st?
446                         if (1 - (totalWidth + currentSizes[i]) > MINIMUM_COLUMN_SIZE) {
447                             currentColumnWidth = currentSizes[i];
448                             $($cells[i + 1]).css('width', (1 - totalWidth - currentSizes[i]) * 100 + '%');
449                         // Shrink the fist to its maximum size
450                         // Make the second its minumum size
451                         } else {
452                             currentColumnWidth = 1 - totalWidth - MINIMUM_COLUMN_SIZE;
453                             $($cells[i + 1]).css('width', MINIMUM_COLUMN_SIZE * 100 + '%');
454                         }
455                     }
456                     $($cells[i]).css('width', currentColumnWidth * 100 + '%');
457                     hasFoundResizedCell = false;
458                 }
459                 totalWidth += currentColumnWidth;
460             }
461         };

463         /**
464          * Sets the height of a row to the heighest column
465          * @param {jQuery} $row jQuery object with class '.contentauthoring_table_row.contentauthoring_cell_container_row'
466          *                      used to search for child cells that can contain content
467          */
468         var setRowHeight = function($row) {
469             var cells = $('.contentauthoring_cell_content', $row);
470             var setDefaultHeight = true;
471             $.each(cells, function(index, cell) {
472                 // Default the height of the cell to auto to avoid that cells stay larger than they should
473                 $('.contentauthoring_cell_content', $row).css('height', 'auto');
474                 // Remove whitespace since jQuery :empty selector doesn't ignore it
475                 var html = $(cell).html().replace(/\s+/, '');
476                 if (html.length || $(html).hasClass('contentauthoring_dummy_element')) {
477                     // There is some content in the row so no default height but the cell height should be considered
478                     setDefaultHeight = false;
479                 }
480             });
481 
482             if (setDefaultHeight) {
483                 // No content in the row, set default height
484                 $('.contentauthoring_cell_content', $row).css('height', 25);
485             } else {
486                 // Some cells have content
487                 // if row is part of the pageviewer than equalheights doesn't need to be set
488                 if (!$('.contentauthoring_cell_content', $row).parents('.pageviewer_widget').length) {
489                     $('.contentauthoring_cell_content', $row).equalHeights();
490                 }
491             }
492         };

566                 var columnWidth = widths[w] * (1 - (newColumnWidth * (totalColumns - widths.length))) * 100 + '%';

581          * @param {Integer} totalColumns    Total of columns we need in the row
582          */
583         var removeColumns = function($row, totalColumns) {

585             var remainingWidth = 1;
586             var $cells = $('.contentauthoring_cell', $row);
587             $row.find('.contentauthoring_dummy_element').remove();
588             // Append the content of the columns that will be removed to the last
589             // column that will be retained
590             for (var i = totalColumns; i < $cells.length; i++) {
591                 var $cell = $($cells[i]);
592                 // De- and re-initialize tinyMCE to avoid errors
593                 killTinyMCEInstances($cell);
594                 var $cellcontent = $('.contentauthoring_cell_content', $cell).children();
595                 initializeTinyMCEInstances($('.contentauthoring_cell_content', $($cells[totalColumns - 1])).append($cellcontent));
596                 $cell.remove();
597                 remainingWidth -= widths[i];
598             }
599             for (var l = 0; l < totalColumns; l++) {
600                 $($cells[l]).css('width', (widths[l] / remainingWidth) * 100 + '%');
601             }
602             checkColumnsEmpty();
603             updateColumnHandles();
604             storeCurrentPageLayout();
605         };
