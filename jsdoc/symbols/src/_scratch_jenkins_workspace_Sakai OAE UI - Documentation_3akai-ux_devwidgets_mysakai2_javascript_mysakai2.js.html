 23 /*global $ */
 24 
 25 require(["jquery", "sakai/sakai.api.core"], function($, sakai) {
 26 
 27     /**
 28      * @name sakai_global.mysakai2
 29      *
 30      * @class mysakai2
 31      *
 32      * @description
 33      * Initialize the mysakai2 w    idget
 34      *
 35      * @version 0.0.1
 36      * @param {String} tuid Unique id of the widget
 37      */
 38     sakai_global.mysakai2 = function(tuid){
 39 
 40 
 41         /////////////////////////////
 42         // Configuration variables //
 43         /////////////////////////////
 44 
 45         var rootel = $("#" + tuid);
 46 
 47         var mysakai2List = "#mysakai2_list";
 48         var mysakai2ListTemplate = "#mysakai2_list_template";
 49         var mysakai2ErrorNosites = "#mysakai2_no_results_container";
 50 
 51 
 52         /**
 53          * Takes a set of json and renders the sakai2 sites.
 54          * @param {Object} newjson sakai 2 list object
 55          */
 56         var doRender = function(resultJson , useDisplayProperties){
 57             // If the user is not registered for any sites, show the no sites error.
 58             if (resultJson.sites.length === 0) {
 59                 $(mysakai2List, rootel).hide();
 60                 $(mysakai2ErrorNosites, rootel).show();
 61             } else {
 62                 $(mysakai2List, rootel).show();
 63                 $(mysakai2ErrorNosites, rootel).hide();
 64                 $(mysakai2List, rootel).html(sakai.api.Util.TemplateRenderer(mysakai2ListTemplate.replace(/#/, ''), resultJson));
 65             }
 66         };
 67 
 68         /**
 69          *
 70          *
 71          */
 72         var loadSakai2SiteList = function(){
 73             // get sakai2favouriteList
 74             sakai.api.Server.loadJSON("/~" + sakai.api.Util.safeURL(sakai.data.me.user.userid) + "/private/sakai2favouriteList",function(savedsuccess,saveddata){
 75                 var url = "/dev/s23/bundles/sites.json";
 76                 if (sakai.config.useLiveSakai2Feeds){
 77                     url = "/var/proxy/s23/sitesUnread.json?unread=true";
 78                 }
 79                 $.ajax({
 80                     url: url,
 81                     type : "GET",
 82                     dataType: "json",
 83                     success: function(fulllist){
 84                         var resultJson = {};
 85                         resultJson.sites = [];
 86                         if(savedsuccess){
 87                             if (fulllist.sites) {
 88                                 for (var ii = 0; ii < fulllist.sites.length; ii++) {
 89                                     for (var i = 0; i < saveddata.id.length; i++) {
 90                                         if (fulllist.sites[ii].id === saveddata.id[i]) {
 91                                             resultJson.sites.push(fulllist.sites[ii]);
 92                                             break;
 93                                         }
 94                                     }
 95                                 }
 96                             }
 97                         } else {
 98                             for (var j = 0; j < fulllist.display; j++){
 99                                 if (fulllist.sites[j]) {
100                                     resultJson.sites.push(fulllist.sites[j]);
101                                 }
102                             }
103                         }
104                         sakai.data.me.sakai2List = resultJson;
105                         doRender(sakai.data.me.sakai2List);
106                     }, 
107                     error: function(){
108                         $("#mysakai2_error_container", rootel).show();
109                     }
110                 });
111             });
112         };
113 
114         ////////////////////
115         // Event Handlers //
116         ////////////////////
117 
118         // Listen for completion of sakai2 favourites site addition
119         // to refresh this widget's sites listing
120         $(window).bind("sakai2-favourites-selected", function() {
121             doInit();
122         });
123 
124         $(".mysakai2_add_files_link", rootel).click(function(ev){
125             // Load the sakai 2 favourites widget.
126             $(window).trigger("init.sakai2favourites.sakai");
127         });
128 
129         /**
130          * Will initiate a request to the my groups service.
131          */
132         var doInit = function(){
133             //get sakai2 list information and then render in my sakai2 widget
134             loadSakai2SiteList();
135             sakai.api.Widgets.widgetLoader.insertWidgets("sakai2favourites_container", false);
136         };
137 
138 
139         // Start the request
140         doInit();
141 
142     };
143 
144     sakai.api.Widgets.widgetLoader.informOnLoad("mysakai2");
145 });
