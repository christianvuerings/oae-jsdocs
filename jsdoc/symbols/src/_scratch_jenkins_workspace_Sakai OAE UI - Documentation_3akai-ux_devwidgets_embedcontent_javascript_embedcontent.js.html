171         /**
172          * Creates an object out of results provided
173          * This object contains valuable information about the file like path, name, type,...
174          * @param {Object} result results provided (eg through a search)
175          * @param {Object} name optional name provided
176          */
177         var createDataObject = function(result, name) {
178             var mimetype = sakai.api.Content.getMimeType(result);
179             var dataObj = {
180                 "value": name || result['_path'],
181                 "name": sakai.api.Security.safeOutput(result['sakai:pooled-content-file-name']),
182                 "type": "file",
183                 "filetype": mimetype.split("/")[0],
184                 "_mimeType": mimetype,
185                 "description": result["sakai:description"] || "",
186                 "path": "/p/" + (name || result['_path']),
187                 "fileSize": sakai.api.Util.convertToHumanReadableFileSize(result["_length"]),
188                 "_path": result['_path'],
189                 "_mimeType/page1-small": result["_mimeType/page1-small"],
190                 "fullresult" : result
191             };
192             var link = sakai.api.Util.safeURL(name || result['_path']) + "/" + sakai.api.Util.safeURL(result['sakai:pooled-content-file-name']);
193             if (dataObj._mimeType === "x-sakai/link"){
194                 dataObj.downloadLink = result["sakai:pooled-content-url"];
195                 dataObj.contentProfileLink = "/content#p=" + link;
196             } else {
197                 dataObj.downloadLink = "/p/" + link;
198                 dataObj.contentProfileLink = "/content#p=" + link;
199             }
200 
201             // if the type is application need to auto check the display name so set ispreviewexist false
202             if(dataObj.filetype === "application") {
203                 isPreviewExist = false;
204             }
205 
206             return dataObj;
207         };

228         /**

231         var setupAutoSuggest = function() {
232             var dataFn = function( query, add ) {
233                 var q = sakai.api.Server.createSearchString(query);
234                 var options = {"page": 0, "items": 15, "q": q, "userid": sakai.data.me.user.userid};
235                 searchUrl = sakai.config.URL.SEARCH_ALL_FILES.replace(".json", ".infinity.json");
236                 sakai.api.Server.loadJSON(searchUrl, function(success, data){
237                     if (success) {
238                         var suggestions = [];
239                         $.each(data.results, function(i) {
240                             var dataObj = createDataObject(data.results[i]);
241                             var doAdd = true;
242                             if (embedConfig.filter) {
243                                 if (dataObj.filetype !== embedConfig.filter) {
244                                     doAdd = false;
245                                 }
246                             }
247                             if (doAdd) {
248                                 suggestions.push(dataObj);
249                             }
250                         });
251                         add( suggestions, query );
252                     }
253                 }, options);
254             };
255             sakai.api.Util.AutoSuggest.setup($embedcontent_content_input, {
256                 asHtmlID: tuid,
257                 retrieveLimit: 10,
258                 selectionLimit: embedConfig.limit,
259                 resultClick: function(data) {
260                     autosuggestSelectionAdded(data.attributes);
261                 },
262                 selectionRemoved: function(elem) {
263                     autosuggestSelectionRemoved(elem);
264                 }
265             }, false, dataFn);
266         };

394         /**
395          * Once the content has been placed on the page it has to be associated with the group
396          * The group is set as a viewer of the content
397          * @param {Object} embeddedItems Array of object containing information about the selected items. Only the path variable is used.
398          */
399         var associatedEmbeddedItemsWithGroup = function(embeddedItems){
400             var data = [];
401             for (var embeddedItem in embeddedItems) {
402                 if (embeddedItems.hasOwnProperty(embeddedItem) && !sakai.api.Content.isContentInLibrary(embeddedItems[embeddedItem].fullresult, sakai_global.group.groupId)) {
403                     data.push(embeddedItems[embeddedItem].value);
404                 }
405             }
406             if (data.length > 0){
407                 sakai.api.Content.addToLibrary(data, sakai_global.group.groupId);
408             }
409         };

411         /**

413          * Call the function that associates the content with this group

415         var doEmbed = function() {
416             var embedContentHTML = "";
417             var formVals = $embedcontent_display_form.serializeObject();
418             var itemsToSave = [];
419             $.each(selectedItems, function(i,item) {
420                 if (item.path) {
421                     itemsToSave.push(item.path);
422                 }
423             });
424             var objectData = {
425                 "layout": selectedItems.length > 1 ? formVals.layout : "single",
426                 "embedmethod": formVals.style,
427                 "title": formVals.title || '',
428                 "description": formVals.description || '',
429                 "items": itemsToSave,
430                 "details": formVals.details ? true : false,
431                 "download": formVals.download ? true : false,
432                 "name": formVals.name ? true : false
433             };
434 
435             if (sakai_global.group && sakai_global.group.groupId) {
436                 // Associate embedded items with the group
437                 associatedEmbeddedItemsWithGroup(selectedItems);
438             }
439 
440             saveWidgetData(objectData);

449         var newItems = [];
450         var processWidget = function(data, callback) {
451             var ret = false;
452             var batchRequests = [];
453             for (var i = 0, j = data.items.length; i < j; i++) {
454                 batchRequests.push({
455                     url: data.items[i] + ".2.json",
456                     method: "GET"
457                 });
458             }
459 
460             if (batchRequests.length > 0) {
461                 sakai.api.Server.batch(batchRequests, function(success, response){
462                     if (success || batchRequests.length === 1) {
463                         $.each(response.results, function(index, item){
464                             if (item.success && item.body){
465                                 var newItem = createDataObject($.parseJSON(item.body));
466                                 newItems.push(newItem);
467                             } else {
468                                 newItems.push({
469                                     type: "notfound",
470                                     name: $embedcontent_item_unavailable_text.text(),
471                                     value: "notfound2" + index,
472                                     path: item.url.replace('.2.json', '')
473                                 });
474                             }
475                         });
476                         wData.items = newItems;
477                         ret = true;
478                     }
479                     if ($.isFunction(callback)) {
480                         callback(ret);
481                     }
482                 });
483             } else if ($.isFunction(callback)) {
484                 callback(ret);
485             }
486         };
