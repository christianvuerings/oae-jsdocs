 18 var sakai = {};
 19 var $ = {};
 20 
 21 // Init google map object with Cambridge location as the center position
 22 var map = new google.maps.Map(document.getElementById('googlemaps_map_canvas'), {
 23     zoom: 8,
 24     center: new google.maps.LatLng(52.2025441, 0.1312368),
 25     mapTypeControlOptions: { style: google.maps.MapTypeControlStyle.DROPDOWN_MENU },
 26     mapTypeId: google.maps.MapTypeId.ROADMAP,
 27     navigationControlOptions: { style: google.maps.NavigationControlStyle.ZOOM_PAN }
 28 });
 29 
 30 // Init google marker object with Cambridge location as the center position
 31 var marker = new google.maps.Marker({
 32     position: new google.maps.LatLng(52.2025441, 0.1312368),
 33     title: 'Point A',
 34     map: map,
 35     draggable: false
 36 });
 37 
 38 // Init a google maps infor window object to open a info window to show the location info
 39 var infoWindow = new google.maps.InfoWindow({
 40     content: ''
 41 });
 42 
 43 // Init a json object which inlcudes the basic properties of the google map
 44 var json = {
 45     'lat': map.getCenter().lat(),
 46     'lng': map.getCenter().lng(),
 47     'mapzoom': map.getZoom(),
 48     'mapinput': '',
 49     'maphtml': 'Cambridge, UK'
 50 };
 51 
 52 /**
 53  * Update the content in infoWindow
 54  */
 55 function updateInfoWindow(html) {
 56     if (infoWindow) {
 57         infoWindow.close();
 58     }
 59 
 60     json.maphtml = html;
 61     infoWindow = new google.maps.InfoWindow({
 62         content: '<h4>' + html + '</h4>'
 63     });
 64     infoWindow.open(map, marker);
 65 }
 66 
 67 /**
 68  * Find the closest position near the marker's location
 69  * @param {Object} position of the closest location
 70  */
 71 function geocodePosition(position) {
 72     var geocoder = new google.maps.Geocoder();
 73     if (geocoder) {
 74         geocoder.geocode({
 75             latLng: position
 76         }, function(responses) {
 77             if (responses && responses.length > 0) {
 78                 var html = responses[0].formatted_address;
 79                 updateInfoWindow(html);
 80             } else {
 81                 $(window).trigger('show.mapsnotification.sakai', {
 82                     subject: sakai.api.i18n.getValueForKey('NO_ADDRESS', 'googlemaps'),
 83                     body: sakai.api.i18n.getValueForKey('CANNOT_DETERMINE_ADDRESS_AT_THIS_LOCATION', 'googlemaps')
 84                 });
 85             }
 86         });
 87     }
 88 }
 89 
 90 /**
 91  * Find the closest position near the marker's location
 92  * @param {Object} address the position of the closest location
 93  */
 94 function geocodeAddress(address) {
 95     var geocoder = new google.maps.Geocoder();
 96     if (geocoder) {
 97         geocoder.geocode({
 98             'address': address
 99         }, function(results, status){
100             if (status == google.maps.GeocoderStatus.OK) {
101                 var position = results[0].geometry.location;
102                 map.setCenter(position);
103                 marker.setMap(map);
104                 marker.setPosition(position);
105                 var html = results[0].formatted_address;
106                 updateInfoWindow(html);
107             } else {
108                 sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey('NO_ADDRESS', 'googlemaps'), 
109                                                  sakai.api.i18n.getValueForKey('CANNOT_DETERMINE_ADDRESS_AT_THIS_LOCATION', 'googlemaps'));
110             }
111         });
112     }
113 }
114 
115 /**
116  * Get the json object
117  */
118 var getJSON = function() {
119     // use marker rather than map because 
120     // map use 14 decimal places while marker uses only 6 decimal places 
121     // causing marker to appear on wrong spot
122     json.lat = marker.getPosition().lat();
123     json.lng = marker.getPosition().lng();
124     json.mapzoom = map.getZoom();
125     return json;
126 };
127 
128 /**
129  * Search the corresponded city between the target location
130  * @param {string} keyword the target location
131  * @param {string} The region where Google will perform the search
132  */
133 function search(keyword, region, _sakai, _$) {
134     sakai = _sakai;
135     $ = _$;
136 
137     // If region is provided, attach it ot the search query
138     // TO DO: This later will need to be moved to the Google API
139     if (region !== '') {
140         geocodeAddress(keyword+', '+region);
141         json.mapinput = keyword+', '+region;
142     } else {
143         geocodeAddress(keyword);
144         json.mapinput = keyword;
145     }
146 }
147 
148 /**
149  * Set the map initial property (center an zoom)
150  * @param {Object} jsonTarget the json object includes map's properties
151  */
152 function setMap(jsonTarget) {
153     json = jsonTarget;
154 
155     var latLng = new google.maps.LatLng(json.lat, json.lng);
156     map.setCenter(latLng);
157     map.setZoom(parseInt(json.mapzoom, 10));
158     marker.setPosition(latLng);
159 
160     // Init the infoWindow object and decide to open it or not
161     if (json.maphtml) {
162         updateInfoWindow(json.maphtml);
163     }
164 }
165 
166 /**
167  * Init the listeners of marker
168  */
169 function init() {
170     google.maps.event.addListener(marker, 'click', function() {
171         geocodePosition(marker.getPosition());
172     });
173 }
174 
175 init();
