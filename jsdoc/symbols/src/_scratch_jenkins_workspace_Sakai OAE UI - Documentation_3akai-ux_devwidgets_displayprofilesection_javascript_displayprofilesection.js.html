 18 /*
 19  * Dependencies
 20  *
 21  * /dev/lib/jquery/plugins/jqmodal.sakai-edited.js
 22  * /dev/lib/misc/trimpath.template.js (TrimpathTemplates)
 23  */
 24 require(["jquery", "sakai/sakai.api.core", "underscore"], function($, sakai, _) {
 25 
 26     /**
 27      * @name sakai_global.displayprofilesection
 28      *
 29      * @class displayprofilesection
 30      *
 31      * @description
 32      * Initialize the displayprofilesection widget
 33      *
 34      * @version 0.0.1
 35      * @param {String} tuid Unique id of the widget
 36      * @param {Boolean} showSettings Show the settings of the widget or not
 37      */
 38     sakai_global.displayprofilesection = function( tuid, showSettings, widgetData ) {
 39         /////////////////////////////
 40         // Configuration variables //
 41         /////////////////////////////
 42 
 43         var editing = false;
 44         var userid = false;
 45         var multiple = false;
 46         var multipleSectionLength = 0;
 47         var sectionData = false;
 48         var allowUpdate = true;
 49 
 50         ///////////////////
 51         // CSS Selectors //
 52         ///////////////////
 53 
 54         var $rootel = $( "#" + tuid ),
 55             $displayprofilesection_container = $( "#displayprofilesection_container", $rootel ),
 56             $displayprofilesection_header = $( "#displayprofilesection_header", $rootel ),
 57             $displayprofilesection_header_template = $( "#displayprofilesection_header_template", $rootel ),
 58             $displayprofilesection_body = $( "#displayprofilesection_body", $rootel ),
 59             $displayprofilesection_body_template = $( "#displayprofilesection_body_template", $rootel ),
 60             $displayprofilesection_view_template = $( "#displayprofilesection_view_template", $rootel ),
 61             $displayprofilesection_edit_template = $( "#displayprofilesection_edit_template", $rootel ),
 62             $displayprofilesection_edit_multiple_template = $( "#displayprofilesection_edit_multiple_template", $rootel ),
 63             $displayprofilesection_view_multiple_template = $( "#displayprofilesection_view_multiple_template" , $rootel ),
 64             $displayprofilesection_view_no_results_template = $( "#displayprofilesection_view_no_results_template", $rootel ),
 65             $profile_message_form_successful = $( "#profile_message_form_successful", $rootel ),
 66             $profile_error_form_error_server = $( "#profile_error_form_error_server", $rootel ),
 67             $displayprofilesection_sections_multiple = false,
 68             $displayprofilesection_add_button = false,
 69             $form = false;
 70 
 71         // Transform the form values from a multiple-assign section into a different data structure
 72         var getMultipleValues = function( values ) {
 73             var uniqueContainers = $( "div.displayprofilesection_multiple_section" );
 74             var multipleValues = {};
 75             $.each( uniqueContainers, function( i, elt ) {
 76                 multipleValues[ $( elt ).attr( "id" ).replace( "form_group_", "" ) ] = {
 77                     order: i
 78                 };
 79             });
 80             $.each( values, function( i, val ) {
 81                 // Each ID is of format fieldtitle_formuniqueid
 82                 var field = i.substring( 0, i.lastIndexOf( "_" ));
 83                 var mvKey = i.substring( i.lastIndexOf( "_" ) + 1 );
 84                 multipleValues[ mvKey ][ field ] = val;
 85             });
 86             values = multipleValues;
 87             return values;
 88         };
 89 
 90         var handleSave = function( success, data ) {
 91             if (success) {
 92                 sakai.api.Util.notification.show("", $profile_message_form_successful.text() , sakai.api.Util.notification.type.INFORMATION);
 93             } else {
 94                 sakai.api.Util.notification.show("", $profile_error_form_error_server.text() , sakai.api.Util.notification.type.ERROR);
 95                 debug.error("The profile data couldn't be saved successfully");
 96             }
 97         };
 98 
 99         /**
100          * Enables the 'Update Profile' button when the user has changed their profile information. 
101          */
102         var enableUpdate = function() {
103             if (allowUpdate) {
104                 $('button.profile-section-save-button', $rootel).removeAttr('disabled');
105             }
106         };
107 
108         var saveValues = function() {
109             // Serialize the data from the form for saving
110             var values = $form.serializeObject();
111             if ( multiple ) {
112                 values = getMultipleValues( values );
113             }
114 
115             // Get tags & categories if they're in this form
116             var tags = [];
117             var $tagfield = $displayprofilesection_body.find( "textarea[data-tag-field]" );
118             if ( $tagfield.length ) {
119                 // Remove the hidden autosuggest input field from the values
120                 $form.find( "input" ).each( function( i, input ) {
121                     if ( $( input ).hasClass( "as-values" ) ) {
122                         delete values[ $( input ).attr("name") ];
123                     }
124                 });
125                 tags = sakai.api.Util.AutoSuggest.getTagsAndCategories( $tagfield, true );
126             }
127             sakai.api.User.updateUserProfile(userid, widgetData.sectionid, values, tags, sectionData, multiple, handleSave);
128             $('button.profile-section-save-button', $rootel).attr('disabled', 'disabled');
129             return false;
130         };
131 
132         // Remove a section from the a multi-assign section
133         var removeSection = function( unique ) {
134             $( "div#form_group_" + unique ).remove();
135             sakai.api.User.deleteUserProfileSection( userid, widgetData.sectionid, unique, handleSave);
136             if ( $( "div.displayprofilesection_multiple_section" ).length === 0 ) {
137                 $( "button.profile-section-save-button", $rootel ).hide();
138             }
139         };
140 
141         // Add a new section to a multi-assign section
142         var addEmptySection = function( section, template ) {
143             var unique = "" + Math.round( Math.random() * 1000000000 );
144             multipleSectionLength++;
145             var sectionHTML = sakai.api.Util.TemplateRenderer( template, {
146                 sectionid: widgetData.sectionid,
147                 section: section,
148                 unique: unique,
149                 order: multipleSectionLength,
150                 data: {}
151             });
152             $displayprofilesection_sections_multiple.append( sakai.api.i18n.General.process( sectionHTML ) );
153             $( "button.profile-section-save-button", $rootel ).show();
154             $( "button#displayprofilesection_remove_link_" + unique, $rootel ).bind( "click", function() {
155                 removeSection( unique );
156             });
157         };
158 
159         var sectionHasElements = function( section ) {
160             var hasElts = false;
161             if ( section ) {
162                 $.each( section, function( i, elt ) {
163                     if ( $.isPlainObject( elt ) ) {
164                         hasElts = true;
165                     }
166                 });
167             }
168             return hasElts;
169         };
170 
171         var renderEmptySection = function( userProfile, section ) {
172             var messageKey = "THIS_PERSON_HASNT_ADDED_INFORMATION";
173             var sectionKey = "THIS_PERSON_HASNT_ADDED_" + widgetData.sectionid.toUpperCase();
174             if ( sakai.api.i18n.getValueForKey( sectionKey, "displayprofilesection") !== sectionKey ) {
175                 messageKey = sectionKey;
176             }
177             var emptyHTML = sakai.api.Util.TemplateRenderer( $displayprofilesection_view_no_results_template, {
178                 userid: userid,
179                 displayName: sakai.api.User.getDisplayName( userProfile ),
180                 errorString: sakai.api.i18n.getValueForKey( messageKey, "displayprofilesection" ),
181                 showMessage: !sakai.api.User.isAnonymous(sakai.data.me)
182             });
183             $displayprofilesection_body.html( sakai.api.i18n.General.process( emptyHTML ) );
184         };
185 
186         // Render a multi-assign section
187         var renderMultiSection = function ( template, section, data ) {
188             multiple = true;
189             var multiTemplate = editing ? $displayprofilesection_edit_multiple_template : $displayprofilesection_view_multiple_template;
190 
191             var multHTML = sakai.api.Util.TemplateRenderer( multiTemplate, {
192                 sectionid: widgetData.sectionid,
193                 section: section
194             });
195             $displayprofilesection_body.html( sakai.api.i18n.General.process( multHTML ) );
196 
197             // Grab the new container to render into
198             $displayprofilesection_sections_multiple = $( "#displayprofilesection_sections_" + widgetData.sectionid );
199             if ( editing ) {
200                 $displayprofilesection_add_button = $( "#displayprofilesection_add_" + widgetData.sectionid );
201                 $displayprofilesection_add_button.bind("click", function() {
202                     addEmptySection( section, template );
203                 });
204             }
205 
206             // If there is some data, render each section
207             if (data[widgetData.sectionid] && data[widgetData.sectionid].elements) {
208                 var subSections = [];
209                 // Convert the sectionData into an array so we can order it
210                 $.each( data[ widgetData.sectionid ].elements, function( uid, sectionData ) {
211                     if ( $.isPlainObject( sectionData ) ) {
212                         var obj = {};
213                         obj[ uid ] = sectionData;
214                         subSections.push( obj );
215                     }
216                 });
217                 // Sort the sections by order
218                 subSections = subSections.sort( function ( a, b ) {
219                     return _.values( a )[ 0 ].order - _.values( b )[ 0 ].order;
220                 });
221                 $.each( subSections, function( i, sectionData ) {
222                     if ( $.isPlainObject( sectionData ) ) {
223                         // Just keep incrementing, since we're not supporting re-ordering yet
224                         multipleSectionLength = _.values( sectionData )[ 0 ].order > multipleSectionLength ? _.values( sectionData )[ 0 ].order : multipleSectionLength;
225                         var uid = _.keys( sectionData )[ 0 ];
226                         var sectionHTML = sakai.api.Util.TemplateRenderer( template, {
227                             section: section,
228                             sakai: sakai,
229                             unique: uid,
230                             data: _.values( sectionData )[ 0 ],
231                             order: _.values( sectionData )[ 0 ].order
232                         });
233                         $displayprofilesection_sections_multiple.append( sakai.api.i18n.General.process( sectionHTML ) );
234                         if ( editing ) {
235                             $( "button#displayprofilesection_remove_link_" + uid, $rootel ).bind( "click", function() {
236                                 removeSection( uid );
237                             });
238                         }
239                     }
240                 });
241                 if ( editing && subSections.length ) {
242                     $( "button.profile-section-save-button", $rootel ).show();
243                 } else if ( !editing && !subSections.length ) {
244                     renderEmptySection( data );
245                 } else if ( !editing ){
246                     $( ".displayprofilesection_multiple_sections hr:last" ).hide();
247                 }
248             } else if ( !editing ) {
249                 renderEmptySection( data );
250             }
251         };
252 
253         var renderSection = function( success, data ) {
254             if ( success ) {
255                 var section = sakai.config.Profile.configuration.defaultConfig[ widgetData.sectionid ];
256 
257                 if ( section ) {
258                     var template = $displayprofilesection_view_template;
259                     if ( editing ) {
260                         template = $displayprofilesection_edit_template;
261                     }
262 
263                     // Render header
264                     var pageTitle = section.label;
265                     if (!editing && section.altLabel) {
266                         pageTitle = sakai.api.i18n.General.process(section.altLabel)
267                             .replace('${user}', sakai.api.User.getFirstName(sakai_global.profile.main.data));
268                     }
269                     var headerHTML = sakai.api.Util.TemplateRenderer( $displayprofilesection_header_template, {
270                         pageTitle: pageTitle
271                     });
272                     $displayprofilesection_header.html( sakai.api.i18n.General.process( headerHTML ) );
273 
274                     // If it is a multiple section, we have to render it with some love
275                     if ( section.multiple ) {
276                         renderMultiSection( template, section, data );
277                     } else {
278                         // data[widgetData.sectionid] won't exist when the user hasn't logged in before
279                         if (editing || (data[widgetData.sectionid] && sectionHasElements(data[widgetData.sectionid].elements))) {
280                             sectionData = data[ widgetData.sectionid ] && data[ widgetData.sectionid ].elements ? data[ widgetData.sectionid ].elements : false;
281                             $.each(section.elements, function(index, element) {
282                                 if (element.altLabel) {
283                                     element.altLabel = sakai.api.i18n.General.process(section.altLabel)
284                                         .replace('${user}', sakai.api.User.getFirstName(sakai_global.profile.main.data));
285                                 }
286                             });
287                             var bodyHTML = sakai.api.Util.TemplateRenderer( template, {
288                                 sectionid: widgetData.sectionid,
289                                 section: section,
290                                 data: sectionData,
291                                 unique: false,
292                                 sakai: sakai
293                             });
294                             $displayprofilesection_body.html( sakai.api.i18n.General.process( bodyHTML ) );
295                             var $tagfield = $displayprofilesection_body.find( "textarea[data-tag-field]" );
296                             if ( $tagfield.length ) {
297                                 allowUpdate = false;
298                                 var autoSuggestOptions = {
299                                     scrollHeight: 120,
300                                     selectionAdded: function() {
301                                         enableUpdate();
302                                     },
303                                     selectionRemoved: function(elem) {
304                                         elem.remove();
305                                         enableUpdate();
306                                     }
307                                 };
308                                 var initialTagsValue = sectionData["sakai:tags"] && sectionData["sakai:tags"].value ? sectionData["sakai:tags"].value : false;
309                                 sakai.api.Util.AutoSuggest.setupTagAndCategoryAutosuggest(
310                                     $tagfield,
311                                     autoSuggestOptions,
312                                     $('.list_categories', $rootel),
313                                     initialTagsValue,
314                                     function() {
315                                         allowUpdate = true;
316                                     }
317                                 );
318                             }
319                         } else {
320                             renderEmptySection( data, section );
321                         }
322                     }
323 
324                     if ( editing ) {
325                         $form = $( "#displayprofilesection_form_" + widgetData.sectionid, $rootel );
326                         var validateOpts = {
327                             submitHandler: saveValues,
328                             messages: {}
329                         };
330                         // Set the custom error messages per field
331                         $.each( section.elements, function( i, elt ) {
332                             if ( elt.errorMessage ) {
333                                 validateOpts.messages[ i ] = {
334                                     required: sakai.api.i18n.General.process( elt.errorMessage )
335                                 };
336                             }
337                         });
338                         sakai.api.Util.Forms.validate( $form, validateOpts );
339                     }
340                 }
341             }
342         };
343 
344         $rootel.on('input change cut paste', function() {
345             enableUpdate();
346         });
347 
348         var getData = function(callback) {
349             if (editing && sakai.data.me.profile && $.isFunction(callback) && sakai.data.me.profile._fullProfileLoaded) {
350                 callback(true, sakai.data.me.profile);
351             } else {
352                 sakai.api.User.getUser(userid, function(success, data) {
353                     if (sakai.data.me.user.userid === data.userid) {
354                         sakai.data.me.profile = data;
355                         sakai.data.me.profile._fullProfileLoaded = true;
356                     }
357                     if ($.isFunction(callback)) {
358                         callback(success, data);
359                     }
360                 });
361             }
362         };
363 
364         var init = function() {
365             userid = sakai_global.profile.main.data.userid;
366             editing = userid && userid === sakai.data.me.user.userid;
367             getData( renderSection );
368         };
369 
370         init();
371     };
372 
373     sakai.api.Widgets.widgetLoader.informOnLoad("displayprofilesection");
374 });
