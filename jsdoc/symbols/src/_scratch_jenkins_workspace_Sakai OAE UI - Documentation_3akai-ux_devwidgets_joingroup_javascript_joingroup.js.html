 19 // load the master sakai object to access all Sakai OAE API methods
 20 require(["jquery", "sakai/sakai.api.core"], function($, sakai) {
 21 
 22     /**
 23      * @name sakai_global.joingroup
 24      *
 25      * @class joingroup
 26      *
 27      * @description
 28      * Overlay for joining groups
 29      *
 30      * @version 0.0.1
 31      * @param {String} tuid Unique id of the widget
 32      * @param {Boolean} showSettings Show the settings of the widget or not
 33      */
 34     sakai_global.joingroup = function (tuid, showSettings) {
 35 
 36         /////////////////////////////
 37         // Configuration variables //
 38         /////////////////////////////
 39 
 40         // DOM jQuery Objects
 41         var $rootel = $("#" + tuid);  // unique container for each widget instance
 42         var $joingroup_hover_template = $("#joingroup_hover_template", $rootel);
 43 
 44         ///////////////////////
 45         // Utility Functions //
 46         ///////////////////////
 47 
 48         /**
 49          * Compare the names of 2 participants
 50          *
 51          * @param {Object} participant a
 52          * @param {Object} participant b
 53          * @return 1, 0 or -1
 54          */
 55         var participantSort = function (a, b) {
 56             if (a.displayname.toLowerCase() > b.displayname.toLowerCase()) {
 57                 return 1;
 58             } else {
 59                 if (a.displayname.toLowerCase() === b.displayname.toLowerCase()) {
 60                     return 0;
 61                 } else {
 62                     return -1;
 63                 }
 64             }
 65         };
 66 
 67         /**
 68          * Adjust the number of participants listed in the search result
 69          *
 70          * @param {String} groupid The group ID
 71          * @param {Integer} value Value to adjust the number of participants by
 72          */
 73         var adjustParticipantCount = function (groupid, value) {
 74             var participantCount = parseInt($("#searchgroups_result_participant_count_" + groupid).text(), 10);
 75             participantCount = participantCount + value;
 76             $("#searchgroups_result_participant_count_" + groupid).text(participantCount);
 77             if (participantCount === 1) {
 78                 $("#searchgroups_text_participant_" + groupid).text(sakai.api.i18n.getValueForKey("PARTICIPANT_LC"));
 79             } else {
 80                 $("#searchgroups_text_participant_" + groupid).text(sakai.api.i18n.getValueForKey("PARTICIPANTS_LC"));
 81             }
 82             $("#searchgroups_result_participant_link_" + groupid).attr("title", $.trim($("#searchgroups_result_participant_link_" + groupid).text()));
 83         };
 84 
 85         /**
 86          * Push the given member to the given list of members to be rendered.
 87          * If the member is a manager, set is_manager to 'true'
 88          *
 89          * @param {Object} member      object returned in batch data results
 90          * @param {Object} list        list of members to add to
 91          * @param {Object} is_manager  optional flag to set whether this user
 92          *     is a manager or not
 93          */
 94         var push_member_to_list = function (member, list, role) {
 95             var link, picsrc, displayname = "";
 96             if (member["sakai:group-id"]) {
 97                 picsrc = sakai.api.Groups.getProfilePicture(member);
 98                 link = "~" + member.groupid;
 99                 displayname = member["sakai:group-title"];
100             } else {
101                 picsrc = sakai.api.User.getProfilePicture(member);
102                 link = "~" + member.userid;
103                 displayname = sakai.api.User.getDisplayName(member);
104             }
105             list.push({
106                 link: link,
107                 picsrc: picsrc,
108                 displayname: displayname,
109                 role: role
110             });
111         };
112 
113         var getGroup = function(groupid, callback) {
114             var group = {};
115             // get batch group data for this group
116             var batchRequests = [
117                 {
118                     url: "/var/joinrequests/list.json?groupId=" + groupid,
119                     method: "GET"
120                 },
121                 {
122                     url: '/system/userManager/group/' + groupid + '.json',
123                     method: "GET"
124                 }
125             ];
126             sakai.api.Server.batch(batchRequests, function (success, data) {
127                 if (success && data && data.results && data.results.length) {
128                     var participants = [];
129                     // join requests
130                     if (data.results[0].body) {
131                         var joinrequests = $.parseJSON(data.results[0].body);
132                         group.joinrequests = joinrequests;
133                     }
134                     // joinability info
135                     if (data.results[1].body) {
136                         var groupdata = $.parseJSON(data.results[1].body);
137                         group.groupProfile = groupdata.properties;
138                         group.joinability = groupdata.properties['sakai:group-joinable'];
139                         group.title = groupdata.properties['sakai:group-title'];
140                         group.id = groupid;
141                     }
142                     sakai.api.Groups.getMembers(groupid, function(success, members) {
143                         members = members[groupid];
144                         group.groupMembers = members;
145 
146                         $.each(members, function(role, users) {
147                             if (users.results) {
148                                 $.each(users.results, function(index, user) {
149                                     push_member_to_list(user, participants, role);
150                                 });
151                             }
152                         });
153 
154                         if (group.groupMembers.Manager && group.groupMembers.Manager.results){
155                             group.managerCount = group.groupMembers.Manager.results.length;
156                         }
157                         group.totalParticipants = participants.length;
158                         if (participants.length > 1) {
159                             participants = participants.sort(participantSort);
160                         }
161                         if (participants.length > 5) {
162                             participants = participants.slice(0, 5);
163                             group.seeAll = true;
164                         } else {
165                             group.seeAll = false;
166                         }
167                         group.participants = participants;
168                         if ($.isFunction(callback)){
169                             callback(group);
170                         }
171                     }, true);
172                 } else {
173                     debug.error("Batch request to fetch group (id: " + id + ") data failed.");
174                 }
175             });
176             return group;
177         };
178 
179         var openTooltip = function (groupid, $item, leaveAllowed) {
180             getGroup(groupid, function(group) {
181                 group.sakai = sakai;
182                 $(document).trigger('init.tooltip.sakai', {
183                     tooltipHTML: sakai.api.Util.TemplateRenderer($joingroup_hover_template, group),
184                     tooltipAutoClose: true,
185                     tooltipArrow: "top",
186                     tooltipTop: $item.offset().top + $item.height(),
187                     tooltipLeft: $item.offset().left + $item.width() + 3,
188                     onShow: function () {
189                         $(document).trigger('init.joinrequestbuttons.sakai', [
190                             {
191                                 "groupProfile": group.groupProfile,
192                                 "groupMembers": group.groupMembers,
193                                 "leaveAllowed": leaveAllowed
194                             },
195                             groupid,
196                             group.joinability,
197                             group.managerCount,
198                             false,
199                             function (renderedButtons) {
200                                 // onShow
201                                 $("#joingroup_joinrequestbuttons").html(
202                                     renderedButtons.html());
203                             },
204                             function (success, id) {
205                                 // requestCallback
206                                 if (success) {
207                                     // reset joinrequest data
208                                     group.joinrequests = false;
209                                 }
210                             },
211                             function (success, id) {
212                                 // joinCallback
213                                 if (success) {
214                                     // re-render tooltip
215                                     resetTooltip(groupid, $item);
216                                     $("#searchgroups_memberimage_" + groupid).show();
217                                     $("#searchgroups_memberimage_" + groupid).parent().removeClass("s3d-actions-addtolibrary");
218                                     adjustParticipantCount(groupid, 1);
219                                 }
220                             },
221                             function (success, id) {
222                                 // leaveCallback
223                                 if (success) {
224                                     // re-render tooltip
225                                     resetTooltip(groupid, $item);
226                                     $("#searchgroups_memberimage_" + groupid).hide();
227                                     $("#searchgroups_memberimage_" + groupid).parent().addClass("s3d-actions-addtolibrary");
228                                     adjustParticipantCount(groupid, -1);
229                                 }
230                             },
231                             group.joinrequests
232                         ]);
233                     }
234                 });
235             });
236         };
237 
238         var resetTooltip = function (groupid, $item) {
239             $(document).trigger('done.tooltip.sakai');
240             openTooltip(groupid, $item);
241         };
242 
243         /////////////////////////////
244         // Initialization function //
245         /////////////////////////////
246 
247         var doInit = function () {
248             $(document).on('click', '.sakai_joingroup_overlay', function(ev) {
249                 var $el = $(this);
250                 var groupid = $el.attr('data-groupid');
251                 if (groupid) {
252                     sakai.api.Groups.isAllowedToLeave(groupid, sakai.data.me, function(leaveAllowed) {
253                         openTooltip(groupid, $el, leaveAllowed[groupid]);
254                     });
255                 }
256                 return false;
257             });
258         };
259 
260         // run the initialization function when the widget object loads
261         doInit();
262 
263     };
264 
265     // inform Sakai OAE that this widget has loaded and is ready to run
266     sakai.api.Widgets.widgetLoader.informOnLoad("joingroup");
267 });
