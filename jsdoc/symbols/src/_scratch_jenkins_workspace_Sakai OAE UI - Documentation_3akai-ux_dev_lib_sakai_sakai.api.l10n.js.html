 18 /**
 19  * @class l10n
 20  *
 21  * @description
 22  * Localisation related functions for general page content, widget
 23  * content and UI elements This should only hold functions
 24  * which are used across multiple pages, and does not constitute functionality
 25  * related to a single area/page
 26  *
 27  * @namespace
 28  * Language localisation
 29  */
 30 define(
 31     [
 32         "config/config_custom",
 33         "misc/l10n/globalize",
 34         "misc/l10n/detect_timezone"
 35     ],
 36     function(sakai_conf) {
 37     return {
 38 
 39         /**
 40          * Get the current logged in user's locale
 41          *
 42          * @return {String} The user's locale string in XXX format
 43          */
 44         getUserLocale : function() {
 45 
 46         },
 47 
 48         getUserDefaultLocale : function() {
 49             var ret = sakai_conf.defaultLanguage;
 50             // Get the browser language preference - IE uses userLanguage, all other browsers user language
 51             var locale = navigator.language ? navigator.language : navigator.userLanguage;
 52             if (locale) {
 53                 var split = locale.split("-");
 54                 if (split.length > 1) {
 55                     split[1] = split[1].toUpperCase();
 56                     var langs = sakai_conf.Languages;
 57                     // Loop over all the available languages - if the user's browser language preference matches
 58                     // then set their locale to that so they don't have to set it manually
 59                     var i;
 60                     for (i=0,j=langs.length; i<j; i++) {
 61                         if (langs[i].language === split[0] && langs[i].country === split[1]) {
 62                             ret = split[0] + "_" + split[1];
 63                             break;
 64                         }
 65                     }
 66                 }
 67             }
 68             return ret;
 69         },
 70 
 71         getUserDefaultTimezone : function() {
 72             var tz = jstz.determine_timezone();
 73             return tz.name();
 74         },
 75 
 76         /**
 77          * Parse a date string into a date object and adjust that date to the timezone
 78          * set by the current user.
 79          * @param {Object} dateString    date to parse in the format 2010-10-06T14:45:54+01:00
 80          * @param {Object} meData the data from sakai.api.User.data.me
 81          */
 82         parseDateString : function(dateString, meData){
 83             var d = new Date();
 84             d.setFullYear(parseInt(dateString.substring(0,4),10));
 85             d.setMonth(parseInt(dateString.substring(5,7),10) - 1);
 86             d.setDate(parseInt(dateString.substring(8,10),10));
 87             d.setHours(parseInt(dateString.substring(11,13),10));
 88             d.setMinutes(parseInt(dateString.substring(14,16),10));
 89             d.setSeconds(parseInt(dateString.substring(17,19),10));
 90             // Localization
 91             if (!isNaN((parseInt(dateString.substring(19,22),10)))){
 92                 d.setTime(d.getTime() - (parseInt(dateString.substring(19,22),10)*60*60*1000));
 93             }
 94             if (meData && meData.user.locale) {
 95                 d.setTime(d.getTime() + meData.user.locale.timezone.GMT * 60 * 60 * 1000);
 96             }
 97             return d;
 98         },
 99         
100         parseDateLong : function(dateLong, meData){
101             var d = new Date(dateLong);
102             return d;
103         },
104 
105         /**
106          * Determines date from an epoch string in UTC and returns a date object and adjust that date to the timezone
107          * set by the current user.
108          * @param {Object} dateString    date epoch
109          * @param {Object} meData the data from sakai.api.User.data.me
110          */
111         fromEpoch : function(dateString, meData){
112             var d = new Date(parseInt(dateString,10));
113             var UTCDate = new Date(Date.UTC(d.getUTCFullYear(), d.getUTCMonth(), d.getUTCDate(), d.getUTCHours(), d.getUTCMinutes(), d.getUTCSeconds()));
114             if (meData && meData.user.locale) {
115                 UTCDate.setHours(UTCDate.getUTCHours() + meData.user.locale.timezone.GMT);
116                 // Check to see if UTC is a day in the future
117                 if (UTCDate.getUTCHours() + meData.user.locale.timezone.GMT < 0) {
118                     UTCDate.setDate(UTCDate.getUTCDate());
119                 }
120             }
121             return UTCDate;
122         },
123 
124         /**
125          * Function that will take in a date object and will transform that
126          * date object into a date only string
127          * @param {Date} date
128          *  JavaScript date object that we would like to transform in a
129          *  date string
130          * @return {String} Localized formatted date string
131          */
132         transformDate : function(date){
133             return Globalize.format(date, 'd');
134         },
135 
136         /**
137          * Function that will take in a date object and will transform that
138          * date object into a time only string
139          * @param {Date} date
140          *  JavaScript date object that we would like to transform in a
141          *  time string
142          * @return {String} Localized formatted time string
143          */
144         transformTime : function(date){
145             return Globalize.format(date, 't');
146         },
147 
148         /**
149          * Function that will take in a date object and will transform that
150          * date object into a date and time string
151          * @param {Date} date
152          *  JavaScript date object that we would like to transform in a
153          *  date and time string
154          * @return {String} Localized fomatted date and time string
155          */
156         transformDateTime : function(date){
157             return Globalize.format(date, 'F');
158         },
159 
160         /**
161          * Function to transform a date into a long date time string
162          * eg. Saturday, November 20, 2010 (for en_US)
163          * @param {Date} date
164          * @return {String} localized string
165          */
166         transformDateTimeLong : function(date) {
167             return Globalize.format(date, "D");
168         },
169 
170         /**
171          * Function to transform a date into a short date time string
172          * eg. 11/20/2010 5:12 PM (for en_US)
173          * @param {Date} date
174          * @return {String} localized string
175          */
176         transformDateTimeShort : function(date) {
177             return Globalize.format(date, "d") + " " + Globalize.format(date, "t");
178         },
179 
180         /**
181          * Function to localize a number including decimal places
182          * @param {Number} num a number
183          * @param {Integer} decimalplaces the number of decimal places to use
184          * @return {Number} the number formatted
185          */
186         transformDecimal : function(num, decimalplaces) {
187             return Globalize.format(num, "n" + decimalplaces);
188         },
189 
190         /**
191          * Function that will take in a date object and will transform that
192          * date object into a GMT date object. This should always be done before
193          * we try to save a date back to a file or the database. The timezone
194          * we are currently in will be determined from our timezone set in the
195          * personal preferences page
196          * @param {Date} date
197          *  JavaScript date object that we would like to transform in a
198          *  GMT date object
199          * @param {Object} meData the data from sakai.api.User.data.me
200          * @return {Date}
201          *  Date object, that will have transformed the given date and time into
202          *  GMT date and time
203          */
204         toGMT : function(date, meData){
205             date.setHours(date.getHours() - meData.user.locale.timezone.GMT);
206             return date;
207         },
208 
209         /**
210          * Function that will take in a GMT date object and will transform that
211          * date object into a local date object. This should always be done after
212          * we load a date back from a file or the database. The timezone
213          * we are currently in will be determined from our timezone set in the
214          * personal preferences page
215          * @param {Date} date
216          *  JavaScript GMT date object that we would like to transform to a local date object
217          * @param {Object} meData the data from sakai.api.User.data.me
218          * @return {Date}
219          *  Date object, that will have transformed the given GMT date and time into
220          *  a local date and time
221          */
222         fromGMT : function(date, meData){
223             date.setHours(date.getHours() + meData.user.locale.timezone.GMT);
224             return date;
225         },
226 
227         /**
228          * Function that will take in a JavaScript Number and will transform it into
229          * a localised number string that complies with decimal points and character used as separator
230          * as specified in the config file
231          * @param {Number} number
232          * Number we want to localise (eg 10000000.442)
233          * @return {String}
234          * Localised string of the number given to this function (eg "10.000.000,442")
235          */
236         transformNumber : function(number){
237             return Globalize.format(number, "n");
238         },
239 
240         getDateFormatString : function() {
241             var pattern = Globalize.cultures[require("sakai/sakai.api.i18n").data.culture].calendar.patterns.d;
242             var split = pattern.split("/");
243             var i;
244             for (i=0, j=split.length; i<j; i++) {
245                 if (split[i] === "m" || split[i] === "M") {
246                     split[i] = "mm";
247                 } else if (split[i] === "d" || split[i] === "D") {
248                     split[i] = "dd";
249                 }
250             }
251             return split.join("/").toUpperCase();
252         }
253     };
254 });
