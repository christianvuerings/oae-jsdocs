 29     /**
 30      * @name sakai_global.deletecontent
 31      *
 32      * @class deletecontent
 33      *
 34      * @description
 35      * Deletecontent widget

 37      * @version 0.0.1
 38      * @param {String} tuid Unique id of the widget
 39      * @param {Boolean} showSettings Show the settings of the widget or not
 40      */
 41     sakai_global.deletecontent = function(tuid, showSettings){
 42 
 43         //////////////////////
 44         // Global variables //
 45         //////////////////////
 46 
 47         var pathsToDelete = false;
 48         var contentIManage = false;
 49         var contentIView = false;
 50         var context = false;
 51         var callback = false;
 52         var contextType = false;
 53 
 54         ///////////////////
 55         // CSS Selectors //
 56         ///////////////////
 57 
 58         var $deletecontent_dialog = $("#deletecontent_dialog");
 59 
 60         ////////////////////////////
 61         // Batch request handling //
 62         ////////////////////////////
 63 
 64         /**
 65          * Once all requests have been collected into a batchRequest array, we can submit them as
 66          * a batch request
 67          * @param {Object} batchRequests    Array that contains all batch requests to be submitted
 68          * @param {Object} successMessage   Id of the dom element that contains the success message to be displayed
 69          */
 70         var sendDeletes = function(batchRequests, successMessage) {
 71             sakai.api.Util.progressIndicator.showProgressIndicator(
 72                 sakai.api.i18n.getValueForKey('REMOVING_CONTENT'),
 73                 sakai.api.i18n.getValueForKey('PROCESSING_REMOVING'));
 74             // Update the inserter
 75             $.each(collectionsToUpdate, function(collectionId, amount) {
 76                 $.each(sakai.api.User.data.me.groups, function(index, group){
 77                     if (group && group.counts && group.groupid === collectionId) {
 78                         group.counts.contentCount -= amount;
 79                         collectionId = collectionId.substring(2,collectionId.length);
 80                         $(window).trigger('updateCount.inserter.sakai', [collectionId, group.counts.contentCount]);
 81                     }
 82                 });
 83             });
 84             sakai.api.Server.batch(batchRequests, function (success, data) {
 85                 if (success) {
 86                     sakai.api.Util.notification.show($("#deletecontent_message_title").html(), $(successMessage).html());
 87                 } else {
 88                     sakai.api.Util.error.show($("#deletecontent_message_title").html(), $("#deletecontent_message_error").html()); 
 89                 }
 90 
 91                 $(document).trigger('done.deletecontent.sakai', [pathsToDelete]);
 92                 if ($.isFunction(callback)) {
 93                     callback(success);
 94                 }
 95                 sakai.api.Util.progressIndicator.hideProgressIndicator();
 96                 sakai.api.Util.Modal.close($deletecontent_dialog);
 97             });
 98         };
 99 
100         //////////////////////////////
101         // Remove from library only //
102         //////////////////////////////
103 
104         /**
105          * Add one request for each item to be removed from the current library
106          * @param {Object} batchRequests    Array to which to add the requests for removing the content
107          * @param {Object} items            Content items to be removed from the current library
108          */
109         var processRemoveFromLibrary = function(batchRequests, items){
110             batchRequests = batchRequests || [];
111             for (var i = 0; i < items.length; i++){
112                 var parameters = {};
113                 if (sakai.api.Content.Collections.isCollection(items[i])) {
114                     var groupId = sakai.api.Content.Collections.getCollectionGroupId(items[i]);
115                     batchRequests.push({
116                         "url": "/system/userManager/group/" + groupId + "-members.update.json",
117                         "method": "POST",
118                         "parameters": {
119                             ":viewer@Delete": context,
120                             ":member@Delete": context
121                         }
122                     });
123                     batchRequests.push({
124                         'url': '/system/userManager/group/' + groupId + '-editors.update.json',
125                         'method': 'POST',
126                         'parameters': {
127                             ':viewer@Delete': context,
128                             ':member@Delete': context
129                         }
130                     });
131                     batchRequests.push({
132                         "url": "/system/userManager/group/" + groupId + "-managers.update.json",
133                         "method": "POST",
134                         "parameters": {
135                             ":viewer@Delete": context,
136                             ":member@Delete": context
137                         }
138                     });
139                 } else {
140                     parameters[':manager@Delete'] = context;
141                     parameters[':editor@Delete'] = context;
142                     parameters[':viewer@Delete'] = context;
143                     batchRequests.push({
144                         "url": "/p/" + items[i]["_path"] + ".members.json",
145                         "method": "POST",
146                         "parameters": parameters
147                     });
148                 }
149             }
150         };
151 
152         /**
153          * Remove the selected items from the current library only and keep them onto the
154          * system.
155          */
156         var removeFromLibrary = function(){
157             var batchRequests = [];
158             processRemoveFromLibrary(batchRequests, contentIView);
159             processRemoveFromLibrary(batchRequests, contentIManage);
160             sendDeletes(batchRequests, (contextType === "collection" ? "#deletecontent_message_from_collection" : "#deletecontent_message_from_library"));
161         };
162 
163         ////////////////////////////
164         // Remove from the system //
165         ////////////////////////////
166 
167         /**
168          * Add one request for each item to delete from the system
169          * @param {Object} batchRequests    Array to which to add the requests for removing the content
170          * @param {Object} items            Content items to be removed from the system
171          */
172         var processRemoveFromSystem = function(batchRequests, items){
173             batchRequests = batchRequests || [];
174             for (var i = 0; i < items.length; i++){
175                 batchRequests.push({
176                     "url": "/p/" + items[i]["_path"],
177                     "method": "POST",
178                     "parameters": {
179                         ":operation": "delete"
180                     }
181                 });
182                 // Remove the pseudoGroups associated to the collection
183                 var collectionGroupId = sakai.api.Content.Collections.getCollectionGroupId(items[i]);
184                 if (sakai.api.Content.Collections.isCollection(items[i])) {
185                     batchRequests.push({
186                         "url": "/system/userManager.delete.json",
187                         "method": "POST",
188                         "parameters": {
189                             ":applyTo": [collectionGroupId, collectionGroupId + "-members", collectionGroupId + "-managers"]
190                         }
191                     });
192                 }
193             }
194         };
195 
196         /**
197          * Remove the selected items from the system and thus from all libraries where this is being
198          * used
199          */
200         var removeFromSystem = function(){
201             // Remove content I manage from the system
202             var batchRequests = [];
203             processRemoveFromLibrary(batchRequests, contentIView);
204             processRemoveFromSystem(batchRequests, contentIManage);
205             sendDeletes(batchRequests, "#deletecontent_message_from_system");
206         };
207 
208         var collectionsToUpdate = {};
209 
210         /**
211          * Check whether any users or groups are either managers or viewers from
212          * any of the selected content items
213          */
214         var checkUsedByOthers = function(){
215             collectionsToUpdate = {};
216             var userGroupIds = [];
217             var collectionsToCheck = [];
218             // Check whether any of the content I manage is managed by or
219             // shared with other people
220             $.each(contentIManage, function(m, contentItem) {
221                 if (sakai.api.Content.Collections.isCollection(contentItem)) {
222                     var collectionGroupId = sakai.api.Content.Collections.getCollectionGroupId(contentItem);
223                     collectionsToCheck.push(collectionGroupId + '-members');
224                     collectionsToCheck.push(collectionGroupId + '-editors');
225                     collectionsToCheck.push(collectionGroupId + '-managers');
226                 } else {
227                     var managers = contentItem['sakai:pooled-content-manager'];
228                     if (managers){
229                         for (var i = 0; i < managers.length; i++){
230                             if ($.inArray(managers[i], userGroupIds) === -1 && managers[i] !== sakai.data.me.user.userid &&
231                             managers[i] !== context){
232                                 userGroupIds.push(managers[i]);
233                             }
234                         }
235                     }
236                     var editors = contentItem['sakai:pooled-content-editor'];
237                     if (editors && editors.length) {
238                         $.each(editors, function(idx, editor) {
239                             if ($.inArray(editor, userGroupIds) === -1 && editor !== sakai.data.me.user.userid && editor !== context) {
240                                 userGroupIds.push(editor);
241                             }
242                         });
243                     }
244                     var viewers = contentItem['sakai:pooled-content-viewer'];
245                     if (viewers){
246                         for (var j = 0; j < viewers.length; j++){
247                             if ($.inArray(viewers[j], userGroupIds) === -1 && viewers[j] !== sakai.data.me.user.userid &&
248                                 viewers[j] !== context && viewers[j] !== "everyone" && viewers[j] !== "anonymous"){
249                                 userGroupIds.push(viewers[j]);
250                                 if (sakai.api.Content.Collections.isCollection(viewers[j])) {
251                                      collectionsToUpdate[viewers[j]] = collectionsToUpdate[viewers[j]] || 0;
252                                      collectionsToUpdate[viewers[j]] += 1;
253                                  }
254                             }
255                         }
256                     }
257                 }
258             });
259             if (collectionsToCheck.length > 0) {
260                 var batchRequest = [];
261                 $.each(collectionsToCheck, function(index, collectiongroup){
262                     batchRequest.push({
263                         "url": "/system/userManager/group/" + collectiongroup + ".members.json",
264                         "method": "GET",
265                         "parameters": {
266                             items: 10000
267                         }
268                     });
269                 });
270                 sakai.api.Server.batch(batchRequest, function (success, data) {
271                     for (var i = 0; i < data.results.length; i++) {
272                         var members = $.parseJSON(data.results[i].body);
273                         for (var ii = 0; ii < members.length; ii++) {
274                             var member = members[ii].userid || members[ii].groupid;
275                             if ($.inArray(member, userGroupIds) === -1 &&
276                                 member !== sakai.data.me.user.userid &&
277                                 member !== context) {
278                                 userGroupIds.push(member);
279                                 if (sakai.api.Content.Collections.isCollection(member)) {
280                                     collectionsToUpdate[member] = collectionsToUpdate[member] || 0;
281                                     collectionsToUpdate[member] += 1;
282                                 }
283                             }
284                         }
285                     }
286                     if (userGroupIds.length > 0) {
287                         setUpUsedByOverlay(userGroupIds);
288                     } else {
289                         removeFromSystem();
290                     }
291                 });
292             } else {
293                 if (userGroupIds.length > 0) {
294                     setUpUsedByOverlay(userGroupIds);
295                 } else {
296                     removeFromSystem();
297                 }
298             }
299         };
300 
301         /**
302          * When the content the user is trying to delete from the system is
303          * being used by others, present an overlay that lists all of the
304          * groups and users that either use or manage the content
305          * @param {Object} userGroupIds    Array that contains the userids and groupids of all
306          *                                 users and groups using the content
307          */
308         var setUpUsedByOverlay = function(userGroupIds){
309             // Show the overview screen of who else is using this
310             $("#deletecontent_used_by_others_container").html("");
311             $("#deletecontent_container").hide();
312             $("#deletecontent_used_by_others").show();
313             // Set up the buttons correctly
314             hideButtons();
315             $("#deletecontent_action_removefromsystem_confirm").show();
316             if (context && contextType === "collection"){
317                 $("#deletecontent_action_removefromcollection_only").show();
318             } else if (context){
319                 $("#deletecontent_action_removefromlibrary_only").show();
320             }
321             // Show the correct overlay title
322             $("#deletecontent_main_content").hide();
323             $("#deletecontent_main_confirm").show();
324             // Get the profile information of who else is using it
325             var batchRequests = [];
326             for (var i = 0; i < userGroupIds.length; i++){
327                 batchRequests.push({
328                     "url": "/~" + userGroupIds[i] + "/public/authprofile.profile.json",
329                     "method": "GET"
330                 });
331             }
332             // Get profile information for each of the users and groups using
333             // this content
334             sakai.api.Server.batch(batchRequests, function (success, data) {
335                 var profileInfo = [];
336                 for (var i = 0; i < data.results.length; i++){
337                     if (data.results[i].success && data.results[i].status !== 404) {
338                         // Process pseudoGroups
339                         var profile = $.parseJSON(data.results[i].body);
340                         profile.showLink = true;
341                         if (sakai.api.Content.Collections.isCollection(profile)){
342                             profile.collectionid = sakai.api.Content.Collections.getCollectionPoolId(profile);
343                         } else if (profile["sakai:excludeSearch"] === "true"){
344                             var splitOnDash = profile.groupid.split("-");
345                             profile["sakai:group-title"] = profile["sakai:parent-group-title"] + " (" + sakai.api.i18n.getValueForKey(profile["sakai:role-title-plural"]) + ")";
346                             profile.groupid = splitOnDash.splice(0, splitOnDash.length - 1).join("-");
347                         }
348                         profileInfo.push(profile);
349                     } else {
350                         var privateProfile = {
351                             showLink: false
352                         };
353                         if (data.results[i].url.substring(0,4) === '/~c-') {
354                             privateProfile.collectionid = true;
355                             privateProfile['sakai:group-title'] =
356                                 sakai.api.i18n.getValueForKey('PRIVATE_COLLECTION', 'deletecontent');
357                         } else {
358                             privateProfile['sakai:group-title'] =
359                                 sakai.api.i18n.getValueForKey('PRIVATE_USER_GROUP', 'deletecontent');
360                         }
361                         profileInfo.push(privateProfile);
362                     }
363                 }
364                 $("#deletecontent_used_by_others_container").html(sakai.api.Util.TemplateRenderer("deletecontent_used_by_others_template", {
365                     "profiles": profileInfo,
366                     "sakai": sakai
367                 }));
368             });
369         };

371         ////////////////////////////
372         // Remove hybrid strategy //
373         ////////////////////////////
374 
375         /**
376          * Check whether the users has chosen to remove the content he manages from his
377          * library only or from the system. If removing from the library only, we can
378          * go ahead and remove the content. If removing from the system, we want to check
379          * first whether the content is being used by anyone else
380          */
381         var selectHybrid = function(){
382             var manageOption = $("input[name='deletecontent_hybrid_options']:checked").val();
383             if (manageOption === "libraryonly") {
384                 removeFromLibrary();
385             } else if (manageOption === "system") {
386                 checkUsedByOthers();
387             }
388         };

391         // Overlay setup //
392         ///////////////////

394         /**
395          * Hide all of the action buttons in the overlay
396          */
397         var hideButtons = function(){
398             $("#deletecontent_action_removefromsystem").hide();
399             $("#deletecontent_action_removefromsystem_nocontext").hide();
400             $("#deletecontent_action_removefromlibrary").hide();
401             $("#deletecontent_action_removefromcollection").hide();
402             $("#deletecontent_action_apply").hide();
403             $("#deletecontent_action_removefromsystem_confirm").hide();
404             $("#deletecontent_action_removefromlibrary_only").hide();
405             $("#deletecontent_action_removefromcollection_only").hide();
406         };

408         /**
409          * Set up the delete overlay depending on the permissions I have on the content
410          * about to be deleted from the overlay
411          * There are 3 scenarios:
412          * 1. I am a manager of some items and a viewer of others
413          * 2. I am a manager of all items
414          * 3. I am a viewer of all items
415          * @param {Object} contentIManage    Array that contains all files about to be
416          *                                   removed from the library that I manage
417          * @param {Object} contentIView      Array that contains all files about to be
418          *                                   removed from the library that I'm a viewer of
419          */
420         var setupOverlay = function(contentIManage, contentIView){
421             hideButtons();
422             var template = "";
423             if (contentIManage.length > 0 && contentIView.length > 0){
424                 // Set up overlay for mixed permissions
425                 template = "deletecontent_template_hybrid";
426                 $("#deletecontent_action_apply").show();
427             } else if (contentIManage.length > 0){
428                 // Set up overlay for full management permissions
429                 template = "deletecontent_template_list";
430                 if (context){
431                     $("#deletecontent_action_removefromsystem").show();
432                     if (contextType === "collection"){
433                         if (sakai_global.content_profile && sakai_global.content_profile.content_data) {
434                             var managerCid = 'c-' + sakai_global.content_profile.content_data.data._path;
435                             collectionsToUpdate[managerCid] = collectionsToUpdate[managerCid] || 0;
436                             collectionsToUpdate[managerCid] += contentIManage.length;
437                         }
438                         $("#deletecontent_action_removefromcollection").show();
439                     } else {
440                         $("#deletecontent_action_removefromlibrary").show();
441                     }
442                 // When no context/library is specified, we assume that the content is being deleted outside
443                 // of a library (e.g. content profile). We thus don't offer the remove from library option
444                 } else {
445                     $("#deletecontent_action_removefromsystem_nocontext").show();
446                 }
447             } else if (contentIView.length > 0){
448                 // Set up overlay for full viewer permissions
449                 template = "deletecontent_template_list";
450                 if (contextType === "collection") {
451                     if (sakai_global.content_profile && sakai_global.content_profile.content_data) {
452                         var viewerCid = 'c-' + sakai_global.content_profile.content_data.data._path;
453                         collectionsToUpdate[viewerCid] =
454                             collectionsToUpdate[viewerCid] || 0;
455                         collectionsToUpdate[viewerCid] += contentIView.length;
456                     }
457                     $("#deletecontent_action_removefromcollection").show();
458                 } else {
459                     $("#deletecontent_action_removefromlibrary").show();
460                 }
461             }
462             $("#deletecontent_container").html(sakai.api.Util.TemplateRenderer(template, {
463                 "contentIManage": contentIManage,
464                 "contentIView": contentIView,
465                 "contextType": contextType,
466                 "sakai": sakai
467             }));
468         };
469 
470         /**
471          * Run over the list of content items to delete and determine whether there
472          * any that I manage and can thus remove from the system
473          * @param {Object} contentList    Response from batch request that retrieved
474          *                                metadata for all content that need to be deleted
475          */
476         var findContentIManage = function(contentList){
477             contentIManage = []; 
478             contentIView = [];
479             $.each(contentList.results, function (i, contentItem) {
480                 var content = $.parseJSON(contentItem.body);
481                 var manage = sakai.api.Content.isUserAManager(content, sakai.data.me);
482                 if (manage){
483                     contentIManage.push(content);
484                 } else {
485                     contentIView.push(content);
486                 }
487             });
488             setupOverlay(contentIManage, contentIView);
489         };
490 
491         /**
492          * Retrieve the metadata of all selected files
493          * @param {Object} paths    Array that contains the paths to all
494          *                          content that needs to be deleted
495          */
496         var getContentInfo = function(paths){
497             var batchRequest = [];
498             $.each(paths, function (i, url) {
499                 batchRequest.push({
500                     url: "/p/" + url + ".json",
501                     method: "GET"
502                 });
503             });
504             sakai.api.Server.batch(batchRequest, function (success, data) {
505                 if (success) {
506                     findContentIManage(data);
507                 }
508             });
509         };

511         ////////////////////
512         // Initialisation //
513         ////////////////////
514 
515         /**
516          * Load the delete content widget with the appropriate data
517          * This function can be called from anywhere within Sakai by triggering the
518          * 'init.deletecontent.sakai' event
519          *
520          * @param {Object} data A JSON object containing the necessary information.
521          *
522          * @example To delete one item:
523          *     $(window).trigger('init.deletecontent.sakai', [{
524          *         "path": [ "/test.jpg" ]
525          *     }, callbackFn]);  // callbackFn is sent one param: success (true if delete succeeded, false otherwise)
526          *
527          * @example To delete multiple items:
528          *     $(window).trigger('init.deletecontent.sakai', [{
529          *         "path": [ "/file1.ext", "/file2.ext", "/file3.ext", "/file4.ext" ]
530          *     }, callbackFn]);  // callbackFn is sent one param: success (true if delete succeeded, false otherwise)
531          */
532         var load = function(ev, data, _callback){
533             context = data.context;
534             contextType = "default";
535             if (context && sakai.api.Content.Collections.isCollection(context)){
536                 contextType = "collection";
537             }
538             callback = _callback;
539             pathsToDelete = data.paths;
540             getContentInfo(data.paths);
541             hideButtons();
542             // Show the appropriate overlay title
543             $("#deletecontent_main_confirm").hide();
544             $("#deletecontent_main_content").show();
545             // Show and clear the main container
546             $("#deletecontent_container").html("");
547             $("#deletecontent_container").show();
548             $("#deletecontent_used_by_others").hide();
549             sakai.api.Util.Modal.open($deletecontent_dialog);
550         };
551 
552         /**
553          * Initialize the delete content widget
554          * All the functionality in here is loaded before the widget is actually rendered
555          */
556         var init = function(){
557             // This will make the widget popup as a layover.
558             sakai.api.Util.Modal.setup($deletecontent_dialog, {
559                 modal: true,
560                 toTop: true
561             });
562         };
563 
564         ////////////////////////////
565         // Internal event binding //
566         ////////////////////////////
567 
568         $("#deletecontent_action_removefromlibrary").bind("click", removeFromLibrary);
569         $("#deletecontent_action_removefromcollection").bind("click", removeFromLibrary);
570         $("#deletecontent_action_removefromsystem").bind("click", checkUsedByOthers);
571         $("#deletecontent_action_apply").bind("click", selectHybrid);
572         $("#deletecontent_action_removefromlibrary_only").bind("click", removeFromLibrary);
573         $("#deletecontent_action_removefromcollection_only").bind("click", removeFromLibrary);
574         $("#deletecontent_action_removefromsystem_confirm").bind("click", removeFromSystem);
575         $("#deletecontent_action_removefromsystem_nocontext").bind("click", checkUsedByOthers);
576 
577         ////////////////////////////
578         // External event binding //
579         ////////////////////////////
580 
581         $(document).on('init.deletecontent.sakai', load);
582 
583         init();
584     };

586     sakai.api.Widgets.widgetLoader.informOnLoad("deletecontent");
