146         /**
147          * Check if the input given by the user is valid
148          * @return {Boolean} true if input is valid
149          */
150         var checkIfInputValid =function(){
151             var pass = $(currentPassTxt).val();
152             var newPass1 = $(newPassTxt).val();
153             var newPass2 = $(newRetypePassTxt).val();
154 
155             // check if the user didn't just fill in some spaces
156             return (pass.replace(/ /g, "") !== "" && newPass1.replace(/ /g, "") !== "" && newPass2.replace(/ /g, "") !== "");
157         };
158 
159         /**
160          * Clears all the password fields
161          */
162         var clearPassFields = function(){
163             $(currentPassTxt).val("");
164             $(newPassTxt).val("");
165             $(newRetypePassTxt).val("");
166         };
167 
168         /**
169          * Makes all the checks
170          * are the new passwords equal
171          *
172          */
173         var changePass = function(){
174             var pass = $(currentPassTxt).val();
175             var newPass1 = $(newPassTxt).val();
176             var newPass2 = $(newRetypePassTxt).val();
177 
178                 /*
179                  * oldPassword : the original password
180                  * password : the new password
181                  */
182                 var requestbody = {"oldPwd" : pass, "newPwd" : newPass1, "newPwdConfirm" : newPass2, "_charset_": "utf-8"};
183 
184                 $.ajax({
185                     url :sakai.config.URL.USER_CHANGEPASS_SERVICE.replace(/__USERID__/, sakai.data.me.user.userid),
186                     type : "POST",
187                     data : requestbody,
188                     success : function(data) {
189 
190                         // show successful password change message through gritter
191                         sakai.api.Util.notification.show($(messagePassChanged).html(), $(messagePassChangedBody).html());
192                         // clear all the fields
193                         clearPassFields();
194                         passwordChanges = false;
195                         finishSave();
196                     },
197                     error: function(xhr, textStatus, thrownError) {
198 
199                         // show error message through gritter
200                         sakai.api.Util.notification.show($(errorFailChangePass).html(), $(errorFailChangePassBody).html());
201                         // clear all the fields
202                         clearPassFields();
203                     }
204                 });
205 
206         };

221             $option.addClass("s3d-outer-shadow-container");

236                 sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("PRIVACY_SETTINGS", "accountpreferences"), sakai.api.i18n.getValueForKey("PRIVACY_SETTINGS_UPDATED", "accountpreferences"));

238                 finishSave();
239             });
240             ev.stopPropagation();
241             return false;

298         /**
299          * Gets all the languages supported and puts them in a combobox
300          */
301         var getLanguages = function(){
302             var langs = $.extend([], sakai.config.Languages);
303             if (sakai.config.displayDebugInfo) {
304                 langs.push({
305                     "country": "GB",
306                     "displayName": "i18n debug",
307                     "language": "lu"
308                 });
309             }
310             var languages = {languages:$.extend(langs, {}, true)};
311             putLangsinCmb(languages);
312         };

314         /**
315          * Saves the regional properties to JCR
316          */
317         var saveRegionalToMe = function(){
318             var language = $(languagesContainer).val();
319             var isAutoTagging = $("input:radio[name='autotagging']:checked").val();
320             var sendTagMsg = $("#tag_msg_info").is(':checked');
321             var locale = {
322                 "locale" : language, 
323                 "timezone" : $(timezonesContainer).val(), 
324                 "_charset_":"utf-8", 
325                 ":sakai:update-profile": false, 
326                 "isAutoTagging": isAutoTagging, 
327                 "sendTagMsg": sendTagMsg
328             };
329 
330             // if regional Setting and langauge is changed only then save the changes
331             //if ( || me.user.properties.isAutoTagging !== isAutoTagging || me.user.properties.sendTagMsg !== sendTagMsg) {
332             $.ajax({
333                 data: locale,
334                 url: "/system/userManager/user/" + me.user.userid + ".update.html",
335                 type: "POST",
336                 success: function(data){
337 
338                     if (language !== me.user.locale.language + "_" + me.user.locale.country || me.user.locale.timezone.name !== $(timezonesContainer).val()) {
339                         sakai.api.Util.notification.show($(messageChangeLangTitle).html(), $(messageChangeLang).html());
340                         // Reload the page if the language for a user has changed
341                         pageReload = true;
342                     } else {
343                         sakai.data.me.user.properties.isAutoTagging = isAutoTagging;
344                         sakai.data.me.user.properties.sendTagMsg = sendTagMsg;
345                         sakai.api.Util.notification.show($(messageChangeLangTitle).html(), $(messageChangeLang).html());
346                     }
347                     preferencesChanges = false;
348                     finishSave();
349 
350                 },
351                 error: function(xhr, textStatus, thrownError){
352                     // show regional setting error message through gritter
353                     sakai.api.Util.notification.show($(messageChangeLangTitle).html(), $(errorFailChangeLang).html());
354                 }
355             });
356         };
357 
358         var saveEmail = function() {
359             var emailVal = $.trim($('#accountpreferences_email').val());
360             var data = {'email': emailVal};
361             sakai.api.User.updateUserProfile(
362                 sakai.data.me.user.userid,
363                 "basic", data, false, {}, false, function(success, data) {
364                     if (success) {
365                         emailChanges = false;
366                         sakai.api.Util.notification.show(
367                             $(messageEmailChanged).html(),
368                             $(messageEmailChangedBody).html());
369                         $(accountPreferencesContainer).jqmHide();
370                     } else {
371                         sakai.api.Util.notification.show(
372                             $(messageEmailChanged).html(),
373                             $(messageEmailChangedBody).html());
374                     }
375                 }
376             );
377         };

379         $(saveButton).bind("click", function(){
380             if (preferencesChanges){
381                 $(accountPreferencesPreferencesForm).submit();
382             }
383             if (privacyChanges){
384                 $("#accountpreferences_privacy_change").submit();
385             }
386             if (passwordChanges){
387                 $(accountPasswordTab).click();
388                 if ($(accountPreferencesPasswordChange).valid()) {
389                     $(accountPreferencesPasswordChange).submit();
390                 }
391             }
392             if (emailChanges) {
393                 $("#accountpreferences_email_change").submit();
394             }
395         });

397         /**
398          * Hides the dialog box when saving is complete and reloads the page if needed
399          */
400         var finishSave = function(){
401             if (!preferencesChanges && !privacyChanges && !passwordChanges){
402                 sakai.api.Util.Modal.close(accountPreferencesContainer);
403                 if (pageReload){
404                     window.setTimeout(function(){
405                         document.location.reload();
406                     },2000);
407                 }
408             }
409         };

411         /**
412          * Initialise form validation
413          */
414         var initValidation = function(){
415             var validateOpts = {
416                 rules: {
417                     curr_pass: {
418                         minlength: 4
419                     },
420                     new_pass: {
421                         minlength: 4,
422                         newpw: true
423                     },
424                     retype_pass: {
425                         minlength: 4,
426                         equalTo: "#new_pass"
427                     }
428                 },
429                 messages: {
430                     retype_pass: {
431                         "equalTo": sakai.api.i18n.getValueForKey("PLEASE_ENTER_PASSWORD_TWICE", "accountpreferences")
432                     }
433                 },
434                 'methods': {
435                     'newpw': {
436                         'method': function(value, element) {
437                             return this.optional(element) || (value !== $('#curr_pass').val());
438                         },
439                         'text': $(errorPassSame).html()
440                     }
441                 },
442                 submitHandler: changePass
443             };
444 
445             // Initialize the validate plug-in
446             sakai.api.Util.Forms.validate($(accountPreferencesPasswordChange), validateOpts);
447 
448             var validatePreferencesOpts = {
449                 submitHandler: saveRegionalToMe
450             };
451 
452             // Initialize the validate plug-in
453             sakai.api.Util.Forms.validate($(accountPreferencesPreferencesForm), validatePreferencesOpts);
454 
455             var validateEmailOpts = {
456                 submitHandler: saveEmail
457             };
458 
459             // Initialize the validate plug-in
460             sakai.api.Util.Forms.validate($(accountPreferencesEmailChange), validateEmailOpts);
461         };

569         var doInit = function(){
570             if (!sakai.data.me.user.anon) {
571                 // An anonymous user shouldn't have access to this page
572                 clearPassFields();
573                 disableElements($(saveButton));
574                 selectTimezone(me.user.locale.timezone);
575                 selectAutoTagging(me.user.properties.isAutoTagging);
576                 selectSendTagMsg(me.user.properties.sendTagMsg);
577 
578                 getLanguages();
579                 initValidation();
580 
581                 // if allowpasswordchange is false then hide the regional setting
582                 if (!sakai.config.allowPasswordChange) {
583                     $(accountPasswordTab).hide();
584                     $(passChangeContainer).hide();
585                 }
586                 if (sakai.config.emailLocation !== 'accountpreferences') {
587                     $(accountEmailTab).hide();
588                     $(emailChangeContainer).hide();
589                 } else {
590                     var emailVal = sakai.api.User.getProfileBasicElementValue(
591                         sakai.data.me.profile,
592                         'email');
593                     $('#accountpreferences_email').val(emailVal);
594                 }
595                 if (sakai.config.displayTimezone || sakai.config.displayLanguage) {
596                     $accountpreferences_langloc_settings.show();
597                     if (sakai.config.displayTimezone) {
598                         $('#accountpreferences_select_timezone').show();
599                     }
600                     if (sakai.config.displayLanguage) {
601                         $('#accountpreferences_select_language').show();
602                     }
603                 }
604             }
605         };
