 27 require(["jquery", "sakai/sakai.api.core"], function($, sakai) {
 28 
 29     /**
 30      * @name sakai.contentpreview
 31      *
 32      * @class contentpreview
 33      *
 34      * @description
 35      * Initialize the contentpreview widget
 36      *
 37      * @version 0.0.1
 38      * @param {String} tuid Unique id of the widget
 39      * @param {Boolean} showSettings Show the settings of the widget or not
 40      */
 41     sakai_global.contentpreview = function(tuid,showSettings){
 42 
 43         var $rootel = $("#" + tuid);
 44         var obj = {};
 45         obj.type = "showpreview";
 46         var contentData = {};
 47 
 48         var determineDataType = function(){
 49             hidePreview();
 50             obj.type = "showpreview";
 51             obj.buttons = "default";
 52             var callback = null;
 53             var user = contentData.data["_bodyLastModifiedBy"];
 54             if (user === "admin") {
 55                 user = contentData.data["sakai:pool-content-created-for"];
 56             } else if (!user) {
 57                 user = contentData.data["_lastModifiedBy"];
 58             }
 59             sakai.api.User.getUser(user, function(success, userdata){
 60                 var mimeType = sakai.api.Content.getMimeType(contentData.data);
 61                 obj.userName = sakai.api.User.getDisplayName(userdata);
 62                 if ($.deparam.querystring().nopreview === 'true'){
 63                     callback = renderDefaultPreview;
 64                     obj.type = "default";
 65                 } else if (mimeType === "x-sakai/link"){
 66                     obj.buttons = "links";
 67                 }
 68                 if (sakai.api.Content.hasPreview(contentData.data)) {
 69                     callback = renderFullSizePreview;
 70                 } else {
 71                     obj.type = "default";
 72                 }
 73                 obj.sakai = sakai;
 74                 obj.contentData = contentData;
 75                 if (sakai_global && sakai_global.content_profile && (sakai_global.content_profile.content_data.data.mimeType !== 'x-sakai/collection' &&
 76                     sakai_global.content_profile.content_data.data.mimeType !== 'x-sakai/document')) {
 77                     $(".collectionviewer_widget .collectionviewer_collection_item_preview").remove();
 78                 }
 79                 sakai.api.Util.TemplateRenderer("contentpreview_widget_main_template", obj, $("#contentpreview_widget_main_container", $rootel));
 80                 if (callback) {
 81                     callback();
 82                 }
 83             });
 84         };
 85 
 86         var renderFullSizePreview = function(){
 87             var fullSizeContainer = $("#contentpreview_fullsize_preview", $rootel);
 88             var tuid = sakai.api.Util.generateWidgetId();
 89             var data = {};
 90             data[tuid] = contentData;
 91             sakai.api.Util.TemplateRenderer($("#contentpreview_fullsize_template", $rootel), {tuid: tuid}, fullSizeContainer);
 92             sakai.api.Widgets.widgetLoader.insertWidgets(fullSizeContainer, false, false, data);
 93         };
 94 
 95         var hidePreview = function(){
 96             $("#contentpreview_widget_main_container", $rootel).html("");
 97             $("#contentpreview_image_preview", $rootel).html("");
 98         };
 99 
100         if (!$rootel.parents(".collectionviewer_collection_item_preview").length){
101             $(window).bind("start.contentpreview.sakai", function(ev, data){
102                 contentData = data;
103                 determineDataType();
104             });
105 
106             $(window).bind("updated.version.content.sakai",function() {
107                 determineDataType();
108             });
109 
110             // Indicate that the widget has finished loading
111             sakai_global.contentpreview.isReady = true;
112             $(window).trigger("ready.contentpreview.sakai", {});
113         } else {
114             $rootel.parents('.collectionviewer_widget').bind('start.collectioncontentpreview.sakai', function(ev, data) {
115                 contentData = {data: data};
116                 determineDataType();
117             });
118 
119             // Indicate that the widget has finished loading
120             sakai_global.contentpreview.isReady = true;
121             $rootel.parents('.collectionviewer_widget').trigger('ready.collectioncontentpreview.sakai');
122         }
123     };
124 
125     sakai.api.Widgets.widgetLoader.informOnLoad("contentpreview");
126 });
