 19 // load the master sakai object to access all Sakai OAE API methods
 20 require(['jquery', 'sakai/sakai.api.core'], function($, sakai) {
 21 
 22     /**
 23      * @name sakai.htmlblock
 24      *
 25      * @class htmlblock
 26      *
 27      * @version 0.0.1
 28      * @param {String} tuid Unique id of the widget
 29      * @param {Boolean} showSettings Show the settings of the widget or not
 30      */
 31     sakai_global.htmlblock = function (tuid, showSettings, widgetData) {
 32 
 33         // Element cache
 34         var $rootel = $('#' + tuid);
 35         var $editor = false;
 36         var $toolbar = false;
 37 
 38         // Configuration variables
 39         var AUTOSAVE_INTERVAL = 5000;
 40         var EVENTS_TO_RESPOND_TO = ['click', 'keyup', 'mouseup', 'paste'];
 41 
 42         // Help variables
 43         var editorId = false;
 44         var autoSavePoll = false;
 45         var lastData = (widgetData && widgetData.htmlblock) ? widgetData.htmlblock.content : '';
 46 
 47         ///////////////////////////
 48         // tinyMCE Functionality //
 49         ///////////////////////////
 50 
 51         /**
 52          * When focus is set on the editor (by clicking inside of it), we hide the draghandle/remove
 53          * button bar, so all the text is visible
 54          */
 55         var editorFocus = function() {
 56             $('.mceExternalToolbar').hide();
 57             $('#inserterbar_widget #inserterbar_tinymce_container').show();
 58             $('.contentauthoring_cell_element_actions').hide();
 59             $toolbar.show();
 60         };
 61 
 62         /**
 63          * Make sure we remove all the insecure content
 64          * @param {Object} editor The editor
 65          * @param {Object} object Object containing all the information for the editor
 66          */
 67         var removeInsecureContent = function(editor, object) {
 68            // Run all the content through the HTML sanitizer
 69            object.content = sakai.api.Security.saneHTML(object.content);
 70            // Update the height - we need a setTimeout for this to work
 71            setTimeout(updateHeight, 100);
 72         };
 73 
 74         var editorSetup = function(ed) {
 75             ed.onClick.add(editorFocus);
 76             ed.onBeforeSetContent.add(removeInsecureContent);
 77         };
 78 
 79         /**
 80          * Convert the textarea in this widget into a tinyMCE editor
 81          */
 82         var loadTinyMCE = function() {
 83             if (window['tinyMCE']) {
 84                 tinyMCE.init({
 85                     language: sakai.api.i18n.getEditorLanguage(),
 86                     mode: 'textareas',
 87                     theme: 'advanced',
 88                     skin: 'sakai',
 89                     formats: {
 90                           'caption': {
 91                               'inline': 'span',
 92                               'classes': 'caption'
 93                           }
 94                     },
 95                     // CSS Files to load in the editor
 96                     content_css: '/dev/css/sakai/main.css',
 97                     // Plugins and toolbar buttons to show
 98                     plugins: 'table,advlink,contextmenu,paste,directionality',
 99                     theme_advanced_blockformats: 'h1,h2,h3,h4,h5,h6,p,blockquote,caption',
100                     theme_advanced_buttons1: 'bold,italic,underline,|,justifyleft,justifycenter,justifyright,justifyfull,|,formatselect,fontsizeselect,|,bullist,numlist,|,forecolor,|,link,table,code',
101                     theme_advanced_buttons2: '',
102                     theme_advanced_buttons3: '',
103                     // Styles to be shown for tables
104                     table_styles: 'Default=default',
105                     table_cell_styles: 'Default=default',
106                     table_row_styles: 'Default=default',
107                     // Specify a toolbar that is not attached to the editor
108                     theme_advanced_toolbar_location: 'external',
109                     theme_advanced_toolbar_align: 'left',
110                     theme_advanced_statusbar_location: 'none',
111                     theme_advanced_resizing: false,
112                     // Constrain the editor loading to the current widget
113                     editor_selector: tuid,
114                     // Resize the editor when a significant change is made
115                     handle_event_callback: tinyMCEEvent,
116                     onchange_callback: tinyMCEEvent,
117                     init_instance_callback: initTinyMCE,
118                     remove_instance_callback: stopAutoSave,
119                     // Additional event handlers
120                     setup : editorSetup
121                 });
122                 // Hide the widget controls
123                 var $containingCell = $('.htmlblock_widget', $rootel).parents('.contentauthoring_cell_element');
124                 $('.contentauthoring_cell_element_actions', $containingCell).addClass('s3d-force-hidden');
125             }
126         };
127 
128         /**
129          * This is executed when the tinyMCE editor has been initialized
130          */
131         var initTinyMCE = function(ui) {
132             editorId = ui.id;
133             // Set focus if there is no "An unsaved version has been found" overlay
134             // showing
135             var ed = tinyMCE.get(editorId);
136             if (!$('.s3d-dialog:visible').length) {
137                 ed.focus();
138             }
139             // Cache the editor elements
140             $editor = $('#' + editorId + '_ifr');
141             $toolbar = $('#' + editorId + '_external').hide();
142             // Move the toolbar to the inserterbar widget
143             $('#inserterbar_widget #inserterbar_tinymce_container').append($toolbar);
144             // Show the toolbar if we are in edit mode
145             if ($('.contentauthoring_edit_mode').length) {
146                 $('.mceExternalToolbar').hide();
147                 $('#inserterbar_widget #inserterbar_tinymce_container').show();
148                 $toolbar.show();
149             }
150             // Set timeOut as tinyMCE seems to need a little bit of additional time before we see all
151             // of the content in the editor
152             setTimeout(function() {
153                 var $containingCell = $('.htmlblock_widget', $rootel).parents('.contentauthoring_cell_element');
154                 $containingCell.removeClass('contentauthoring_init');
155                 updateHeight();
156                 $('.contentauthoring_cell_element_actions', $containingCell).removeClass('s3d-force-hidden').hide();
157             }, 1000);
158             startAutoSave();
159         };
160 
161         /**
162          * Function executed every time an event is fired in the tinyMCE editor. We only respond to a limited
163          * amount of events (i.e., the ones that can cause the height of the editor to change)
164          */
165         var tinyMCEEvent = function(ev, ui) {
166             if ($editor && (!ev || !ev.type || $.inArray(ev.type, EVENTS_TO_RESPOND_TO) !== -1)) {
167                 updateHeight();
168             }
169         };
170 
171         /////////////////////
172         // Editor resizing //
173         /////////////////////
174 
175         /**
176          * Update the height of the editor iframe so it's the same as its internal
177          * content. In other words, the editor should be the same height as the text,
178          * avoiding a scrollbar to be shown
179          */
180         var updateHeight = function() {
181             if ($editor.length) {
182                 try {
183                     var docHt = 0;
184                     var frame = $editor[0];
185                     $editor.contents().scrollTop(0);
186                     var innerDoc = frame.contentDocument ? frame.contentDocument : frame.contentWindow.document;
187                     $editor.css('height', '25px');
188                     docHt = innerDoc.body.scrollHeight;
189                     // Set a minimum internal height of 25px
190                     if (docHt < 25) {
191                         docHt = 25;
192                     }
193                     // Only update the editor height if it has changed
194                     if ($editor.height() !== docHt) {
195                         $editor.css('height', docHt + 'px');
196                         $(window).trigger("updateheight.contentauthoring.sakai");
197                     }
198                 } catch (err) {
199                     return false;
200                 }
201             }
202         };
203 
204         /////////////////////////
205         // Autosave and saving //
206         /////////////////////////
207 
208         /**
209          * When the page is being saved, we have to save the widget's content
210          * and also render the current text in view mode
211          */
212         var fullSave = function() {
213             // Only save when the widget is on the current page
214             if ($rootel.is(':visible')) {
215                 autoSave();
216                 if (editorId && tinyMCE.get(editorId)) {
217                     // Render the page in view mode
218                     var currentText = tinyMCE.get(editorId).getContent();
219                     currentText = sakai.api.Security.saneHTML(currentText);
220                     $('#htmlblock_view_container', $rootel).html(currentText);
221                     sakai.api.Util.renderMath($rootel);
222                 }
223             }
224         };
225 
226         /**
227          * Automatically save the widget's content from time to time, but only
228          * do this when the widget content has changed
229          */
230         var autoSave = function() {
231             // Only save when the widget is on the current page
232             if ($editor.is(':visible')) {
233                 var currentText = tinyMCE.get(editorId).getContent();
234                 if (currentText !== lastData) {
235                     lastData = currentText;
236                     sakai.api.Widgets.saveWidgetData(tuid, {'content': currentText});
237                 }
238             }
239         };
240 
241         /**
242          * Start auto saving the widget data
243          */
244         var startAutoSave = function() {
245             // Kill the current autosave interval
246             stopAutoSave();
247             // Start the autosave
248             autoSavePoll = setInterval(autoSave, AUTOSAVE_INTERVAL);
249         };
250 
251         /**
252          * Stop the auto save interval for the current widget
253          */
254         var stopAutoSave = function() {
255             if (autoSavePoll) {
256                 clearInterval(autoSavePoll);
257                 autoSavePoll = false;
258             }
259         };
260 
261         ///////////////////
262         // Event binding //
263         ///////////////////
264 
265         /**
266          * Every time the contentauthoring widget thinks a resize might be
267          * necessary, this event will be thrown
268          */
269         $(window).on('resize.contentauthoring.sakai', updateHeight);
270 
271         /**
272          * This event will be sent out by the contentauthoring widget when
273          * the user is trying to save the page. At this point, the htmlblock
274          * widget needs to store its changes as well
275          */
276         $(window).on('save.contentauthoring.sakai', fullSave);
277 
278         ////////////////////
279         // Initialization //
280         ////////////////////
281 
282         /**
283          * Load the HTMLBlock widget
284          */
285         var doInit = function () {
286             // Set the name attribute of the textarea to the widget id, so we can restrict
287             // tinyMCE loading to this widget
288             var $textarea = $('textarea', $rootel).attr('name', tuid).addClass(tuid);
289             // Fill up the textarea
290             if (widgetData && widgetData.htmlblock) {
291                 var processedContent = sakai.api.i18n.General.process(widgetData.htmlblock.content);
292                 processedContent = sakai.api.Security.saneHTML(processedContent);
293                 $('#htmlblock_view_container', $rootel).html(processedContent);
294                 sakai.api.Util.renderMath($rootel);
295                 $textarea.val(processedContent);
296             }
297             // Set the height of the textarea to be the same as the height of the view mode,
298             // so tinyMCE picks up on this initial height
299             $textarea.css('height', $('#htmlblock_view_container', $rootel).height());
300             loadTinyMCE();
301         };
302         
303         // run the initialization function when the widget object loads
304         doInit();
305     };
306 
307     // inform Sakai OAE that this widget has loaded and is ready to run
308     sakai.api.Widgets.widgetLoader.informOnLoad('htmlblock');
309 });
