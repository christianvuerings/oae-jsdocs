 83         var updateCounts = function(pageid, value, add) {
 84             // Adjust the count value by the specified value for the page ID
 85             if (add !== false) {
 86                 add = true;
 87             }
 88             var subpage = false;
 89             if (pageid.indexOf('/') !== -1) {
 90                 var parts = pageid.split('/');
 91                 pageid = parts[0];
 92                 subpage = parts[1];
 93             }

 95             var adjustCount = function(pageStructure, pageid, subpage, value) {
 96                 var listitem = 'li[data-sakai-addcontextoption="user"][data-sakai-path=\'';
 97                 var count;
 98                 var element;
 99                 if (subpage) {
100                     count = pageStructure.items[pageid][subpage];
101                     listitem = $(listitem + pageid + '/' + subpage + '\']');
102                     element = '.lhnavigation_sublevelcount';
103                 } else {
104                     count = pageStructure.items[pageid];
105                     listitem = $(listitem + pageid + '\']');
106                     element = '.lhnavigation_levelcount';
107                 }
108                 if (add) {
109                     count._count = (count._count || 0) + value;
110                 } else {
111                     count._count = value;
112                 }
113                 if (listitem.length) {
114                     if (!listitem.find('.lhnavigation_levelcount').length) {
115                         listitem.find('.lhnavigation_item_content').prepend(
116                             sakai.api.Util.TemplateRenderer('lhnavigation_counts_template', {
117                             'count': 0
118                             })
119                         );
120                     }
121                     $(element, listitem).text(count._count);
122                     if (count._count <= 0) {
123                         $(element, listitem).hide();
124                     } else {
125                         $(element, listitem).show();
126                     }
127                 }
128                 return pageStructure;
129             };
130 
131             if (pubstructure && pubstructure.items[pageid]) {
132                 pubstructure = adjustCount(pubstructure, pageid, subpage, value);
133             } else if (privstructure && privstructure.items[pageid]) {
134                 privstructure = adjustCount(privstructure, pageid, subpage, value);
135             }
136         };

159         var getPage = function(path, structure) {
160             if (path.indexOf('/') > -1) {
161                 structure = structure[path.split('/')[0]];
162                 path = path.substring(path.indexOf('/')+1);
163                 return getPage(path, structure);
164             } else if (structure[path]) {
165                 return structure[path];
166             } else {
167                 return null;

171         var getPageCount = function(pagestructure, pageCount) {
172             if (!pageCount) {
173                 pageCount = 0;
174             }
175             for (var tl in pagestructure) {
176                 if (pagestructure.hasOwnProperty(tl) && tl.substring(0, 1) !== '_') {
177                     pageCount++;
178                     if (pageCount >= 3) {
179                         return 3;
180                     }
181                     pageCount = getPageCount(pagestructure[tl], pageCount);
182                 }
183             }
184             return pageCount;

206                         dataType: 'json',
207                         success: function(data) {
208                             infinityStructuresPulled.push(ref);
209                             sakai.api.Server.convertObjectToArray(data, null, null);
210                             if (data && data.rows && data.rows.length) {
211                                 $.each(data.rows, function(index, row) {
212                                     if (!$.isPlainObject(row)) {
213                                         data.rows[index] = $.parseJSON(row);
214                                     }
215                                 });
216                             }
217                             if (privstructure.pages.hasOwnProperty(toplevelref + '-_lastModified')) {
218                                 privstructure.pages[ref] = data;
219                             } else {
220                                 pubstructure.pages[ref] = data;
221                             }
222                             callback();
223                         }
224                     });
225                 } else {
226                     callback();
227                 }
228             } else {
229                 callback();
230             }

233         var includeChildCount = function(structure) {
234             var childCount = 0;
235             for (var level in structure) {
236                 if (level && level.substring(0,1) !== '_') {
237                     childCount++;
238                     structure[level] = includeChildCount(structure[level]);
239                 } else if (level && level === '_title') {
240                     structure[level] = sakai.api.i18n.General.process(structure[level]);
241                 } else if (level && level === '_altTitle') {
242                     structure[level] = sakai.api.i18n.General.process(structure[level]);
243                     structure[level] = structure[level].replace('${user}', sakai.api.User.getFirstName(contextData.profile));
244                 }
245             }
246             structure._childCount = childCount;
247             return structure;
248         };

250         var finishProcessData = function(structure, data, callback) {

252             structure.items = includeChildCount(structure.items);
253             for (var page in data) {
254                 if (page.substring(0,9) !== 'structure' && page.substring(0,1) !== '_') {
255                     structure.pages[page] = data[page];
256                 }
257             }

261         var processData = function(data, docURL, callback) {
262             var structure = {};
263             structure.items = {};
264             structure.pages = {};
265             if (data['structure0']) {
266                 if (typeof data['structure0'] === 'string') {
267                     structure.items = $.parseJSON(data['structure0']);
268                 } else {
269                     structure.items = data['structure0'];
270                 }
271                 for (var i in structure.items) {
272                     if (structure.items.hasOwnProperty(i)) {
273                         structure.items[i] = addDocUrlIntoStructure(structure.items[i], docURL);
274                     }
275                 }
276                 // Get a list of all Sakai Docs that have to be 'added'
277                 var pids = collectPoolIds(structure.items, []);
278                 if (pids.length === 0) {
279                     finishProcessData(structure, data, callback);
280                 } else {
281                     continueProcessData(structure, data, pids, callback);
282                 }
283             } else {
284                 finishProcessData(structure, data, callback);
285             }
286         };

292         var collectPoolIds = function(structure, refs) {
293             for (var level in structure) {
294                 if (level && level.substring(0, 1) !== '_') {
295                     refs = collectPoolIds(structure[level], refs);
296                 } else if (level && level === '_pid' && structure['_canView'] !== false) {
297                     if ($.inArray(structure[level], refs) == -1) {
298                         refs.push(structure[level]);
299                     }
300                 }
301             }
302             return refs;

305         var insertDocStructure = function(structure0, docInfo, pid) {
306             for (var level in structure0) {
307                 if (structure0[level]._pid && structure0[level]._pid === pid) {
308                     var docStructure = docInfo.structure0;
309                     if (typeof docStructure === 'string') {
310                         docStructure = $.parseJSON(docStructure);
311                     }
312                     structure0[level] = $.extend(true, structure0[level], docStructure);
313                     for (var sublevel in structure0[level]) {
314                         if (structure0[level][sublevel]._ref) {
315                             structure0[level][sublevel]._ref = pid + '-' + structure0[level][sublevel]._ref;
316                         }
317                     }
318                     for (var subpage in docStructure) {
319                         if (docStructure[subpage]._ref) {
320                             structure0[level]._ref = pid + '-' + docStructure[subpage]._ref;
321                             break;
322                         }
323                     }
324                 }
325             }
326             return structure0;
327         };

338         var addDocUrlIntoStructure = function(structure, url) {
339             structure._poolpath = url;
340             for (var i in structure) {
341                 if (structure.hasOwnProperty(i) && i.substring(0,1) !== '_' && typeof structure[i] !== 'string') {
342                     structure[i] = addDocUrlIntoStructure(structure[i], url);
343                 }
344             }
345             return structure;

348         var continueProcessData = function(structure, data, pids, callback) {
349             // Prepare a batch request
350             var batchRequests = [];
351             for (var i = 0; i < pids.length; i++) {
352                 batchRequests.push({
353                     'url': '/p/' + pids[i] + '.json',
354                     'method': 'GET'
355                 });
356             }
357             sakai.api.Server.batch(batchRequests, function(success, data) {
358                 if (success) {
359                     for (var i = 0; i < pids.length; i++) {
360                         if (data.results[i].status === 404 || data.results[i].status === 403) {
361                             for (var level in structure.items) {
362                                 if (structure.items[level]._pid && structure.items[level]._pid === pids[i]) {
363                                     structure.items[level]._canView = false;
364                                     structure.items[level]._canEdit = false;
365                                     structure.items[level]._canSubedit = false;
366                                 }
367                             }
368                         } else {
369                             var docInfo = sakai.api.Server.cleanUpSakaiDocObject($.parseJSON(data.results[i].body));
370                             docInfo.orderedItems = orderItems(docInfo.structure0);
371                             sakaiDocsInStructure['/p/' + pids[i]] = docInfo;
372                             addDocUrlIntoStructure(docInfo.structure0, '/p/' + pids[i]);
373                             structure.items = insertDocStructure(structure.items, docInfo, pids[i]);
374                             structure = insertDocPages(structure, docInfo, pids[i]);
375                         }
376                     }
377                 }
378                 finishProcessData(structure, data, callback);
379             });
380         };

386         var getLowestOrderItem = function(items, alreadyAdded) {
387             var ret = false,
388                 lowest = false;
389             $.each(items, function(idx, item) {
390                 idx = ''+idx;
391                 // if it is a valid property to order
392                 if (idx.substring(0,1) !== '_' && item.hasOwnProperty('_order')) {
393                     // and it is the lowest in the list and we haven't already ordered it
394                     if ((lowest === false || item._order < lowest) && $.inArray(idx, alreadyAdded) === -1) {
395                         lowest = item._order;
396                         ret = [idx,item];
397                     }
398                 }
399             });
400             return ret;
401         };
402 
403         var orderItems = function(items) {
404             var orderedItems = [],
405                 alreadyAdded = [],
406                 order = 0;
407             if (items) {
408                 $.each(items, function(idx, item) {
409                     var toAdd = getLowestOrderItem(items, alreadyAdded);
410                     var itemToAdd = toAdd[1], itemID = toAdd[0];
411                     if (toAdd) {
412                         itemToAdd._order = order;
413                         order++;
414                         itemToAdd._id = itemID;
415                         itemToAdd._elements = orderItems(itemToAdd);
416                         orderedItems.push(itemToAdd);
417                         alreadyAdded.push(itemID);
418                     }
419                 });
420             }
421             return orderedItems;
422         };

427             }
428             if (pubstructure && pubstructure.items) {
429                 pubstructure.orderedItems = orderItems(pubstructure.items);
430             }

472         var getFirstSelectablePage = function(structure) {
473             var selected = false;
474             if (structure.orderedItems) {
475                 for (var i = 0; i < structure.orderedItems.length; i++) {
476                     if (structure.orderedItems[i]._canView !== false) {
477                         if (structure.orderedItems[i]._childCount > 1) {
478                             for (var ii = 0; ii < structure.orderedItems[i]._elements.length; ii++) {
479                                 selected = structure.orderedItems[i]._id + '/' + structure.orderedItems[i]._elements[ii]._id;
480                                 break;
481                             }
482                         } else {
483                             selected = structure.orderedItems[i]._id;
484                         }
485                         break;
486                     }
487                 }
488             }
489             return selected;

493             for (var i = 0; i < structure.orderedItems.length; i++) {
494                 if (structure.orderedItems[i]._canView !== false && structure.orderedItems[i]._id === selected) {
495                     for (var ii = 0; ii < structure.orderedItems[i]._elements.length; ii++) {

504         var checkPageExists = function(structure, selected) {
505             var structureFoundIn = false;
506             if (selected.indexOf('/') !== -1) {
507                 var splitted = selected.split('/');
508                 if (structure.items[splitted[0]] && structure.items[splitted[0]][splitted[1]]) {
509                     structureFoundIn = structure;
510                 }
511             } else {
512                 if (structure.items[selected]) {
513                     structureFoundIn = structure;
514                 }
515             }
516             return structureFoundIn;
517         };

519         /**
520          * Check if a user can view the specific page.
521          * If the page has no _canView property than the user can view the page.
522          *
523          * @param {Object} structure The structure containing the page
524          * @param {String} selected The path to the page, ie. 'syllabus/week1'
525          */
526         var canViewPage = function(structure, selected) {
527             if (structure) {
528                 var pageSelected = selected;
529                 var subPage = false;
530 
531                 // Check if this is a subpage
532                 if (selected.indexOf('/') !== -1) {
533                     pageSelected = selected.split('/')[0];
534                     subPage = selected.split('/')[1];
535                 }
536 
537                 // Check if the page has a _canView property and it's set to false, if it is undefined the user should still be able to see the page
538                 if ((structure.items[pageSelected] && structure.items[pageSelected]._canView === false) ||
539                     (subPage && structure.items[pageSelected][subPage] && structure.items[pageSelected][subPage]._canView === false)) {
540                     return false;
541                 }
542             }
543             return true;
544         };

546         var selectPage = function(newPageMode) {
547             if (contextData.forceOpenPage) {
548                 $.bbq.pushState({
549                     'l': contextData.forceOpenPage
550                 }, 0);
551                 contextData.forceOpenPage = false;
552             } else {
553                 var state = $.bbq.getState('l');
554                 var selected = state || false;
555                 var structureFoundIn = false;
556                 // Check whether this page exist and if we have permission to view
557                 if (selected) {
558                     structureFoundIn = checkPageExists(privstructure, selected) || checkPageExists(pubstructure, selected);
559                     if (!structureFoundIn || !canViewPage(structureFoundIn, selected)) {
560                         selected = false;
561                     }
562                 }
563                 // If it exists, check whether it has more than 1 subpage
564                 if (selected && selected.indexOf('/') === -1) {
565                     if (structureFoundIn.items[selected]._childCount > 1) {
566                         selected = getFirstSubPage(structureFoundIn, selected);
567                     }
568                 }
569                 // If no page is selected, select the first one from the nav
570                 if (!selected) {
571                     selected = getFirstSelectablePage(privstructure) || getFirstSelectablePage(pubstructure);
572                 }
573                 if (selected) {
574                     // update links in all menus with subnav with the selected page, so they wont trigger handleHashChange and cause weirdness
575                     $('#lhnavigation_container').find('a.lhnavigation_toplevel_has_subnav').attr('href', '#l=' + selected);
576                     // Select correct item
577                     var menuitem = $('li[data-sakai-path=\'' + selected + '\']');
578                     if (menuitem.length) {
579                         if (selected.split('/').length > 1) {
580                             var par = $('li[data-sakai-path=\'' + selected.split('/')[0] + '\']');
581                             showHideSubnav(par, true);
582                         }
583                         var ref = menuitem.data('sakai-ref');
584                         var savePath = menuitem.data('sakai-savepath') || false;
585                         var pageSavePath = menuitem.data('sakai-pagesavepath') || false;
586                         var canEdit = menuitem.data('sakai-submanage') || false;
587                         var nonEditable = menuitem.data('sakai-noneditable') || false;
588                         if (!menuitem.hasClass(navSelectedItemClass)) {
589                             selectNavItem(menuitem, $(navSelectedItem));
590                         }
591 
592                         getPageContent(ref, function() {
593                             preparePageRender(ref, selected, savePath, pageSavePath, nonEditable, canEdit, newPageMode === true);
594                         });
595                     }
596                 } else {
597                     renderPageUnavailable();
598                 }
599             }
600         };

602         var preparePageRender = function(ref, path, savePath, pageSavePath, nonEditable, canEdit, newPageMode) {
603             var content = returnStructure(ref);
604             var pageContent = content ? content : sakai.config.defaultSakaiDocContent;
605             var lastModified = content && content._lastModified ? content._lastModified : null;
606             var autosave = content && content.autosave ? content.autosave : null;
607             var pageTitle = $.trim($('.lhnavigation_selected_item .lhnavigation_page_title_value').text());
608             var saveRef = ref;
609             if (saveRef.indexOf('-') !== -1) {
610                 saveRef = saveRef.substring(saveRef.indexOf('-') + 1);
611             }
612             currentPageShown = {
613                 'ref': ref,
614                 'path': path,
615                 'content': pageContent,
616                 'savePath': savePath,
617                 'pageSavePath': pageSavePath,
618                 'saveRef': saveRef,
619                 'canEdit': canEdit,
620                 'nonEditable': nonEditable,
621                 '_lastModified': lastModified,
622                 'autosave': autosave,
623                 'title': pageTitle
624             };
625             if (newPageMode) {
626                 $(window).trigger('editpage.contentauthoring.sakai', [currentPageShown, newPageMode]);
627                 contextMenuHover = {
628                     path: currentPageShown.path,
629                     ref: currentPageShown.ref,
630                     pageSavePath: currentPageShown.pageSavePath,
631                     savePath: currentPageShown.savePath,
632                     content: currentPageShown.content
633                 };
634                 editPageTitle();
635                 $(document).on('click', '#inserterbar_action_add_page', addNewPage);
636             } else {
637                 $(window).trigger('showpage.contentauthoring.sakai', [currentPageShown]);
638             }
639         };

978             var pagePath = pageToDelete.path.toString();

1006             renderData();
1007             addParametersToNavigation();
1008             enableSorting();

1028                     'l': '',

1036         /*
1037          * Update the page reference if it belonged to a subpage that was just deleted
1038          */
1039         var updatePageReference = function(structure, path) {
1040             if (path.indexOf('/') !== -1) {
1041                 var parts = path.split('/');
1042                 var checkRef = structure[parts[0]]._ref;
1043                 if (checkRef.indexOf('-') !== -1) {
1044                     var newRef = false;
1045                     var checkRefPage = checkRef.split('-')[0];
1046                     var checkRefId = checkRef.split('-')[1];
1047                     // The page ref should be the first sub page
1048                     if (structure[parts[0]]._elements &&
1049                         structure[parts[0]]._elements[0] &&
1050                         structure[parts[0]]._elements[0].main &&
1051                         structure[parts[0]]._elements[0].main._ref &&
1052                         checkRefId !== structure[parts[0]]._elements[0].main._ref) {
1053                         if (structure[parts[0]]._elements[0].main._ref.indexOf(checkRefPage) !== -1) {
1054                             checkRefPage = '';
1055                         } else {
1056                             checkRefPage = checkRefPage + '-';
1057                         }
1058                         structure[parts[0]]._ref = checkRefPage + structure[parts[0]]._elements[0].main._ref;
1059                     }
1060                 }
1061             }
1062         };

1064         var updateCountsAfterDelete = function(structure, pageslist, ref, path) {
1065             orderedItems = orderItems(structure);
1066             var oldOrder = 0;
1067             if (path.indexOf('/') !== -1) {
1068                 var parts = path.split('/');
1069                 oldOrder = structure[parts[0]][parts[1]]._order;
1070                 delete structure[parts[0]][parts[1]];
1071                 for (var i = 0; i < orderedItems.length; i++) {
1072                     if (orderedItems[i]._pid === ref.split('-')[0]) {
1073                         orderedItems[i]._elements.splice(oldOrder, 1);
1074                         orderedItems[i]._childCount--;
1075                         for (var o = oldOrder; o < orderedItems[i]._elements.length; o++) {
1076                             orderedItems[i]._elements[o]._order = o;
1077                             structure[orderedItems[i]._id][orderedItems[i]._elements[o]._id]._order = o;
1078                         }
1079                     }
1080                 }
1081             } else {
1082                 oldOrder = structure[path]._order;
1083                 delete structure[path];
1084                 orderedItems.splice(oldOrder, 1);
1085                 for (var z = oldOrder; z < orderedItems.length; z++) {
1086                     orderedItems[z]._order = z;
1087                     structure[orderedItems[z]._id]._order = z;
1088                 }
1089             }
1090             delete pageslist[ref];
1091         };
