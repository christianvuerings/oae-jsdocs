232         /**
233          * Adds to the count of items in the collection's library
234          * @param {String} collectionId The id of the collection to increase the count of (cached variable)
235          * @param {int} amount Total amount of dropped items to add to the count
236          * @param {Boolean} inLibrary Indicates if the dropped content is already in the personal library
237          */
238         var addToCollectionCount = function(collectionId, amount, inLibrary) {
239             // If no content was uploaded previously, initialize contentCount variable
240             sakai.data.me.user.properties.contentCount = sakai.data.me.user.properties.contentCount || 0;
241 
242             // If content was not in library yet, update counts for the library
243             if (!inLibrary) {
244                 sakai.data.me.user.properties.contentCount += amount;
245                 // Display library the counts in the UI
246                 var $libraryCountEl = $('#inserter_init_container ul li[data-collection-id="library"] .inserter_item_count_container', $rootel);
247                 $libraryCountEl.text(sakai.data.me.user.properties.contentCount);
248                 // Update the left hand nav library count
249                 $(window).trigger('lhnav.updateCount', ['library', sakai.data.me.user.properties.contentCount, false]);
250             }
251 
252             // We need to update collection variables if that is where it was dropped
253             if (collectionId !== 'library' && sakai.data.me.user.userid !== collectionId) {
254                 $.each(sakai.data.me.groups, function(index, group) {
255                     if (group['sakai:category'] === 'collection' && group.groupid === 'c-' + collectionId) {
256                         // Display the collection counts in the UI
257                         updateCollectionCount(false, collectionId, group.counts.contentCount);
258                         // Update the header of a collection if necessary
259                         if (inCollection) {
260                             $('#inserter_header_itemcount > #inserter_header_itemcount_count', $rootel).text(
261                                 group.counts.contentCount);
262                         }
263                         libraryData[collectionId].counts.contentCount = group.counts.contentCount;
264                     }
265                 });
266             } else {
267                 var currentCount = parseInt($('#inserter_header_itemcount > #inserter_header_itemcount_count', $rootel).text(), 10);
268                 // Update the header of the library if necessary
269                 if (inCollection) {
270                     $('#inserter_header_itemcount > #inserter_header_itemcount_count', $rootel).text(
271                         currentCount + amount);
272                 }
273             }
274         };

276         /**
277          * Creates a new, empty, collections with the given name and opens it in the inserter
278          * @param {String} title Title to give to the new collection
279          */
280         var createNewCollection = function(title) {
281             sakai.api.Util.progressIndicator.showProgressIndicator(sakai.api.i18n.getValueForKey('CREATING_YOUR_COLLECTION', 'inserter'), sakai.api.i18n.getValueForKey('WONT_BE_LONG', 'inserter'));

283             var permissions = sakai.config.Permissions.Collections.defaultaccess;
284             sakai.api.Content.Collections.createCollection(title, '', permissions, [], contentToAdd, [], function(success, path) {
285                 sakai.api.Server.loadJSON('/p/' + path + '.json', function(success, collection) {
286                     if (success) {
287                         $(document).trigger('done.newaddcontent.sakai', [[collection], 'user']);
288                     }
289                 });
290                 contentToAdd = [];
291                 $(window).trigger('sakai.collections.created');
292                 sakai.api.Util.progressIndicator.hideProgressIndicator();
293                 sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey('COLLECTION_CREATED'), sakai.api.i18n.getValueForKey('COLLECTION_CREATED_LONG'));
294                 $(inserterCreateCollectionInput, $rootel).val('');
295                 $(window).trigger('sakai.mylibrary.createdCollections', {
296                     items: ['newcollection']
297                 });
298                 addToCollectionCount('library', 1, false);
299             });

345         /**
346          * Add a batch of dropped items to a given collection
347          * @param {String} collectionId ID of the collection to add the items to
348          * @param {Array} collectedCollections Array of collected collection IDs
349          * @param {Array} collectedContent Array of collected content IDs
350          */
351         var addDroppedToIndependentCollection = function(collectionId, collectedCollections, collectedContent) {
352             // Add dropped content to the collection
353             sakai.api.Content.Collections.addToCollection(collectionId, collectedContent, function() {
354                 // Share the collections that were dropped
355                 sakai.api.Content.Collections.shareCollection(collectedCollections,
356                     sakai.api.Content.Collections.getCollectionGroupId(collectionId), false, function() {
357                     addToCollectionCount(collectionId, collectedCollections.length, true);
358                     sakai.api.Util.progressIndicator.hideProgressIndicator();
359                     if (inCollection) {
360                         $.each(sakai.data.me.groups, function(index, item) {
361                             if (item['sakai:category'] === 'collection' &&
362                                 !item['sakai:pseudoGroup'] &&
363                                 item['sakai:group-id'] === 'c-' + collectionId) {
364                                 contentListDisplayed = item;
365                                 contentListDisplayed._path = collectionId;
366                             }
367                         });
368                         infinityContentScroll.prependItems(itemsDropped);
369                     } else {
370                         showUIElements('reset');
371                     }
372                 });
373             });
374         };

376         /**
377          * Add a batch of dropped items to my library
378          * @param {String} collectionId ID of the collection to add the items to
379          * @param {Array} collectedCollections Array of collected collection IDs
380          * @param {Array} collectedContent Array of collected content IDs
381          */
382         var addDroppedToMyLibrary = function(collectionId, collectedCollections, collectedContent) {
383             $.each(collectedCollections, function(i, collection) {
384                 sakai.api.Content.addToLibrary(collection, sakai.data.me.user.userid, false, function() {
385                     addToCollectionCount(collectionId, 1, false);
386                 });
387             });
388             $.each(collectedContent, function(i, content) {
389                 sakai.api.Content.addToLibrary(content, sakai.data.me.user.userid, false, function() {
390                     addToCollectionCount(collectionId, 1, false);
391                 });
392             });
393             if (inCollection) {
394                 infinityContentScroll.prependItems(itemsDropped);
395             }
396             sakai.api.Util.progressIndicator.hideProgressIndicator();
397         };

399         /**
400          * Add a dropped content item to the collection (used for drag and drop inside of window, not from desktop)
401          * @param {Object} ev Event fired by dropping a content item onto the list
402          * @param {Object} data The data received from the event
403          * @param {Object} target jQuery object indicating the drop target
404          */
405         var addDroppedToCollection = function(ev, data, target) {
406             var collectionId = target.attr('data-collection-id');
407             var collectedContent = [];
408             var collectedCollections = [];
409             var isCollection = (collectionId !== 'library' && collectionId !== sakai.data.me.user.userid);
410             $.each(data, function(index, item) {
411                 if (!isCollection || item.canshare) {
412                     if (item.collection) {
413                         // We don't need to send an extra POST since we can't add a collection to itself
414                         if (collectionId !== item.entityid) {
415                             collectedCollections.push(item.entityid);
416                         } else {
417                             sakai.api.Util.notification.show(
418                                 sakai.api.i18n.getValueForKey('ADD_COLLECTION', 'inserter'),
419                                 sakai.api.i18n.getValueForKey('CANT_ADD_A_COLLECTION_TO_ITSELF', 'inserter'),
420                                 sakai.api.Util.notification.type.ERROR);
421                         }
422                     }
423                     else {
424                         collectedContent.push(item.entityid);
425                     }
426                 }
427             });
428             if (collectedContent.length + collectedCollections.length > 0) {
429                 sakai.api.Util.progressIndicator.showProgressIndicator(
430                     sakai.api.i18n.getValueForKey('UPLOADING_CONTENT_ADDING_TO_COLLECTION', 'inserter'),
431                     sakai.api.i18n.getValueForKey('WONT_BE_LONG', 'inserter'));
432                 // If the collection the content was added to is not the user's library
433                 // share the content with that collection
434                 // If it is the library execute different API functions
435                 if (isCollection) {
436                     addDroppedToIndependentCollection(collectionId, collectedCollections, collectedContent);
437                 } else {
438                     addDroppedToMyLibrary(collectionId, collectedCollections, collectedContent);
439                 }
440             } else {
441                 sakai.api.Util.notification.show(
442                     sakai.api.i18n.getValueForKey('ADD_CONTENT'),
443                     sakai.api.i18n.getValueForKey('COULDNT_ADD_CONTENT', 'inserter'),
444                     sakai.api.Util.notification.type.ERROR);
445             }
446         };

448         /**
449          * Upload a set of files dropped onto the inserter lists
450          * @param {String} collectionId the ID of the collection to associate the content with
451          * @param {String} permissions Permissions for the newly uploaded content (default to public)
452          * @param {Array} itemsDropped Array of files dropped on the inserter
453          */
454         var setDataOnDropped = function(collectionId, permissions, itemsDropped) {
455             var batchRequests = [];
456             var itemIDs = [];
457             $.each(itemsDropped, function(index, item) {
458                 $.extend(item, item.item);
459                 var splitOnDot = item['sakai:pooled-content-file-name'].split('.');
460                 // Set initial version
461                 batchRequests.push({
462                     'url': '/p/' + item.poolId + '.save.json',
463                     'method': 'POST'
464                 });
465                 batchRequests.push({
466                     'url': '/p/' + item.poolId,
467                     'method': 'POST',
468                     'parameters': {
469                         'sakai:permissions': permissions,
470                         'sakai:copyright': 'creativecommons',
471                         'sakai:allowcomments': 'true',
472                         'sakai:showcomments': 'true',
473                         'sakai:fileextension': splitOnDot[splitOnDot.length -1]
474                     }
475                 });
476 
477                 itemIDs.push(item.poolId);
478                 item.hashpath = item.poolId;
479                 item.permissions = permissions;
480             });
481             sakai.api.Server.batch(batchRequests, function(success, response) {
482                 // Set the correct file permissions
483                 sakai.api.Content.setFilePermissions(itemsDropped, function() {
484                     // Add it to the collection
485                     if (collectionId !== 'library') {
486                         sakai.api.Content.Collections.addToCollection(collectionId, itemIDs, function() {
487                             addToCollectionCount(collectionId, itemsDropped.length, false);
488                             if (inCollection) {
489                                 infinityContentScroll.prependItems(itemsDropped);
490                             }
491                             $(document).trigger('done.newaddcontent.sakai', [itemsDropped, 'user']);
492                             sakai.api.Util.progressIndicator.hideProgressIndicator();
493                         });
494                     } else {
495                         addToCollectionCount(collectionId, itemsDropped.length, false);
496                         if (inCollection) {
497                             infinityContentScroll.prependItems(itemsDropped);
498                         }
499                         $(document).trigger('done.newaddcontent.sakai', [itemsDropped, 'user']);
500                         sakai.api.Util.progressIndicator.hideProgressIndicator();
501                     }
502                 });
503             });
504         };

506         /**
507          * Handles dropping items and applying the fileupload functionality
508          */
509         var addDnDToElements = function() {
510             // Initialize drag and drop from desktop
511             $('#inserter_collector', $rootel).fileupload({
512                 url: '/system/pool/createfile',
513                 formData: {
514                     '_charset_': 'utf-8'
515                 },
516                 drop: function(ev, data) {
517                     $dropTarget = $(ev.currentTarget);
518                     var error = false;
519                     $.each(data.files, function(index, file) {
520                         if (file.size > 0) {
521                             numberOfItemsDropped++;
522                         } else {
523                             error = true;
524                         }
525                     });
526                     if (error) {
527                         sakai.api.Util.notification.show(
528                             sakai.api.i18n.getValueForKey('DRAG_AND_DROP_ERROR', 'inserter'),
529                             sakai.api.i18n.getValueForKey('ONE_OR_MORE_DROPPED_FILES_HAS_AN_ERROR', 'inserter'));
530                     }
531                     if (numberOfItemsDropped) {
532                         sakai.api.Util.progressIndicator.showProgressIndicator(
533                             sakai.api.i18n.getValueForKey('UPLOADING_CONTENT_ADDING_TO_COLLECTION', 'inserter'),
534                             sakai.api.i18n.getValueForKey('WONT_BE_LONG', 'inserter'));
535                     }
536                 },
537                 dropZone: $('#inserter_collector ul li,#inserter_collector .s3d-no-results-container', $rootel),
538                 done: function(ev, data) {
539                     var result = $.parseJSON(data.result);
540                     itemsDropped.push(result[data.files[0].name]);
541                     if (itemsDropped.length && itemsDropped.length === numberOfItemsDropped) {
542                         var collectionId = $dropTarget.attr('data-collection-id');
543                         setDataOnDropped(collectionId, 'public', itemsDropped);
544                         numberOfItemsDropped = 0;
545                         itemsDropped = [];
546                     }
547                 }
548             });
549         };

556         /**
557          * Processes an empty infinite scroll list and displays the appropriate message
558          */
559         var emptyCollectionList = function() {
560             var mimetype = $inserterMimetypeFilter.val() || '';
561             disableEnableHeader(!$.trim($(inserterCollectionContentSearch, $rootel).val()) && !mimetype);
562             var query = $.trim($(inserterCollectionContentSearch, $rootel).val());
563             var libraryId = false;
564             if (!contentListDisplayed.length) {
565                 libraryId = sakai.data.me.user.userid;
566             }
567             if (!$inserterMimetypeFilter.val() || query) {
568                 sakai.api.Util.TemplateRenderer(inserterNoResultsTemplate, {
569                     'search': query,
570                     'collection': libraryId || sakai.api.Content.Collections.getCollectionGroupId(contentListDisplayed).replace('c-', '')
571                 }, $inserterNoResultsContainer);
572                 $inserterNoResultsContainer.show();
573             } else {
574                 query = $.trim($(inserterCollectionContentSearch, $rootel).val());
575                 sakai.api.Util.TemplateRenderer(inserterNoResultsTemplate, {
576                     'search': 'mimetypesearch',
577                     'collection': libraryId || sakai.api.Content.Collections.getCollectionGroupId(contentListDisplayed).replace('c-', '')
578                 }, $inserterNoResultsContainer);

605          * @param {Object} item Contains data about the collection to be loaded

607         var showCollection = function(item) {
608             inCollection = true;
609             var query = $.trim($(inserterCollectionContentSearch, $rootel).val());
610             var mimetype = $inserterMimetypeFilter.val() || '';
611 
612             var params = {
613                 sortOn: '_lastModified',
614                 sortOrder: 'desc',
615                 q: query,
616                 mimetype: mimetype
617             };
618             if (item === 'library' || library) {
619                 library = true;
620                 params.userid = sakai.data.me.user.userid;
621             } else {
622                 library = false;
623                 contentListDisplayed = item._path || item;
624                 params.userid = sakai.api.Content.Collections.getCollectionGroupId(contentListDisplayed);
625             }
626 
627             // Disable the previous infinite scroll
628             killInfiniteScroll();
629             infinityContentScroll = $inserterCollectionItemsList.infinitescroll(
630                 sakai.config.URL.POOLED_CONTENT_SPECIFIC_USER,
631                 params,
632                 function(items, total) {
633                     disableEnableHeader(false);
634                     return sakai.api.Util.TemplateRenderer(inserterCollectionContentTemplate, {
635                         items: items,
636                         collection: params.userid.replace('c-', ''),
637                         sakai: sakai
638                     });
639                 },
640                 emptyCollectionList,
641                 sakai.config.URL.INFINITE_LOADING_ICON,
642                 handleLibraryItems,
643                 collectionListPostRender,
644                 sakai.api.Content.getNewList(contentListDisplayed),
645                 false,
646                 $inserterContentInfiniteScrollContainer
647             );
648         };

759         /**
760          * Initialize the inserter widget
761          */
762         var doInit = function() {
763             $inserterCollectionContentContainer.hide();
764             $inserterWidget.css('top', topMargin);
765             $inserterWidget.draggable({
766                 cancel: 'div#inserter_collector',
767                 stop: function(ev) {
768                     // Calculate the position of the widget and reset its position when
769                     // it goes out of bounds
770                     var elOffset = $(ev.target).offset();
771                     var wHeight = $(window).height();
772                     var wWidth = $(window).width();
773                     var iHeight= $inserterWidget.height();
774                     var iWidth = $inserterWidget.width();
775                     var borderMargin = 15;
776                     if (elOffset) {
777                         // Overlaps left window border
778                         if (elOffset.left <= 0) {
779                             $inserterWidget.css('left', borderMargin);
780                         }
781                         // Overlaps right window border
782                         if (elOffset.left > wWidth - iWidth) {
783                             $inserterWidget.css('left', wWidth - iWidth - borderMargin);
784                         }
785                         // Overlaps top window border or topnavigation
786                         if (elOffset.top < topMargin) {
787                             $inserterWidget.css('top', topMargin);
788                         }
789                         // Overlaps bottom window border
790                         if (elOffset.top > $(window).scrollTop() + wHeight - iHeight) {
791                             $inserterWidget.css('top', wHeight - iHeight - borderMargin);
792                         }
793                         // Store new position
794                         widgetPos = {
795                             'left': $inserterWidget.css('left'),
796                             'top': $inserterWidget.css('top')
797                         };
798                     } else {
799                         $inserterWidget.css({
800                             'left': widgetPos.left,
801                             'top': widgetPos.top
802                         });
803                     }
804                 }
805             });
806             renderHeader('init');
807             // SAKIII-6013 Update the me feed before opening up Collection Viewer.
808             sakai.api.User.loadMeData(function(success, data) {
809                 if (success) {
810                     sakai.api.Util.TemplateRenderer('inserter_init_prescroll_template', {
811                         sakai: sakai
812                     }, $inserterCollectionInfiniteScrollContainer);
813                     $inserterCollectionInfiniteScrollContainerList = $($inserterCollectionInfiniteScrollContainerList, $rootel);
814                     validateNewCollectionForm();
815                     fetchLibrary();
816                     if (focusCreateNew) {
817                         $(inserterCreateCollectionInput).focus();
818                     }
819                 } else {
820                     debug.error('Error in updating me data before opening Collection Viewer.');
821                 }
822             });
823         };
824 
825         addBinding();
826     };
