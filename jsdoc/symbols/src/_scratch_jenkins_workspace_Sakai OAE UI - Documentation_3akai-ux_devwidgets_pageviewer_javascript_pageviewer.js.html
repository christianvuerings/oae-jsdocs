 19 // load the master sakai object to access all Sakai OAE API methods
 20 require(['jquery', 'sakai/sakai.api.core'], function($, sakai) {
 21 
 22     /**
 23      * @name sakai_global.pageviewer
 24      *
 25      * @class pageviewer
 26      *
 27      * @description
 28      * The page viewer widget shows a Sakai Doc in a simplified way with 
 29      * a custom left hand navigation and no sublevel items
 30      *
 31      * @version 0.0.1
 32      * @param {String} tuid Unique id of the widget
 33      * @param {Boolean} showSettings Show the settings of the widget or not
 34      */
 35     sakai_global.pageviewer = function (tuid, showSettings) {
 36 
 37 
 38         /////////////////////////////
 39         // Configuration variables //
 40         /////////////////////////////
 41 
 42         var $rootel = $('#' + tuid);
 43         var docPath = '';
 44         var tempDocData = {};
 45         var tempItem = false;
 46         var docData = {};
 47         var storePath = '';
 48         var selectedPage = '';
 49 
 50         // Containers
 51         var $pageViewerContentContainer = $('#pageviewer_content_container', $rootel);
 52         var $pageViewerLHNavContainer = $("#pageviewer_lhnav_container", $rootel);
 53 
 54         // Templates
 55         var pageViewerContentTemplate = 'pageviewer_content_template';
 56         var pageViewerLHNavTemplate = 'pageviewer_lhnav_template';
 57 
 58 
 59         ///////////////////////
 60         // Utility functions //
 61         ///////////////////////
 62 
 63         /**
 64          * Check whether any of the columns in the current page are empty (i.e., they have no widgets
 65          * inside of them). If so, we add the placeholder widget
 66          */
 67         var checkColumnsEmpty = function() {
 68             var hasContent = false;
 69             $.each($('.contentauthoring_cell_content:visible', $rootel), function(i, cellcontainer) {
 70                 if ($(cellcontainer).find('.contentauthoring_cell_element').length) {
 71                     hasContent = true;
 72                 }
 73             });
 74             if (!hasContent) {
 75                 $(sakai.api.Util.TemplateRenderer('pageviewer_dummy_element_template', {}, $pageViewerContentContainer));
 76             }
 77         };
 78 
 79         /**
 80          * Renders the page in the widget
 81          */
 82         var renderContainer = function() {
 83             // Check to see that we're not recursively embedding this page
 84             if (sakai.api.Widgets.isRecursivelyEmbedded($rootel, docPath, tempItem._ref)) {
 85                 return;
 86             }
 87             $('.pageviewer_widget', $rootel).show();
 88             sakai.api.Util.TemplateRenderer(pageViewerContentTemplate, {
 89                 data: docData,
 90                 selectedPage: selectedPage
 91             }, $pageViewerContentContainer);
 92             sakai.api.Widgets.widgetLoader.insertWidgets('pageviewer_content_container', false, storePath, false);
 93             sakai.api.Util.renderMath($pageViewerContentContainer);
 94             checkColumnsEmpty();
 95         };
 96 
 97         /**
 98          * Renders the left hand navigation in the pageviewer widget
 99          */
100         var renderNavigation = function() {
101             sakai.api.Util.TemplateRenderer(pageViewerLHNavTemplate, {
102                 data: docData,
103                 selectedPage: selectedPage
104             }, $pageViewerLHNavContainer, false);
105 
106             $pageViewerLHNavContainer.show();
107 
108             $(".pageviewer_lhnav_item button", $rootel).on('click', selectPage);
109             $(".pageviewer_lhnav_item:first button", $rootel).click();
110         };
111 
112         /**
113          * Parse cells in a column in a row on a page
114          * @param {Object} cellIndex Index of the cell
115          * @param {Object} cell data for a cell in a column
116          */
117         var parseCells = function(cellIndex, cell) {
118             if ($.isPlainObject(cell)) {
119                 if (tempDocData[tempItem._ref][cell.id]) {
120                     docData[tempItem._ref][cell.id] = {};
121                     var cellData = tempDocData[tempItem._ref][cell.id][cell.type];
122                     if (cell.type !== "htmlblock" && cell.type !== "pagetitle") {
123                         docData[tempItem._ref][cell.id][cell.type] = {
124                             'embedmethod': cellData.embedmethod,
125                             'sakai:indexed-fields': cellData['sakai:indexed-fields'],
126                             'download': cellData.download,
127                             'title': cellData.title,
128                             'details': cellData.details,
129                             'description': cellData.description,
130                             'name': cellData.name,
131                             'layout': cellData.layout,
132                             'items': {}
133                         };
134                         if (cellData.items) {
135                             $.each(cellData.items, function(itemsIndex, cellItem) {
136                                 docData[tempItem._ref][cell.id][cell.type].items[itemsIndex] = cellItem;
137                             });
138                         }
139                     } else {
140                         docData[tempItem._ref][cell.id][cell.type] = {
141                             'content': cellData.content
142                         };
143                     }
144                 }
145             }
146         };
147 
148         /**
149          * Parse columns in a row on a page
150          * @param {Object} columnIndex Index of the column
151          * @param {Object} column data for a column
152          */
153         var parseColumns = function(columnIndex, column) {
154             if ($.isPlainObject(column)) {
155                 $.each(column.elements, parseCells);
156             }
157         };
158 
159         /**
160          * Parse rows on a page
161          * @param {Object} rowIndex Index of the row
162          * @param {Object} row data for a row
163          */
164         var parseRows = function(rowIndex, row) {
165             if ($.isPlainObject(row)) {
166                 $.each(row.columns, parseColumns);
167             }
168         };
169 
170         /**
171          * Parses the document structure before sending it to the renderer
172          */
173         var parseStructure = function() {
174             var json = $.parseJSON(tempDocData.structure0);
175             var pageCount = 0;
176             $.each(json, function(index, item) {
177                 if ($.isPlainObject(item)) {
178                     tempItem = item;
179                     if (!selectedPage) {
180                         selectedPage = tempItem._ref;
181                     }
182                     docData[tempItem._ref] = {};
183                     docData[tempItem._ref].pageTitle = item._title;
184                     docData[tempItem._ref]['rows'] = tempDocData[tempItem._ref].rows;
185                     storePath = 'p/' + docPath + '/' + tempItem._ref + '/';
186 
187                     // Go through rows, columns and cells and extract any content
188                     $.each(docData[tempItem._ref]['rows'], parseRows);
189 
190                     pageCount++;
191                 }
192             });
193             docData.template = 'all';
194 
195             // Render the pages and navigation in the widget
196             if (pageCount > 1) {
197                 renderNavigation();
198             } else {
199                 renderContainer();
200             }
201         };
202 
203         /**
204          * Selects a page from the left hand navigation
205          */
206         var selectPage = function() {
207             $rootel.find('.tinyMCE').each(function() {
208                 tinyMCE.execCommand( 'mceRemoveControl', false, $(this).attr('id') );
209             });
210             var $li = $(this).parent();
211             $(".pageviewer_lhnav_item", $rootel).removeClass("selected");
212             $li.addClass("selected");
213             selectedPage = $li.attr("data-index");
214             storePath = 'p/' + docPath + '/' + selectedPage + '/';
215             renderContainer();
216         };
217 
218         /**
219          * Fetches the page content
220          */
221         var fetchPages = function() {
222             sakai.api.Server.loadJSON("/p/" + docPath + ".infinity.json", function(success, data){
223                 tempDocData = data;
224                 parseStructure();
225             });
226         };
227 
228 
229         //////////////////////////////
230         // Initialization functions //
231         //////////////////////////////
232 
233         /**
234          * Initializes the pageviewer widget
235          */
236         var doInit = function() {
237             fetchPages();
238         };
239 
240         $(window).off('start.pageviewer.sakai').on('start.pageviewer.sakai', function(ev, data) {
241             docPath = data.id;
242             doInit();
243         });
244 
245         $(window).trigger('ready.pageviewer.sakai');
246 
247     };
248 
249     sakai.api.Widgets.widgetLoader.informOnLoad('pageviewer');
250 });
