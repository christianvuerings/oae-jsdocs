193         /**
194          * converts the xml-feed to a json-object
195          * @param {Object} feed
196          */
197         var printFeed = function(feed) {
198             try {
199                 // Make the json-object where the rss-data will be saved
200                 var rss = {"items" : []};
201                 var xmlobject = feed;
202 
203                 // retrieve data from the xmlobject and put it in the JSON-object
204                 var channel = $(xmlobject).find("channel");
205 
206                 // set fields to check
207                 var itemField = "item";
208                 var dateField = "pubDate";
209                 var contentField = "description";
210                 var atomFeed = false;
211 
212                 // check if this is an atom feed instead of an rss feed
213                 if (!channel.length) {
214                     channel = $(xmlobject).find("feed");
215                     itemField = "entry";
216                     dateField = "published";
217                     contentField = "content";
218                     atomFeed = true;
219                 }
220 
221                 // put all the nodes in JSON-props
222                 rss.title = $("title:eq(0)",channel).text();
223                 rss.link = $("link:eq(0)",channel).text();
224                 rss.id = feedUrl;
225                 rss.description = $("description:eq(0)",channel).text();
226                 $(xmlobject).find(itemField).each(function() {
227                     var item = $(this);
228                     var pubDate = "";
229                     var pubDateObj = new Date();
230                     var dateText = $(dateField,item).text().replace("  ", " ");
231                     if (dateText.length > 0){
232                         if (!atomFeed){
233                             pubDateObj = sakai.api.Util.parseRFC822Date(dateText);
234                         } else {
235                             pubDateObj.setTime(Date.parse(dateText));
236                         }
237                         if (pubDateObj.valueOf()) {
238                             // we have a valid date
239                             pubDate = sakai.api.l10n.transformDateTimeShort(pubDateObj);
240                         } else {
241                             // invalid date
242                             pubDate = $(dateField,item).text();
243                             pubDateObj = new Date();  // can't sort on date...
244                         }
245                     }
246                     rss.items.push({
247                         "title" : $("title",item).text(),
248                         "link" : $("link",item).text(),
249                         "description" : $(contentField,item).text(),
250                         "pubDate" : pubDate,
251                         "guid" : $("guid",item).text(),
252                         "pubDateObj" : pubDateObj
253                     });
254               });
255               return rss;
256             } catch(ex){
257                 sakai.api.Util.notification.show($(rssIncorrectRssFeed).html(), $(rssPasteValidRssAddress).html());
258             }
259             // return false if some kind of error occured
260             // this will be mostly rss format errors and will also work as a tester to see if the rss format is correct
261             return false;
262         };

264         /**
265          * Request a feed
266          * @param {String} url: the url for the rss-feed
267          * @param {function} The function where the response will be send to;
268          */
269         var getFeed = function(url, onResponse) {
270             // put the url to a module variable
271             // later on this will also be added to the json-object
272             // See if we have http:// in url
273             if (url.search("http://") === -1 && url.search("https://") === -1) {
274                 url = "http://" + url;
275             }
276 
277             feedUrl = url;
278 
279             $.ajax({
280                 url : sakai.config.URL.PROXY_RSS +  url,
281                 type : "GET",
282                 success : function(data) {
283                     $(rssTxtUrl, rootel).val("");
284                     onResponse(printFeed(data));
285                 },
286                 error: function(xhr, textStatus, thrownError) {
287                     sakai.api.Util.notification.show($(rssUnableToConnect).html(), $(rssCannotConnectToRssFeed).html());
288                 }
289             });
290         };

300         var currentSort = "dateA";

302         /**
303          * sorts an array of feeds on the pubDate, this can be used with the JavaScript sort function
304          * @param {Object} a  a rss feed item
305          * @param {Object} b  another rss feed item
306          */
307         var sortByDatefunction = function(a, b) {
308             var ret = -1;
309             if (currentSort === "dateD") {
310                 ret = 1;
311             }
312             if (a.pubDateObj >  b.pubDateObj) {
313                 return ret;
314             } else if (b.pubDateObj >  a.pubDateObj) {
315                 return -(ret);
316             }
317             return 0;
318         };
319 
320         /**
321          * sorts an array of feeds on the source, this can be used with the javascript sort function
322          * @param {Object} a
323          * @param {Object} b
324          */
325         var sortBySourcefunction =  function(a, b){
326             var ret = -1;
327             if (currentSort === "sourceD") {
328                 ret = 1;
329             }
330             if (a.feed.title >  b.feed.title) {
331                 return -(ret);
332             } else if (b.feed.title >  a.feed.title) {
333                 return ret;
334             }
335             return 0;
336         };

339         /**
340          * Converts checkbox display settings to bools (they get sent back
341          * from the server as strings). This function is only effective if the
342          * resultJSON object contains the displaySource and displayHeadlines settings
343          */
344         var convertDisplaySettingsToBool = function () {
345             if (resultJSON) {
346                 if (resultJSON.displaySource) {
347                     if (!resultJSON.displaySource || resultJSON.displaySource === "false") {
348                         resultJSON.displaySource = false;
349                     } else {
350                         resultJSON.displaySource = true;
351                     }
352                 }
353                 if (resultJSON.displayHeadlines) {
354                     if (!resultJSON.displayHeadlines || resultJSON.displayHeadlines === "false") {
355                         resultJSON.displayHeadlines = false;
356                     } else {
357                         resultJSON.displayHeadlines = true;
358                     }
359                 }
360             }
361         };

392         /**
393          * returns an array of all the entries that are shown on this pagenumber
394          * @param {int} pagenumber
395          * @return {array} entries
396          */
397         var getShownEntries = function(page) {
398             page = page-1;
399             // if you don't make a clone, the splice-method will also remove entries form the original JSON-object
400             var shownEntries = cloneObject(resultJSON);
401             // only 3 entries per page
402             shownEntries.entries = shownEntries.entries.splice(page*3, 3);
403 
404             return shownEntries.entries;
405         };

407         /**
408          * Is executed when the user navigates to a different page in the rss widget
409          * Will the entries of the page where the user is currently at,
410          * and connects the pager again
411          * @param {Object} pageClicked
412          */
413         var pagerClickHandler = function(clicked) {
414             pageClicked = parseInt(clicked, 10);
415             // first get the entries that need to be shown on this page
416             resultJSON.shownEntries = getShownEntries(clicked);
417             // render these entries
418             $(rssOutput, rootel).html(sakai.api.Util.TemplateRenderer(rssOutputTemplate, resultJSON));
419             // change the pageNumeber
420             $(rssPager,rootel).pager({
421                 pagenumber: clicked,
422                 pagecount: Math.ceil(resultJSON.entries.length / 3),
423                 buttonClickCallback: pagerClickHandler
424             });
425             addBinding();
426         };

428         /**
429          * gets the entries from all subscribed feeds
430          * @param {Object} json
431          */
432         var fillRssOutput = function() {
433             // retrieve the rss-feeds
434             getFeed(resultJSON.urlFeeds[resultJSON.feeds.length], function(rssFeed) {
435                 for (var i = 0; i < rssFeed.items.length; i++) {
436                     // Some info needs to be shown in every entry
437                     rssFeed.items[i].feed = {};
438                     rssFeed.items[i].feed.title = rssFeed.title;
439                     rssFeed.items[i].feed.link = rssFeed.link;
440                     rssFeed.items[i].feed.id = rssFeed.id;
441                     rssFeed.items[i].feed.description = rssFeed.description;
442                     resultJSON.entries.push(rssFeed.items[i]);
443                 }
444                 resultJSON.feeds.push(rssFeed);
445                 // not all the feeds have been retrieved so call this function again
446                 if(resultJSON.feeds.length < resultJSON.urlFeeds.length){
447                     fillRssOutput();
448                 }
449                 // all the feeds have been retrieved
450                 else {
451                     // sort by date
452                     resultJSON.entries.sort(sortByDatefunction);
453                     // make sure the array only has the requested number of entries (example : 20 latest entries)
454                     resultJSON.entries.splice(resultJSON.numEntries,resultJSON.entries.length - resultJSON.numEntries);
455                     // show the first page only
456                     pagerClickHandler(1);
457                 }
458             });
459         };

462         /**
463          * Loads the settings screen
464          * @param {Object} exists
465          */
466         var loadSettings = function(exists) {
467             $(rssOutput,rootel).hide();
468             $(rssSettings,rootel).show();
469 
470             if (exists) {
471                 resultJSON.feeds = resultJSON.feeds || [];
472                 $(rssTxtTitle,rootel).val(resultJSON.title);
473                 $(rssNumEntries,rootel).val(resultJSON.numEntries);
474                 if (resultJSON.displaySource) {
475                     $(rssDisplaySource, rootel).attr("checked", "checked");
476                 } else {
477                     $(rssDisplaySource, rootel).removeAttr("checked");
478                 }
479                 if (resultJSON.displayHeadlines) {
480                     $(rssDisplayHeadlines, rootel).attr("checked", "checked");
481                 } else {
482                     $(rssDisplayHeadlines, rootel).removeAttr("checked");
483                 }
484                 resultJSON.feeds = [];
485                 if (resultJSON.urlFeeds && resultJSON.urlFeeds.length) {
486                     fillRssFeed(resultJSON.urlFeeds);
487                 }
488             }
489 
490         };

493         /**
494          * checks if a feed is already added
495          * @param {Object} rssFeedUrl
496          */
497         var checkIfRssAlreadyAdded = function(rssFeedUrl) {
498             resultJSON.feeds = resultJSON.feeds || [];
499             for (var i = 0; i < resultJSON.feeds.length; i++) {
500                 if ($.trim(resultJSON.feeds[i].id) === $.trim(rssFeedUrl)) {
501                     return true;
502                 }
503             }
504             return false;
505         };
506         /**
507          * executed when feed-request responses
508          * @param {Object} rssFeed
509          */
510         var getFeedResponse = function(rssFeed) {
511             if (rssFeed !== false) {
512                 resultJSON.feeds = resultJSON.feeds || [];
513                 resultJSON.feeds.push(rssFeed);
514                 $(rssFeedListContainer, rootel).html(sakai.api.Util.TemplateRenderer(rssFeedListTemplate, resultJSON));
515                 bindFeedRemove();
516             }
517         };

519         /**
520          * adds a feed to the widget
521          */
522         var addRssFeed = function() {
523             var rssURL = $(rssTxtUrl,rootel).val();
524             rssURL = rssURL.replace("feed://","http://");
525             if (!checkIfRssAlreadyAdded(rssURL)) {
526                 getFeed(rssURL, getFeedResponse);
527             } else {
528                 sakai.api.Util.notification.show($(rssFeedAlreadyAdded).html(), $(rssFeedAlreadyEntered).html());
529             }
530         };
531 
532         /**
533          * gets the settings object
534          */
535         var getSettingsObject = function() {
536             resultJSON.feeds = resultJSON.feeds || [];
537             resultJSON.title = $(rssTxtTitle,rootel).val();
538             resultJSON.numEntries = parseInt($(rssNumEntries,rootel).val(),10);
539             resultJSON.displaySource = $(rssDisplaySource, rootel).is(":checked");
540             resultJSON.displayHeadlines = $(rssDisplayHeadlines, rootel).is(":checked");
541             resultJSON.urlFeeds = [];
542             for (var i= 0; i< resultJSON.feeds.length; i++) {
543                 resultJSON.urlFeeds.push(resultJSON.feeds[i].id);
544             }
545             resultJSON.feeds = [];
546             return resultJSON;
547         };
548 
549         var saveRssOptions = function() {
550             var settingsObj = getSettingsObject();
551             if (settingsObj !== false) {
552                 sakai.api.Widgets.saveWidgetData(tuid, settingsObj, function(success, data){
553                     if ($(".sakai_dashboard_page").is(":visible")) {
554                         showSettings = false;
555                         showHideSettings(showSettings);
556                     } else {
557                         sakai.api.Widgets.Container.informFinish(tuid, "rss");
558                     }
559                 }, true);
560             }
561         };

567         /**
568          * Shows or hides the settings screen
569          * @param {Object} show
570          */
571         var showHideSettings = function(show) {
572             if (show) {
573                 sakai.api.Widgets.loadWidgetData(tuid, function(success, data){
574                     if (success) {
575                         resultJSON = data;
576                         convertDisplaySettingsToBool();
577                         loadSettings(true);
578                     } else {
579                         loadSettings(false);
580                     }
581                 });
582             } else {
583                 $(rssSettings,rootel).hide();
584                 $(rssOutput,rootel).show();
585 
586                 sakai.api.Widgets.loadWidgetData(tuid, function(success, data){
587                     if (success) {
588                         resultJSON = data;
589                         resultJSON.entries = [];
590                         resultJSON.feeds = [];
591                         convertDisplaySettingsToBool();
592                         if (resultJSON.urlFeeds && resultJSON.urlFeeds.length) {
593                             fillRssOutput();
594                         } else {
595                             $("#rss_no_feeds").show();
596                         }
597                     } else {
598                         $("#rss_no_feeds").show();
599                     }
600                 });
601             }
602         };
603 
604         showHideSettings(showSettings);
605 
606         addBinding();
607 
608         // Inserts the sendmessage-widget
609         sakai.api.Widgets.widgetLoader.insertWidgets(tuid);
610     };
