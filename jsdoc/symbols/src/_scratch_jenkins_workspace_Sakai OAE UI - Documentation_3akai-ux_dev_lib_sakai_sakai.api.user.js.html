 61                 "_charset_": "utf-8",
 62                 "locale": sakai_l10n.getUserDefaultLocale(),
 63                 "timezone": sakai_l10n.getUserDefaultTimezone(),
 64                 "pwd": password,
 65                 "pwdConfirm": passwordConfirm,
 66                 "firstName": firstName,
 67                 "lastName": lastName,
 68                 "email": email,
 69                 ":name": username,
 70                 ":sakai:profile-import": JSON.stringify(profileData)
 71             };
 72             for (var i in extraOptions) {
 73                 if (extraOptions.hasOwnProperty(i)) {
 74                     switch(i) {
 75                         case "recaptcha":
 76                             user[":create-auth"] = "reCAPTCHA.net";
 77                             user[":recaptcha-challenge"] = extraOptions[i].challenge;
 78                             user[":recaptcha-response"] = extraOptions[i].response;

 81                             user["locale"] = extraOptions[i];

 90             }

115         /**
116          * Update a user's profile
117          *
118          * @param {String} userid The userid of the user to update their profile
119          * @param {String} section The profile section (ie basic, publications)
120          * @param {Object} data The data to save on this section
121          * @param {Array} tags The tags and categories on this user
122          * @param {Object} sectionData The current data for this section, before any updates.
123          *                             Used for saving tags on the user.
124          * @param {Boolean} multiple If this is a multi-assign section, like publications
125          * @param {Function} callback The callback function for after the data has been saved
126          */
127         updateUserProfile: function( userid, section, data, tags, sectionData, multiple, callback ) {
128             var url = "/~" + userid + "/public/authprofile",
129                 saveJSONURL = url + "/" + section + ".profile.json";
130 
131             var postData = {
132                 elements: {}
133             };
134             $.each( data, function( key, value ) {
135                 if ( multiple ) {
136                     postData.elements[ key ] = {};
137                     $.each( value, function( subkey, subvalue ) {
138                         //oOrder is special, save it without a value sub-element
139                         if ( subkey === "order" ) {
140                             postData.elements[ key ][ subkey ] = subvalue;
141                         } else {
142                             postData.elements[ key ][ subkey ] = {
143                                 value: subvalue
144                             };
145                         }
146                     });
147                     // TODO set the data nested-like
148                 } else {
149                     postData.elements[ key ] = {
150                         value: value
151                     };
152                 }
153 
154             });
155 
156             var savedFunction = function(success, data) {
157                 if (success) {
158                     var oldDisplayName = sakaiUserAPI.getDisplayName(sakaiUserAPI.data.me.profile);
159                     // Update the users profile to reflect changed data
160                     $.extend(true, sakaiUserAPI.data.me.profile[section], postData);
161                     var newDisplayName = sakaiUserAPI.getDisplayName(sakaiUserAPI.data.me.profile);
162                     if (oldDisplayName !== newDisplayName) {
163                         $(window).trigger("displayName.profile.updated.sakai");
164                     }
165                 }
166                 if ($.isFunction(callback)) {
167                     callback(success, data);
168                 }
169             };
170 
171             var existingTags = sectionData["sakai:tags"] ? sectionData["sakai:tags"].value : false;
172             sakai_util.tagEntity( url, tags, existingTags, function( success, final_tags ) {
173                 sectionData["sakai:tags"] = {
174                     value: final_tags
175                 };
176                 sakai_serv.saveJSON( saveJSONURL, postData, savedFunction, true );
177             });

258          * @param {Array} userArray Array of userIds to fetch the profiles for
259          * @param {Function} callback Callback function to call when the request is complete
260          */
261         getMultipleUsers: function(userArray, callback){
262             var uniqueUserArray = [];
263             var batchRequests = [];
264 
265             for (var i in userArray) {
266                 if (userArray.hasOwnProperty(i) && $.inArray(userArray[i], uniqueUserArray) === -1) {
267                     uniqueUserArray.push(userArray[i]);
268                 }
269             }
270             for (var ii in uniqueUserArray) {
271                 if (uniqueUserArray.hasOwnProperty(ii)) {
272                     batchRequests.push({
273                         "url": "/~" + uniqueUserArray[ii] + "/public/authprofile.profile.json",
274                         "method": "GET",
275                         "dataType": "json"
276                     });
277                 }
278             }
279 
280             sakai_serv.batch(batchRequests, function(success, reqData) {
281                 var users = {};
282                 if (success) {
283                     $.each(reqData.results, function(index, val) {
284                         var data = $.parseJSON(val.body);
285                         if (data && data.userid) {
286                             users[data.userid] = data;
287                         }
288                     });
289                 }
290                 callback(users);
291             });

416         /**
417          * Retrieves all available information about a logged in user and stores it under data.me object. When ready it will call a specified callback function
418          *
419          * @param {Function} [callback] A function which will be called when the information is retrieved from the server.
420          * The first argument of the callback is a boolean whether it was successful or not, the second argument will contain the retrieved data or the xhr object
421          * @return {Void}
422          */
423         loadMeData : function(callback) {
424             var cache = true;
425             // don't use cache for IE8
426             if ($.browser.msie && parseInt($.browser.version, 10) < 9) {
427                 cache = false;
428             }
429 
430             // Start a request to the service
431             $.ajax({
432                 // Get the service url from the config file
433                 url: sakai_conf.URL.ME_SERVICE,
434                 cache: cache,
435                 success: function(data) {
436                     sakaiUserAPI.data.me = sakai_serv.convertObjectToArray(data, null, null);
437 
438                     // Check for firstName and lastName property - if not present use "rep:userId" for both (admin for example)
439                     if (sakaiUserAPI.getProfileBasicElementValue(sakaiUserAPI.data.me.profile, "firstName") === "") {
440                         sakaiUserAPI.setProfileBasicElementValue(sakaiUserAPI.data.me.profile, "firstName", sakaiUserAPI.data.me.profile["rep:userId"]);
441                     }
442                     if (sakaiUserAPI.getProfileBasicElementValue(sakaiUserAPI.data.me.profile, "lastName") === "") {
443                         sakaiUserAPI.setProfileBasicElementValue(sakaiUserAPI.data.me.profile, "lastName", sakaiUserAPI.data.me.profile["rep:userId"]);
444                     }
445 
446                     // SAKIII-2419 server isn't saving basic access param
447                     if (sakaiUserAPI.data.me.profile.basic.access === undefined){
448                         sakaiUserAPI.data.me.profile.basic.access = "everybody";
449                     }
450 
451                     if (sakaiUserAPI.data.me.user.properties) {
452                         if (sakaiUserAPI.data.me.user.properties.isAutoTagging) {
453                             if (sakaiUserAPI.data.me.user.properties.isAutoTagging === "true") {
454                                 sakaiUserAPI.data.me.user.properties.isAutoTagging = true;
455                             } else if (sakaiUserAPI.data.me.user.properties.isAutoTagging === "false") {
456                                 sakaiUserAPI.data.me.user.properties.isAutoTagging = false;
457                             }
458                         }
459                         if (sakaiUserAPI.data.me.user.properties.sendTagMsg) {
460                             if (sakaiUserAPI.data.me.user.properties.sendTagMsg === "true") {
461                                 sakaiUserAPI.data.me.user.properties.sendTagMsg = true;
462                             } else if (sakaiUserAPI.data.me.user.properties.sendTagMsg === "false") {
463                                 sakaiUserAPI.data.me.user.properties.sendTagMsg = false;
464                             }
465                         }
466                     }
467 
468                     // Call callback function if set
469                     if ($.isFunction(callback)) {
470                         callback(true, sakaiUserAPI.data.me);
471                     }
472                 },
473                 error: function(xhr, textStatus, thrownError) {
474 
475                     // Log error
476                     debug.error("sakai.api.User.loadMeData: Could not load logged in user data from the me service!");
477 
478                     if (xhr.status === 500 && window.location.pathname !== "/dev/500.html" && window.location.pathname !== "/500") {
479                         document.location = "/500";
480                     }
481 
482                     // Call callback function if set
483                     if ($.isFunction(callback)) {
484                         callback(false, xhr);
485                     }
486                 }
487             });
488         },

522         /**
523          * Retrieves the display name to use for the user from config
524          * and parses it from the profile elements
525          *
526          * @param {Object} profile the user's profile (data.me.profile for the current user)
527          * @return {String} the name to show for a user
528          */
529         getDisplayName : function(profile) {
530             var configDisplayName = [sakai_conf.Profile.userNameDisplay, sakai_conf.Profile.userNameDefaultDisplay];
531             var nameToReturn = "";
532             var done = false;
533             var idx = 0;
534 
535             var parseName = function(i,key) {
536                 if (profile &&
537                     profile.basic &&
538                     profile.basic.elements &&
539                     profile.basic.elements[key] !== undefined &&
540                     profile.basic.elements[key].value !== undefined &&
541                     $.trim(profile.basic.elements[key].value) !== "") {
542                    nameToReturn += profile.basic.elements[key].value + " ";
543                    done = true;
544                }
545             };
546 
547             // iterate over the configDisplayName object until a valid non-empty display name is found
548             while (!done && idx < 2) {
549                 if (configDisplayName[idx] !== undefined && configDisplayName[idx] !== "") {
550                     var configEltsArray = configDisplayName[idx].split(" ");
551                     $(configEltsArray).each(parseName);
552                 }
553                 idx++;
554             }
555 
556             if(!done){
557                 if(profile && profile["rep:userId"]){
558                     return profile["rep:userId"];
559                 } else {
560                     return "";
561                 }
562             } else {
563                 return sakai_util.Security.safeOutput($.trim(nameToReturn));
564             }
565             return false;
566         },

568         /**
569          * Get a user's short description from their profile
570          * This is based off of the configuration in config.js
571          * Example: "${role} in ${department}" could translate to "Undergraduate Student in Computer Science"
572          *           based on the configuration in config.js and the user's profile information
573          * If the user doesn't have the profile information requested by config.js, the function
574          * will remove the token from the string and any modifiers before the token after the previous token
575          * In the above example, if the user only had a department, the resulting string would be "Computer Science"
576          *
577          * @param {Object} profile The user's profile to get a description from
578          * @return {String} the user's short description
579          */
580         getShortDescription : function(profile) {
581             var shortDesc = sakai_conf.Profile.shortDescription || "";
582             var tokenRegex = /\$\{[A-Za-z]+\}/gi;
583             var tokens = shortDesc.match(tokenRegex);
584             var lastReplacementValue = "";
585             $(tokens).each(function(i, val) {
586                 var profileNode = val.match(/[A-Za-z]+/gi)[0];
587                 if (profile.basic.elements[profileNode] && $.trim(profile.basic.elements[profileNode].value) !== "") {
588                     /*if (lastReplacementValue === "" && tokens[i-1]) {
589                         // replace everything before this and after the last token
590                     } */
591                     if (sakai_conf.Profile.configuration.defaultConfig.basic.elements[profileNode].type === "select") {
592                         lastReplacementValue = profile.basic.elements[profileNode].value;
593                         lastReplacementValue = sakai_conf.Profile.configuration.defaultConfig.basic.elements[profileNode].select_elements[lastReplacementValue];
594                         lastReplacementValue = sakai_i18n.General.process(lastReplacementValue);
595                     } else {
596                         lastReplacementValue = profile.basic.elements[profileNode].value;
597                     }
598 
599                     shortDesc = shortDesc.replace(val, lastReplacementValue);
600                 } else {
601                     if (tokens[i-1]) { // this is not the first time through
602                         var indexToStart = 0;
603                         // if the previous token's replaced value exists
604                         if (lastReplacementValue !== "" && shortDesc.indexOf(shortDesc.indexOf(lastReplacementValue)) !== -1) {
605                             // the index to start replacing at is the end of the last replacement
606                             indexToStart = shortDesc.indexOf(shortDesc.indexOf(lastReplacementValue)) + lastReplacementValue.length;
607                         }
608                         var indexToEnd = shortDesc.indexOf(val) + val.length;
609                         shortDesc = $.trim(shortDesc.replace(shortDesc.substring(indexToStart, indexToEnd), ""));
610                     } else {
611                         shortDesc = $.trim(shortDesc.replace(val, ""));
612                     }
613                 }
614             });
615             return $.trim(shortDesc);
616         },

792         /**
793          * Function to process search results for users
794          *
795          * @param {Object} results Search results to process
796          * @param {Object} meData User object for the user
797          * @returns {Object} results Processed results
798          */
799         preparePeopleForRender: function(results, meData) {
800             $.each(results, function(i, item){
801                 // The My Contacts feed comes back with everything wrapped inside of
802                 // a target object
803                 if (item.target){
804                     item = item.profile;
805                 }
806                 if (item && item["rep:userId"] && item["rep:userId"] !== "anonymous") {
807                     item.id = item["rep:userId"];
808                     item.userid = item["rep:userId"];
809                     item.picture = sakaiUserAPI.getProfilePicture(item);
810                     item.name = sakaiUserAPI.getDisplayName(item);
811                     item.nameShort = sakai_util.applyThreeDots(item.name, 580, {max_rows: 1,whole_word: false}, "s3d-bold", true);
812                     item.nameShorter = sakai_util.applyThreeDots(item.name, 150, {max_rows: 1,whole_word: false}, "s3d-bold", true);
813 
814                     // use large default user icon on search page
815                     if (item.picture === sakai_conf.URL.USER_DEFAULT_ICON_URL){
816                         item.pictureLarge = sakai_conf.URL.USER_DEFAULT_ICON_URL_LARGE;
817                     }
818                     if (item["sakai:tags"] && item["sakai:tags"].length > 0){
819                         item.tagsProcessed = sakai_util.formatTags(item["sakai:tags"]);
820                     } else if (item.basic && item.basic.elements && item.basic.elements["sakai:tags"]) {
821                         item.tagsProcessed = sakai_util.formatTags(item.basic.elements["sakai:tags"].value);
822                     }
823 
824                     item.connected = false;
825                     item.accepted = false;
826                     item.invited = item.invited !== undefined ? item.invited : false;
827                     // Check if this user is a friend of us already.
828                     var connectionState = false;
829                     if (item["sakai:state"] || results[i]["details"]) {
830                         connectionState = item["sakai:state"] || results[i]["details"]["sakai:state"];
831                         item.connected = true;
832                         // if invited state set invited to true
833                         if(connectionState === "INVITED"){
834                             item.invited = true;
835                         } else if(connectionState === "PENDING"){
836                             item.pending = true;
837                         } else if(connectionState === "ACCEPTED"){
838                             item.accepted = true;
839                         } else if(connectionState === "NONE"){
840                             //user.none = true;
841                             item.connected = false;
842                         }
843                     }
844 
845                     // Check if the user you found in the list isn't the current
846                     // logged in user
847                     if (item.userid === meData.user.userid) {
848                         item.isMe = true;
849                     }
850                     results[i] = item;
851                 }
852             });
853             return results;
854         },
