 18 require(["jquery", "sakai/sakai.api.core"], function($, sakai){
 19 
 20     /**
 21      * @name sakai_global.carousel
 22      *
 23      * @class carousel
 24      *
 25      * @description
 26      *
 27      * @version 0.0.1
 28      * @param {String} tuid Unique id of the widget
 29      * @param {Boolean} showSettings Show the settings of the widget or not
 30      */
 31     sakai_global.carousel = function(tuid, showSettings){
 32 
 33         /////////////////////////////
 34         // Configuration variables //
 35         /////////////////////////////
 36 
 37         // Containers
 38         var carouselContainer = "#carousel_container";
 39 
 40         // Templates
 41         var carouselSingleColumnTemplate = "carousel_single_column_template";
 42 
 43 
 44         /////////////////////
 45         // RENDER CAROUSEL //
 46         /////////////////////
 47 
 48         var updateViewAfterAnimation = function(carousel, li, index, state){
 49             if (index > carousel.options.size || index < 1) {
 50                 index = index % carousel.options.size;
 51                 if (!index) {
 52                     index = carousel.options.size;
 53                 }
 54                 if (index < 1) {
 55                     index = carousel.options.size + index;
 56                 }
 57             }
 58             $("#carousel_container .carousel_view_toggle li").removeClass("carousel_view_toggle_selected");
 59             $("#carousel_view_toggle_" + carousel.last).removeClass("carousel_view_toggle_selected");
 60             $("#carousel_view_toggle_" + index).addClass("carousel_view_toggle_selected");
 61 
 62             $(window).bind("sakai.addToContacts.requested", function(evObj, user){
 63                 var addbutton = $.grep($("#carousel_container .sakai_addtocontacts_overlay"), function(value, index) {
 64                     return $(value).attr("sakai-entityid") === user.userid;
 65                 });
 66                 $(addbutton).remove();
 67             });
 68         };
 69 
 70         var stopAutoScrolling = function(carousel){
 71             carousel.startAuto(0);
 72         };
 73 
 74         var carouselBinding = function(carousel){
 75             // Pause autoscrolling if the user moves with the cursor over the clip.
 76             carousel.clip.hover(function(){
 77                 carousel.stopAuto();
 78             }, function(){
 79                 carousel.startAuto();
 80             });
 81 
 82             // Disable autoscrolling if the user clicks the prev or next button.
 83             carousel.buttonNext.bind('click', function(){
 84                 stopAutoScrolling(carousel);
 85             });
 86 
 87             carousel.buttonPrev.bind('click', function(){
 88                 stopAutoScrolling(carousel);
 89             });
 90 
 91             $('.carousel_view_toggle li').bind('click', function(){
 92                 stopAutoScrolling(carousel);
 93                 carousel.scroll($.jcarousel.intval($(this)[0].id.split("carousel_view_toggle_")[1]));
 94                 $("#carousel_container .carousel_view_toggle li").removeClass("carousel_view_toggle_selected");
 95                 $(this).addClass("carousel_view_toggle_selected");
 96                 return false;
 97             });
 98 
 99             $(window).bind(tuid + ".shown.sakai", {"carousel": carousel}, toggleCarousel);
100         };
101 
102         var toggleCarousel = function(e, showing){
103             if (showing) {
104                 e.data.carousel.startAuto();
105             } else {
106                 e.data.carousel.stopAuto();
107             }
108         };
109 
110         var renderCarousel = function(dataArr){
111             sakai.api.Util.TemplateRenderer(carouselSingleColumnTemplate, {
112                 "data": dataArr,
113                 "sakai": sakai
114             }, $(carouselContainer), false);
115             applyThreeDots();
116             $(carouselContainer).jcarousel({
117                 auto: 8,
118                 animation: "slow",
119                 scroll: 1,
120                 easing: "swing",
121                 size: dataArr.length,
122                 initCallback: carouselBinding,
123                 wrap: "circular",
124                 itemFirstInCallback: {
125                     onAfterAnimation: updateViewAfterAnimation
126                 }
127             });
128         };
129 
130         var applyThreeDots = function(){
131             $.each($(".carousel_apply_threedots"), function(index, item){
132                 var maxrows = 1;
133                 if (item && item.className) {
134                     var classes = item.className.split(" ");
135                     $.each(classes, function(i, cl){
136                         if (cl && cl.indexOf("threedots_allow_") === 0) {
137                             maxrows = parseInt(cl.split("threedots_allow_")[1], 10);
138                             return false;
139                         }
140                     });
141                 }
142                 $(item).text(sakai.api.Util.applyThreeDots($(item).text(), $(item).width(), {max_rows:maxrows}, "carousel_content_tags s3d_action", true));
143             });
144         };
145 
146         var parseContent = function(data, dataArr, callback){
147             var noPreviewArr = [];
148             var previewArr = [];
149 
150             if (data && data.content && data.content.results) {
151                 sakai.api.Content.prepareContentForRender(data.content.results, sakai.data.me, function(results){
152                     $.each(results, function(index, item) {
153                         if (item.thumbnail){
154                             previewArr.push(item);
155                         } else {
156                             noPreviewArr.push(item);
157                         }
158                     });
159                     // Prefer items with previews
160                     var suggested = {
161                         contentType: "suggestedContent",
162                         suggestions: previewArr.concat(noPreviewArr)
163                     };
164                     dataArr.push(suggested);
165                     callback(data, dataArr);
166                 });
167             }
168         };
169 
170         var parseGroups = function(data, dataArr){
171             var picDescTags = [];
172             var picDesc = [];
173             var picTags = [];
174             var descTags = [];
175             var desc = [];
176             var tags = [];
177             var noPic = [];
178             $.each(data.groups.results, function (index, group){
179                 var obj = {};
180 
181                 if (group["sakai:group-description"] && group["sakai:group-description"].length) {
182                     obj.description = group["sakai:group-description"];
183                 }
184                 if (group["sakai:tags"] && group["sakai:tags"].length) {
185                     obj.tags = sakai.api.Util.formatTags(group["sakai:tags"]);
186                 }
187                 if (group.picture){
188                     obj.picture = sakai.api.Groups.getProfilePicture(group);
189                 }
190                 obj.counts = group.counts;
191 
192                 obj.contentType = "group";
193                 obj.groupid = group["sakai:group-id"];
194                 obj.title = group["sakai:group-title"];
195 
196                 if (obj.picture && obj.description && obj.tags) {
197                     picDescTags.push(obj);
198                 } else if (obj.picture && obj.description) {
199                     picDesc.push(obj);
200                 } else if (obj.picture && obj.tags) {
201                     picTags.push(obj);
202                 } else if (obj.description && obj.tags) {
203                     descTags.push(obj);
204                 } else if (obj.description) {
205                     desc.push(obj);
206                 } else if (obj.tags) {
207                     tags.push(obj);
208                 } else {
209                     noPic.push(obj);
210                 }
211             });
212             var suggested = {
213                 contentType: "suggestedGroups",
214                 suggestions: picDescTags.concat(picDesc, picTags, descTags, desc, tags, noPic).splice(0,6)
215             };
216 
217             dataArr.push(suggested);
218 
219         };
220 
221         var parseUsers = function(data, dataArr){
222             var hasPicAndTag = [];
223             var hasPic = [];
224             var hasTag = [];
225             var noPicAndTag = [];
226 
227             sakai.api.User.getContacts(function() {
228                 $.each(data.users.results, function (index, user){
229                     var obj = {};
230 
231                     obj.userid = user.profile.userid;
232                     obj.contentType = "user";
233                     obj.displayName = sakai.api.User.getDisplayName(user.profile);
234                     obj.counts = user.profile.counts;
235 
236                     user = user.profile.basic.elements;
237                     if (user["sakai:tags"] && user["sakai:tags"].value && user["sakai:tags"].value.length){
238                         obj.tags = sakai.api.Util.formatTags(user["sakai:tags"].value);
239                     }
240                     if (user.aboutme){
241                         obj.aboutme = user.aboutme.elements.aboutme.value;
242                     }
243                     if (user.picture && user.picture.value && user.picture.value.length){
244                         obj.picture = $.parseJSON(user.picture.value);
245                     }
246                     // is the user a contact or pending contact
247                     if ($.grep(sakai.data.me.mycontacts, function(value, index){return value.target === obj.userid;}).length !== 0){
248                         obj.connected = true;
249                     }
250 
251                     if (obj.picture && obj.tags){
252                         hasPicAndTag.push(obj);
253                     } else if (obj.picture) {
254                         hasPic.push(obj);
255                     } else if (obj.tags) {
256                         hasTag.push(obj);
257                     } else {
258                         noPicAndTag.push(obj);
259                     }
260                 });
261 
262                 var suggested = {
263                     contentType: "suggestedUsers",
264                     suggestions: hasPicAndTag.concat(hasPic, hasTag, noPicAndTag).splice(0, 8)
265                 };
266                 dataArr.push(suggested);
267             });
268         };
269 
270         var parseData = function(data){
271             var dataArr = [];
272             parseContent(data, dataArr, function(data, dataArr){
273                 parseGroups(data, dataArr);
274                 parseUsers(data, dataArr);
275                 if (dataArr.length) {
276                     renderCarousel(dataArr);
277                 }
278             });
279         };
280 
281         var loadFeatured = function(){
282             var dataArr = {
283                 "content": false,
284                 "groups": false,
285                 "users": false
286             };
287 
288             var reqs = [
289                 {
290                     url: "/var/search/pool/me/related-content.json",
291                     method: "GET",
292                     parameters: {
293                         "items": 11
294                     }
295                 },
296                 {
297                     url: "/var/contacts/related-contacts.json",
298                     method: "GET",
299                     parameters: {
300                         "items": 11
301                     }
302                 },
303                 {
304                     url: "/var/search/myrelatedgroups.json",
305                     method: "GET",
306                     parameters: {
307                         "items": 11
308                     }
309                 }
310             ];
311 
312             sakai.api.Server.batch(reqs, function(success, data) {
313                 if (success) {
314                     //content
315                     dataArr.content = $.parseJSON(data.results[0].body);
316                     //users
317                     dataArr.users = $.parseJSON(data.results[1].body);
318                     //groups
319                     dataArr.groups = $.parseJSON(data.results[2].body);
320                 }
321                 parseData(dataArr);
322             });
323 
324         };
325 
326         var doInit = function(){
327             loadFeatured();
328         };
329 
330         doInit();
331     };
332 
333     sakai.api.Widgets.widgetLoader.informOnLoad("carousel");
334 });
