164         sakai_global.newaddcontent.getNewContent = function(library) {
165             var newContentLibrary = [];
166             // grab all of the newly uploaded content, regardless of target library
167             if (!library) {
168                 newContentLibrary = allNewContent;
169             } else if (brandNewContent[library]) {
170                 newContentLibrary = brandNewContent[library];
171             }
172             // return a copy
173             return $.merge([], newContentLibrary);
174         };

176         var deleteContent = function(e, paths) {
177             if (paths && paths.length) {
178                 $.each(paths, function(i, path) {
179                     $.each(allNewContent, function(j, newContent) {
180                         if (newContent && newContent._path === path) {
181                             allNewContent.splice(j,1);
182                         }
183                     });
184                     $.each(brandNewContent, function(lib, items) {
185                         $.each(items, function(k, item) {
186                             if (item && item._path === path) {
187                                 items.splice(k,1);
188                             }
189                         });
190                     });
191                 });
192             }
193         };

220         /**

267         /**
268          * Add an item to the queue
269          * @param {Object/Array} contentToAdd Object or array of objects containing data about the object to be added to the queue
270          * @param {Boolean} disableRender Disable rendering of the queue.
271          */
272         var addContentToQueue = function(contentToAdd, disableRender, append) {
273             if ($.isArray(contentToAdd)) {
274                 itemsToUpload = itemsToUpload.concat(contentToAdd);
275             } else {
276                 itemsToUpload.push(contentToAdd);
277             }
278 
279             disableAddToQueue();
280             enableStartUpload();
281             if (!disableRender) {
282                 renderQueue(append, contentToAdd);
283             }
284         };

286         /**
287          * Remove an item from the queue
288          */
289         var removeItemToAdd = function() {
290             $newaddcontentSelectedItemsEditPermissionsContainer.hide();
291             $newaddcontentSelecteditemsEditDataContainer.hide();
292 
293             var index = $(this).parent()[0].id.split('newaddcontent_selecteditems_')[1];
294             var obj = itemsToUpload[index];
295 
296             switch (obj.type) {
297                 case 'content':
298                     var $found = $('*:contains(\'' + obj.originaltitle + '\')');
299                     $found.last().prev('a').click();
300                     // If the user removes an item that was selected through browsing the OS reduce the file count to avoid bug (https://jira.sakaiproject.org/browse/SAKIII-3269)
301                     if(obj.origin == 'user') {
302                         numberOfBrowsedFiles--;
303                     }
304                     break;
305                 case 'existing':
306                     var $existing = $('input#' + obj['_path']);
307                     $existing.removeAttr('disabled');
308                     $existing.removeAttr('checked');
309                     $existing.parent().removeClass(newaddcontentExistingItemsListContainerDisabledListItem);
310                     break;
311             }
312 
313             itemsToUpload.splice(index,1);
314 
315             if (!itemsToUpload.length) {
316                 disableStartUpload();
317             }
318 
319             renderQueue();
320         };

322         /**
323          * Construct an item to add to the queue
324          * Depending on the type of the item to add construct a different object
325          */
326         var constructItemToAdd = function() {
327             var uniqueId = sakai.api.Util.generateWidgetId();
328             var tags = sakai.api.Util.AutoSuggest.getTagsAndCategories($autoSuggestElt, true);
329             var $thisForm = $(this).parents($newaddcontentNewItemContainer).children(newAddContentForm);
330             if ($(this).attr('id') === 'newaddcontent_container_newitem_raquo_right') {
331                 $thisForm = $(this).prev().children(':visible').find(newAddContentForm);
332             }
333 
334             switch ($thisForm.attr('id')) {
335 
336                 //////////////////////////
337                 // Uploading a new file //
338                 //////////////////////////
339 
340                 case 'newaddcontent_upload_content_form':
341                     var originalTitle = $thisForm.find(newaddcontentUploadContentOriginalTitle)[0].id;
342 
343                     // Calculate the file extension
344                     var splitOnDot = originalTitle.split('.');
345                     var contentObj = {
346                         'sakai:pooled-content-file-name': $thisForm.find(newaddcontentUploadContentTitle).val(),
347                         'sakai:description': $thisForm.find(newaddcontentUploadContentDescription).val(),
348                         'sakai:permissions': $thisForm.find(newaddcontentUploadContentPermissions).val(),
349                         'sakai:copyright': $('#newaddcontent_upload_content_copyright').val(),
350                         'sakai:originaltitle': originalTitle,
351                         'sakai:tags': tags,
352                         'sakai:fileextension': splitOnDot[splitOnDot.length - 1],
353                         'css_class': sakai.config.MimeTypes[sakai.config.Extensions[(originalTitle).slice(originalTitle.lastIndexOf('.') + 1, originalTitle.length).toLowerCase()] || 'other'].cssClass || 's3d-icon-unknown',
354                         'type': 'content',
355                         'origin':'user' // 'origin' tells Sakai that this file was selected from the users hard drive
356                     };
357                     // Store the temporary browsed file in the upload array
358                     filesList.push(tmpBrowsedFile);
359                     addContentToQueue(contentObj);
360                     multifileQueueAddAllowed = true;
361                     $thisForm.find(newaddcontentUploadContentTitle + ', ' + newaddcontentUploadContentDescription + ', ' + newaddcontentUploadContentTags).val('');
362                     // Increase the number of files that the user browsed for and added to the list
363                     numberOfBrowsedFiles++;
364                     break;
365 
366                 ///////////////////
367                 // Adding a link //
368                 ///////////////////
369 
370                 case 'newaddcontent_add_link_form':
371                     var linkObj = {
372                         'sakai:pooled-content-url': $thisForm.find(newaddcontentAddLinkURL).val(),
373                         'sakai:pooled-content-file-name': $thisForm.find(newaddcontentAddLinkTitle).val() || $thisForm.find(newaddcontentAddLinkURL).val(),
374                         'sakai:description': $thisForm.find(newaddcontentAddLinkDescription).val(),
375                         'sakai:tags': tags,
376                         'sakai:permissions': sakai.config.Permissions.Links.defaultaccess,
377                         'sakai:copyright': sakai.config.Permissions.Copyright.defaults['links'],
378                         'css_class': 's3d-icon-url',
379                         'type':'link'
380                     };
381                     addContentToQueue(linkObj);
382                     $thisForm[0].reset();
383                     break;
384 
385                 /////////////////////////////
386                 // Creating a new document //
387                 /////////////////////////////
388 
389                 case 'newaddcontent_add_document_form':
390                     if ($thisForm.valid()) {
391                         var documentObj = {
392                             'sakai:pooled-content-file-name': $thisForm.find(newaddcontentAddDocumentTitle).val(),
393                             'sakai:permissions': $thisForm.find(newaddcontentAddDocumentPermissions).val(),
394                             'sakai:description': $thisForm.find(newaddcontentAddDocumentDescription).val(),
395                             'sakai:tags': tags,
396                             'sakai:copyright': sakai.config.Permissions.Copyright.defaults['sakaidocs'],
397                             'css_class': 's3d-icon-sakaidoc',
398                             'type': 'document'
399                         };
400                         addContentToQueue(documentObj);
401                         $thisForm[0].reset();
402                     }
403                     break;
404 
405                 ///////////////////////////////
406                 // Re-using existing content //
407                 ///////////////////////////////
408 
409                 case 'newaddcontent_existing_content_form':
410                     var contentToAdd = [];
411                     $.each($thisForm.find('.newaddcontent_existingitems_select_checkbox:checked'), function(index, item) {
412                         if (!$(item).is(':disabled')) {
413                             var viewers = [];
414                             if ($(item).data('sakai-pooled-content-viewer')) {
415                                 viewers = ('' + $(item).data('sakai-pooled-content-viewer')).split(',');
416                             }
417                             var managers = [];
418                             if ($(item).data('sakai-pooled-content-manager')) {
419                                 managers = ('' + $(item).data('sakai-pooled-content-manager')).split(',');
420                             }
421                             var contentObj = {
422                                 'sakai:pooled-content-file-name': $(item).next().text(),
423                                 'sakai:pooled-content-viewer': viewers,
424                                 'sakai:pooled-content-manager': managers,
425                                 '_path': item.id,
426                                 '_mimeType': $(item).data('mimetype'),
427                                 'canshare': $(item).attr('data-canshare'),
428                                 'type': 'existing',
429                                 'css_class': $(item).next().children(newaddcontentExistingItemsListContainerListItemIcon)[0].id
430                             };
431                             contentToAdd.push(contentObj);
432                             $(item).attr('disabled', 'disabled');
433                             $(item).parent().addClass(newaddcontentExistingItemsListContainerDisabledListItem);
434                         }
435                     });
436                     addContentToQueue(contentToAdd, false, true);
437                     break;
438 
439             }
440             sakai.api.Util.AutoSuggest.reset( $autoSuggestElt );
441             sakai.api.Util.AutoSuggest.setupTagAndCategoryAutosuggest( $autoSuggestElt, null, $autoSuggestListCatElt );
442         };

645          * @param {Object} documentObj    Content object that represents the original zip file upload
646          * @param {Object} fileUploadObj  Content object that was returned when uploading the zip file
647          */
648         var setIMSCPContent = function(documentObj, fileUploadObj) {
649             // Use the filename and description provided by the package
650             documentObj['sakai:pooled-content-file-name'] = fileUploadObj['sakai:pooled-content-file-name'];
651             documentObj['sakai:description'] = fileUploadObj['sakai:description'];
652             // Set page content for all pages in the package
653             var resources = $.parseJSON(documentObj.resources);
654             var content = {};
655             var resourceIds = {};
656             for (var i = 0; i < resources.length; i++) {
657                 resourceIds[i] = resources[i]._id;
658                 var widgetId = sakai.api.Util.generateWidgetId();
659                 content[resourceIds[i]] = {
660                     'rows': [{
661                         'id': sakai.api.Util.generateWidgetId(),
662                         'columns': [{
663                             'width': 1,
664                             'elements': [{
665                                 'id': widgetId,
666                                 'type': 'htmlblock'
667                             }]
668                         }]
669                     }]
670                 };
671                 content[resourceIds[i]][widgetId] = {
672                     'htmlblock': {
673                         'content': resources[i].page
674                     }
675                 };
676             }
677             finishSakaiDoc(documentObj, content);

835                     'method': 'POST',
836                     'parameters': {
837                         'sakai:pooled-content-file-name': obj['sakai:pooled-content-file-name'],
838                         'sakai:description': obj['sakai:description'],
839                         'sakai:permissions': obj['sakai:permissions'],
840                         'sakai:copyright': obj['sakai:copyright'],
841                         'sakai:allowcomments': 'true',
842                         'sakai:showcomments': 'true',
843                         'sakai:fileextension': obj['sakai:fileextension']

845                 });
846 
847                 // Set initial version
848                 if (obj['_mimeType'] !== 'x-sakai/document') {
849                     contentDataBatch.push({
850                         'url': '/p/' + obj['_path'] + '.save.json',
851                         'method': 'POST'

859                         checkUploadCompleted();
860                     });
861                 });

865                 $.each(contentObj, function(index, obj) {
866                     setContent(obj);
867                     lastUpload.push(obj);
868                 });
869             } else {
870                 setContent(contentObj);
