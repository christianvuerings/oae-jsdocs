 25 require(["jquery", "sakai/sakai.api.core"], function($, sakai) {
 26 
 27     /**
 28      * @name sakai_global.personinfo
 29      *
 30      * @class personinfo
 31      *
 32      * @description
 33      * Personinfo dialog box
 34      *
 35      * @version 0.0.1
 36      * @param {String} tuid Unique id of the widget
 37      * @param {Boolean} showSettings Show the settings of the widget or not
 38      */
 39     sakai_global.personinfo = function(tuid, showSettings) {
 40 
 41         var $rootel = $("#" + tuid);
 42         var $personinfo_widget = $("#personinfo_widget", $rootel);
 43         var $personinfo_container = $("#personinfo_container", $rootel);
 44         var $personinfo_template = $("#personinfo_template", $rootel);
 45         var $personinfo_close = $(".personinfo_close", $rootel);
 46         var $personinfo_message = $("#personinfo_message", $rootel);
 47         var $personinfo_invite = $("#personinfo_invite", $rootel);
 48         var $personinfo_invited = $("#personinfo_invited", $rootel);
 49         var $personinfo_pending = $("#personinfo_pending", $rootel);
 50         var dataCache = {};
 51         var open = false;
 52         var userId;
 53 
 54         $personinfo_widget.jqm({
 55             modal: false,
 56             overlay: 0,
 57             zIndex: 900,
 58             toTop: true
 59         });
 60 
 61         /**
 62          * hidePersonInfo
 63          * Hides the widget
 64          */
 65         var hidePersonInfo = function() {
 66             open = false;
 67             $personinfo_widget.jqmHide();
 68 
 69             // unbind the close event
 70             $(document).unbind("click.personinfo_close");
 71         };
 72 
 73         /**
 74          * showPersonInfo
 75          * Shows the widget
 76          */
 77         var showPersonInfo = function($clickedEl) {
 78             var personinfoTop = $clickedEl.offset().top + $clickedEl.height();
 79             var personinfoLeft = $clickedEl.offset().left + $clickedEl.width() / 2 - 125;
 80 
 81             $personinfo_widget.css({
 82                 top: personinfoTop,
 83                 left: personinfoLeft
 84             });
 85 
 86             $personinfo_widget.jqmShow();
 87         };
 88 
 89         /**
 90          * showConnectionButton
 91          * Display appropriate connection button according to users connection state
 92          */
 93         var showConnectionButton = function() {
 94             $personinfo_message.hide();
 95             $personinfo_invite.hide();
 96             $personinfo_invited.hide();
 97             $personinfo_pending.hide();
 98             if (userId !== sakai.data.me.user.userid && !sakai.data.me.user.anon) {
 99                 $personinfo_message.show();
100                 if (!dataCache[userId].connectionState || dataCache[userId].connectionState === "NONE") {
101                     $personinfo_invite.show();
102                 } else if (dataCache[userId].connectionState === "PENDING") {
103                     $personinfo_pending.show();
104                 } else if (dataCache[userId].connectionState === "INVITED") {
105                     $personinfo_invited.show();
106                 }
107             }
108         };
109 
110         /**
111          * togglePersonInfo
112          * Displays the widget
113          */
114         var togglePersonInfo = function($clickedEl) {
115             showConnectionButton();
116 
117             var json = {
118                 "user": dataCache[userId],
119                 "me": sakai.data.me,
120                 "sakai": sakai
121             };
122 
123             $($personinfo_container).html(sakai.api.Util.TemplateRenderer("#personinfo_template", json));
124             showPersonInfo($clickedEl);
125         };
126 
127         /**
128          * fetchPersonInfo
129          * Fetches data about the user
130          */
131         var fetchPersonInfo = function($clickedEl) {
132             if (dataCache[userId]) {
133                 togglePersonInfo($clickedEl);
134             } else {
135                 // display loading message
136                 $($personinfo_container).html(sakai.api.Util.TemplateRenderer("#personinfo_loading_template", {"me": sakai.data.me}));
137                 showPersonInfo($clickedEl);
138                 sakai.api.User.getUser(userId, function(success, data) {
139                     if (success) {
140                         dataCache[userId] = data;
141 
142                         // check if user is a contact and their connection state
143                         sakai.api.User.getConnectionState(userId, function(state) {
144                             dataCache[userId].connectionState = state;
145                         });
146 
147                         // get display pic
148                         var displayPicture = sakai.api.Util.constructProfilePicture(dataCache[userId]);
149                         if (!displayPicture) {
150                             displayPicture = sakai.config.URL.USER_DEFAULT_ICON_URL;
151                         }
152                         dataCache[userId].displayPicture = displayPicture;
153                         dataCache[userId].displayName = sakai.api.User.getDisplayName(dataCache[userId]);
154 
155                         // get content items for the user
156                         $.ajax({
157                             url: sakai.config.URL.POOLED_CONTENT_SPECIFIC_USER,
158                             data: {
159                                 "page": 0,
160                                 "items": 20,
161                                 "userid": userId
162                             },
163                             success: function(data){
164                                 // Truncate long filenames
165                                 if (data && data.results) {
166                                     for (var item in data.results) {
167                                         if (data.results.hasOwnProperty(item)) {
168                                             if (data.results[item]["sakai:pooled-content-file-name"]) {
169                                                 data.results[item]["sakai:pooled-content-file-name"] = sakai.api.Util.applyThreeDots(data.results[item]["sakai:pooled-content-file-name"], 165, {
170                                                     max_rows: 1,
171                                                     whole_word: false
172                                                 }, "s3d-bold");
173                                             }
174                                         }
175                                     }
176                                 }
177 
178                                 // add user content to their data object
179                                 dataCache[userId].contentItems = data;
180 
181                                 // check user still has widget open before rendering results
182                                 if (open) {
183                                     togglePersonInfo($clickedEl);
184                                 }
185                             }
186                         });
187                     } else {
188                         dataCache[userId] = {
189                             displayName: sakai.api.i18n.getValueForKey('PRIVATE_USER', 'personinfo'),
190                             displayPicture: sakai.config.URL.USER_DEFAULT_ICON_URL,
191                             isPrivate: true
192                         };
193                         if (open) {
194                             togglePersonInfo($clickedEl);
195                         }
196                     }
197                 });
198             }
199         };
200 
201         // bind personinfo cancel
202         $personinfo_close.live("click", function(){
203             hidePersonInfo();
204         });
205 
206         // bind personinfo message button
207         $personinfo_message.live("click", function () {
208             var sendMessageUserObj = {};
209             sendMessageUserObj.uuid = userId;
210             sendMessageUserObj.username = sakai.api.User.getDisplayName(dataCache[userId]);
211             sendMessageUserObj.type = "user";
212             // initialize the sendmessage-widget
213             $(document).trigger('initialize.sendmessage.sakai', [sendMessageUserObj, false, false, null, null, null]);
214         });
215 
216         // bind personinfo request connection button
217         $personinfo_invite.live("click", function(){
218             $(document).trigger('initialize.addToContacts.sakai', [dataCache[userId]]);
219         });
220 
221         // bind personinfo request connection button
222         $personinfo_invited.live("click", function(){
223             sakai.api.User.acceptContactInvite(userId, function(success) {
224                 if (success) {
225                     $personinfo_invited.hide();
226                 }
227             });
228         });
229 
230         // bind hashchange to close dialog
231         $(window).bind("hashchange hashchanged.inbox.sakai", function(){
232             hidePersonInfo();
233         });
234 
235         // bind click trigger
236         $(".personinfo_trigger_click").live("click", function(){
237             doInit($(this));
238         });
239 
240         // bind addtocontact contact request
241         $(window).bind("sakai.addToContacts.requested", function(ev, userToAdd){
242             if (dataCache[userToAdd.userid]){
243                 dataCache[userToAdd.userid].connectionState = "PENDING";
244                 $personinfo_invite.hide();
245                 $personinfo_pending.show();
246             }
247         });
248 
249 
250         /////////////////////////////
251         // Initialization function //
252         /////////////////////////////
253 
254         /**
255          * Initialization function that is run when the widget is triggered.
256          * Determines which events to bind to and fetches user data if the widget
257          * is not already opened.
258          */
259         var doInit = function ($clickedEl) {
260             userId = $clickedEl.data("userid");
261 
262             // bind outside click to close widget
263             $(document).bind("click.personinfo_close", function (e) {
264                 var $clicked = $(e.target);
265                 // Check if one of the parents is the tooltip
266                 if (!$clicked.parents().is("#personinfo") && $personinfo_widget.is(":visible")) {
267                     hidePersonInfo();
268                 }
269             });
270 
271             if (!open && userId){
272                 open = true;
273                 fetchPersonInfo($clickedEl);
274             }
275         };
276     };
277 
278     sakai.api.Widgets.widgetLoader.informOnLoad("personinfo");
279 });
