 24 /*global $ */
 25 
 26 require(["jquery", "sakai/sakai.api.core"], function($, sakai) {
 27     
 28     /**
 29      * @name sakai_global.mygroups
 30      *
 31      * @class mygroups
 32      *
 33      * @description
 34      * Initialize the mygroups widget
 35      *
 36      * @version 0.0.1
 37      * @param {String} tuid Unique id of the widget
 38      */
 39     sakai_global.mygroups = function(tuid){
 40 
 41 
 42         /////////////////////////////
 43         // Configuration variables //
 44         /////////////////////////////
 45 
 46         var rootel = $("#" + tuid);
 47 
 48         // IDs
 49         var mygroupsList = "#mygroups_list";
 50         var mygroupsErrorNoSettings = "#mygroups_error_nosettings";
 51         var mygroupsCreateNewGroup = "#create_new_group_link";
 52         var ellipsisContainer = ".mygroups_ellipsis_container";
 53         var mygroupsItemsList = ".mygroup_items_list";
 54 
 55         var mygroups_error_class = "mygroups_error";
 56 
 57 
 58         ///////////////////////
 59         // Utility functions //
 60         ///////////////////////
 61 
 62         /**
 63          * Takes a set of json and renders the groups.
 64          * @param {Object} newjson group list object
 65          */
 66         var doRender = function(newjson){
 67             // Sort the groups by their name
 68             for (var group in newjson.entry) {
 69                 if (newjson.entry.hasOwnProperty(group)) {
 70                     newjson.entry[group]["sakai:group-title"] = sakai.api.Util.applyThreeDots(newjson.entry[group]["sakai:group-title"], $(".my_groups_widget .s3d-widget-content").width() - 50, {max_rows: 1,whole_word: false}, "s3d-bold");
 71                 }
 72             }
 73             sakai.api.Util.TemplateRenderer("mygroups_list_template", newjson, $(mygroupsList, rootel));
 74         };
 75 
 76         /**
 77          * Will initiate a request to the my groups service.
 78          */
 79         var doInit = function(){
 80             //get groups list info from me object, filter and then render groups
 81             doRender(sakai.api.Groups.getMemberships(sakai.data.me.groups));
 82         };
 83 
 84 
 85         ////////////////////
 86         // Event Handlers //
 87         ////////////////////
 88 
 89         $(".mygroups_create_new_group", rootel).die("click");
 90         $(".mygroups_create_new_group", rootel).live("click", function(ev){
 91             $(window).trigger("sakai.overlays.createGroup"); 
 92         });
 93 
 94         // Start the request
 95         doInit();
 96 
 97     };
 98 
 99     sakai.api.Widgets.widgetLoader.informOnLoad("mygroups");
100 });
