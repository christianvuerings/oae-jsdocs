 44     var sakaiCommunicationsAPI =  {
 45         /**
 46          * Sends a Sakai message to one or more users. If a group id is received, the
 47          * message is sent to users that are members of that group.
 48          *
 49          * @param {Array|String} to Array with the ids of the users or groups to post a
 50          *   message to or a String with one user or group id.
 51          * @param {String} meData Who the message is from - sakai.data.me (or equivolent profile)
 52          * @param {String} subject The subject for this message
 53          * @param {String} body The text that this message will contain
 54          * @param {String} [category="message"] The category for this message
 55          * @param {String} [reply] The id of the message you are replying on
 56          * @param {Function} [callback] A callback function which is executed at the end of the operation
 57          * @param {Boolean} [sendMail] True if a mail needs to be sent, False if no mail is needed. Unles specified false the default will be true and a mail will be sent
 58          * @param {Boolean|String} [context] String used in switch to set sakai:templatePath and sakai:templateParams
 59          * @param {Object} [optionalParams] Passed in when out of the group context to provide data necessary to send the message (Ids, titles) that can't be retrieved from the global object
 60          *
 61          */
 62         sendMessage : function(to, meData, subject, body, category, reply, callback, sendMail, context, optionalParams) {
 63 
 64             var toUsers = "";              // aggregates all message recipients
 65             var sendDone = false;          // has the send been issued?
 66 
 67             ///////////////////////
 68             // UTILITY FUNCTIONS //
 69             ///////////////////////
 70 
 71             /**
 72              * Adds the given userids (String or Array) to the current list of recipients
 73              * @param {Array|String} userids Either a single userid (String) or a list
 74              * of userids (Array) to be added to the current list of recipients
 75              * @return None
 76              */
 77             var addRecipient = function(userids) {
 78                 // append comma if the list already exists
 79                 if(toUsers) {
 80                     toUsers += ",";
 81                 }
 82                 if(typeof(userids) === "string" && $.trim(userids) !== "") {
 83                     userids = $.trim(userids);
 84                     toUsers += "internal:" + userids;
 85                 } else if(typeof(userids) === "object") {
 86                     toUsers += "internal:" + userids.join(",internal:");
 87                     toUsers = toUsers.replace(/internal\:\,/g, "");
 88                 }
 89             };
 90 
 91             var buildEmailParams = function(){
 92                 var toSend = {
 93                     "sakai:type": "smtp",
 94                     "sakai:sendstate": "pending",
 95                     "sakai:messagebox": "pending",
 96                     "sakai:to": toUsers,
 97                     "sakai:from": meData.user.userid,
 98                     "sakai:subject": subject,
 99                     "sakai:body": body,
100                     "sakai:category": "message",
101                     "_charset_": "utf-8"
102                 };
103 
104                 // These checks are needed to work in every area (created group or on group creation)
105                 var groupTitle = "";
106                 var groupId = "";
107                 if(sakai_global.group && sakai_global.group.groupData && sakai_global.group.groupData["sakai:group-title"]){
108                     groupTitle = sakai_global.group.groupData["sakai:group-title"];
109                 } else if(optionalParams && optionalParams.groupTitle){
110                     groupTitle = optionalParams.groupTitle;
111                 }
112                 if (sakai_global.group && sakai_global.group.groupData && sakai_global.group.groupData["sakai:group-id"]) {
113                     groupId = sakai_global.group.groupData["sakai:group-id"];
114                 } else if (optionalParams && optionalParams.groupId){
115                     groupId = optionalParams.groupId;
116                 }
117                 var sender = sakai_user.getDisplayName(meData.profile);
118                 switch(context){
119                     case "new_message":
120                         toSend["sakai:templatePath"] = "/var/templates/email/new_message";
121                         toSend["sakai:templateParams"] = "sender=" + sender +
122                         "|system=" + sakai_i18n.getValueForKey("SAKAI") + "|subject=" + subject + "|body=" + body + "|link=" + sakai_conf.SakaiDomain + sakai_conf.URL.INBOX_URL;
123                         break;
124                     case "group_invitation":
125                         toSend["sakai:templatePath"] = "/var/templates/email/group_invitation";
126                         toSend["sakai:templateParams"] = "sender=" + sender +
127                         "|system=" + sakai_i18n.getValueForKey("SAKAI") + "|name=" + groupTitle +
128                         "|body=" + body +
129                         "|link=" + sakai_conf.SakaiDomain + "/~" + groupId;
130                         break;
131                     case "shared_content":
132                         toSend["sakai:templatePath"] = "/var/templates/email/shared_content";
133                         toSend["sakai:templateParams"] = "sender=" + sender +
134                         "|system=" + sakai_i18n.getValueForKey("SAKAI") +
135                         "|body=" + body;
136                         break;
137                     case "contact_invitation":
138                         toSend["sakai:templatePath"] = "/var/templates/email/contact_invitation";
139                         toSend["sakai:templateParams"] = "sender=" + sender +
140                         "|system=" + sakai_i18n.getValueForKey("SAKAI") + "|body=" + body +
141                         "|link=" + sakai_conf.SakaiDomain + sakai_conf.URL.INVITATIONS_URL;
142                         break;
143                 }
144                 return toSend;
145             };
146 
147             var doSendMail = function(){
148                 // Basic message details
149                 var toSend = buildEmailParams();
150                 // Send message
151                 $.ajax({
152                     url: "/~" + sakai_util.safeURL(meData.user.userid) + "/message.create.html",
153                     type: "POST",
154                     data: toSend,
155                     success: function(data) {
156                         if ($.isFunction(callback)) {
157                             callback(true, data);
158                         }
159                     },
160                     error: function(xhr, textStatus, thrownError) {
161                         if ($.isFunction(callback)) {
162                             callback(false, xhr);
163                         }
164                     }
165                 });
166                 sendDone = true;
167             };
168 
169             var doSendMessage = function() {
170                 // Basic message details
171                 var toSend = {
172                     "sakai:type": "internal",
173                     "sakai:sendstate": "pending",
174                     "sakai:messagebox": "outbox",
175                     "sakai:to": toUsers,
176                     "sakai:from": meData.user.userid,
177                     "sakai:subject": subject,
178                     "sakai:body":body,
179                     "_charset_":"utf-8"
180                 };
181 
182                 // Message category
183                 if (category) {
184                     toSend["sakai:category"] = category;
185                 } else {
186                     toSend["sakai:category"] = "message";
187                 }
188 
189                 // See if this is a reply or not
190                 if (reply) {
191                     toSend["sakai:previousmessage"] = reply;
192                 }
193                 // Send message
194                 $.ajax({
195                     url: "/~" + sakai_util.safeURL(meData.user.userid) + "/message.create.html",
196                     type: "POST",
197                     data: toSend,
198                     success: function(data){
199                         if (sendMail) {
200                             doSendMail();
201                         }else if($.isFunction(callback)) {
202                             callback(true, data);
203                         }
204                     },
205                     error: function(xhr, textStatus, thrownError){
206                         if ($.isFunction(callback)) {
207                             callback(false, xhr);
208                         }
209                     }
210                 });
211                 // the send has been issued
212                 sendDone = true;
213             };
214 
215             //////////////////
216             // MAIN ROUTINE //
217             //////////////////
218 
219             var reqs = [];
220             if (typeof(to) === "string") {
221                 var id = to;
222                 to = [];
223                 to[0] = id;
224             }
225 
226             if (typeof(to) === "object") {
227                 for (i = 0; i < to.length; i++) {
228                     reqs[reqs.length] = {
229                         "url": "/~" + to[i] + "/public/authprofile.profile.json",
230                         "method": "GET"
231                     };
232                 }
233             } else {
234                 // unrecognized type
235                 debug.warn("sakai.api.Communication.sendMessage(): invalid argument ('to' not an Array or String).");
236 
237                 if ($.isFunction(callback)) {
238                     callback(false, xhr);
239                 }
240             }
241             sakai_server.batch(reqs, function(success, data) {
242                 if (success) {
243                     // array of recipients
244                     addRecipient(to);
245                     // send now if we have only a list of users ("thread" safe?)
246                     if (!sendDone) {
247                         doSendMessage();
248                     }
249                 }
250             });
251         },

253         deleteMessages : function(messages, hardDelete, callback) {
254             var requests = [],
255                 params = {},
256                 unreads = [];
257 
258             if (hardDelete) {
259                 params = {":operation": "delete"};
260             } else {
261                 params = {"sakai:messagebox": "trash"};
262             }
263             if (!$.isArray(messages)) {
264                 messages = [messages];
265             }
266             $.each(messages, function(i, val){
267                 if (!val.read || val.read === "false") {
268                     unreads.push(val);
269                 }
270                 var req = {
271                     "url": val.path || val._path,
272                     "method": "POST",
273                     "parameters": params
274                 };
275                 requests.push(req);
276             });
277             sakaiCommunicationsAPI.updateLocalCounts(unreads);
278             sakai_server.batch(requests, function(success, data) {
279                 if ($.isFunction(callback)) {
280                     if (success) {
281                         callback(true, data);
282                     } else {
283                         callback(false, {});
284                     }
285                 }
286             });
287         },

289         updateLocalCounts : function(messages) {
290             sakai_user.data.me.messages.unread -= $.grep(messages, function(message, index){
291                 return message.box === "inbox";
292             }).length;
293             if (sakai_user.data.me.messages.countOverview) {
294                 $.each(sakai_user.data.me.messages.countOverview.count, function(index, countObj){
295                     var catFilter = function(message, index){
296                         return (message.box === "inbox" && message.category === countObj.group);
297                     };
298                     var messageCount = $.grep(messages, catFilter).length;
299                     var count = {
300                         count: countObj.count - messageCount,
301                         group: countObj.group
302                     };
303                     sakai_user.data.me.messages.countOverview.count[index] = count;
304                 });
305             }
306             $(window).trigger("updated.messageCount.sakai");
307         },

309         markMessagesAsRead : function(messages, callback) {
310             var requests = [];
311 
312             if (! $.isArray(messages)){
313                 messages = [messages];
314             }
315 
316             $.each(messages, function(i, message) {
317                 var path = message.path;
318                 if (path.substring(0, 2) === "a:"){
319                     path = "~" + path.substring(2);
320                 }
321                 var req = {url: path + ".json", method: "POST", parameters: {"sakai:read": "true"}};
322                 requests.push(req);
323             });
324             sakai_server.batch(requests, function(success, data) {
325                 if (success) {
326                     $.each(messages, function(i, message) {
327                         message.read = true;
328                     });
329                     sakaiCommunicationsAPI.updateLocalCounts(messages);
330                     if ($.isFunction(callback)) {
331                         callback(true, data);
332                     }
333                 } else {
334                     if ($.isFunction(callback)) {
335                         callback(false, {});
336                     }
337                 }
338             });
339         },

341         /**
342          * Processes the messages from the server, stripping out everything we don't need
343          */
344         processMessages : function(results) {
345             var messages = [];//,
346             $.each(results, function(i, msg) {
347                 if (!$.isEmptyObject(msg)) {
348                     var newMsg = {};
349                     newMsg.replyAll = [];
350                     var userFrom = _.first(msg.userFrom);
351                     newMsg.from = {
352                         name:  userFrom.userid ? sakai_user.getDisplayName(userFrom) : sakai_util.Security.safeOutput(userFrom["sakai:group-title"]),
353                         picture: sakai_util.constructProfilePicture(userFrom),
354                         connectionState: userFrom["sakai:state"] ? userFrom["sakai:state"] : false,
355                         userObj : {
356                             uuid: userFrom.userid ? userFrom.userid : userFrom.groupid,
357                             username: userFrom.userid ? sakai_user.getDisplayName(userFrom) : sakai_util.Security.safeOutput(userFrom["sakai:group-title"]),
358                             type: userFrom.userid ? "user" : "group"
359                         }
360                     };
361                     if (newMsg.from.userObj.uuid !== sakai_user.data.me.user.userid) {
362                         newMsg.replyAll.push(newMsg.from.userObj);
363                     }
364                     newMsg.to = [];
365                     newMsg.toList = [];
366                     $.each(msg.userTo, function(i, user) {
367                         var tmpUsr = {
368                             name : user.userid ? sakai_user.getDisplayName(user) : sakai_util.Security.safeOutput(user["sakai:group-title"]),
369                             picture : sakai_util.constructProfilePicture(user),
370                             userObj : {
371                                 uuid: user.userid ? user.userid : user.groupid,
372                                 username: user.userid ? sakai_user.getDisplayName(user) : sakai_util.Security.safeOutput(user["sakai:group-title"]),
373                                 type: user.userid ? "user" : "group"
374                             }
375                         };
376                         if (user.userid !== sakai_user.data.me.user.userid) {
377                             newMsg.replyAll.push(tmpUsr.userObj);
378                         }
379                         newMsg.toList.push(tmpUsr.name);
380                         newMsg.to.push(tmpUsr);
381                     });
382                     // We are adding the div to force jQuery to interpret this
383                     // as html and not a selector (in case there are no tags
384                     // in the messsage body).
385                     msg["sakai:body"] = sakai_util.Security.safeOutput(msg["sakai:body"]);
386                     var bodyToAutolink = $("<div>"+sakai_util.Security.safeOutput(msg["sakai:body"]).replace(/\n/gi, "<br />")+"</div>");
387                     newMsg.body = bodyToAutolink.autolink().html();
388                     newMsg.body_nolinebreaks = $.trim(msg["sakai:body"].replace(/\n/gi, " "));
389                     newMsg.subject = msg["sakai:subject"];
390                     newMsg.box = msg["sakai:messagebox"];
391                     newMsg.category = msg["sakai:category"];
392                     newMsg.date = sakai_l10n.transformDateTimeShort(sakai_l10n.fromEpoch(msg["_created"], sakai_user.data.me));
393                     newMsg.timeago = $.timeago(new Date(msg["_created"]));
394                     newMsg.id = msg.id || msg['sakai:id'];
395                     newMsg.read = msg["sakai:read"];
396                     newMsg.path = msg['_path'];
397                     if (msg['_path'].substring(0,2) === 'a:') {
398                         newMsg.path = msg['_path'].replace('a:', '/~');
399                     }
400                     if (msg.previousMessage) {
401                         newMsg.previousMessage = sakaiCommunicationsAPI.processMessages([msg.previousMessage]);
402                         $.each(newMsg.previousMessage, function(i,val){
403                             newMsg.previousMessage = val;
404                         });
405                     }
406                     messages.push(newMsg);
407                 }
408             });
409             return messages;
410         },

412         /**
413         * Gets all messages from a box
414         *
415         * @param {String} box The name of the box to get messages from
416         * @param {String} category The type of messages to get from the box
417         * @param {Number} messagesPerPage The number of messages to fetch
418         * @param {Number} currentPage The page offset to start from
419         * @param {String} sortBy The name of the field to sort on
420         * @param {String} sortOrder Sort messages asc or desc
421         * @param {Function} callback The function that will be called on completion
422         * @param {Boolean} doProcessing process the messages after they come back to make them easier to deal with
423         *                               defaults to true
424         * @param {Boolean} doFlip Flip the to and from
425         */
426         getAllMessages : function(box, category, search, messagesPerPage, currentPage, sortBy, sortOrder, callback, doProcessing, doFlip) {
427             var parameters = {
428                 "box": box,
429                 "items": messagesPerPage,
430                 "page": currentPage,
431                 "sortOn": sortBy,
432                 "sortOrder": sortOrder
433             };
434 
435             // Set up optional, additional params
436             if (search) {
437                 parameters.q = search;
438             }
439             if (category) {
440                 parameters.category = category;
441             }
442 
443             // Set the base URL for the search
444             if (search) {
445                 url = sakai_conf.URL.MESSAGE_BOXCATEGORY_SERVICE;
446             } else {
447                 url = sakai_conf.URL.MESSAGE_BOXCATEGORY_ALL_SERVICE;
448             }
449             $.ajax({
450                 url: url,
451                 data: parameters,
452                 cache: false,
453                 success: function(data){
454                     if (box === "inbox") {
455                         sakaiCommunicationsAPI.getUnreadMessageCount(box, function() {
456                             sakaiCommunicationsAPI.getUnreadMessagesCountOverview(box, function(){
457                                 $(window).trigger("updated.messageCount.sakai");
458                             }, false);
459                         });
460                     }
461                     if (doProcessing !== false) {
462                         data.results = sakaiCommunicationsAPI.processMessages(data.results, doFlip);
463                     }
464                     if ($.isFunction(callback)) {
465                         callback(true, data);
466                     }
467                 },
468                 error: function(xhr, textStatus, thrownError){
469                     if ($.isFunction(callback)) {
470                         callback(false, {});
471                     }
472                 }
473             });
474         },

476         /**
477          * Retrieve a message based on its box and its id. This function will also
478          * include the user that has sent the message
479          * @param {String} id            Unique id of the message
480          * @param {String} box           Message box the message lives in. The possible options are inbox, outbox and trash
481          * @param {Object} meData        Me object
482          * @param {Function} callback    Function to call once the message has been retrieved
483          */
484         getMessage : function(id, box, meData, callback){
485             var url = "/~" + sakai_util.safeURL(sakai_user.data.me.user.userid) + "/message/" + box + "/" + id + ".json";
486             $.ajax({
487                 url: url,
488                 cache: false,
489                 dataType: "json",
490                 success: function(data){
491                     var useridToLookup = data["sakai:from"];
492                     if (data["sakai:from"] === meData.user.userid){
493                         useridToLookup = data["sakai:to"].split(":")[1];
494                     }
495                     sakai_user.getUser(useridToLookup, function(success,profiledata){
496                         if (data["sakai:from"] === meData.user.userid){
497                             data.userFrom = [meData.profile];
498                             data.userTo = [profiledata];
499                         } else {
500                             data.userFrom = [profiledata];
501                             data.userTo = [meData.profile];
502                         }
503                         callback(sakaiCommunicationsAPI.processMessages([data])[0]);
504                     });
505                 },
506                 error: function(xhr, textStatus, thrownError){
507                     callback(false);
508                 }
509             });
510         },

539         /**
540          * Gets a count of the unread messages in a box belonging
541          * to the current user
542          */
543         getUnreadMessageCount : function(box, callback, category) {
544             if (box === "inbox" && sakai_user.data.me.messages.unread) {
545                 if ($.isFunction(callback)) {
546                     callback(true, sakai_user.data.me.messages.unread);
547                 }
548             } else {
549                 var url = "/~" + sakai_util.safeURL(sakai_user.data.me.user.userid) + "/message.count.json?filters=sakai:messagebox,sakai:read&values=" + box + ",false&groupedby=sakai:category";
550                 $.ajax({
551                     url: url,
552                     cache: false,
553                     success: function(data){
554                         var count = 0;
555                         if (category){
556                             /*
557                              * Data format for this return is:
558                              * {"count":[{"group":"message","count":3},{"group":"invitation","count":2}]}
559                              */
560                             if (data.count && data.count.length){
561                                 for (var i = 0; i < data.count.length; i++){
562                                     if (data.count[i].group && data.count[i].group === category && data.count[i].count){
563                                         count = data.count[i].count;
564                                     }
565                                 }
566                             }
567                         } else if (data.count && data.count[0] && data.count[0].count) {
568                             count = data.count[0].count;
569                         }
570                         if (box === "inbox") {
571                             sakai_user.data.me.messages.unread = count;
572                         }
573                         if ($.isFunction(callback)) {
574                             callback(true, count);
575                         }
576                     },
577                     error: function(xhr, textStatus, thrownError) {
578                         if ($.isFunction(callback)) {
579                             callback(false,{});
580                         }
581                     }
582                 });
583             }
584         },
