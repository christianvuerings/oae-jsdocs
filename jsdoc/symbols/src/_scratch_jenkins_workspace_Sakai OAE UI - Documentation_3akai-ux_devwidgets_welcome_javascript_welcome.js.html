 19 // load the master sakai object to access all Sakai OAE API methods
 20 require(["jquery", "sakai/sakai.api.core"], function($, sakai) {
 21 
 22     /**
 23      * @name sakai_global.welcome
 24      *
 25      * @class welcome
 26      *
 27      * @version 0.0.1
 28      * @param {String} tuid Unique id of the widget
 29      * @param {Boolean} showSettings Show the settings of the widget or not
 30      */
 31     sakai_global.welcome = function (tuid, showSettings) {
 32 
 33         var $rootel = $('#' + tuid);
 34         var $welcomeWidget = $('.welcome_widget', $rootel);
 35         var welcomeTemplate = "welcome_template";
 36 
 37         var addBinding = function() {
 38             $welcomeWidget.on('click', 'button[data-trigger]', function() {
 39                 $(window).trigger($(this).attr('data-trigger'));
 40             });
 41         };
 42 
 43         var renderWidget = function(){
 44             $welcomeWidget.html(sakai.api.Util.TemplateRenderer(welcomeTemplate, {
 45                 "anon": sakai.data.me.user.anon || false
 46             }));
 47         };
 48 
 49         renderWidget();
 50         addBinding();
 51 
 52     };
 53 
 54     sakai.api.Widgets.widgetLoader.informOnLoad("welcome");
 55 });
