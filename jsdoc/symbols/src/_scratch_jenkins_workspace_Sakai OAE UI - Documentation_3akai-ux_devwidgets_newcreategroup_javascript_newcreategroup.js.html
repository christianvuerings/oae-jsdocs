 19 require(["jquery", "sakai/sakai.api.core"], function($, sakai) {
 20 
 21     /**
 22      * @name sakai_global.newcreategroup
 23      *
 24      * @class newcreategroup
 25      *
 26      * @description
 27      * newcreategroup widget
 28      *
 29      * @version 0.0.1
 30      * @param {String} tuid Unique id of the widget
 31      * @param {Boolean} showSettings Show the settings of the widget or not
 32      */
 33     sakai_global.newcreategroup = function(tuid, showSettings, widgetData){
 34 
 35     /////////////////////////////
 36     // Configuration variables //
 37     /////////////////////////////
 38     
 39     var $rootel = $("#" + tuid);
 40 
 41     // Containers
 42     var $newcreategroupContainer = $("#newcreategroup_container", $rootel);
 43     var $newcreategroupGroupMembersNoneAddedContainer = $("#newcreategroup_group_members_none_added_container", $rootel);
 44     var $newcreategroupMembersAddedContainer = $("#newcreategroup_group_members_added_container", $rootel);
 45 
 46     // Elements
 47     var $newcreategroupCancelCreateButton = $("#newcreategroup_cancel_create", $rootel);
 48     var $newcreategroupGroupTitle = $("#newcreategroup_title", $rootel);
 49     var $newcreategroupSuggestedURL = $("#newcreategroup_suggested_url", $rootel);
 50     var $newcreategroupGroupDescription = $("#newcreategroup_description", $rootel);
 51     var $newcreategroupGroupTags = $("#newcreategroup_tags", $rootel);
 52     var $newcreategroupSuggestedURLBase = $("#newcreategroup_suggested_url_base", $rootel);
 53     var $newcreategroupCanBeFoundIn = $("#newcreategroup_can_be_found_in", $rootel);
 54     var $newcreategroupGroupMembership = $("#newcreategroup_membership", $rootel);
 55     var $newcreategroupAddPeople = $(".newcreategroup_add_people", $rootel);
 56     var newcreategroupMembersMessage = "#newcreategroup_members_message";
 57     var $newcreategroup_members_message_template_unprocessed = $("#newcreategroup_members_message_template_unprocessed", $rootel);
 58 
 59     // Forms
 60     var $newcreategroupGroupForm = $("#newcreategroup_group_form", $rootel);
 61 
 62     // Templates
 63     var newcreategroupMembersSelectedTemplate = "newcreategroup_group_members_selected_template";
 64     var newcreategroupMembersMessageTemplate = "newcreategroup_members_message_template";
 65 
 66     var selectedUsers = {},
 67         currentTemplate = false,
 68         templatePath = false,
 69         translatedRoles = {};
 70 
 71     var renderShareMessage = function(){
 72         $(newcreategroupMembersMessage, $rootel).html(sakai.api.Util.TemplateRenderer(newcreategroupMembersMessageTemplate, {
 73             "creatorName" : sakai.api.User.getDisplayName(sakai.data.me.profile),
 74             "groupName" : sakai.api.Security.safeOutput($newcreategroupGroupTitle.val() || ""),
 75             "link": sakai.config.SakaiDomain + "/~" + sakai.api.Util.makeSafeURL($newcreategroupSuggestedURL.val() || "")
 76         }));
 77     };
 78 
 79     var getTranslatedRoles = function() {
 80         if (currentTemplate.roles) {
 81             var roles = [];
 82             $.each(currentTemplate.roles, function(i,role) {
 83                 translatedRoles[role.id] = role.title;
 84             });
 85         }
 86     };
 87 
 88     var createUsersToAddObject = function() {
 89         var users = [];
 90         $.each(selectedUsers, function(index, item){
 91             users.push({
 92                 "name": item.name,
 93                 "firstName": item.firstName || item.name,
 94                 "userid": item.userid,
 95                 "role": item.permission,
 96                 "roleString": translatedRoles[item.permission]
 97             });
 98         });
 99         users.push({
100             "name": sakai.api.User.getDisplayName(sakai.data.me.profile),
101             "userid": sakai.data.me.user.userid,
102             "role": currentTemplate.creatorRole,
103             "roleString": translatedRoles[currentTemplate.creatorRole],
104             "creator": true
105         });
106         return users;
107     };
108 
109     var checkDefaultContentAdded = function(contentToAdd, count){
110         return !$.isArray(contentToAdd) || contentToAdd.length - 1 === count;
111     };
112 
113     var setDefaultContent = function(groupid){
114         var contentToAdd = $.bbq.getState("contentToAdd");
115         if(contentToAdd.length > 1 && !$.isArray(contentToAdd)){
116             contentToAdd = contentToAdd.split(",");
117         }
118         var count = 0;
119         $.each(contentToAdd, function(i, contentId){
120             sakai.api.Content.addToLibrary(contentId, groupid, false, function(contentId, entityId) {
121                 if(checkDefaultContentAdded(contentToAdd, count)){
122                     window.location = "/~" + groupid;
123                 } else {
124                     count++;
125                 }
126             });
127         });
128     };
129 
130     /**
131      * Create a simple group and execute the tagging and membership functions
132      */
133     var doCreateSimpleGroup = function() {
134         sakai.api.Util.progressIndicator.showProgressIndicator(
135             sakai.api.i18n.getValueForKey('CREATING_YOUR_GROUP', 'newcreategroup').replace(/\$\{type\}/, sakai.api.i18n.getValueForKey(currentTemplate.title)),
136             sakai.api.i18n.getValueForKey('PROCESSING_GROUP', 'newcreategroup')
137         );
138         var grouptitle = $newcreategroupGroupTitle.val() || "";
139         var groupdescription = $newcreategroupGroupDescription.val() || "";
140         var groupid = sakai.api.Util.makeSafeURL($newcreategroupSuggestedURL.val(), "-");
141         var grouptags = sakai.api.Util.AutoSuggest.getTagsAndCategories( $newcreategroupGroupTags, true );
142         var users = createUsersToAddObject();
143         var subject = sakai.api.i18n.getValueForKey("USER_HAS_ADDED_YOU_AS_A_ROLE_TO_THE_GROUP_GROUPNAME", "newcreategroup").replace("<\"Role\">", "${role}");
144         var body = $.trim($newcreategroup_members_message_template_unprocessed.text().replace("<\"Role\">", "${role}").replace("<\"First Name\">", "${firstName}"));
145         var joinable = $newcreategroupGroupMembership.val();
146         var visible = $newcreategroupCanBeFoundIn.val();
147         sakai.api.Groups.createGroup(groupid, grouptitle, groupdescription, grouptags, users, joinable, visible, templatePath, subject, body, sakai.data.me, function(success, groupData, nameTaken){
148             if (success) {
149                 if($.bbq.getState("contentToAdd")){
150                     setDefaultContent(groupid);
151                 } else {
152                     window.location = "/~" + groupid;
153                 }
154             } else {
155                 $newcreategroupContainer.find("select, input, textarea:not([class*='as-input']), button").removeAttr("disabled");
156                 sakai.api.Util.progressIndicator.hideProgressIndicator();
157                 sakai.api.Util.notification.show(sakai.api.i18n.getValueForKey("AN_ERROR_OCCURRED", "newcreategroup"),
158                     sakai.api.i18n.getValueForKey("GROUP_NOT_SUCCESSFULLY_CREATED", "newcreategroup").replace(/\$\{title\}/, grouptitle),
159                     sakai.api.Util.notification.type.ERROR);
160             }
161         });
162     };
163 
164     /**
165      * Intialize the add people widget
166      * @param {Boolean} openDialog Whether you want to open the dialog
167      */
168     var initializeAddPeople = function(openDialog) {
169         $(document).trigger('init.addpeople.sakai', {
170             editingGroup: false,
171             openDialog: openDialog
172         });
173     };
174 
175     /**
176      * Add binding to the elements and validate the forms on submit
177      */
178     var addBinding = function(){
179         var validateOpts = {
180             'methods': {
181                 'uniquegroupname': {
182                     'method': function(value, element) {
183                         return !sakai.api.Groups.checkIfGroupExists(value);
184                     },
185                     'text': sakai.api.i18n.getValueForKey('THIS_GROUP_HAS_BEEN_TAKEN', 'newcreategroup')
186                 }
187             },
188             rules: {
189                 newcreategroup_title: {
190                     maxlength: 255
191                 },
192                 newcreategroup_suggested_url: {
193                     maxlength: 255            
194                 }
195             },
196             submitHandler: function(form){
197                 $newcreategroupContainer.find("select, input, textarea:not([class*='as-input']), button").attr("disabled", "disabled");
198                 doCreateSimpleGroup();
199             }
200         };
201         // Initialize the validate plug-in
202         sakai.api.Util.Forms.validate($newcreategroupGroupForm, validateOpts, true);
203         sakai.api.Util.AutoSuggest.setupTagAndCategoryAutosuggest($newcreategroupGroupTags, null, $(".list_categories", $rootel));
204         $newcreategroupGroupTitle.bind("keyup", function(){
205             var suggestedURL = sakai.api.Util.makeSafeURL($(this).val().toLowerCase(), "-");
206             $newcreategroupSuggestedURL.val(suggestedURL);
207             $newcreategroupSuggestedURLBase.attr("title", window.location.protocol + "//" + window.location.host + "/~" + suggestedURL);
208             renderShareMessage();
209         });
210 
211         $newcreategroupSuggestedURL.bind("blur", function(){
212             var suggestedURL = sakai.api.Util.makeSafeURL($(this).val(), "-");
213             $newcreategroupSuggestedURL.val(suggestedURL);
214             renderShareMessage();
215         });
216 
217         $(document).on('click', '.newcreategroup_add_people', initializeAddPeople);
218 
219         // We also need to intialize the add people widget but not show the dialog
220         // when there is a query parameter called 'members'
221         if ($.bbq.getState('members')) {
222             initializeAddPeople(false);
223         }
224 
225     };
226 
227     /**
228      * Initialize the create group widget
229      */
230     var doInit = function(){
231         sakai.api.Groups.getTemplate(widgetData.category, widgetData.id, function(success, template, templates) {
232             if (success) {
233                 currentTemplate = $.extend(true, {}, template);
234                 currentTemplate.roles = sakai.api.Groups.getRoles(currentTemplate, true);
235                 getTranslatedRoles();
236                 templatePath = '/var/templates/worlds/' + widgetData.category + '/' + widgetData.id;
237                 $('.newcreategroup_template_name', $rootel).text(sakai.api.i18n.getValueForKey(currentTemplate.title));
238                 if (widgetData.singleTemplate === true) {
239                     $newcreategroupCancelCreateButton.hide();
240                 }
241                 $newcreategroupSuggestedURLBase.text(
242                     sakai.api.Util.applyThreeDots(window.location.protocol + '//' + window.location.host + '/~', 105, {
243                         'middledots': true
244                     }, null, true)
245                 );
246                 $newcreategroupSuggestedURLBase.attr('title', window.location.protocol + '//' + window.location.host + '/~');
247 
248                 var category = false;
249                 for (var i = 0; i < templates.length; i++) {
250                     if (templates[i].id === widgetData.category) {
251                         category = templates[i];
252                         break;
253                     }
254                 }
255                 var defaultaccess = currentTemplate.defaultaccess || sakai.config.Permissions.Groups.defaultaccess;
256                 var defaultjoin = currentTemplate.defaultjoin || sakai.config.Permissions.Groups.defaultjoin;
257 
258                 $('#newcreategroup_can_be_found_in option[value="' + defaultaccess + '"]', $rootel).attr('selected', 'selected');
259                 $('#newcreategroup_membership option[value="' + defaultjoin + '"]', $rootel).attr('selected', 'selected');
260 
261                 sakai_global.selecttemplate.currentTemplate = currentTemplate;
262 
263                 $newcreategroupContainer.show();
264                 addBinding();
265             } else {
266                 debug.error('Could not get the group template');
267             }
268         });
269     };
270 
271     $newcreategroupCancelCreateButton.bind("click", function(){
272         $.bbq.pushState({"_r": Math.random()});
273     });
274 
275     $(window).bind("toadd.addpeople.sakai", function(ev, initTuid, users){
276         selectedUsers = $.extend(true, {}, users);
277         $newcreategroupMembersAddedContainer.html(sakai.api.Util.TemplateRenderer(newcreategroupMembersSelectedTemplate, {
278             "users": selectedUsers,
279             "roles": currentTemplate.roles,
280             "sakai": sakai
281         }));
282         var count = 0;
283         for (var item in selectedUsers) {
284             count++;
285         }
286         if (count) {
287             renderShareMessage();
288             $newcreategroupGroupMembersNoneAddedContainer.hide();
289             $newcreategroupMembersAddedContainer.show();
290         } else {
291             $newcreategroupGroupMembersNoneAddedContainer.show();
292             $newcreategroupMembersAddedContainer.hide();
293         }
294     });
295 
296     doInit();
297 
298     };
299 
300     sakai.api.Widgets.widgetLoader.informOnLoad("newcreategroup");
301
