 19 // load the master sakai object to access all Sakai OAE API methods
 20 require(["jquery", "sakai/sakai.api.core"], function($, sakai) {
 21 
 22     /**
 23      * @name sakai.worldsettings
 24      *
 25      * @class worldsettings
 26      *
 27      * @description
 28      * WIDGET DESCRIPTION
 29      *
 30      * @version 0.0.1
 31      * @param {String} tuid Unique id of the widget
 32      * @param {Boolean} showSettings Show the settings of the widget or not
 33      */
 34     sakai_global.worldsettings = function (tuid, showSettings) {
 35 
 36         /////////////////////////////
 37         // Configuration variables //
 38         /////////////////////////////
 39         var $rootel = $("#" + tuid);
 40 
 41         // Data Items in the Form
 42         var worldsettingsTitle = "#worldsettings_title";
 43         var worldsettingsDescription = "#worldsettings_description";
 44         var worldsettingsTags =  "#worldsettings_tags";
 45         var $worldsettingsTags = false;
 46         var worldsettingsCanBeFoundIn = "#worldsettings_can_be_found_in";
 47         var worldsettingsMembership = "#worldsettings_membership";
 48 
 49         // Page Structure Elements
 50         var $worldsettingsContainer = $("#worldsettings_container", $rootel);
 51         var $worldsettingsDialog = $(".worldsettings_dialog", $rootel);
 52         var $worldsettingsForm = $("#worldsettings_form", $rootel);
 53         var $worldsettingsApplyButton = $("#worldsettings_apply_button");
 54 
 55         var visibility = "",
 56             worldId = "";
 57 
 58         var showWarning = function(){
 59             var newVisibility = $(worldsettingsCanBeFoundIn);
 60             var newVisibilityVal = $.trim(newVisibility.val());
 61             var oldVisibilityVal = sakai_global.group.groupData['sakai:group-visible'];
 62             var oldVisibilityIndex = parseInt(newVisibility.find("option[value='" + sakai_global.group.groupData["sakai:group-visible"] + "']").attr("index"), 10);
 63             if (sakai_global.group.groupData['sakai:group-visible'] === newVisibilityVal || parseInt(newVisibility.attr('selectedIndex'), 10) > oldVisibilityIndex || newVisibilityVal === 'members-only' || oldVisibilityVal === 'public') {
 64                 $worldsettingsForm.submit();
 65             } else {
 66                 $("#worldsettings_warning_container_text").html(sakai.api.Util.TemplateRenderer("worldsettings_warning_container_text_template", {
 67                     "visibility": newVisibilityVal,
 68                     "group": sakai_global.group.groupData['sakai:group-title']
 69                 }));
 70 
 71                 sakai.api.Util.Modal.open('#worldsettings_warning_container');
 72             }
 73         };
 74 
 75         ////////////////////
 76         // Event Handlers //
 77         ////////////////////
 78 
 79         var handleSubmit = function(form) {
 80             $worldsettingsContainer.find("select, input").attr("disabled","disabled");
 81             var worldTitle = $.trim($(worldsettingsTitle).val());
 82             var worldDescription = $.trim($(worldsettingsDescription).val());
 83             var foundIn = $.trim($(worldsettingsCanBeFoundIn).val());
 84             var membership = $.trim($(worldsettingsMembership).val());
 85 
 86             var worldTags = sakai.api.Util.AutoSuggest.getTagsAndCategories( $worldsettingsTags, true );
 87 
 88             var worldData = {
 89                 "sakai:group-title":  worldTitle,
 90                 "sakai:group-description": worldDescription,
 91                 "sakai:group-visible": foundIn,
 92                 "sakai:group-joinable": membership
 93             };
 94 
 95             sakai.api.Groups.updateGroupProfile(worldId, worldData, worldTags, sakai_global.group.groupData, function( success ) {
 96                 $worldsettingsContainer.find("select, input").removeAttr("disabled");
 97 
 98                 $(window).trigger('updatedTitle.worldsettings.sakai', worldTitle);
 99                 sakai.api.Util.notification.show($("#worldsettings_success_title").html(), $("#worldsettings_success_body").html());
100                 sakai.api.Util.Modal.close($worldsettingsDialog);
101                 sakai.api.Util.Modal.close('#worldsettings_warning_container');
102             });
103         };
104 
105         var bindEvents = function(worldId) {
106             $worldsettingsApplyButton.die("click").live("click", function() {
107                 showWarning();
108             });
109             $("#worldsettings_proceedandapply").die("click");
110             $("#worldsettings_proceedandapply").live("click", function(){
111                 $worldsettingsForm.submit();
112             });
113 
114             var validateOpts = {
115                 submitHandler: handleSubmit
116             };
117             // Initialize the validate plug-in
118             sakai.api.Util.Forms.validate($worldsettingsForm, validateOpts, true);
119         };
120 
121         /////////////////////////////
122         // Initialization function //
123         /////////////////////////////
124 
125         var renderWorldSettings = function(){
126             var profile = sakai_global.group.groupData;
127             $("#worldsettings_form_container").html(sakai.api.Util.TemplateRenderer("worldsettings_form_template",{
128                 "title": profile['sakai:group-title'],
129                 "description": profile['sakai:group-description'],
130                 "foundin": profile['sakai:group-visible'],
131                 "membership": profile['sakai:group-joinable']
132             }));
133             var initialTagsValue = profile[ "sakai:tags" ] ? profile[ "sakai:tags" ] : false;
134             $worldsettingsTags = $( worldsettingsTags );
135             sakai.api.Util.AutoSuggest.setupTagAndCategoryAutosuggest( $worldsettingsTags, null, $(".list_categories", $rootel), initialTagsValue );
136         };
137 
138         /**
139          * Initialization function DOCUMENTATION
140          */
141         var doInit = function (_worldId) {
142             worldId = _worldId;
143             renderWorldSettings();
144             bindEvents();
145             sakai.api.Util.Modal.setup($worldsettingsDialog, {
146                 modal: true,
147                 overlay: 20,
148                 toTop: true,
149                 zIndex: 3000
150             });
151             sakai.api.Util.Modal.setup('#worldsettings_warning_container', {
152                 modal: true,
153                 overlay: 20,
154                 toTop: true,
155                 zIndex: 4000
156             });
157             sakai.api.Util.Modal.open($worldsettingsDialog);
158         };
159 
160         // run the initialization function when the widget object loads
161         $(document).on('init.worldsettings.sakai', function(e, _worldId) {
162             doInit(_worldId);
163         });
164     };
165 
166     // inform Sakai OAE that this widget has loaded and is ready to run
167     sakai.api.Widgets.widgetLoader.informOnLoad("worldsettings");
168 });
